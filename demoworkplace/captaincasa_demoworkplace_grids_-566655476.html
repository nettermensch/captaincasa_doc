The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Independent Application Marking

Select items of the grid by click/ shift-click/ ctrl-click. The behavior is:
1. every 5th item is not select-able
2. the selection is not changed when navigating with the keyboard.


By default the grid manages selection issues "on its own" - the grid row, which the user currently navigates to, is the one selected. - Sometimes this automatic selection does not fit to the corresponding scenario.

Examples:
1. you want the user to navigate (e.g. with the keyboard) through the grid items but you do not want to update the selection.
2. you want the user to navigate through all items, but you have certain items which are not "select-able" from application point of view. (In this example every 5th item is not "select-able")

Consequence: you leave the grid selection as it is - it highlights the item the user just navigates to. But: you define some own "marking" on item level: the marking comes with some background coloring of the item.
So the marking is some application-marking - and is decoupled from the current navigation status.

The grid provides the information if the user pressed ctrl or shift when navigating through the grid, so you can apply some corresponding multiple marking.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoGridMarking}")

/*
 * By default the grid manages selection isseus "on its own" - the grid row, which
 * the user currently navigates to, it the one selected. There is a corresponding
 * visual selection of the row. - Sometimes this automatic selection does not
 * fit to the corresponding scenario, esp. if you want to define some rows not to be
 * selectable: in this case you want the user still to navigate to this row (e.g.
 * with keyboard) but the navigation should not automatically be bound to an
 * "application selection" of the correspdongin row.
 * Consequence: you leave the grid selection as it is - it highlights the item
 * the user just navigates to. But: you define some own "marking" in item level:
 * the marking comes with some background coloring of the item.
 */
public class DemoGridMarking
    extends DemoPageBean 
    implements Serializable
{
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {    
        /*
         * Definition is marking is possible or not.
         */
        boolean m_markable = true;
        
        /*
         * Application marking.
         */
        boolean m_marked = false;
        public boolean isMarked() { return m_marked; }
        public void setMarked(boolean marked) 
        {
            if (m_markable == false) return;
            m_marked = marked; 
        }

        String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { this.m_lastName = value; }

        String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { this.m_firstName = value; }
        
        /*
         * The background painting of a row - following the marked-indicator.
         * And: the last selected item is marked a bit different to the previous
         * selected items. The BGPAINT attribute of the cell controls binds
         * to this value (e.g. LABEL-BGPAINT = ".{bgpaint}").
         */
        public String getBgpaint()
        {
            int myIndex = m_grid.getItems().indexOf(this);
            if (m_marked == true)
            {
                if (myIndex == m_firstShiftClickIndex && m_secondShiftClickIndex >= 0)
                    return "rectangle(0,0,100%,100%,#FF6600A0,#FF660070,vertical)";
                else
                    return "rectangle(0,0,100%,100%,#FF660060,#FF660030,vertical)";
            }
            else
                return null;
        }
        
        @Override
        /*
         * This is the normal selection - representing the current navigation
         * of the user. No marking is done here.
         */
        public void onRowSelect()
        {
        }
        
        @Override
        /*
         * The grid is used in the mode FIXGRID-SINGLECICKEXECUTE. It is executed
         * when the user (single-)clicks into one row or when the user presses the
         * return key within one row. It is NOT executed when the user navigates
         * with the keyboard up/down.
         */
        public void onRowExecute()
        {
            int myIndex = m_grid.getItems().indexOf(this);
            if (m_grid.getShiftPressedOfLastSelection() == true)
            {
                // shift clicked
                if (m_firstShiftClickIndex >= 0)
                {
                    if (m_secondShiftClickIndex >= 0)
                        applyFromToMarked(m_firstShiftClickIndex,m_secondShiftClickIndex,false);
                    m_secondShiftClickIndex = myIndex;
                    applyFromToMarked(m_firstShiftClickIndex,m_secondShiftClickIndex,true);
                }
                else
                {
                    m_firstShiftClickIndex = myIndex;
                }
            }
            else if (m_grid.getCtrlPressedOfLastSelection())
            {
                // ctrl clicked
                applyOneMarked(this);
            }
            else
            {
                // "default"
                applyOnlyOneMarked(this);
            }
        }
        
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    int m_firstShiftClickIndex = -1;
    int m_secondShiftClickIndex = -1;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridMarking(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        // fill grid - every 5th item is not markable
        for (int i=0; i<300; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First " + i);
            gi.setLastName("Last " + i);
            if (i%5 == 0)
                gi.m_markable = false;
            m_grid.getItems().add(gi);
        }
    }

    // ------------------------------------------------------------------------

    /*
     * Detail information for the bottom textarea of the screen - summary
     * of all marked items
     */
    public String getDetail()
    {
        StringBuffer sb = new StringBuffer();
        for (GridItem gi: m_grid.getItems())
        {
            if (gi.isMarked() == true)
            {
                sb.append(gi.getLastName() + ", " + gi.getFirstName() + "\n");
            }
        }
        return sb.toString();
    }
    
    private void applyOneMarked(GridItem gi)
    {
        gi.setMarked(!gi.m_marked);
    }
    
    private void applyOnlyOneMarked(FIXGRIDItem item)
    {
        m_firstShiftClickIndex = -1;
        m_secondShiftClickIndex = -1;
        int index = 0;
        for (GridItem gi: m_grid.getItems())
        {
            if (gi == item)
            {
                gi.setMarked(true);
                m_firstShiftClickIndex = index;
            }
            else
            {
                gi.setMarked(false);
            }
            index++;
        }
    }
    
    private void applyFromToMarked(int index1, int index2, boolean marked)
    {
        int from = index1;
        int to = index2;
        if (from > to)
        {
            to = index1;
            from = index2;
        }
        for (int i=from; i<=to; i++)
        {
            m_grid.getItems().get(i).setMarked(marked);
        }
    }
}
