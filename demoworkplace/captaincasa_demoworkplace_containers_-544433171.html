The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Scroll Pane - Extended Functions
  • Synchronized Scrolling (Java Client only)

Scroll the central scroll pane - either the scroll pane above or the scroll pane left will scroll as well.

You may use the attributes SCROLLREFERENCEHORIZONTAL and/or SCROLLREFERENCEVERTICAL in order to bind the own scrolling state of a SCROLLPANE to the scrolling of another SCROLLPANE.

Please pay attention: you must assure that the sizing of the corresponding SCROLLPANEs is correct.

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoScrollPaneReference}")

public class DemoScrollPaneReference
    extends PageBean
    implements Serializable
{
    String m_bgpaintVertical = "";
    public String getBgpaintVertical() { return m_bgpaintVertical; }
    public void setBgpaintVertical(String value) { this.m_bgpaintVertical = value; }

    String m_bgpaintHorizontal = "";
    public String getBgpaintHorizontal() { return m_bgpaintHorizontal; }
    public void setBgpaintHorizontal(String value) { this.m_bgpaintHorizontal = value; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    private IListener m_listener;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoScrollPaneReference(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<20; i++)
        {
            int y = i * 100;
            if (i != 0) m_bgpaintVertical += "line(0,"+y+",100,"+y+",1,#000000);";
            m_bgpaintVertical += "write(5,"+(y+5)+","+y+",lefttop);";
            int x = i * 100;
            if (i != 0) m_bgpaintHorizontal += "line("+x+",0,"+x+",100,1,#000000);";
            m_bgpaintHorizontal += "write("+(x+5)+",5,"+x+",lefttop);";
        }
    }

    public String getPageName() { return "/workplace/demoscrollpanereference.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoScrollPaneReference}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
