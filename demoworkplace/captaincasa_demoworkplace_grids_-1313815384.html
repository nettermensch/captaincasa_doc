The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Usage variant management
  • Usage variants with "Bean data grid"

The page bean component CCDataGridView2 also provides an implementation of the interface "IExtractApply" - that is the base for the usage variant management.

The data that is stored per usage variant is:

1. all the runtime configuration data of the grid itself (column sequence, widths, sorting)
2. all the filter definitions

By placing component USAGEVARIANTBUTTON into the layout and binding it to the CCDataGridView2 instance, the variant management is activated.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.ccaddons.pbc.usagevariants.CCUsageVariantButton;
import org.eclnt.ccee.usagevariants.UsageVariantManager;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoPBCCCUsageVariantDgv2}")

public class DemoPBCCCUsageVariantDgv2
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataGridView2 m_dataGrid = new CCDataGridView2<Person>(Person.class);
    CCUsageVariantButton m_usageVariant = new CCUsageVariantButton();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCUsageVariantDgv2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillGrid();
        UsageVariantManager uvm = new UsageVariantManager("CCDATAGRIDVIEW2","DEMOPBCCCUSAGEVARIANTDGV2",m_dataGrid.getExtractApplyServiceBridge());
        m_usageVariant.prepare(uvm,null);
        uvm.activate(true);
    }

    public String getPageName() { return "/workplace/demopbcccusagevariantdgv2.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCUsageVariantDgv2}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public CCDataGridView2 getDataGrid() { return m_dataGrid; }
    public CCUsageVariantButton getUsageVariant() { return m_usageVariant; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        List<Person> ps = new ArrayList<Person>();
        for (int i=0; i<100; i++)
            ps.add(RandomLogic.createPerson());
        m_dataGrid.prepare(ps,null);
    }

}
