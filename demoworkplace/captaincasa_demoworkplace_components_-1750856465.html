The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Drag & Drop
  • Drag over info text + cursor

By using attribute DROPRASTERTEXT you can output different type of information once some information is dragged over your component.

The current options are:

(.) output pixel/percentage position
(.) output explicit text that depends on position

In addition you may use attribute DROPSHAPE in order to influence the rendering of the drag over cursor. The current possibilities are:

(.) output crosshair
(.) output veritcal/horizontal line

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalTime;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.util.DropRasterTextX;
import org.eclnt.jsfserver.elements.util.DropRasterTextY;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDropRaster}")

public class DemoDropRaster
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DropRasterTextX m_complexRaster = new DropRasterTextX();
    DropRasterTextY m_complexRasterY = new DropRasterTextY();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDropRaster(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // prepare "complex" raster for schedule with quarter of hours
        int counter = -1;
        for (int i=0; i<24; i++)
        {
            for (int j=0; j<4; j++)
            {
                counter++;
                double pos = counter * 100d / (4 * 24); 
                LocalTime lt = LocalTime.of(i,j*15);
                m_complexRaster.addText(pos,lt.toString());
                m_complexRasterY.addText(pos,lt.toString());
            }
        }
    }

    public String getPageName() { return "/workplace/demodropraster.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDropRaster}"; }

    public DropRasterTextX getComplexRaster() { return m_complexRaster; }
    public DropRasterTextY getComplexRasterY() { return m_complexRasterY; }

    public void onDropAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventDrop)
        {
            BaseActionEventDrop e = (BaseActionEventDrop)event;
            float xPos = e.getPercentageHorizontaAsFloat() + e.getPercentageDeltaX();
            float yPos = e.getPercentageVerticalAsFloat() + e.getPercentageDeltaY();
            Statusbar.outputSuccess("DROP was registered at position: " + xPos + ", " + yPos);
        }
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

}
