The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Page Beans
  • Master-Detail Navigation

This demo demonstrates a master-detail scenario.

When CLICKING an item on the left then a corresponding detail screen is shown on the right. The detail screen is in display mode.

When DOUBLE-CLICKING an item on the left then the detail screen is shown in edit mode. The user may change the data. While editing the grid-navigation on the left is blocked so that there is no confusion between editing on the right and navigation on the left. After pressing OK/Cancel the right detail screen is brought back to display mode - and the blocking on the left side is released.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.pagebean.IPageBean;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import workplace.DemoMasterLogic.Employee;
import workplace.DemoMasterLogic.PermanentEmployee;

@CCGenClass (expressionBase="#{d.DemoMaster}")

public class DemoMaster
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        Employee i_employee;
        public GridItem(Employee employee)
        {
            i_employee = employee;
        }
        public boolean isPermanent() { return i_employee.getClass() == PermanentEmployee.class; }
        public Employee getEmployee() { return i_employee; }
        // single click / cursor selection
        public void onRowSelect()
        {
            showEmployee(this,i_employee,false);
        }
        // double click / return selection
        public void onRowExecute()
        {
            showEmployee(this,i_employee,true);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    IPageBean m_detailUI;
    boolean m_justEditingDetail = false;
    Trigger m_animationTrigger = new Trigger();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMaster(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        List<Employee> employees = DemoMasterLogic.findAllEmployees();
        for (Employee employee: employees)
        {
            GridItem gi = new GridItem(employee);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demomaster.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMaster}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public IPageBean getDetailUI() { return m_detailUI; }
    public boolean isJustEditingDetail() { return m_justEditingDetail; }
    public Trigger getAnimationTrigger() { return m_animationTrigger; }

    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void showEmployee(final GridItem gridItem, Employee employee, final boolean editMode)
    {
        // in case of editing: create a clone so that the user does not work
        // on the original object
        if (editMode == true)
        {
            employee = employee.createClone();
            m_justEditingDetail = true;
        }
        final Employee detailEmployee = employee;
        // open up the right detail screen
        final DemoMasterDetailEmployee ui = createUIForEmplyoee(employee);
        ui.prepare(detailEmployee, editMode, new DemoMasterDetailEmployee.IListener()
        {
            public void reactOnSave()
            {
                // take over cloned instance into grid
                gridItem.i_employee = detailEmployee;
                ui.setEditMode(false);
                m_justEditingDetail = false;
            }
            public void reactOnCancel()
            {
                // put the original instance of the grid item into the detail
                ui.setEmployee(gridItem.i_employee);
                ui.setEditMode(false);
                m_justEditingDetail = false;
            }
        });
        m_detailUI = ui;
        m_animationTrigger.trigger();
    }
    
    private DemoMasterDetailEmployee createUIForEmplyoee(Employee employee)
    {
        if (employee instanceof PermanentEmployee)
            return new DemoMasterDetailPermanentEmployee();
        else
            return new DemoMasterDetailWorkingStudent();
    }
    
}
