The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Touch Screen
  • Stable virtual keyboard

The TOUCHVIRUTALKEYBOARD is a software keyboard component that is not bound to a specific field/component but that operates indenpently as a vitrual keyboard.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTouchKeyboard}")

public class DemoTouchKeyboard
    extends PageBean
    implements Serializable
{
    protected String m_lastName;
    protected String m_firstName;
    protected int m_shoeSize;
    protected BigDecimal m_height;
    protected boolean m_married;
    String m_department;
    
    public DemoTouchKeyboard(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("touchvirtualkeyboard");
    }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { m_lastName = value; }

    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { m_firstName = value; }
    
    public void setShoeSize(int value) { m_shoeSize = value; }
    public int getShoeSize() { return m_shoeSize; }
    
    public void setHeight(BigDecimal value) { m_height = value; }
    public BigDecimal getHeight() { return m_height; }

    public void setMarried(boolean value) { m_married = value; }
    public boolean getMarried() { return m_married; }
    
    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    public void onAction(ActionEvent event) 
    {
        getOwningDispatcher().getStatusbar().outputSuccess("Input: " + m_firstName + "/" + m_lastName + "/" + m_shoeSize + "/" + m_height);
    }

}
