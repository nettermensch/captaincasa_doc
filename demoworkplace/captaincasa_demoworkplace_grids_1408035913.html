The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • Column Spanning

By using the component SPANGRID you can flexibly arrange components within the cells of the grid. Each cell component may define an explicit horizontal column spanning.

The definition of cell components needs to be done within the code of the page (i.e. within the corresponding Java program), because each row now may contain an individual set of components.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoGridColSpan}" rowdistance="10" >
<t:row id="g_2" >
<t:spangrid id="g_3" bordercolor="#00000060" borderheight="1" borderwidth="0" dynamicheightsizing="true" height="100%" objectbinding="#{d.DemoGridColSpan.grid}" rowbgpaint="rectangle(0,0,100%,50%,#ffffff10);rectangle(0,50%,100%,50%,#00000010)" rowheight="23" rowpopupmenu="GRID" sbvisibleamount="25" selectioncolor1="#FF000080" selectioncolor2="#FF000020" selectorcolumn="1" width="100%" >
<t:spangridcol id="g_4" sortenabled="false" text="Title" width="60" />
<t:spangridcol id="g_5" text="First Name" width="50%;100" />

</t:popupmenu>

</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.CHECKBOXNode;
import org.eclnt.jsfserver.elements.componentnodes.COLDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.elements.impl.SPANGRIDItem;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridColSpan}")

public class DemoGridColSpan
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // Some "business classes"
    // ------------------------------------------------------------------------
    
    public class Summary
    {
        boolean m_checked;
        String m_text;
        public void setText(String value) { m_text = value; }
        public String getText() { return m_text; }
        public boolean isChecked() { return m_checked; }
        public void setChecked(boolean checked) { m_checked = checked; }
    }
    
    public class Person
    {
        String m_title;
        String m_firstName;
        String m_lastName;
        public void setTitle(String value) { m_title = value; }
        public String getTitle() { return m_title; }
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
    }
    
    public class CostCenter
    {
        String m_name;
        public void setName(String value) { m_name = value; }
        public String getName() { return m_name; }
    }
    
    // ------------------------------------------------------------------------
    // Inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends SPANGRIDItem
    {
        Object m_o;
        public GridItem(Object o)
        {
            m_o = o;
            renderContent();
        }
        private void renderContent()
        {
            ComponentNode cn;
            if (m_o instanceof CostCenter)
            {
                cn = new LABELNode()
                    .setText(".{o.name}")
                    .setWidth("100%")
                    .setBackground("#c0c0ff80")
                    .setHeight("100%;40")
                    .setFocusable(true)
                    .setAlign("center")
                    .setFont("weight:bold");
                addContent(3,cn);
            }
            else if (m_o instanceof Person)
            {
                cn = new FIELDNode()
                    .setText(".{o.title}")
                    .setBorder("left:1;color:#00000030")
                    .setBackground("C0FFC080")
                    .setRounding(0);
                addContent(1,cn);
                cn = new FIELDNode()
                    .setText(".{o.firstName}")
                    .setBorder("left:1;color:#00000030")
                    .setBackground("FFFFFF80")
                    .setRounding(0);
                addContent(1,cn);
                cn = new FIELDNode()
                    .setText(".{o.lastName}")
                    .setBorder("left:1;color:#00000030")
                    .setBackground("FFFFFF80")
                    .setRounding(0);
                addContent(1,cn);
            }   
            else if (m_o instanceof Summary)
            {
                cn = new COLDISTANCENode();
                addContent(1,cn);
                cn = new PANENode()
                    .setPadding("left:5");
                ROWNode rn = new ROWNode();
                cn.addSubNode(rn);
                CHECKBOXNode cbn = new CHECKBOXNode()
                    .setSelected(".{o.checked}")
                    .setText(".{o.text}");
                rn.addSubNode(cbn);
                addContent(2,cn);
            }   
        }
        public Object getO() { return m_o; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    boolean m_differentRowheights = false;

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoGridColSpan(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher); 
        for (int i=0; i<30; i++)
        {
            CostCenter cc = new CostCenter();
            cc.setName("Cost Center " + i);
            m_grid.getItems().add(new GridItem(cc));
            for (int j=0; j<5; j++)
            {
                Person p = new Person();
                p.setTitle("Dr");
                p.setFirstName("First " + j);
                p.setLastName("Last " + j);
                m_grid.getItems().add(new GridItem(p));
            }
            Summary s = new Summary();
            s.setText("5 persons listed above");
            m_grid.getItems().add(new GridItem(s));
        }
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public boolean getDifferentRowheights() { return m_differentRowheights; }
    public void setDifferentRowheights(boolean differentRowheights) { m_differentRowheights = differentRowheights; }

    public void onDifferentRowheightsAction(ActionEvent event)
    {
        if (m_differentRowheights == false)
        {
            for (GridItem gi: m_grid.getItems())
                gi.setRowHeight(20);
        }       
        else
        {
            for (GridItem gi: m_grid.getItems())
            {
                if (gi.getO() instanceof CostCenter)
                    gi.setRowHeight(30);
                else if (gi.getO() instanceof Summary)
                    gi.setRowHeight(16);
                else
                    gi.setRowHeight(20);
            }
        }
    }
    
    
}
