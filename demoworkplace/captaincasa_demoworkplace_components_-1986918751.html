The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Text Input/Output
  • Text Diff Viewer

The page bean CCTextDiff compares two texts with one another and renders the comparison result.

Change the text on the left side - or change the text on the right side. Shortly after finishing your input the comparison will be updated.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCTextDiff;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPBCTextDiff}")

public class DemoPBCTextDiff
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_rightText = "Right text. The weather outside is not nice at all. Please remind me.";
    String m_leftText = "Left Text. The weather outside is nice. Please remind me.";
    CCTextDiff m_textDiffBean = new CCTextDiff();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCTextDiff(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);  
        m_textDiffBean.prepareTexts(m_leftText,m_rightText);
    }

    public String getPageName() { return "/workplace/demopbctextdiff.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCTextDiff}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getRightText() { return m_rightText; }
    public void setRightText(String value) { this.m_rightText = value; }

    public String getLeftText() { return m_leftText; }
    public void setLeftText(String value) { this.m_leftText = value; }

    public CCTextDiff getTextDiffBean() { return m_textDiffBean; }
    public void setTextDiffBean(CCTextDiff value) { this.m_textDiffBean = value; }

    public void onLeftTextAction(javax.faces.event.ActionEvent event) 
    {
        m_textDiffBean.prepareTexts(m_leftText,m_rightText);
    }

    public void onRIghtTextActioi(javax.faces.event.ActionEvent event)
    {
        m_textDiffBean.prepareTexts(m_leftText,m_rightText);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
