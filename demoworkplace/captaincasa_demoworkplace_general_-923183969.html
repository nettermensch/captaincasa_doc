The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Guided tour

By pressing the button "Start guided tour" a sequence of popup dialogs is started that explains certain parts of the screen.

The demo shows how to use certain features in the area of opening dialogs on top of components.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.elements.util.DefaultModalPopupListener;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGuidedDialog}")

public class DemoGuidedDialog
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String[] m_areaIds = new String[] 
    {
        "PART1",
        "PART2",
        "PART3"
    };
    String[] m_areaInfoTexts = new String[] 
    {
        "The first name and the last name are the most important fields to input!",
        "The address part is only needed in case of shipping goods to the user. You may input or not input - it's up to you.",
        "Here you can specify any additional comment."
    };
    int m_currentAreaIndex = 0;
    DemoGuidedDialogInfo m_currentInfo = null;
    String m_positioning = "ONTOP";
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGuidedDialog(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demoguideddialog.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGuidedDialog}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onStartAction(javax.faces.event.ActionEvent event) 
    {
        m_currentAreaIndex = 0;
        showCurrentDialog();
    }

    public String getPositioning() { return m_positioning; }
    public void setPositioning(String value) { this.m_positioning = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void showCurrentDialog()
    {
        if (m_currentInfo != null)
        {
            closePopup(m_currentInfo);
        }
        m_currentInfo = new DemoGuidedDialogInfo();
        m_currentInfo.prepare(
           m_areaInfoTexts[m_currentAreaIndex],
           new DemoGuidedDialogInfo.IListener()
           {
               public void reactOnNavigation(int delta)
               {
                   m_currentAreaIndex += delta;
                   m_currentAreaIndex = (m_currentAreaIndex + m_areaIds.length) % m_areaIds.length;
                   showCurrentDialog();
               }
           }
        );
        ModalPopup p = openModalPopup(m_currentInfo,"",0,0,new DefaultModalPopupListener(this,m_currentInfo));
        p.setUndecorated(true);
        p.setCloseonclickoutside(true);
        if ("ONTOP".equals(m_positioning))
        {
            p.setWidthHeightToSizeOfReference();
            p.setLeftTopReferenceComponentIdCentered(m_areaIds[m_currentAreaIndex]);
            p.setScreenBackgroundColor("#FFFFFF00"); // transparent
        }
        else
        {
            p.setWidth(400);
            p.setHeight(200);
            p.setLefTopReferenceComponentIdBottom(m_areaIds[m_currentAreaIndex]);
            p.setTop(-30);
            p.setLeft(100);
            p.setHighlightReference(true);
            p.setScreenBackgroundColor("#00000060"); // darker gray
        }
    }
}
