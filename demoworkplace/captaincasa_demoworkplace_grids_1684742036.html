The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid Header Cell
  • Title Label with Icons

You can define additional components to be placed into the column header.

The way to do so, is:

(.) the first component within a GRIDCOL component is the cell-component
(.) all the rest of the components are components that are added into the title header

Please note: even though the title-components (the 2nd, 3rd, ... ones) are arranged below a GRIDCOL component, they are not part of the grid item! They are controls that are NOT "multiplied out" into the cells, but they are controls that are rendered one time within the title area.
As result you must not use any relative expressions (".{...}") for these components - because they do not refer to the item. You need to use absolute expressions ("#{...}").

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.jsfserver.util.IClientLogAware;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridHeaderLabel}")

public class DemoGridHeaderLabel
    extends PageBean 
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }

    }

    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    boolean m_iconAddRendered = true;
    boolean m_iconAcceptRendered = true;

    public DemoGridHeaderLabel(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("FN " + i);
            gi.setLastName("LN " + i);
            m_grid.getItems().add(gi);
        }
    }

    public boolean isIconAddRendered() { return m_iconAddRendered; }
    public void setIconAddRendered(boolean iconAddRendered) { m_iconAddRendered = iconAddRendered; }

    public boolean isIconAcceptRendered() { return m_iconAcceptRendered; }
    public void setIconAcceptRendered(boolean iconAcceptRendered) { m_iconAcceptRendered = iconAcceptRendered; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onHeaderIconPressed(ActionEvent event)
    {
        BaseActionEvent bae = (BaseActionEvent)event;
        Statusbar.outputSuccess("Header Icon was pressed: " + bae.getSourceReference());
        ClientTestLog.addClientTestLogMessage("Header Icon was pressed: " + bae.getSourceReference());
    }
    
    
}
