The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Issues
  • Fix columns with tree

This example shows how to bind two tree in a way that they form a fix-column-scenario.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoTreeFixColumns}")

public class DemoTreeFixColumns
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyTreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public MyTreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        public void onToggle()
        {
        }
        public String getLastName()
        {
            return i_lastName;
        }
        public String getFirstName()
        {
            return i_firstName;
        }
    }


    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDTreeBinding<MyTreeNode> m_treeRight = new FIXGRIDTreeBinding<MyTreeNode>();
    FIXGRIDTreeBinding<MyTreeNode> m_treeLeft = new FIXGRIDTreeBinding<MyTreeNode>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTreeFixColumns(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        MyTreeNode root = new MyTreeNode(null);
        m_treeLeft.setRootNode(root);
        m_treeRight.setRootNode(root);
        m_treeLeft.connectWithBinding(m_treeRight);
        for (int i=0; i<10; i++)
        {
            MyTreeNode top = new MyTreeNode(root);
            top.setText("Top " + i);
            top.setStatus(MyTreeNode.STATUS_CLOSED);
            for (int j=0; j<5; j++)
            {
                MyTreeNode sub = new MyTreeNode(top);
                sub.setText("Sub " + i + "/" + j);
                sub.i_firstName = "First Name " + i + "/" + j;
                sub.i_lastName = "Last Name " + i + "/" + j;
                sub.setStatus(MyTreeNode.STATUS_ENDNODE);
            }
        }
    }

    public String getPageName() { return "/workplace/demotreefixcolumns.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTreeFixColumns}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDTreeBinding<MyTreeNode> getTreeRight() { return m_treeRight; }
    public FIXGRIDTreeBinding<MyTreeNode> getTreeLeft() { return m_treeLeft; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
