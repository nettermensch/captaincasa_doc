The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Open Street Map
  • Dynamic Way Points

By clicking into the map you add new way points into the grid.

This example shows how to dynamically use the OSMVIEWER and OSMVIEWERWAYPOINT components, based on the usage of the DYNAMICCONTENT component.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventOsmviewerSelection;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoOsmViewerDynamic}")

public class DemoOsmViewerDynamic
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {    
        protected double m_latitude;
        public double getLatitude() { return m_latitude; }
        public void setLatitude(double value) { m_latitude = value; }

        protected double m_longitude;
        public double getLongitude() { return m_longitude; }
        public void setLongitude(double value) { m_longitude = value; }

        protected String m_name;
        public String getName() { return m_name; }
        public void setName(String value) { m_name = value; }

        public void onRemove(ActionEvent event) 
        {
            m_grid.getItems().remove(this);
            onTakeOver(null);
        }
        
        public void onSelectAction(ActionEvent event)
        {
            m_grid.deselectCurrentSelection();
            m_grid.selectAndFocusItem(this);
        }
    }

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    DYNAMICCONTENTBinding m_waypointContent = new DYNAMICCONTENTBinding();
    double m_longitude = 8.715119361877441;
    double m_latitude = 49.411112809289975;

    public DemoOsmViewerDynamic(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        initRefComponent("osmviewer");
        GridItem gi = new GridItem();
        gi.setLatitude(49.411112809289975);
        gi.setLongitude(8.715119361877441);
        gi.setName("Heidelberg, Castle");
        m_grid.getItems().add(gi);
        gi = new GridItem();
        gi.setLatitude(49.31717765464028);
        gi.setLongitude(8.442220687866211);
        gi.setName("Speyer, Dome");
        m_grid.getItems().add(gi);
//        for (int i=0; i<100; i++)
//        {
//            gi = new GridItem();
//            gi.setLatitude(49.31717765464028 + i * 0.05);
//            gi.setLongitude(8.442220687866211 + i * 0.05);
//            gi.setName("Waypoint " + i);
//            m_grid.getItems().add(gi);
//        }
        onTakeOver(null);
    }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public DYNAMICCONTENTBinding getWaypointContent() { return m_waypointContent; }
    
    public double getLongitude() { return m_longitude; }
    public void setLongitude(double longitude) { m_longitude = longitude; }

    public double getLatitude() { return m_latitude; }
    public void setLatitude(double latitude) { m_latitude = latitude; }

    public void onCreateNewLocation(ActionEvent event)
    {
        GridItem gi = new GridItem();
        m_grid.getItems().add(gi);
        m_grid.deselectCurrentSelection();
        m_grid.selectAndFocusItem(gi);
    }
    
    public void onMapSelection(ActionEvent event)
    {
        if (event instanceof BaseActionEventOsmviewerSelection)
        {
            BaseActionEventOsmviewerSelection e = (BaseActionEventOsmviewerSelection)event;
            try
            {
                GridItem gi = new GridItem();
                gi.setLatitude(e.getLatitude());
                gi.setLongitude(e.getLongitude());
                m_grid.getItems().add(gi);
                m_grid.deselectCurrentSelection();
                m_grid.selectAndFocusItem(gi);
                onTakeOver(null);
            }
            catch (Throwable t)
            {
            }
        }
    }
    
    public void onTakeOver(ActionEvent event)
    {
        List<ComponentNode> nodes = new ArrayList<ComponentNode>();
        String[] bgpaints = new String[] 
        {
            "rectangle(7,7,100%-7,100%-7,#00000080)",
            "oval(0,0,15,15,#000000)",
            "oval(3,3,9,9,#FFFFFF)",
            null                
        };
        int counter = 0;
        for (GridItem gi: m_grid.getItems())
        {
            ComponentNode node = new ComponentNode("t:osmviewerwaypoint");
            node.addAttribute("latitude",""+gi.m_latitude);
            node.addAttribute("longitude",""+gi.m_longitude);
            bgpaints[3] = ValueManager.encodeMethod("write",new String[] {"15","10",gi.getName(),"14","#FFFFC0","bold","lefttop","0","1","1"});
            String bgpaint = ValueManager.encodeCSV(bgpaints);
            node.addAttribute("bgpaint",bgpaint);
            node.addAttribute("height","32");
            node.addAttribute("width","150");
            node.addAttribute("actionListener","#{d.DemoOsmViewerDynamic.grid.items["+counter+"].onSelectAction}");
            node.addAttribute("text",gi.getName());
            if (counter%2 == 0)
                node.addAttribute("markerimage","/images.iconssvg.factory.#FF0000.64x64.ccsvg");
            else
                node.addAttribute("markerimage","/images.iconssvg.factory.#00FF00.64x64.ccsvg");
            node.addAttribute("hideinfopopup","true");
            nodes.add(node);
            counter++;
        }
        m_waypointContent.setContentNodes(nodes);
    }

}
