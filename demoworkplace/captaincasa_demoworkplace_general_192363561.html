The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Many images in flexible grid

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowbodypane id="g_4" >
<t:row id="g_5" >
<t:pane id="g_6" height="100%" width="100%" >
<t:sizetransfer id="g_8" actionListener="#{d.DemoFlexImageListViaGrid.onPaneSizeAction}" flush="true" horizontalcategories="SMALL:800;MIDDLE:1500;LARGE:10000" sizevalue="#{d.DemoFlexImageListViaGrid.paneSize}" verticalcategories="ALL:10000" />
<t:rowdynamiccontent id="g_11" contentbinding="#{d.DemoFlexImageListViaGrid.dynContent}" />
</t:pane>
</t:row>
</t:rowbodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.IMAGENode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoFlexImageListViaGrid}")

public class DemoFlexImageListViaGrid
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    ROWDYNAMICCONTENTBinding m_dynContent = new ROWDYNAMICCONTENTBinding();
    public ROWDYNAMICCONTENTBinding getDynContent() { return m_dynContent; }
    public void setDynContent(ROWDYNAMICCONTENTBinding value) { this.m_dynContent = value; }

    public void onPaneSizeAction(javax.faces.event.ActionEvent event) 
    {
        renderGrid();
    }

    String m_paneSize;
    public String getPaneSize() { return m_paneSize; }
    public void setPaneSize(String value) { this.m_paneSize = value; }

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        List<String> i_images = new ArrayList<String>();
        public List<String> getImages() { return i_images; }
    }

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;
    private int m_nofImages = 1000;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFlexImageListViaGrid(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_paneSize = "SMALL;ALL";
        renderGrid();
    }

    public String getPageName() { return "/workplace/demofleximagelistviagrid.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFlexImageListViaGrid}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderGrid()
    {
        // render grid
        FIXGRIDNode f = new FIXGRIDNode()
            .setWidth("100%")
            .setHeight("100%")
            .setSuppressheadline(true)
            .setBordercolor("#00000010")
            .setBorderwidth(1)
            .setBorderheight(1)
            .setSbvisibleamount(20)
            .setRowheight("60")
            .setObjectbinding(pbx("grid"));
        int nofCols = 5;
        if (m_paneSize.startsWith("MIDDLE"))
            nofCols = 10;
        else if (m_paneSize.startsWith("LARGE"))
            nofCols = 20;
        for (int i=0; i<nofCols; i++)
        {
            GRIDCOLNode c = new GRIDCOLNode()
                .setWidth("100%");
            IMAGENode in = new IMAGENode()
                .setImage(".{images["+i+"]}");
            f.addSubNode(c);
            c.addSubNode(in);
        }
        m_dynContent.setContentNode(f);
        // build items
        m_grid.getItems().clear();
        int nofRows = m_nofImages / nofCols;
        if (m_nofImages % nofCols > 0)
            nofRows++;
        for (int iRow=0;iRow<nofRows;iRow++)
        {
            GridItem gi = new GridItem();
            m_grid.getItems().add(gi);
            for (int iCol=0;iCol<nofCols; iCol++)
            {
                int index = iRow*nofCols + iCol;
                if (index > m_nofImages)
                    break;
                gi.i_images.add("/images/icons/bigspeaker.png");
            }
        }
    }
}
