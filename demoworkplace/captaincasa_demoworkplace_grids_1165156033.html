The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • Column rendering

The visibility of columns can be controlled using the RENDERED attribute that is available with the GRIDCOL component.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

<t:rowdemobodypane id="g_1" objectbinding="#{d.demoGridcolrender}" >

<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" borderwidth="1" drawoddevenrows="true" objectbinding="#{d.demoGridcolrender.rows}" sbvisibleamount="10" width="100%" >

This is the Java code:

package workplace;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.demoGridcolrender}")

public class DemoGridcolrender extends PageBean
{
    public class MyRow extends FIXGRIDItem
    {
        String m_firstName;
        String m_lastName;
        String m_street;
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
        public void setStreet(String value) { m_street = value; }
        public String getStreet() { return m_street; }
    }
    
    FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    boolean m_firstNameRendered = true;
    boolean m_lastNameRendered = true;
    boolean m_streetRendered = true;
    
    public DemoGridcolrender(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<25; i++)
        {
            MyRow r = new MyRow();
            r.setFirstName("FN " + i);
            r.setLastName("LN " + i);
            r.setStreet("ST " + i);
            m_rows.getItems().add(r);
        }
    }
    
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    
    public void setFirstNameRendered(boolean value) { m_firstNameRendered = value; }
    public boolean isFirstNameRendered() { return m_firstNameRendered; }
    public void setLastNameRendered(boolean value) { m_lastNameRendered = value; }
    public boolean getLastNameRendered() { return m_lastNameRendered; }
    public void setStreetRendered(boolean value) { m_streetRendered = value; }
    public boolean getStreetRendered() { return m_streetRendered; }
    
}
