This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventBrowserEventMessage;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.workplace.IWorkpageDispatcher;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;

@CCGenClass (expressionBase="#{d.DemoPooledBrowserEventMessage}")

public class DemoPooledBrowserEventMessage
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_message;
    Trigger m_eventMessageTrigger = new Trigger();
    String m_pageSource;
    String m_receivedMessage = "";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPooledBrowserEventMessage(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        readPageSource();
        buildMessage();
    }

    public String getPageName() { return "/workplace/DemoPooledBrowserEventMessage.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPooledBrowserEventMessage}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getMessage() { return m_message; }
    public void setMessage(String value) { this.m_message = value; }

    public Trigger getEventMessageTrigger() { return m_eventMessageTrigger; }
    
    public void onSendEventAction(javax.faces.event.ActionEvent event) 
    {
        m_eventMessageTrigger.trigger();
    }

    public void onBrowserAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventBrowserEventMessage)
        {
            BaseActionEventBrowserEventMessage e = (BaseActionEventBrowserEventMessage)event;
            m_receivedMessage += e.getEventMessageAsJSON() + "\n";
            Statusbar.outputSuccess("Message received from browser: " + System.currentTimeMillis());
        }
    }

    public String getPageSource() { return m_pageSource; }
    public void setPageSource(String value) { this.m_pageSource = value; }

    public String getReceivedMessage() { return m_receivedMessage; }
    public void setReceivedMessage(String value) { this.m_receivedMessage = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildMessage()
    {
        Map<String,String> data = new HashMap<String,String>();
        data.put("mswp_type","message");
        data.put("app_command","showOrder");
        data.put("app_customer","4711");
        data.put("app_order","4500002434");
        Gson gs = new GsonBuilder().setPrettyPrinting().create();
        m_message = gs.toJson(data);
    }

    private void readPageSource()
    {
        m_pageSource = WebResourceReader.readUTF8FileIntoString("workplace/browserevents/demo1.html");
    }

}
