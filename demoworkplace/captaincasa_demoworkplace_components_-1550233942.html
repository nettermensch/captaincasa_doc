The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Images
  • Dynamic SVG content

Just like other images (JPG, ...), SVG-images can also be dynamically loaded.

You may either use the component HEXIMAGE - or you may use any other component and pass a pseudo URL ("hex(...)") into the IMAGE-attribute.

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.util.file.FileManager;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoSVGImagesDynamic}")

public class DemoSVGImagesDynamic
    extends PageBean 
    implements Serializable
{
    String m_hexData;
    public DemoSVGImagesDynamic(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        load("/images/iconssvg/svglogo.svg");
    }

    public String getPageName() { return "/workplace/demosvgimagesdynamic.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSVGImagesDynamic}"; }

    public String getHexData() { return m_hexData; }
    
    private void load(String resource)
    {
        byte[] bytes = new WebResourceReader().readFileIntoByteArray(resource);
        m_hexData = ValueManager.encodeHexString(bytes);
    }

    public void onLoadBirthdayCakeAction(javax.faces.event.ActionEvent event) 
    {
        load("/images/iconssvg/cake_withsize.svg");
    }

    public void onLoadSVGLogoAction(javax.faces.event.ActionEvent event) 
    {
        load("/images/iconssvg/svglogo.svg");
    }
    
    public String getUrlHexData() { return "/hex(" + getHexData() + ")"; }
    
}
