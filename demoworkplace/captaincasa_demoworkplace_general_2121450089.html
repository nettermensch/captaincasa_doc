The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Client Integration
  • Start Browser, dynamic content

This demo shows how to pass content dynamically into the browser that is opened up by the JSHOWURL component.

Either you pass HTML text, or you pass some content, e.g. a PDF byte array. When passing non-HTML content it is recommended to also pass the extension of the content ("pdf") so that the browser is able to select the right application to internally load.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" >
<t:jshowurl id="g_2" trigger="#{d.DemoStartBrowserDynamicContent.trigger}" url="#{d.DemoStartBrowserDynamicContent.browserURL}" usedesktop="true" />
</t:beanprocessing>
<t:rowdemobodypane id="g_3" objectbinding="#{d.DemoStartBrowserDynamicContent}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >
<t:row id="g_4" >
<t:button id="g_5" actionListener="#{d.DemoStartBrowserDynamicContent.onPDF}" text="Start with PDF" width="150" />
</t:row>
<t:row id="g_6" >
<t:button id="g_7" actionListener="#{d.DemoStartBrowserDynamicContent.onHTML}" text="Start with HTML" width="150" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.BufferedContentMgr;
import org.eclnt.jsfserver.bufferedcontent.DefaultBufferedContent;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.IWorkpageLifecycleListener;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoStartBrowserDynamicContent}")

public class DemoStartBrowserDynamicContent
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class ResourceBufferedContent extends DefaultBufferedContent
    {
        String i_resourceName;
        public void passContent(String resourceName, String extension)
        {
            i_resourceName = resourceName;
            setExtension(extension);
        }
        public byte[] getContent()
        {
            return (WebResourceReader.readFileIntoByteArray(i_resourceName));
        }
        public String getContentType()
        {
            return null;
        }
    }
    
    public class HTMLBufferedContent extends DefaultBufferedContent
    {
        public byte[] getContent()
        {
            try
            {
                StringBuffer sb = new StringBuffer();
                sb.append("<html>");
                sb.append("This page was dynamically created on server side.<br>");
                sb.append("This page was dynamically created on server side.<br>");
                sb.append("This page was dynamically created on server side.<br>");
                sb.append("This page was dynamically created on server side.<br>");
                sb.append("This page was dynamically created on server side.<br>");
                sb.append("</html>");
                return sb.toString().getBytes("UTF-8");
            }
            catch (Throwable t)
            {
                return ("Error occurred: " + t.toString()).getBytes();
            }
        }
        public String getContentType()
        {
            return null;
        }
    }

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoStartBrowserDynamicContent(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        BufferedContentMgr.add(m_bufferedContent1);
        BufferedContentMgr.add(m_bufferedContent2);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            public void reactOnDestroyed()
            {
                BufferedContentMgr.remove(m_bufferedContent1);
                BufferedContentMgr.remove(m_bufferedContent2);
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    Trigger m_trigger = new Trigger();
    String m_url = null;
    ResourceBufferedContent m_bufferedContent1 = new ResourceBufferedContent();
    HTMLBufferedContent m_bufferedContent2 = new HTMLBufferedContent();

    public String getBrowserURL() { return m_url; }
    public Trigger getTrigger() { return m_trigger; }

    public void onPDF(ActionEvent event) 
    {
        m_bufferedContent1.passContent("/pdf/pdftest1.pdf","pdf");
        m_url = m_bufferedContent1.getURLForBROWSERUsage();
        m_trigger.trigger();
    }

    public void onHTML(ActionEvent event) 
    {
        m_url = m_bufferedContent2.getURLForBROWSERUsage();
        m_trigger.trigger();
    }

}
