The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Load popup menu dynamically

By right clicking on the labels you see that each label has its own set of menu items.

In this case this is NOT implemented by providing different popup menu definitions, one per label.

In this case the attribute POPUPMENULOADROUNDTRIP is used:
(.) When the user right-clicks onto the label then a roundtrip is triggered to the label's actionListener
(.) The actionListener code builds up the popup menu items by feeding a corresponding DYNAMICCONTENT component

Advantage of this implementation: there is only one popup menu definition which is very flexibly control-able from server side.

Disadvantage: a roundtrip is triggered on right mouse click.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoPopupMenuLoadRoundtrip}" rowdistance="20" >
<t:row id="g_2" >
<t:label id="g_3" actionListener="#{d.DemoPopupMenuLoadRoundtrip.onLabelTopAction}" stylevariant="ccbig" popupmenu="CENTRALPOPUP" popupmenuloadroundtrip="true" text="Right Click onto this top label" />
</t:row>
<t:row id="g_4" >
<t:label id="g_5" text="or" />
</t:row>
<t:row id="g_6" >
<t:label id="g_7" actionListener="#{d.DemoPopupMenuLoadRoundtrip.onLabelBottomAction}" stylevariant="ccbig" popupmenu="CENTRALPOPUP" popupmenuloadroundtrip="true" text="Right Click onto this bottom label" />
</t:row>
</t:rowdemobodypane>
<t:popupmenu id="CENTRALPOPUP" >
<t:dynamiccontent id="g_8" contentbinding="#{d.DemoPopupMenuLoadRoundtrip.dynMenuContent}" />
</t:popupmenu>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuLoad;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoPopupMenuLoadRoundtrip}")

public class DemoPopupMenuLoadRoundtrip
    extends PageBean 
    implements Serializable
{
     protected DYNAMICCONTENTBinding m_dynMenuContent = new DYNAMICCONTENTBinding();

    public DemoPopupMenuLoadRoundtrip(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        initRefComponent("popupmenu");
    }

    public DYNAMICCONTENTBinding getDynMenuContent() { return m_dynMenuContent; }
    
    public void onLabelTopAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuLoad)
        {
            List<ComponentNode> nodes = new ArrayList<ComponentNode>();
            {
                MENUITEMNode node = new MENUITEMNode();
                node.setText("First of top");
                node.setCommand("TOP1");
                nodes.add(node);
            }
            {
                MENUITEMNode node = new MENUITEMNode();
                node.setText("Second of top");
                node.setCommand("TOP1");
                nodes.add(node);
            }
            m_dynMenuContent.setContentNodes(nodes);
        }
        else if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            Statusbar.outputSuccess("Command of popup menu item: " + e.getCommand());
        }
    }
    
    public void onLabelBottomAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuLoad)
        {
            List<ComponentNode> nodes = new ArrayList<ComponentNode>();
            {
                MENUITEMNode node = new MENUITEMNode();
                node.setText("First of bottom");
                node.setCommand("BOT1");
                nodes.add(node);
            }
            {
                MENUITEMNode node = new MENUITEMNode();
                node.setText("Second of bottom");
                node.setCommand("BOT2");
                nodes.add(node);
            }
            {
                MENUITEMNode node = new MENUITEMNode();
                node.setText("Third of bottom");
                node.setCommand("BOT3");
                nodes.add(node);
            }
            m_dynMenuContent.setContentNodes(nodes);
        }
        else if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            Statusbar.outputSuccess("Command of popup menu item: " + e.getCommand());
        }
    }

}
