The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Repeat Processing
  • Image Selection

This demo shows some combination out of the components...

ROWFLEXLINE
REPEAT.

Select an image on the left and it will be shown in the right area.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoRepeatImages}")

public class DemoRepeatImages
    extends PageBean 
    implements Serializable
{
    List<ImagesItem> m_images = new ArrayList<ImagesItem>();
    public List<ImagesItem> getImages() { return m_images; }
    public void setImages(List<ImagesItem> value) { this.m_images = value; }

    public class ImagesItem extends Object implements java.io.Serializable
    {    
        public void onClickAction(ActionEvent event) 
        {
            for (ImagesItem ii: m_images)
            {
                if (ii == this)
                {
                    ii.setSelected(true);
                    m_selectedImage = ii.getImage();
                }
                else
                {
                    ii.setSelected(false);
                }
            }
        }

        public String getBackground() 
        { 
            return m_selected ? "#FFCC0050" : null; 
        }

        boolean m_selected;
        public boolean getSelected() { return m_selected; }
        public void setSelected(boolean value) { this.m_selected = value; }

        String m_image;
        public String getImage() { return m_image; }
        public void setImage(String value) { this.m_image = value; }

    }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_selectedImage = null;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRepeatImages(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);      
        for (int i=0; i<50; i++)
        {
            ImagesItem ii = new ImagesItem();
            ii.setImage("/images/pictures/ph"+(i%8)+".jpg");
            m_images.add(ii);
        }
    }

    public String getPageName() { return "/workplace/demorepeatimages.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRepeatImages}"; }
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getSelectedImage()
    {
        return m_selectedImage;
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
