The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Reporting, PDF, SVG
  • Chart.js - With labels

By using BRIDGEPAGE "eclnt/risc/ext_chartjs/ext_chartjs_280_datalabels_060.html" you use chartjs together with its label-extension (see https://chartjs-plugin-datalabels.netlify.com).

Result: numeric values behind the charts are directly output as part of the chart - and not only available as tooltip on user request.

By default you do not have to add additional configuration data, but you may: please take a look at the JSON example "Stacked Bar Chart, explicit labels": the positioning and rendering of the label is directly controlled in this example.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventChartJsPointSelection;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoChartJsWithLabels}")

public class DemoChartJsWithLabels
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_config;
    ValidValuesBinding m_resourceVVS = new ValidValuesBinding();
    String m_resource;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoChartJsWithLabels(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_resource = "workplace/DemoChartJsLineChart.json";
        m_resourceVVS.addValidValue("workplace/DemoChartJsLineChart.json","Line Chart");
        m_resourceVVS.addValidValue("workplace/DemoChartJsPieChart.json","Pie Chart");
        m_resourceVVS.addValidValue("workplace/DemoChartJsPieChart2.json","Double Pie Chart");
        m_resourceVVS.addValidValue("workplace/DemoChartJsBarStacked.json","Stacked Bar Chart");
        m_resourceVVS.addValidValue("workplace/DemoChartJsBarStackedWithLabels.json","Stacked Bar Chart, explicit labels");
        updateConfigFromResource();
    }

    public String getPageName() { return "/workplace/demochartjswithlabels.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoChartJsWithLabels}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getConfig() { return m_config; }
    public void setConfig(String value) { this.m_config = value; }

    public ValidValuesBinding getResourceVVS() { return m_resourceVVS; }
    public void setResourceVVS(ValidValuesBinding value) { this.m_resourceVVS = value; }

    public String getResource() { return m_resource; }
    public void setResource(String value) { this.m_resource = value; }

    public void onResourceAction(ActionEvent event) 
    {
        updateConfigFromResource();
    }

    public void onChartJsAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventChartJsPointSelection)
        {
            BaseActionEventChartJsPointSelection e = (BaseActionEventChartJsPointSelection)event;
            Statusbar.outputSuccess("Point selection: dataset " + e.getDatasetIndex() + ", point " + e.getPointIndex());
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void updateConfigFromResource()
    {
        m_config = (new ClassloaderReader(getClass().getClassLoader()).readUTF8File(m_resource,true));
    }
}
