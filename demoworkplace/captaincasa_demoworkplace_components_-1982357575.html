The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Text Input/Output
  • Code / XML Editor
  • Code editor

The CODEEDITOR component internally wraps the ACE editor - please check the page https://ace.c9.io for detailed information.

The editor can be configured with more than 100 different code formats, that it correctly colors and observes while keying in the text.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoCodeEditor}" />
<t:rowdemobodypane id="g_2" objectbinding="#{d.DemoCodeEditor}" rowdistance="10" >
<t:row id="g_3" coldistance="5" >
<t:button id="g_4" actionListener="#{d.DemoCodeEditor.onLoadText}" reference="xml" text=".xml" width="100+" />
<t:button id="g_5" actionListener="#{d.DemoCodeEditor.onLoadText}" reference="java" text=".java" width="100+" />
<t:button id="g_6" actionListener="#{d.DemoCodeEditor.onLoadText}" reference="javascript" text=".js" width="100+" />
<t:button id="g_7" actionListener="#{d.DemoCodeEditor.onLoadText}" reference="xmas" text=".xmas" width="100+" />
</t:row>
<t:row id="g_8" >
<t:codeeditor id="g_9" codetype="#{d.DemoCodeEditor.codeType}" enabled="true" height="100%" text="#{d.DemoCodeEditor.text}" width="100%" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoCodeEditor}")

public class DemoCodeEditor
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_codeType = "xml";
    Map<String,String> m_texts = new HashMap<String, String>(); // codeType, text
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCodeEditor(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);   
        readText();
    }

    public String getPageName() { return "/workplace/democodeeditor.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCodeEditor}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getText() { return m_texts.get(m_codeType); } 
    public void setText(String text) { m_texts.put(m_codeType,text); }
    
    public String getCodeType() 
    { 
        if ("xmas".equals(m_codeType)) return "java"; // ;-)
        return m_codeType; 
    }
    public void setCodeType(String codeType) { m_codeType = codeType; }

    public void onLoadText(ActionEvent event)
    {
        m_codeType = ((BaseActionEvent)event).getSourceReference();
        readText();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void readText()
    {
        if (m_texts.containsKey(m_codeType) == false)
        {
            String text = new ClassloaderReader(true).readUTF8File("workplace/DemoCodeEditor_"+m_codeType+".txt",true);
            m_texts.put(m_codeType,text);
        }
    }
}
