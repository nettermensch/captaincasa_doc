The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Auto Completion
  • Passing Text with Id

The AUTOCOMPLETE component is also able to manage text information that is associated with a certain id. In this case the component is to be used in the following way:

(.) By binding the VALUE-attribute you just as normal receive the text content of the component. By in addition binding the VALUEID-attribute you receive the id that is associated with the text.

(.) The AutoCompleteProvider on server side needs to extend class DefaultAutoCompleteProviderTextId - and pass both text and id information.

The behaviour of the components is:
When a proposed value is picked from the auto-complete-list then an id is passed back to the processing, because then the text is associated with a certain id. - If the user just inputs some text without picking a value from the list then only the text is passed (VALUE attribute), no id is passed (VALUEID attribute).

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.util.AutoCompleteMgr;
import org.eclnt.jsfserver.util.DefaultAutoComplete2Provider;
import org.eclnt.jsfserver.util.IAutoCompleteProvider2.AutoCompleteItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoAutoCompleteIdText}")

public class DemoAutoCompleteIdText
    extends PageBean
    implements Serializable
{
      // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    class MyAutoCompleteProvider extends DefaultAutoComplete2Provider
    {
        public List<AutoCompleteItem> getProposals(String value)
        {
            List<AutoCompleteItem> result = new ArrayList<AutoCompleteItem>();
            if ("ABC".equals(value))
            {
                result.add(new AutoCompleteItem("ABC Text 1","1"));
                result.add(new AutoCompleteItem("ABC Text 2","2"));
                result.add(new AutoCompleteItem("ABC Text 3","3"));
                result.add(new AutoCompleteItem("ABC Text 4","4"));
                result.add(new AutoCompleteItem("ABC Text 5","5"));
                result.add(new AutoCompleteItem("ABC Text 6","6"));
                result.add(new AutoCompleteItem("ABC Text 7","7"));
                result.add(new AutoCompleteItem("ABC Text 8","8"));
                result.add(new AutoCompleteItem("ABC Text 9","9-1"));
                result.add(new AutoCompleteItem("ABC Text 9","9-2"));
                result.add(new AutoCompleteItem("ABC Text 9","9-3"));
            }
            else if ("ABC Text 9".equals(value))
            {
                result.add(new AutoCompleteItem("ABC Text 9","9-1"));
                result.add(new AutoCompleteItem("ABC Text 9","9-2"));
                result.add(new AutoCompleteItem("ABC Text 9","9-3"));
            }
            else
            {
                result.add(new AutoCompleteItem("No information found",""));
            }
            return result;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyAutoCompleteProvider m_acp;
    protected String m_value;
    protected String m_valueId;
    
  // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoAutoCompleteIdText(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("autocomplete"); 
        m_acp = new MyAutoCompleteProvider();
        AutoCompleteMgr.add(m_acp);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener() 
        {
            public void reactOnDestroyed()
            {
                super.reactOnDestroyed();
                AutoCompleteMgr.remove(m_acp);
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getValueURL() { return m_acp.getURL(); }

    public String getValue() 
    {
        if (m_value != null && m_value.startsWith("a"))
            return null;
        else
            return m_value; 
    }
    public void setValue(String value) { m_value = value; }

    public String getValueId() { return m_valueId; }
    public void setValueId(String value) { m_valueId = value; }

    public void onValueFlush(ActionEvent event) 
    {
        Statusbar.outputMessage("Passed value: " + m_value + " / Id: " + m_valueId);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
