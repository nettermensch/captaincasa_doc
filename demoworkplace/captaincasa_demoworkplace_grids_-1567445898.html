The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Sorting
  • Explicit sort- and export-value for column

Take a look into the definition of the last column: it is not sorted by its "value" but by its "sortValue". And if you call one of the export functions, then not the "value" is exported but the "exportValue".

A grid cell value is primarily referenced by the corresponding component inside the grid. But it is also used by default for sorting the grid and for exporting the grid.

There are situations in which you want to sort the grid by some different value - and there are situations in which you want to export some different value for the corresponding column.

An GRIDCOL component level there are two attributes to control this:

1. SORTREFERENCE
2. EXPORTREFERENCE

An EXPORTREFERENCE only needs to be defined if is different to the SORTREFERENCE.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.ColumnInfo;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.READTYPE;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridColValueMgmt}")

public class DemoGridColValueMgmt
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_exportValue;
        int i_sortValue;
        String i_value;
        public String getExportValue() { return i_exportValue; }
        public void setExportValue(String value) { this.i_exportValue = value; }
        public int getSortValue() { return i_sortValue; }
        public void setSortValue(int value) { this.i_sortValue = value; }
        public String getValue() { return i_value; }
        public void setValue(String value) { this.i_value = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    String m_protocol = "";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridColValueMgmt(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        fillGrid();
    }

    public String getPageName() { return "/workplace/demogridcolvaluemgmt.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridColValueMgmt}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public String getProtocol() { return m_protocol; }

    public void onOutputTextAction(javax.faces.event.ActionEvent event) 
    {
        StringBuffer sb = new StringBuffer();
        List<ColumnInfo> cols = m_grid.getVisibleColumns();
        for (GridItem gi: m_grid.getItems())
        {
            int counter = 0;
            for (ColumnInfo col: cols)
            {
                if (counter != 0) sb.append(", ");
                String cv = m_grid.getItemColumnClientStringValue(gi,col,READTYPE.EXPORT);
                sb.append(cv);
                counter++;
            }
            sb.append("\n");
        }
        m_protocol = sb.toString();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setValue(""+i);
            gi.setSortValue(i);
            gi.setExportValue("EXP "  + i);
            m_grid.getItems().add(gi);
        }
    }

}
