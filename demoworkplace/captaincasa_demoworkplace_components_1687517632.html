The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Organization Tree via PageBeanComponent

The Page Bean Component "CCGraphicalTree" which is part of the ccaddons-package allows a simple definition of an interactive, graphical tree.

Basically you just have to pass the hierarchy of nodes - that's it. All issues like collapsing/opening of sub nodes and selection of nodes are managed internally.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoPBCGraphicalTree}" >
<t:row id="g_2" >
<t:scrollpane id="g_3" background="#FFFFFF" border="#c0c0c0" height="100%" padding="20" width="100%" >
<t:row id="g_4" >
<t:pagebeancomponent id="g_5" pagebeanbinding="#{d.DemoPBCGraphicalTree.graphicalTree}" pagebeanclass="org.eclnt.ccaddons.pbc.CCGraphicalTree" pagebeaninitdata="boxWidth:120;boxHeight:50" />
</t:row>
</t:scrollpane>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Random;

import org.eclnt.ccaddons.pbc.CCDiagramBox;
import org.eclnt.ccaddons.pbc.CCDiagramBoxTitleAndText;
import org.eclnt.ccaddons.pbc.CCGraphicalTree;
import org.eclnt.ccaddons.pbc.util.GraphicalTreeNode;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPBCGraphicalTree}")

public class DemoPBCGraphicalTree
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static String[] FIRSTNAMES = new String[] {"Joseph","Jack","Barbara","Heinz","Claus","Henry","Josephine"};
    static String[] LASTNAMES = new String[] {"Miller","Brown","Carter","Meyer","Schmitz","McDonald"};
    
    private IListener m_listener;
    CCGraphicalTree m_graphicalTree;
    GraphicalTreeNode m_topNode;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCGraphicalTree(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        buildUpTree();
    }

    public String getPageName() { return "/workplace/demopbcgraphicaltree.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCGraphicalTree}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    public CCGraphicalTree getGraphicalTree() { return m_graphicalTree; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildUpTree()
    {
        Random r = new Random();
        CCDiagramBoxTitleAndText topBox = new CCDiagramBoxTitleAndText();
        topBox.setTitle("Company xyz");
        topBox.setText("Munich, Germany");
        topBox.setImage("/images/icons/package.png");
        topBox.setImageWidth(16);
        m_topNode = new GraphicalTreeNode(topBox);
        m_topNode.setCollapsed(false);
        int maxi = 4 + r.nextInt(4);
        for (int i=0; i<maxi; i++)
        {
            CCDiagramBoxTitleAndText locationBox = new CCDiagramBoxTitleAndText();
            locationBox.setTitle("Location " + i);
            locationBox.setText(r.nextInt(1000) + "employees");
            locationBox.setImage("/images/icons/chart_pie.png");
            locationBox.setImageWidth(16);
            GraphicalTreeNode locationNode = new GraphicalTreeNode(locationBox);
            locationNode.setCollapsed(true);
            m_topNode.addChildNode(locationNode);
            int maxj = r.nextInt(5) + 3;
            for (int j=0; j<maxj; j++)
            {
                CCDiagramBoxTitleAndText depBox = new CCDiagramBoxTitleAndText();
                depBox.setTitle("Department " + j);
                depBox.setText(buildName(r));
                depBox.setImage("/images/icons/person.png");
                depBox.setImageWidth(16);
                GraphicalTreeNode depNode = new GraphicalTreeNode(depBox);
                locationNode.addChildNode(depNode);
            }
        }
        m_graphicalTree = new CCGraphicalTree();
        m_graphicalTree.setTopNode(m_topNode);
        // listen to selection events
        m_graphicalTree.addListener(new CCGraphicalTree.IListener()
        {
            public void reactOnSelection(GraphicalTreeNode node, CCDiagramBox nodePageBean)
            {
                if (nodePageBean instanceof CCDiagramBoxTitleAndText)
                    Statusbar.outputSuccess("Selection of node: " + ((CCDiagramBoxTitleAndText)nodePageBean).getTitle());
            }
        });
    }
    
    private String buildName(Random r)
    {
        String firstName = FIRSTNAMES[r.nextInt(FIRSTNAMES.length)];
        String lastName = LASTNAMES[r.nextInt(LASTNAMES.length)];
        return lastName + ", " + firstName;
    }
}
