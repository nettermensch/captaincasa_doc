The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Bean data grid (PBC)
  • With own controls
  • With own header / footer

This demo shows how to add additional header and footer elements in the data grid view.

Please pay attention: the number of components that you add to the header/footer must exactly match the number of columns.

Please also take a look onto the calculation of the average height in the footer: the corresponding calculation is triggered by some IListener-event ("reactOnGridFiltered") and it uses method "getCurrentlyShownObjects" to obtain the list of items that is currently filtered.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPBCCCDataGridView2HeaderFooter}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPBCCCDataGridView2HeaderFooter}" >
<t:row id="g_25" >
<tpbc:datagridview2 id="g_2" pagebeanbinding="#{d.DemoPBCCCDataGridView2HeaderFooter.dataView}" shownullcontent="true" >
<tpbc:datagridview2column id="g_35" propertyName="firstName" title="First name" />
<tpbc:datagridview2column id="g_27" propertyName="lastName" title="Last  name" />
<tpbc:datagridview2column id="g_29" propertyName="height" title="Height" />
<tpbc:datagridview2column id="g_31" propertyName="birthDate" title="Date of birth" />
<tpbc:datagridview2column id="g_33" propertyName="married" title="Married" />
</tpbc:datagridview2>
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.COLDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDFOOTERNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDHEADERNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoPBCCCDataGridView2HeaderFooter}")

public class DemoPBCCCDataGridView2HeaderFooter
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /** 
     * The bean... This normally is an own class. In the demo it
     * is an inner class so that all code is kept within the
     * demo class.
     */
    public static class TestPerson
    {
        String i_firstName;
        String i_lastName;
        int i_height;
        LocalDate i_birthDate;
        boolean i_married;
        public String getFirstName() { return i_firstName; }
        public String getLastName() { return i_lastName; }
        public int getHeight() { return i_height; }
        public LocalDate getBirthDate() { return i_birthDate; }
        public boolean isMarried() { return i_married; }
        public void setFirstName(String firstName) { i_firstName = firstName; }
        public void setLastName(String lastName) { i_lastName = lastName; }
        public void setHeight(int height) { i_height = height; }
        public void setBirthDate(LocalDate birthDate) { i_birthDate = birthDate; }
        public void setMarried(boolean married) { i_married = married; }
    }

    /**
     * Listener to the data grid view's listener so that in order to 
     * react on events of the grid processing.
     */
    public class MyDGVListener extends CCDataGridView2.DefaultListener<TestPerson>
    {
        @Override
        public void reactOnGridSelect(TestPerson object)
        {
            reactOnItemSelection(object);
        }
        @Override
        public void reactOnGridFiltered()
        {
            refreshAverageHeight();
        }
    }
    
    public class MyDGV extends CCDataGridView2<TestPerson>
    {
        public MyDGV()
        {
            super(TestPerson.class);
        }
        @Override
        protected void beforeRenderHeader(FIXGRIDNode grid)
        {
            // add 2 additional rows before the filter header line
            for (int i=0; i<2; i++)
            {
                GRIDHEADERNode ghn = new GRIDHEADERNode();
                grid.addSubNode(ghn);
                // add components for the inidvidual components
                String background = "#C0C0FF";
                ghn.addSubNode(new BUTTONNode().setText("This").setBackground(background));
                ghn.addSubNode(new BUTTONNode().setText("is").setBackground(background));
                ghn.addSubNode(new BUTTONNode().setText("an").setBackground(background));
                ghn.addSubNode(new BUTTONNode().setText("additional").setBackground(background));
                ghn.addSubNode(new BUTTONNode().setText("header!").setBackground(background));
            }
        }
        @Override
        protected void beforeRenderFooter(FIXGRIDNode grid)
        {
            // add an additional row before the filter header line
            GRIDFOOTERNode gfn = new GRIDFOOTERNode();
            grid.addSubNode(gfn);
            // add components for the inidvidual components
            String background = "#C0FFC0";
            gfn.addSubNode(new PANENode().setBackground(background));
            gfn.addSubNode(new PANENode().setBackground(background));
            gfn.addSubNode(new LABELNode()
                    .setText(DemoPBCCCDataGridView2HeaderFooter.this.pbx("averageHeight"))
                    .setFont("weight:bold")
                    .setFormat("int")
                    .setAlign("right")
                    .setBackground(background));
            gfn.addSubNode(new PANENode().setBackground(background));
            gfn.addSubNode(new PANENode().setBackground(background));
        }
        
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyDGV m_dataView = new MyDGV();
    int m_averageHeight = 0;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDataGridView2HeaderFooter(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillData();
        refreshAverageHeight();
    }

    public String getPageName() { return "/workplace/demopbcccdatagridview2headerfooter.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCDataGridView2HeaderFooter}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public MyDGV getDataView() { return m_dataView; }
    public int getAverageHeight() { return m_averageHeight; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void reactOnItemSelection(TestPerson object)
    {
        Statusbar.outputSuccess("Selection of item: " + object.getLastName() + ", " + object.getFirstName());
    }

    private void fillData()
    {
        List<TestPerson> ps = new ArrayList<TestPerson>();
        for (int i=0; i<100; i++)
        {
            TestPerson p = new TestPerson();
            p.setFirstName(RandomLogic.getFirstName());
            p.setLastName(RandomLogic.getLastName());
            p.setHeight(RandomLogic.getHeight());
            p.setMarried(i%2 == 0);
            p.setBirthDate(RandomLogic.getBirthDate());
            ps.add(p);
        }
        m_dataView.prepare(ps,new MyDGVListener());
    }
    
    private void refreshAverageHeight()
    {
        m_averageHeight = 0;
        List<TestPerson> tps = m_dataView.getCurrentlyShownObjects();
        if (tps.size() > 0)
        {
            int totalHeight = 0;
            for (TestPerson tp: tps)
                totalHeight += tp.getHeight();
            m_averageHeight = Math.round((float)totalHeight/(float)tps.size());
        }
    }

}
