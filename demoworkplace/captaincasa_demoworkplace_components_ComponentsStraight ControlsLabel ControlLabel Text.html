The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Label Control
  • Label Text

The attribute LABELTEXT arranges a small, descriptive text on top of the component. It is consistently available for all components that are typically representing attribute data.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoExtLabelText}")

public class DemoExtLabelText
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_department = "0002";
    ValidValuesBinding m_departmentVVS = new ValidValuesBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoExtLabelText(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_departmentVVS.addValidValue("0001","Sales");
        m_departmentVVS.addValidValue("0002","Marketing");
        m_departmentVVS.addValidValue("0003","Production");
    }

    public String getPageName() { return "/workplace/demoextlabeltext.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoExtLabelText}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    public ValidValuesBinding getDepartmentVVS() { return m_departmentVVS; }

}
