The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Triggered by Server

There is a special file download component FILEDOWNLOAD. This component is not optically visible. You can trigger from the server side processing that this component should start the download of a certain URL into a certain file.

Result: when the server response is processed in the client, then a popup opens up asking the user to download a certain file.

An analogue component exists for the upload direction: FILEUPLOADASYNCHRONOUS - allowing to upload certain file content, triggered by the server side processing.

The triggering is done by using the TRIGGER attribute - pointing to a server side Trigger-instance.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.BufferedContentMgr;
import org.eclnt.jsfserver.bufferedcontent.DefaultBufferedContent;
import org.eclnt.jsfserver.bufferedcontent.DefaultUploadContent;
import org.eclnt.jsfserver.bufferedcontent.UploadContentMgr;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoFileDownloadAuto}")

public class DemoFileDownloadAuto 
    extends PageBean
    implements Serializable
{
    public class MyBufferedContent extends DefaultBufferedContent
    {
        public byte[] getContent()
        {
            byte[] pdfBytes = (new WebResourceReader()).readFileIntoByteArray("pdf/pdftest1.pdf");
            return pdfBytes;
        }
        public String getContentType()
        {
            return "application/pdf";
        }
    }
    
    public class MyUploadContent extends DefaultUploadContent
    {
        public void beginPassing()
        {
        }
        public void passClientFile(String fileName, byte[] content)
        {
            CLog.L.log(CLog.LL_INF,"Upload of content: is processed.");
            CLog.L.log(CLog.LL_INF,"Upload of content: ");
            m_uploadInfo = "Upload was executed:\n" +
                           "File name: " + fileName + "\n" +
                           "Content length: " + content.length;
        }
        public void endPassing()
        {
        }
    }
    
    Trigger m_downloadTrigger = new Trigger();
    Trigger m_uploadTrigger = new Trigger();
    String m_uploadInfo = "(No upload info yet.)";
    MyUploadContent m_uploadContent;
    MyBufferedContent m_bufferedContent;
    
    public DemoFileDownloadAuto(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_uploadContent = new MyUploadContent();
        m_bufferedContent = new MyBufferedContent();
        UploadContentMgr.add(m_uploadContent);
        BufferedContentMgr.add(m_bufferedContent);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnDestroyed()
            {
                UploadContentMgr.remove(m_uploadContent);
                m_uploadContent = null;
                m_bufferedContent = null;
            }
        });
    }
    
    public String getDownloadUrl() { return m_bufferedContent.getURL(); }
    public String getDownloadFileName() { return "some test document.pdf"; }
    public Trigger getDownloadTrigger() { return m_downloadTrigger; }
    
    public String getUploadUrl() { return m_uploadContent.getURL(); }
    public Trigger getUploadTrigger() { return m_uploadTrigger; }
    public String getUploadInfo() { return m_uploadInfo; }

    public void onCallAppUseTrigger(ActionEvent event)
    {
        // do application processing
        // ...
        // ...
        // ...
        m_downloadTrigger.trigger();
        // ...
        // ...
        // ...
    }
    
    public void onCallUploadTrigger(ActionEvent event)
    {
        // do application processing
        // ...
        // ...
        // ...
        m_uploadTrigger.trigger();
        // ...
        // ...
        // ...
    }
    
    public void onUploadFinished(ActionEvent event)
    {
        Statusbar.outputSuccess("Upload finished!");
    }
}
