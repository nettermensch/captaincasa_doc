The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Combining popup menus
  • Example

Press the right mouse button inside the fields.

It is possible two combine several popup menus by referencing them via a semicolon separated list of POPUPMENU-ids.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuLoad;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupMenuSequence}")

public class DemoPopupMenuSequence
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DYNAMICCONTENTBinding m_dynMenuContent = new DYNAMICCONTENTBinding();
    int m_counter = 0;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupMenuSequence(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demopopupmenusequence.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupMenuSequence}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynMenuContent() { return m_dynMenuContent; }
    
    public void onLabel1Action(javax.faces.event.ActionEvent event) { processAction(event); }
    public void onLabel2Action(javax.faces.event.ActionEvent event) { processAction(event); }
    public void onLabel12Action(javax.faces.event.ActionEvent event) { processAction(event); }

    public void onLabel3Action(javax.faces.event.ActionEvent event) { processAction(event); }
    public void onLabel123Action(javax.faces.event.ActionEvent event) { processAction(event); }

    public void onMenuItemAction(javax.faces.event.ActionEvent event) 
    {
        String ref = ((BaseActionEvent)event).getSourceReference();
        Statusbar.outputSuccess("Menu item action: " + ref);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void processAction(ActionEvent event)
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            Statusbar.outputSuccess("Component action: " + e.getCommand());
        }
        else if (event instanceof BaseActionEventPopupMenuLoad)
        {
            renderDynMenu();
            Statusbar.outputSuccess("Menu load action");
        }
    }
    
    private void renderDynMenu()
    {
        m_counter++;
        List<ComponentNode> nodes = new ArrayList<ComponentNode>();
        for (int i=0; i<3; i++)
        {
            MENUITEMNode m = new MENUITEMNode()
                    .setCommand("DYN"+i)
                    .setText("Dynamic text "  + m_counter);
            nodes.add(m);
        }
        m_dynMenuContent.setContentNodes(nodes);
    }

}
