The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Wizards

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowbodypane id="g_3" >
  <t:row id="g_4" >
    <t:button id="g_5" actionListener="#{d.demoWizard.onStartWizard}" text="Start Wizard inside Page" />
    <t:button id="g_6" actionListener="#{d.demoWizard.onStartWizardPopup}" text="Start Wizard as Popup" />
  </t:row>
  <t:rowdistance id="g_7" height="20" />
  <t:row id="g_8" >
    <t:pane id="g_9" border="#C0C0C0" height="100%" padding="1" width="100%" >
      <t:rowinclude id="g_10" page="#{d.demoWizard.pageName}" />
    </t:pane>
  </t:row>
</t:rowbodypane>
<t:pageaddons/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.Wizard;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup.IModelessPopupListener;
import org.eclnt.jsfserver.defaultscreens.Wizard.IWizardLogicProvider;
import org.eclnt.jsfserver.defaultscreens.Wizard.IWizardPageInfo;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoWizard extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class MyWizardPageInfo extends Wizard.WizardPageInfo
    {
        public MyWizardPageInfo(String arg0, String arg1, boolean arg2,
                boolean arg3)
        {
            super(arg0, arg1, arg2, arg3);
        }
    }
    
    public class WizardLogic implements IWizardLogicProvider, Serializable
    {
        public IWizardPageInfo[] getWizardPageInfos()
        {
            return new IWizardPageInfo[]
                                       {
                    new MyWizardPageInfo("Hello World!","/workplace/demohelloworld.jsp",false,true),
                    new MyWizardPageInfo("Panes","/workplace/demopanes.jsp",false,true),
                    new MyWizardPageInfo("Outlook bar","/workplace/demooutlookbar.jsp",true,true)
                                       };
        }
        public void exit()
        {
            m_modelessPopup.close();
        }
        public boolean checkIfNavigationIsPossible(IWizardPageInfo fromPage,
                                                   IWizardPageInfo toPage,
                                                   int directionOfNavigation)
        {
            return true;
        }
        public void prepareNextPage(IWizardPageInfo wp,
                                    int directionOfNavigation)
        {
            System.out.println("PREPARE CALLED FOR " + wp.getStep());
            System.out.println("PREPARE CALLED FOR " + wp.getStep());
            System.out.println("PREPARE CALLED FOR " + wp.getStep());
        }
        public void finish() 
        {
            if (m_modelessPopup != null)
            {
                m_modelessPopup.close();
                m_modelessPopup = null;
            }
        }
        public void cancel()
        {
            if (m_modelessPopup != null)
            {
                m_modelessPopup.close();
                m_modelessPopup = null;
            }
        }
        public boolean checkIfGridNavigationIsEnabled()
        {
            return true;
        }
         
    }
    
    private class MyWizardLogic extends WizardLogic
    {
        public void exit()
        {
            if (m_modelessPopup != null)
            {
                m_modelessPopup.close();
                m_modelessPopup = null;
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    String m_pageName = "/empty.jsp";
    ModelessPopup m_modelessPopup;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoWizard(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public void setPageName(String value) { m_pageName = value; }
    public String getPageName() { return m_pageName; }
    
    public void onStartWizard(ActionEvent ae)
    {
        m_pageName = "/eclntjsfserver/popups/wizard.jsp";
        Wizard.getInstance().prepare(new MyWizardLogic());
    }
    
    public void onStartWizardPopup(ActionEvent ae)
    {
        m_pageName = "/empty.jsp";
        Wizard.getInstance().prepare(new MyWizardLogic());
        if (m_modelessPopup == null)
        {
            m_modelessPopup = ModelessPopup.createInstance();
            IModelessPopupListener mpl = new IModelessPopupListener() {
                public void reactOnPopupClosedByUser()
                {
                    m_modelessPopup.close();
                    m_modelessPopup = null;
                }
            };
            m_modelessPopup.open("/eclntjsfserver/popups/wizard.jsp","Wizard",1000,600,mpl);
        }
    }
}
