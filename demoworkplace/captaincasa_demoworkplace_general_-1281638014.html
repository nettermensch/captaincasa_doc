The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Asynchronous Processing
  • Long Polling

This demo shows how to actively send events from the server to the client side processing. With a certain periodicity there is an event on server side, that causes the client to pick new data by executing a roundtrip.

The central element is the LONGPOLLING component, which is connected to a URL on the one hand and to an actionListener on the other hand. The URL is provided by a obect of type DefaultLongPolling, which you need to register and deregister. The request that is processed by the URL is waiting for a wake up signal on the server side. In the wake up you can pass the information if to continue with request processing or not.

Whenever the request from the LONGPOLLINGURL is passing back a result, the corresponding actionListener is invoked - which is then responsible for the "real roundtrip". - Imagine the LONGPOLLING component to be a virtual button that is pressed not by the user, but by a response coming from an http communication to the server.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.polling.DefaultLongPolling;
import org.eclnt.jsfserver.polling.LongPollingMgr;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.IWorkpageLifecycleListener;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoLongPolling}")

public class DemoLongPolling
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * This is the server side processing thread (as simulation...).
     * From time to time it calls the long polling wakup function in order
     * to trigger a long polling "event". (Event meaning, that a response
     * is passed back the client, which then triggeres a calling of
     * the LONGPOLLING's actionListener.
     */
    class ProcessingThread extends Thread
    {
        int i_counter = 0;
        @Override
        public void run()
        {
            while (i_counter <= 20)
            {
                try
                {
                    Thread.currentThread().sleep(1000); // simulation of "heavy processing"
                    // trigger long polling response
                    m_longPolling.wakeup(true);
                    i_counter++;
                }
                catch (Throwable t)
                {
                    CLog.L.log(CLog.LL_ERR,"Error occurred when waking up thread",t);
                    break;
                }
            }
            // end with polling
            m_processingThread = null;
            m_longPolling.wakeup(true);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    DefaultLongPolling m_longPolling = new DefaultLongPolling();
    ProcessingThread m_processingThread;
    String m_message = new String();
    int m_counter = 0;
    boolean m_longPollingRendered = false;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoLongPolling(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        LongPollingMgr.add(m_longPolling);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener() 
        {
            public void reactOnDestroyed()
            {
                LongPollingMgr.remove(m_longPolling); // clean up long polling
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getLongPollingURL() 
    { 
        return m_longPolling.getURL(); 
    }
    public boolean getLongPollingRendered() { return m_longPollingRendered; }

    public void onClear(ActionEvent event)
    {
        m_message = "";
    }
    
    /**
     * This method is called by the start button.
     */
    public void onStartPorcessing(ActionEvent event)
    {
        if (m_processingThread != null)
        {
            Statusbar.outputError("Server side process already started - please wait until current processing has finished.");
            return;
        }
        m_longPollingRendered = true;
        m_processingThread = new ProcessingThread();
        m_processingThread.start();
    }
    
    /**
     * This method is called by the LONGPOLLING component on client side, when
     * it receives a response.
     */
    public void onLongPollingAction(ActionEvent event)
    {
        if (m_processingThread != null)
        {
            int counter = m_processingThread.i_counter; 
            m_message = "Current counter in porocessing thread: " + m_processingThread.i_counter + "\n" + m_message;
        }
        else
        {
            m_message = "No processing thread available anymore.\n" +
                        "Stopping long polling.";
            m_longPollingRendered = false;
        }
    }   
    
    public String getMessage() { return m_message; }
    
}
