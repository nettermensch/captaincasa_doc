The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Client Integration
  • Barcode Scan Demo

This demo is an example for implementing a screen that is used with a barcode scanner that is connected by simulating keyboard input.

The focus is always redirected to the article id field so that the scanner input is directed into the right component.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.session.RequestFocusManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTouchScan}")

public class DemoTouchScan
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    public DemoTouchScan(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    protected Trigger m_soundTrigger = new Trigger();
    public Trigger getSoundTrigger() { return m_soundTrigger; }
    public void setSoundTrigger(Trigger value) { m_soundTrigger = value; }

    protected FIXGRIDListBinding<OrderRow> m_orderRows = new FIXGRIDListBinding<OrderRow>();
    public FIXGRIDListBinding<OrderRow> getOrderRows() { return m_orderRows; }
    public void setOrderRows(FIXGRIDListBinding<OrderRow> value) { m_orderRows = value; }

    public class OrderRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected int m_quantity;
        public int getQuantity() { return m_quantity; }
        public void setQuantity(int value) { m_quantity = value; }

        protected String m_articleText;
        public String getArticleText() { return m_articleText; }
        public void setArticleText(String value) { m_articleText = value; }

        protected String m_articleId;
        public String getArticleId() { return m_articleId; }
        public void setArticleId(String value) { m_articleId = value; }

    }

    protected int m_quantity = 1;
    public int getQuantity() { return m_quantity; }
    public void setQuantity(int value) { m_quantity = value; }

    public void onArticleIdAction(ActionEvent event) 
    {
        m_articleIdFocus = RequestFocusManager.getNewRequestFocusCounter()+"";   
        OrderRow or = new OrderRow();
        or.setArticleId(m_articleId);
        or.setArticleText("Text for " + m_articleId);
        or.setQuantity(m_quantity);
        m_orderRows.getItems().add(or);
        m_orderRows.deselectCurrentSelection();
        m_orderRows.selectItem(or);
        m_quantity = 1;
        m_articleId = "";
        m_soundTrigger.trigger();
    }
    
    public void onQuantityAction(ActionEvent event) 
    {
        m_articleIdFocus = RequestFocusManager.getNewRequestFocusCounter()+"";
    }
    
    public void onSetBarcodeFocus(ActionEvent event) 
    {
        m_articleIdFocus = RequestFocusManager.getNewRequestFocusCounter()+"";
    }

    protected String m_articleIdFocus = RequestFocusManager.getNewRequestFocusCounter()+"";
    public String getArticleIdFocus() { return m_articleIdFocus; }
    public void setArticleIdFocus(String value) { m_articleIdFocus = value; }

    protected String m_articleId;
    public String getArticleId() { return m_articleId; }
    public void setArticleId(String value) { m_articleId = value; }

}
