The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Adapter Binding
  • By annotation, with bean property

The base class "ComponentAdapterByAnnotationForBeanProperty" extends the functions of "ComponentAdapterByAnnotation" by adding one additional feature.

You define one component attribute to be resolved by a property of a bean. The bean itself is passed either directly or by an interface "IBeanAccess".

Advantage: if you e.g. use this adapter for a field component, then the value of the field is directly taken from e.g. a Pojo-bean's property - while all the other values for the component are added as additional properties of the adapter. This makes the usage of Pojo-beans much easier.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.adapter.ComponentAdapterByAnnotationForBeanProperty;
import org.eclnt.jsfserver.elements.adapter.ComponentAttribute;
import org.eclnt.jsfserver.elements.adapter.ComponentAttributeType;
import org.eclnt.jsfserver.elements.adapter.IBeanAccess;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;

@CCGenClass (expressionBase="#{d.Democabreflection2}")

public class Democabreflection2
    extends PageBean 
    implements Serializable, IBeanAccess<Person>
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    /**
     * In the demo: inner class. In real scenarios: real class that is used
     * by many page beans.
     */
    public class PropertyAdapter extends ComponentAdapterByAnnotationForBeanProperty<Person>
    {
        String i_label;
        boolean i_containsError;
        String i_errorText;
        boolean i_mandatory = false;
        public PropertyAdapter(String componentAttribute, 
                               IBeanAccess<Person> beanAccess, 
                               String property,
                               String label)
        {
            super(componentAttribute, beanAccess, property);
            i_label = label;
        }
        @ComponentAttribute(type=ComponentAttributeType.FIX,attribute="labeltext")
        public String getLabel() { return i_label; }
        @ComponentAttribute
        public String getBgpaint()
        {
            if (i_containsError)
                return "error()";
            else if (i_mandatory)
                return "mandatory()";
            else
                return null;
        }
        @ComponentAttribute
        public String getTooltip()
        {
            if (i_containsError == true)
                return i_errorText;
            else
                return null;
        }
        public void indicateError(String text) 
        { 
            i_containsError = true;
            i_errorText = text;
        }
        public void resetError() 
        { 
            i_containsError = false;
            i_errorText = null;
        }
        public void setMandatory(boolean value) { i_mandatory = value; }
    }

    /**
     * In the demo: inner class. In real scenarios: real class that is used
     * by many page beans.
     */
    public class PropertyComboAdapter extends PropertyAdapter
    {
        ValidValuesBinding i_vvb = new ValidValuesBinding();
        public PropertyComboAdapter(String componentAttribute, 
                                    IBeanAccess<Person> beanAccess, 
                                    String property,
                                    String label)
        {
            super(componentAttribute, beanAccess, property, label);
        }
        @ComponentAttribute(attribute="validvaluesbinding")
        public ValidValuesBinding getVvb() { return i_vvb; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Person m_person;
    Map<String,PropertyAdapter> m_adapters = new HashMap<String,PropertyAdapter>();
    
    PropertyAdapter m_firstName; 
    PropertyAdapter m_lastName; 
    PropertyAdapter m_title; 
    PropertyAdapter m_gender;
    PropertyComboAdapter m_department;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public Democabreflection2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_person = new Person();
        m_firstName = createPropertyAdapter("text",Person.PROP_FIRSTNAME,"First name");
        m_firstName.setMandatory(true);
        m_lastName = createPropertyAdapter("text",Person.PROP_LASTNAME,"Last name");
        m_lastName.setMandatory(true);
        m_title = createPropertyAdapter("text",Person.PROP_TITLE,"Title");
        m_gender = createPropertyAdapter("value",Person.PROP_GENDER,"Gender");
        m_department = createPropertyComboAdapter("value",Person.PROP_DEPARTMENT,"Department");
        m_department.setMandatory(true);
        m_department.getVvb().addValidValue("0001","Sales");
        m_department.getVvb().addValidValue("0002","Marketing");
        m_department.getVvb().addValidValue("0003","Production");
        m_department.getVvb().addValidValue("0004","Human resources");
    }

    public String getPageName() { return "/workplace/democabreflection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCabReflection}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    @Override
    public Person getBean() { return m_person; }

    public Person getPerson() { return m_person; } 
    
    public PropertyAdapter getFirstName() { return m_firstName; } 
    public PropertyAdapter getLastName() { return m_lastName; } 
    public PropertyAdapter getTitle() { return m_title; } 
    public PropertyAdapter getGender() { return m_gender; }
    public PropertyComboAdapter getDepartment() { return m_department; }

    public void onCheckAction(ActionEvent event)
    {
        clearAdapterErrors();
        processChecks();
        if (checkIfError() == true)
            Statusbar.outputError("Person contains error - please check.");
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    /**
     * In demo: checks are done directly in page bean (which means: on view layer).
     * Better: checks to be done on logic layer!
     */
    private void processChecks()
    {
        if (m_person.getFirstName() == null || m_person.getFirstName().trim().length() == 0)
            m_firstName.indicateError("First name does not contain any value.");
        if (m_person.getLastName() == null || m_person.getLastName().trim().length() == 0)
            m_lastName.indicateError("Last name does not contain any value.");
        if (m_person.getDepartment() == null || m_person.getDepartment().trim().length() == 0)
            m_department.indicateError("Department does not contain any value.");
    }
    
    /**
     * In real scenario: this method is part of some base class.
     */
    private PropertyAdapter createPropertyAdapter(String componentAttribute, 
                                                  String property,
                                                  String label)
    {
        PropertyAdapter result = new PropertyAdapter(componentAttribute,this,property,label);
        m_adapters.put(property,result);
        return result;
    }
    
    /**
     * In real scenario: this method is part of some base class.
     */
    private PropertyComboAdapter createPropertyComboAdapter(String componentAttribute, 
                                                            String property,
                                                            String label)
    {
        PropertyComboAdapter result = new PropertyComboAdapter(componentAttribute,this,property,label);
        m_adapters.put(property,result);
        return result;
    }

    /**
     * In real scenario: this method is part of some base class.
     */
    private void clearAdapterErrors()
    {
        for (PropertyAdapter pa: m_adapters.values())
            pa.resetError();
    }

    /**
     * In real scenario: this method is part of some base class.
     */
    private boolean checkIfError()
    {
        for (PropertyAdapter pa: m_adapters.values())
            if (pa.i_containsError == true)
                return true;
        return false;
    }

}
