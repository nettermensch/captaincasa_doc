The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Large Output Lists
  • Data Types in Array Grid

This demo shows how to define different data types within an array grid.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.Date;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDItem;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoArrayGridDataTypes}")

public class DemoArrayGridDataTypes
    extends PageBean 
    implements Serializable
{
    public class GridItem extends ARRAYGRIDItem implements java.io.Serializable
    {
    }

    public DemoArrayGridDataTypes(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_grid.setTitles(new String[] {"Date of Birth","Town of Birth","Weight at birth (kg)","Height at birth (cm)","female"});
        m_grid.setAligns(new String[] {"left","left","right","right","left"});
        m_grid.setWidths(new String[] {"120","100%;200","100","100","50"});
        m_grid.setFormats(new String[] {"datetime",null,"bigdecimal","int","boolean"});
        m_grid.setFormatmasks(new String[] {"medium",null,"dec3",null,null});
        onFillGrid(null);
    }

    protected ARRAYGRIDListBinding<GridItem> m_grid = new ARRAYGRIDListBinding<GridItem>();
    public ARRAYGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onFillGrid(ActionEvent event) 
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            // somtimes data, sometimes null
            if (i%3 != 2)
                gi.setValues(new Object[] {new Date(),"Town " + i,new BigDecimal("3.23").add(new BigDecimal(i)),new Integer(50+i),true});
            else
                gi.setValues(new Object[] {null,"Town " + i,null,null,false});
            // sometimes images
            if (i%4 == 0)
            {
                gi.setImages(new String[] {null,"/images/icons/add.png",null,null,null});
            }
            m_grid.getItems().add(gi);
        }
    }

}
