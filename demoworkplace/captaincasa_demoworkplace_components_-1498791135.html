The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Menubars

Menus are easily defined: they are an arrangement of MENUBAR, MENU and MENUITEM components.

The top menu bar of this page (the one which is direclty below the titlebar) is statically defined. Please have a look into the XML layout. Each item is associated with an action listener - just like with normal button or icon controls.

The second menu bar is dynamically built. Via the attribute COMPONENTBINDING it points into a server side implementation of the menu bar. Check the Java code and you will see that the components are assembled by the server side program.

package workplace;

import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseComponent;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.MENUNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.MENUBARComponent;
import org.eclnt.jsfserver.elements.impl.MENUComponentTag;
import org.eclnt.jsfserver.elements.impl.MENUITEMComponentTag;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoMenu extends PageBean
{    
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Invoker
    {
        String i_text;
        public Invoker(String text)
        {
            i_text = text;
        }
        public void onInvoke(ActionEvent event) 
        {
            Statusbar.outputSuccess("Menu item was pressed: " + i_text);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    DYNAMICCONTENTBinding m_menuContent = new DYNAMICCONTENTBinding();
    List<Invoker> m_invokers = new ArrayList<Invoker>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoMenu(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("menubar");
        renderDynamicMenuContent();
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public DYNAMICCONTENTBinding getMenuContent() { return m_menuContent; }
    public List<Invoker> getInvokers() { return m_invokers; }

    public void onClose(ActionEvent ae)
    {
        Statusbar.outputMessage("Close was pressed");
    }
    
    public void onSave(ActionEvent ae)
    {
        Statusbar.outputMessage("Save was pressed");
    }

    public void onHelp(ActionEvent ae)
    {
        Statusbar.outputMessage("Help was pressed");
    }

    private void renderDynamicMenuContent()
    {
        m_invokers.clear();
        List<ComponentNode> contentNodes = new ArrayList<ComponentNode>();
        for (int i=0; i<5; i++)
        {
            MENUNode menuNode = new MENUNode();
            contentNodes.add(menuNode);
            menuNode.setText("Menu " + i);
            menuNode.setFont("weight:bold");
            for (int j=0; j<5; j++)
            {
                String itemText = "Item "+j+"/"+i;
                Invoker invoker = new Invoker(itemText);
                m_invokers.add(invoker);
                MENUITEMNode menuItemNode = new MENUITEMNode();
                menuNode.addSubNode(menuItemNode);
                menuItemNode.setForeground("#800000");
                menuItemNode.setFont("size:"+(12+(j*2)));
                menuItemNode.addAttribute("text",itemText);
                menuItemNode.addAttribute("actionListener","#{d.DemoMenu.invokers["+(m_invokers.size()-1)+"].onInvoke}");
            }
        }
        m_menuContent.setContentNodes(contentNodes);

//            }

    }

}
