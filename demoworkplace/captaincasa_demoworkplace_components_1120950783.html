The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Date Input
  • From - to validity check

With the CALENDARFIELD component you can define a FROMDATE and TODATE attribute - restricting the date input to a certain time range.

Please note: it's extremely important to pay attention to the milliseconds of a date representation - because that's the actual level the comparison is executed on.
In case of using java.util.Date you may set the milliseconds to 0 (or some other defined value) by using the Calendar-object, as shown in the example.
In case of using java.time.LocalDateTime you may set the milliseconds to 0 by using method ".withNano(0)".

This is the Java code:

package workplace;

import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoCalendarFromTo}")

public class DemoCalendarFromTo extends PageBean
{
    public DemoCalendarFromTo(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_date = new Date();
        TimeZone tz = TimeZone.getDefault();
        m_timezone = tz.getID();
        Calendar cal = Calendar.getInstance(tz);
        // right size from date
        m_fromDate = new Date();
        cal.setTime(m_fromDate);
        cal.add(Calendar.DAY_OF_MONTH,-5);
        cal.set(Calendar.HOUR_OF_DAY,0);
        cal.set(Calendar.MINUTE,0);
        cal.set(Calendar.SECOND,0);
        cal.set(Calendar.MILLISECOND,0);
        m_fromDate = cal.getTime();
        // right size to date
        m_toDate = new Date();
        cal.setTime(m_toDate);
        cal.add(Calendar.DAY_OF_MONTH,5);
        cal.set(Calendar.HOUR_OF_DAY,23);
        cal.set(Calendar.MINUTE,59);
        cal.set(Calendar.SECOND,59);
        cal.set(Calendar.MILLISECOND,999);
        m_toDate = cal.getTime();
    }

    protected String m_timezone;
    public String getTimezone() { return m_timezone; }
    public void setTimezone(String value) { m_timezone = value; }

    protected Date m_toDate;
    public Date getToDate() { return m_toDate; }
    public void setToDate(Date value) { m_toDate = value; }

    protected Date m_fromDate;
    public Date getFromDate() { return m_fromDate; }
    public void setFromDate(Date value) { m_fromDate = value; }

    protected Date m_date;
    public Date getDate() { return m_date; }
    public void setDate(Date value) { m_date = value; }

}
