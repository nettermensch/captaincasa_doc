The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Adapter Binding
  • By annotation, "self configuring"

This example of using the adapter binding shows two important apsects:

1.

The adapters are implementing interface IComponentAdapter2. This interface extends IComponentAdapter by an additional method, in which at runtime the component, that uses the adapter binding, is passed (method: initComponent(IBaseComponent component)).

AS consequence the definition of the attribute name that serves the component can be directly derived from the component name. Of course any other behavior now can be defined dependent on the component as well.

2.

The adapters are not kept in fix member variables, but in a map (m_adapters). The map is a "lazy loading" map, i.e. its content is created at this point of time when it is required. In other words: the map automatically fills itself during usage.


Please note: this example assembles all aspects into one class, which are normally arranged in some base class - and some extending class!

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.IBaseComponent;
import org.eclnt.jsfserver.elements.adapter.ComponentAdapterByAnnotationForBeanProperty;
import org.eclnt.jsfserver.elements.adapter.ComponentAttribute;
import org.eclnt.jsfserver.elements.adapter.ComponentAttributeType;
import org.eclnt.jsfserver.elements.adapter.IBeanAccess;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.util.map.LazyLoadingMap;
import org.eclnt.util.map.LazyLoadingMap.ILazyLoader;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;

@CCGenClass (expressionBase="#{d.Democabreflection4}")

public class Democabreflection4
    extends PageBean 
    implements Serializable, IBeanAccess<Person>
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    /**
     * In the demo: inner class. In real scenarios: real class that is used
     * by many page beans.
     * <br><br>
     * In the method initComponent at runtime the component is passed. Dependent
     * on the component the attribute name is initialized.
     */
    public static class PropertyAdapter extends ComponentAdapterByAnnotationForBeanProperty<Person>
    {
        String i_label;
        boolean i_containsError;
        String i_errorText;
        boolean i_mandatory = false;
        ValidValuesBinding i_vvb = null;
        public PropertyAdapter(IBeanAccess<Person> beanAccess,String property)
        {
            super(beanAccess, property);
            i_label = property.toUpperCase();
        }
        @Override
        public void initComponent(IBaseComponent component)
        {
            super.initComponent(component);
            String tagName = component.getTagNameWithPrefix();
            switch (tagName)
            {
                case "t:field":
                case "t:combofield":
                case "t:textarea":
                case "t:label":
                    initComponentAttribute("text");
                    break;
                case "t:formattedfield":
                case "t:calendarfield":
                case "t:combobox":
                case "t:radiobutton":
                    initComponentAttribute("value");
                    break;
                case "t:checkbox":
                    initComponentAttribute("selected");
                    break;
                default:
                    throw new Error("Adapter not prepared to be used for component: " + tagName);
            }
        }
        @ComponentAttribute(type=ComponentAttributeType.FIX,attribute="labeltext")
        public String getLabel() { return i_label; }
        @ComponentAttribute
        public String getBgpaint()
        {
            if (i_containsError)
                return "error()";
            else if (i_mandatory)
                return "mandatory()";
            else
                return null;
        }
        @ComponentAttribute
        public String getTooltip()
        {
            if (i_containsError == true)
                return i_errorText;
            else
                return null;
        }
        public void indicateError(String text) 
        { 
            i_containsError = true;
            i_errorText = text;
        }
        public void resetError() 
        { 
            i_containsError = false;
            i_errorText = null;
        }
        public void setMandatory(boolean value) { i_mandatory = value; }
        @ComponentAttribute(attribute="validvaluesbinding")
        public ValidValuesBinding getVvb() 
        { 
            if (i_vvb == null) i_vvb = new ValidValuesBinding();
            return i_vvb; 
        }
        @ComponentAttribute
        public boolean getWithnullitem() { return true; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Person m_person;
    LazyLoadingMap<PropertyAdapter> m_adapters = new LazyLoadingMap<PropertyAdapter>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public Democabreflection4(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_adapters.setLoader(new ILazyLoader<PropertyAdapter>()
        {
            @Override
            public PropertyAdapter lazyLoad(String key)
            {
                return new PropertyAdapter(Democabreflection4.this,key);
            }
        });
        // configure property adapters
        m_adapters.get("firstName").setMandatory(true);
        m_adapters.get("lastName").setMandatory(true);
        m_adapters.get("department").setMandatory(true);
        m_adapters.get("department").getVvb().addValidValue("0001","Sales");
        m_adapters.get("department").getVvb().addValidValue("0002","Marketing");
        m_adapters.get("department").getVvb().addValidValue("0003","Production");
        m_adapters.get("department").getVvb().addValidValue("0004","Human resources");
        m_adapters.get("title").getVvb().addValidValue("DR","Dr");
        m_adapters.get("title").getVvb().addValidValue("PROF","Prof");
        // create some bean
        m_person = new Person();
        m_person.setFirstName("Captain");
        m_person.setLastName("Casa");
        m_person.setTitle("DR");
    }

    public String getPageName() { return "/workplace/democabreflection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCabReflection}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    @Override
    public Person getBean() { return m_person; }
    public Person getPerson() { return m_person; } 
    public LazyLoadingMap<PropertyAdapter> getAdapters() { return m_adapters; }

    public void onCheckAction(ActionEvent event)
    {
        clearAdapterErrors();
        processChecks();
        if (checkIfError() == true)
            Statusbar.outputError("Person contains error - please check.");
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    /**
     * In demo: checks are done directly in page bean (which means: on view layer).
     * Better: checks to be done on logic layer!
     */
    private void processChecks()
    {
        if (m_person.getFirstName() == null || m_person.getFirstName().trim().length() == 0)
            indicateError("First name does not contain any value.","firstName");
        if (m_person.getLastName() == null || m_person.getLastName().trim().length() == 0)
            indicateError("Last name does not contain any value.","lastName");
        if (m_person.getDepartment() == null || m_person.getDepartment().trim().length() == 0)
            indicateError("Department does not contain any value.","department");
    }
    
    /**
     * In real scenario: this method is part of some base class.
     */
    private void clearAdapterErrors()
    {
        for (PropertyAdapter pa: m_adapters.values())
            pa.resetError();
    }

    /**
     * In real scenario: this method is part of some base class.
     */
    private boolean checkIfError()
    {
        for (PropertyAdapter pa: m_adapters.values())
            if (pa.i_containsError == true)
                return true;
        return false;
    }

    private void indicateError(String message, String... properties)
    {
        for (String property: properties)
        {
            PropertyAdapter pa = m_adapters.getDirectly(property);
            if (pa != null) pa.indicateError(message);
        }
    }
}
