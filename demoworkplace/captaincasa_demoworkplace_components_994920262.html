The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Hot Keys
  • Hotkey bar

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

<t:rowbodypane id="g_16" padding="0" popupmenu="BODY" >
<t:row id="g_2" >
<t:coldistance id="g_7" width="#{ccstylevalue[&#39;@defaultcoldistance@&#39;]}" />

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoHotkeyBar}")

public class DemoHotkeyBar
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    public ValidValuesBinding getKeyFilterVVB() { return m_keyFilterVVB; }
    String m_keyFilter = "all";
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoHotkeyBar(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_keyFilterVVB.addValidValue("all","All keys");
        m_keyFilterVVB.addValidValue("fkeysonly","Function keys only");
    }

    public String getPageName() { return "/workplace/demohotkeybar.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoHotkeyBar}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    ValidValuesBinding m_keyFilterVVB = new ValidValuesBinding();

    public String getKeyFilter() { return m_keyFilter; }
    public void setKeyFilter(String value) { this.m_keyFilter = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
