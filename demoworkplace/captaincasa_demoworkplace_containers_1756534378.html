The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Positioning in free style HTML
  • Outside-in sizing "Excel"

This HTML was created by some Excel export. The controls placed inside are multiline TEXTEDITOR components.

Please observe: the table resizes the height of the components on input.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoTextWithControlsOISExcel}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoTextWithControlsOISExcel}" >
<t:row id="g_10" >
<t:combobox id="g_13" actionListener="#{d.DemoTextWithControlsOISExcel.onHtmlTemplateAction}" flush="true" labeltext="HTML template" validvaluesbinding="#{d.DemoTextWithControlsOISExcel.htmlTemplateVVS}" value="#{d.DemoTextWithControlsOISExcel.htmlTemplate}" width="300" />
<t:coldistance id="g_2" width="50" />
<t:button id="g_15" actionListener="#{d.DemoTextWithControlsOISExcel.onReloadAction}" hotkey="alt-82" text="Reload HTML" width="100+" />
</t:row>
<t:row id="g_6" >

</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.TEXTEDITORNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTextWithControlsOISExcel}")

public class DemoTextWithControlsOISExcel
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ValidValuesBinding m_htmlTemplateVVS = new ValidValuesBinding();
    String m_htmlTemplate = "oissampleexcel1.html";
    String m_htmlText;
    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();
    Map<String,String> m_data = new HashMap<String,String>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTextWithControlsOISExcel(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_htmlTemplateVVS.addValidValue("oissampleexcel1.html","oissampleexcel1.html");
        m_htmlTemplateVVS.addValidValue("oissampleexcel2.html","oissampleexcel2.html");
        m_htmlTemplateVVS.addValidValue("oissampleexcel3.html","oissampleexcel3.html");
        readHTMLText();
        createFields();
    }

    public String getPageName() { return "/workplace/demotextwithcontrolsoisexcel.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTextWithControlsOISExcel}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getHtmlTemplateVVS() { return m_htmlTemplateVVS; }
    public String getHtmlTemplate() { return m_htmlTemplate; }
    public void setHtmlTemplate(String value) { this.m_htmlTemplate = value; }
    public String getHtmlText() { return m_htmlText; }
    public Map<String, String> getData() { return m_data; } 
    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    public void onReloadAction(javax.faces.event.ActionEvent event) 
    {
        m_htmlText = m_htmlText + " ";
    }

    public void onHtmlTemplateAction(javax.faces.event.ActionEvent event)
    {
        readHTMLText();
        createFields();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void readHTMLText()
    {
        m_htmlText = new ClassloaderReader(true).readUTF8File("/workplace/resources/"+m_htmlTemplate,true);
    }

    private void createFields()
    {
        m_data.clear();
        int counter = 0;
        List<String> ids = findIds();
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (String id: ids)
        {
            TEXTEDITORNode f = new TEXTEDITORNode()
//            FIELDNode f = new FIELDNode()
                .setId(id)
                .setText(pbx("data."+id))
                .setAlign("left");
            l.add(f);
            m_data.put(id,"Value " + counter);
            counter++;
        }
        m_dynContent.setContentNodes(l);
    }
    
    private List<String> findIds()
    {
        List<String> result = new ArrayList<String>();
        int currentIndex = 0;
        while (true)
        {
            int pos1 = m_htmlText.indexOf("@@",currentIndex);
            if (pos1 < 0) break;
            int pos2 = m_htmlText.indexOf("@@",pos1+2);
            if (pos2 < 0) break;
            result.add(m_htmlText.substring(pos1+2,pos2));
            currentIndex = pos2+2;
        }
        return result;
    }
    
}
