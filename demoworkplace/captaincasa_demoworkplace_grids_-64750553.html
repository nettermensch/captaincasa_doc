The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • Dynamic Grid - Column level

You may place a DYNAMICCONTENT component into the FIXGRID - and as result render all or some columns dynamically.

In the example the number of columns can be set, after pressing the button "Render" the grid is updated accordingly.

The dynamically created columns access a map "data" which is available on item level.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoGridDynContentColumns}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoGridDynContentColumns}" rowdistance="#{ccstylevalue[&#39;@defaultrowdistance@&#39;]}" >
<t:row id="g_5" >
<t:formattedfield id="g_6" format="int" labeltext="Number of dynamic columns" value="#{d.DemoGridDynContentColumns.numberOfColumns}" width="250" />
<t:button id="g_7" actionListener="#{d.DemoGridDynContentColumns.onRenderAction}" text="Render columns" width="150" />
</t:row>
<t:row id="g_11" >
<t:fixgrid id="g_2" height="100%" objectbinding="#{d.DemoGridDynContentColumns.grid}" width="100%" >
<t:gridcol id="g_3" text="Fix column 1" width="100%;100" >
<t:label id="g_8" text=".{text1}" />
</t:gridcol>
<t:dynamiccontent id="g_9" contentbinding="#{d.DemoGridDynContentColumns.dymColumns1}" />
<t:gridcol id="g_13" text="Fix column 2" width="100%;100" >
<t:label id="g_14" text=".{text2}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridDynContentColumns}")

public class DemoGridDynContentColumns
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_text1;
        String i_text2;
        Map<String,String> m_data = new HashMap<String,String>();
        public GridItem()
        {
            Random r = new Random();
            for (int i=0; i<20; i++)
                m_data.put("dyn"+i,"Dyn " + i + "/" + r.nextInt(100));
        }
        public String getText1() { return i_text1; }
        public void setText1(String value) { this.i_text1 = value; }
        public String getText2() { return i_text2; }
        public void setText2(String value) { this.i_text2 = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        public Map<String, String> getData() { return m_data; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DYNAMICCONTENTBinding m_dymColumns1 = new DYNAMICCONTENTBinding();
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    int m_numberOfColumns = 3;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridDynContentColumns(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        renderColumns();
        fillData();
    }

    public String getPageName() { return "/workplace/demogriddyncontentcolumns.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridDynContentColumns}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDymColumns1() { return m_dymColumns1; }
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public int getNumberOfColumns() { return m_numberOfColumns; } 
    public void setNumberOfColumns(int numberOfColumns) { m_numberOfColumns = numberOfColumns; }

    public void onRenderAction(javax.faces.event.ActionEvent event) 
    {
        renderColumns();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderColumns()
    {
        List<ComponentNode> cs = new ArrayList<ComponentNode>();
        for (int i=0; i<m_numberOfColumns; i++)
        {
            GRIDCOLNode gcn = new GRIDCOLNode()
                .setText("Dyn " + i)
                .setWidth("100%;100");
            cs.add(gcn);
            LABELNode ln = new LABELNode()
                .setText(".{data.dyn"+i+"}");
            gcn.addSubNode(ln);
        }
        m_dymColumns1.setContentNodes(cs);
    }

    private void fillData()
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setText1("Text 1 " + i);
            gi.setText2("Text 2 " + i);
            m_grid.getItems().add(gi);
        }
    }

}
