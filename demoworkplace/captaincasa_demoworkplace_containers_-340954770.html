The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Label opening up detail area
  • Label flip - events

By defining attribute WITHFLIPEVENTCONTENT and WITHFLIPEVENTLABEL you can control if events will be sent to the server side when the display status of the LABELFLIP component is changed by the user.

There are two event types: "BaseActionEventFlipContent" and "BaseActionEventFlipLabel" on server side.

In the event processing you may e.g. prepare the detail data that is shown in the content area when the user flips from label to content.
Vice versa you may process or store input data of the detail content when the user flips back to the label.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventFlipContent;
import org.eclnt.jsfserver.elements.events.BaseActionEventFlipLabel;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoLabelFlipEvents}")

public class DemoLabelFlipEvents
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_zipCode = "69245";
    String m_town = "Bammental";
    String m_street = "Hindemithweg";

    Trigger m_triggerShowLabel = new Trigger();
    Trigger m_triggerShowContent = new Trigger();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLabelFlipEvents(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demolabelflipevents.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLabelFlipEvents}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getZipCode() { return m_zipCode; }
    public void setZipCode(String value) { this.m_zipCode = value; }

    public String getTown() { return m_town; }
    public void setTown(String value) { this.m_town = value; }

    public String getStreet() { return m_street; }
    public void setStreet(String value) { this.m_street = value; }

    public String getAddress() { return m_zipCode + " " + m_town + ", " + m_street; }

    public Trigger getTriggerShowLabel() { return m_triggerShowLabel; }

    public Trigger getTriggerShowContent() { return m_triggerShowContent; }

    public void onTriggerShowLabel(javax.faces.event.ActionEvent event) 
    {
        m_triggerShowLabel.trigger();
    }

    public void onTriggerShowContent(javax.faces.event.ActionEvent event) 
    {
        m_triggerShowContent.trigger();
    }

    
    public void onFlipAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventFlipContent)
        {
            // you may e.g. load data here that is needed for editing
            Statusbar.outputSuccess("Now showing the editable content.");
        }
        else if (event instanceof BaseActionEventFlipLabel)
        {
            // you may e.g. store data here that was edited
            Statusbar.outputSuccess("Now showing the label.");
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
