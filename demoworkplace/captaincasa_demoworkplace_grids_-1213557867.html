The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Repeat Processing
  • Grid inside REPEAT

This demo shows a FIXGRID component that is nested within a REPEAT component.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoRepeatWithGrid}")

public class DemoRepeatWithGrid
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        
    }
    
    public class ListItem extends Object implements java.io.Serializable
    {
        String i_departmentName;
        FIXGRIDListBinding<GridItem> i_grid = new FIXGRIDListBinding<GridItem>();
        public String getDepartmentName() { return i_departmentName; }
        public FIXGRIDListBinding<GridItem> getGrid() { return i_grid; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<ListItem> m_list = new ArrayList<ListItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRepeatWithGrid(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        Random r = new Random();
        for (int i=0; i<5; i++)
        {
            ListItem li = new ListItem();
            li.i_departmentName = "Department " + i;
            m_list.add(li);
            int jMax = r.nextInt(10);
            for (int j=0; j<jMax; j++)
            {
                GridItem gi = new GridItem();
                li.i_grid.getItems().add(gi);
                gi.i_firstName = "First Name " + i + "/" + j;
                gi.i_lastName = "Last Name " + i + "/" + j;
            }
        }
    }

    public String getPageName() { return "/workplace/demorepeatwithgrid.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRepeatWithGrid}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<ListItem> getList() { return m_list; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
