The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Field with floating label

By setting attribute LABELTEXT within all FIELD-components (FIELD, CALENDARFIELD, COMBOFIELD, CALENDARFIELD, SPINNER, ...) the label text will be shown as so called floating text: if first is shown as background text. Once the user defines some value, then the text moves up into the left top corner of the field.

When setting LABELTEXTALWAYSHOWN to "true" then there is no floating text, but the label text is always shown on the left top.

In order to harmonize the field's look and feel with checkboxes and radio buttons there is the component LABELCONTAINER: it arranges the content component into a container, that is in sync with the field's look and feel.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoLabelContainer}")

public class DemoLabelContainer
    extends PageBean 
    implements Serializable
{
    static Map<String,String> DEPARTMENTS = new HashMap<String, String>();
    
    static
    {
        DEPARTMENTS.put("0001","Production");
        DEPARTMENTS.put("0002","Sales");
        DEPARTMENTS.put("0003","Research");
        DEPARTMENTS.put("0004","Human Resources");
    }
    
    public String getDepartmentText() 
    { 
        if (m_department == null)
            return null;
        String result = DEPARTMENTS.get(m_department);
        if (result == null)
            result = "(unknown)";
        return result;
    }

    public void onDepartmentAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            IdTextSelection idts = IdTextSelection.createInstance();
            for (String key: DEPARTMENTS.keySet())
            {
                idts.addLine(key,DEPARTMENTS.get(key));
            }
            idts.sortItemsById();
            idts.setCallBack(new ISetId()
            {
                @Override
                public void setId(String id)
                {
                    m_department = id;
                }
            });
        }
    }

    String m_department;
    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    Integer m_weight = null;
    public int getWeight() { return m_weight; }
    public void setWeight(int value) { this.m_weight = value; }

    String m_lastName;
    String m_firstName;
    String m_gender;
    boolean m_archived;
    String m_country;
    LocalDate m_birthDate;
    
    public DemoLabelContainer(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demolabelcontainer.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLabelContainer}"; }

    public String getGender() { return m_gender; }
    public void setGender(String value) { this.m_gender = value; }

    public boolean getArchived() { return m_archived; }
    public void setArchived(boolean value) { this.m_archived = value; }

    public String getCountry() { return m_country; }
    public void setCountry(String value) { this.m_country = value; }

    public LocalDate getBirthDate() { return m_birthDate; }
    public void setBirthDate(LocalDate value) { this.m_birthDate = value; }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { this.m_lastName = value; }

    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { this.m_firstName = value; }

}
