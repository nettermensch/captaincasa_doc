This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.ModalPopup.IModalPopupListener;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup.IModelessPopupListener;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.pagebean.PageBean;

@CCGenClass (expressionBase="#{d.DemoPopupOncloseoutsideNestingPopup1}")

public class DemoPopupOncloseoutsideNestingPopup1
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public interface IListener
    {
        public void reactOnClose();
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    IListener m_listener;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupOncloseoutsideNestingPopup1()
    {
    }

    public String getPageName() { return "/workplace/demopopuponcloseoutsidenestingpopup1.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupOncloseoutsideNestingPopup1}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void prepare(IListener listener)
    {
        m_listener = listener;
    }
    
    public void onReopenModalAction(javax.faces.event.ActionEvent event) 
    {
        final DemoPopupOncloseoutsideNestingPopup1 ui = new DemoPopupOncloseoutsideNestingPopup1();
        ui.prepare(new DemoPopupOncloseoutsideNestingPopup1.IListener()
        {
            @Override
            public void reactOnClose() { closePopup(ui); }
        });
        ModalPopup modalPopup = openModalPopup(ui,"Level 1",0,0,new IModalPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser() { closePopup(ui); }
        });
    }
    
    public void onReopenModalWithClickOutsideAction(javax.faces.event.ActionEvent event) 
    {
        final DemoPopupOncloseoutsideNestingPopup1 ui = new DemoPopupOncloseoutsideNestingPopup1();
        ui.prepare(new DemoPopupOncloseoutsideNestingPopup1.IListener()
        {
            @Override
            public void reactOnClose() { closePopup(ui); }
        });
        ModalPopup modalPopup = openModalPopup(ui,"Level 1",0,0,new IModalPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser() { closePopup(ui); }
        });
        modalPopup.setCloseonclickoutside(true);
    }
    
    public void onReopenModelessWithClickOutsideAction(javax.faces.event.ActionEvent event) 
    {
        final DemoPopupOncloseoutsideNestingPopup1 ui = new DemoPopupOncloseoutsideNestingPopup1();
        ui.prepare(new DemoPopupOncloseoutsideNestingPopup1.IListener()
        {
            @Override
            public void reactOnClose() { closePopup(ui); }
        });
        ModelessPopup modelessPopup = openModelessPopup(ui,"Level 1",0,0,new IModelessPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser() { closePopup(ui); }
        });
        modelessPopup.setCloseonclickoutside(true);
    }

    public void onReopenModelessAction(javax.faces.event.ActionEvent event) 
    {
        final DemoPopupOncloseoutsideNestingPopup1 ui = new DemoPopupOncloseoutsideNestingPopup1();
        ui.prepare(new DemoPopupOncloseoutsideNestingPopup1.IListener()
        {
            @Override
            public void reactOnClose() { closePopup(ui); }
        });
        ModelessPopup modelessPopup = openModelessPopup(ui,"Level 1",0,0,new IModelessPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser() { closePopup(ui); }
        });
    }

    public void onOpeneStatusbarPopup(javax.faces.event.ActionEvent event) 
    {
        Statusbar.outputMessageWithPopup("Some message via status bar.");
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
