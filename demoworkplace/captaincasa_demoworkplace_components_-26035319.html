The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Text Input/Output
  • Wiki Type Texts

This demo shows how a Wiki text that you enter in the top area is translated on servier side into an HTML text. The translation is done using the bliki.jar library - which is an open source library available through sourceforge.

The translation is done when pressing the button or when pressing ctrl-U on the keyboard.

This is the Java code:

package workplace;

import info.bliki.wiki.filter.WikiModel;

import java.io.ByteArrayInputStream;
import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoWiki}")

public class DemoWiki 
    extends PageBean
    implements Serializable
{
    public DemoWiki(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("textarea");
        onUpdate(null);
    }

    protected String m_htmlText;
    public String getHtmlText() { return m_htmlText; }
    public void setHtmlText(String value) { m_htmlText = value; }

    protected String m_wikiText =
        "=A headline=\n" +
        "This is some straight text.\n" +
        "*first\n" +
        "*second\n" +
        "*third\n" +
        "Now a mini table\n" +
        "{| border=\"1\"\n" +
        "!Column1\n" +
        "!Columns2\n" +
        "|-\n" +
        "|Value1\n" +
        "|Value2\n" +
        "|}";
    public String getWikiText() { return m_wikiText; }
    public void setWikiText(String value) { m_wikiText = value; }

    public void onUpdate(ActionEvent event) 
    {
        try
        {
            WikiModel wikiModel = new WikiModel("${image}", "${title}");
            m_htmlText = wikiModel.render(m_wikiText);
        }
        catch (Exception e)
        {
            CLog.L.log(CLog.LL_ERR,"",e);
        }
    }

}
