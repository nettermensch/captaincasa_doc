The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Basic Controls

This page is a summary of basic controls that are provided.

This is the Java code:

package workplace;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.managedbean.IDispatcher;

public class DemoBasicControls extends PageBean
{
    boolean m_checkboxvalue;
    String m_fieldvalue = "Peter";
    String m_textareavalue = "This is a nice description.\nPlease carefully read it.";
    String m_radiobuttonvalue = "left";
    String m_comboboxvalue = "1";

    String m_all;
    
    public DemoBasicControls(IDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public void onOutputAll(ActionEvent ae)
    {
        m_all = "";
        m_all += "Field: " + m_fieldvalue + "\r\n";
        m_all += "Checkbox: " + m_checkboxvalue + "\r\n";
        m_all += "Radiobutton: " + m_radiobuttonvalue + "\r\n";
        m_all += "Combobox: " + m_comboboxvalue + "\r\n";
        m_all += "Textarea: " + m_textareavalue + "\r\n";
    }
    
    // ------------------------------------------------------------------------
    
    public String getAll()
    {
        return m_all;
    }
    public void setAll(String all)
    {
        m_all = all;
    }
    public boolean getCheckboxvalue()
    {
        return m_checkboxvalue;
    }
    public void setCheckboxvalue(boolean checkboxvalue)
    {
        m_checkboxvalue = checkboxvalue;
    }
    public String getFieldvalue()
    {
        return m_fieldvalue;
    }
    public void setFieldvalue(String fieldvalue)
    {
        m_fieldvalue = fieldvalue;
    }
    public String getTextareavalue()
    {
        return m_textareavalue;
    }
    public void setTextareavalue(String textareavalue)
    {
        m_textareavalue = textareavalue;
    }
    public String getRadiobuttonvalue()
    {
        return m_radiobuttonvalue;
    }
    public void setRadiobuttonvalue(String radiobuttonvalue)
    {
        m_radiobuttonvalue = radiobuttonvalue;
    }
    public String getComboboxvalue()
    {
        return m_comboboxvalue;
    }
    public void setComboboxvalue(String comboboxvalue)
    {
        m_comboboxvalue = comboboxvalue;
    }    
}
