The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid in Grid
  • Grid in Grid Cell

You can define grids to be contained in grid cells.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridInGrid}")

public class DemoGridInGrid 
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Kid extends FIXGRIDItem implements java.io.Serializable
    {    
        Person i_owner;
        
        public Kid(Person owner)
        {
            i_owner = owner;
        }
        
        public void onRemoveKid(ActionEvent event) 
        {
            i_owner.getKids().getItems().remove(this);
        }

        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }

    }

    public class Person extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
    
        protected FIXGRIDListBinding<Kid> m_kids = new FIXGRIDListBinding<Kid>();
        public FIXGRIDListBinding<Kid> getKids() { return m_kids; }
        public void setKids(FIXGRIDListBinding<Kid> value) { m_kids = value; }

        public void createKid() 
        {
            Kid k = new Kid(this);
            m_kids.getItems().add(k);
        }

    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    protected FIXGRIDListBinding<Person> m_persons = new FIXGRIDListBinding<Person>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoGridInGrid(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
        
    
    public FIXGRIDListBinding<Person> getPersons() { return m_persons; }

    public void onAddPerson(ActionEvent event)
    {
        Person p = new Person();
        p.setFirstName("New first name");
        p.setLastName("New last name");
        m_persons.getItems().add(p);
    }
    
    public void onAddKid(ActionEvent event) 
    {
        Person p = m_persons.getSelectedItem();
        if (p == null)
            return;
        p.createKid();
    }


}
