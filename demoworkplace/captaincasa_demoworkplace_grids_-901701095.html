The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Drag & Drop
  • Grid Drag Drop

Drag & drop items from the left to the right, and vice versa. When dropping onto an item you will see that the dropped item is either positioned in front of or behind - dependent from the vertical position where you dropped.

This example shows the simplest way to implement Drag & Drop for grids. The FIXGRID component provides DRAGSEND and DROPRECEIVE attributes - these are the ones which are valid for the whole grid, e.g. when the user drags/drops from/to an empty area of the grid which is not covered by a row.

And the grid provides for a ROWDRAGSEND and ROWDROPRECEIVE attribute, these are the ones that are located on item level - so that you can drop information onto individual items.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Collection;
import java.util.Set;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.managedbean.IDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridDragDrop}")

public class DemoGridDragDrop
    extends PageBean
    implements Serializable
{
    protected boolean m_grid2Enabled = true;
    public boolean getGrid2Enabled() { return m_grid2Enabled; }
    public void setGrid2Enabled(boolean value) { m_grid2Enabled = value; }

    protected boolean m_grid1Enabled = true;
    public boolean getGrid1Enabled() { return m_grid1Enabled; }
    public void setGrid1Enabled(boolean value) { m_grid1Enabled = value; }

    public class StockItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_articleText;
        public String getArticleText() { return m_articleText; }
        public void setArticleText(String value) { m_articleText = value; }

        protected String m_articleId;
        public String getArticleId() { return m_articleId; }
        public void setArticleId(String value) { m_articleId = value; }

        @Override
        public void onRowDrop(BaseActionEventDrop event)
        {
            String dragSend = event.getDragInfo(); 
            // drag & drop on item level ==> move dropped item at right position!
            boolean inFront = true;
            if (event.getPercentageVertical() >= 50)
                inFront = false;
            if (dragSend.startsWith("grid1:"))
                moveItemsBetweenGrids(m_stockItems1,m_stockItems2,m_stockItems2.getItems().indexOf(this),inFront);
            else if (dragSend.startsWith("grid2:"))
                moveItemsBetweenGrids(m_stockItems2,m_stockItems1,m_stockItems1.getItems().indexOf(this),inFront);
        }
    }
    
    public class StockItemGrid extends FIXGRIDListBinding<StockItem>
    {
        @Override
        protected void onDrop(BaseActionEventDrop dropEvent)
        {
            String dragsend = dropEvent.getDragInfo();
            if (dragsend.startsWith("grid1:"))
                moveItemsBetweenGrids(m_stockItems1,m_stockItems2,-1,true);
            else if (dragsend.startsWith("grid2:"))
                moveItemsBetweenGrids(m_stockItems2,m_stockItems1,-1,true);
        }
    }
    
    
    
    public DemoGridDragDrop(IDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("fixgrid");
        for (int i=0; i<5; i++)
        {
            StockItem si = new StockItem();
            si.setArticleId("4711"+i);
            si.setArticleText("Text for " + i);
            m_stockItems1.getItems().add(si);
            si = new StockItem();
            si.setArticleId("0815"+i);
            si.setArticleText("Text for " + i);
            m_stockItems2.getItems().add(si);
        }
    }

    protected StockItemGrid m_stockItems2 = new StockItemGrid();
    public StockItemGrid getStockItems2() { return m_stockItems2; }
    
    protected StockItemGrid m_stockItems1 = new StockItemGrid();
    public StockItemGrid getStockItems1() { return m_stockItems1; }

    private void moveItemsBetweenGrids(StockItemGrid gridFrom, 
                                       StockItemGrid gridTo, 
                                       int toIndex,
                                       boolean inFront)
    {
        if (inFront == false)
            toIndex++;
        Collection<StockItem> sis = gridFrom.getSelectedItemsAsSequence();
        for (StockItem si: sis)
        {
            if (toIndex >= 0)
            {
                gridTo.getItems().add(toIndex,si);
                toIndex++;
            }
            else
            {
                gridTo.getItems().add(si);
            }
            gridFrom.getItems().remove(si);
        }
    }
    
}
