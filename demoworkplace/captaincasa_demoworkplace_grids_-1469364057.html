The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid in Grid
  • Flexible Tree

Each line of the grid provides a dynamic content which is filled according to the object that is kept in the line.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.COLLINEComponentTag;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.jsfserver.elements.impl.LABELComponentTag;
import org.eclnt.jsfserver.elements.impl.ROWComponent;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTreeFlex}")

public class DemoTreeFlex
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class CostCenter
    {
        String m_name;
        String m_head;
        BigDecimal m_budget;
        public String getName() { return m_name; }
        public void setName(String name) { m_name = name; }
        public String getHead() { return m_head; }
        public void setHead(String head) { m_head = head; }
        public BigDecimal getBudget() { return m_budget; }
        public void setBudget(BigDecimal budget) { m_budget = budget; }
    }
    
    public class Employee
    {
        String m_firstName;
        String m_lastName;
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
    }
    
    public class FlexTreeItem extends FIXGRIDTreeItem implements java.io.Serializable
    {
        Object i_object;
        ROWDYNAMICCONTENTBinding i_dynContent = new ROWDYNAMICCONTENTBinding();
        public FlexTreeItem(FIXGRIDTreeItem parentNode, Object object)
        {
            super(parentNode);
            i_object = object;
            renderContent();
        }
        public Object getObject()
        {
            return i_object;
        }
        @Override
        public String getText()
        {
            if (i_object instanceof CostCenter) return ((CostCenter)i_object).getName();
            else if (i_object instanceof Employee) return ((Employee)i_object).getLastName();
            else return "?";
        }
        public String getBackground() 
        { 
            if (i_object instanceof CostCenter) return "#F0FFF0";
            else if (i_object instanceof Employee) return "#F0F0FF";
            else return "#FFFFFF";
        }
        public ROWDYNAMICCONTENTBinding getDynContent() { return i_dynContent; }
        public void renderContent()
        {
            List<ComponentNode> cns = new ArrayList<ComponentNode>();
            if (i_object instanceof CostCenter)
            {
                CostCenter o = (CostCenter)i_object;
                ComponentNode cn;
                cn = new ComponentNode("t:label"); // -----
                cns.add(cn);
                cn.addAttribute("text",o.getName());
                cn.addAttribute("width","100");
                cn = new ComponentNode("t:colline"); // -----
                cns.add(cn);
                cn.addAttribute("height","100%");
                cn.addAttribute("background","#c0c0c0");
                cn = new ComponentNode("t:label"); // -----
                cns.add(cn);
                cn.addAttribute("text",o.getHead());
                cn.addAttribute("width","100");
                cn = new ComponentNode("t:colline"); // -----
                cns.add(cn);
                cn.addAttribute("height","100%");
                cn.addAttribute("background","#c0c0c0");
                cn = new ComponentNode("t:label"); // -----
                cns.add(cn);
                cn.addAttribute("text",o.getBudget()+"");
                cn.addAttribute("width","100%");
                cn.addAttribute("align","right");
                cn.addAttribute("format","bigdecimal");
                cn.addAttribute("formatmask","dec2");
            }
            else 
            {
                Employee o = (Employee)i_object;
                ComponentNode cn;
                cn = new ComponentNode("t:label"); // -----
                cns.add(cn);
                cn.addAttribute("text",o.getFirstName());
                cn.addAttribute("width","50%");
                cn = new ComponentNode("t:colline"); // -----
                cns.add(cn);
                cn.addAttribute("height","100%");
                cn.addAttribute("background","#c0c0c0");
                cn = new ComponentNode("t:label"); // -----
                cns.add(cn);
                cn.addAttribute("text",o.getLastName());
                cn.addAttribute("width","50%");
            }
            i_dynContent.setContentNodes(cns);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDTreeBinding<FlexTreeItem> m_tree = new FIXGRIDTreeBinding<FlexTreeItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoTreeFlex(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<10; i++)
        {
            CostCenter cc = new CostCenter();
            cc.setBudget(new BigDecimal((i+1)*1000));
            cc.setName("Cost Center " + i);
            cc.setHead("Head " + i);
            FlexTreeItem top = new FlexTreeItem(m_tree.getRootNode(),cc);
            top.setStatus(top.STATUS_CLOSED);
            for (int j=0; j<8; j++)
            {
                Employee emp = new Employee();
                emp.setFirstName("First " + i + "/" + j);
                emp.setLastName("Last " + i + "/" + j);
                FlexTreeItem sub = new FlexTreeItem(top,emp);
                sub.setStatus(sub.STATUS_ENDNODE);
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDTreeBinding<FlexTreeItem> getTree() { return m_tree; }
    public void setTree(FIXGRIDTreeBinding<FlexTreeItem> value) { m_tree = value; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    

}
