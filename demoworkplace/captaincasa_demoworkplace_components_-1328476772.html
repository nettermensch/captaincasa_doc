The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Drag & Drop
  • Update row sequence in grid

You may drag items from the grid and re-position them in the grid.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoGridDragDropSequence}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" height="100%" objectbinding="#{d.DemoGridDragDropSequence.grid}" rowdragsend="#{d.DemoGridDragDropSequence.rowDragSend}" rowdropreceive="#{d.DemoGridDragDropSequence.rowDropReceive}" sbvisibleamount="40" width="100%" >
<t:gridcol id="g_4" text="Some column title" width="100%" >
<t:label id="g_5" text=".{text}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.util.valuemgmt.UniqueIdCreator;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridDragDropSequence}")

public class DemoGridDragDropSequence
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_text;
        public String getText() { return i_text; }
        public void setText(String value) { this.i_text = value; }
        @Override
        public void onRowDrop(BaseActionEventDrop event)
        {
            super.onRowDrop(event);
            if (event.getDragInfo().startsWith(m_dragDropType))
            {
                GridItem selectedItem = m_grid.getSelectedItem();
                if (selectedItem == this) return; // drag on same node as drop
                m_grid.getItems().remove(selectedItem);
                int newIndex = m_grid.getItems().indexOf(this);
                if (event.getPercentageVertical() >= 50)
                    newIndex++;
                m_grid.getItems().add(newIndex,selectedItem);
                m_grid.selectAndFocusItem(selectedItem);
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    String m_dragDropType;
    String m_rowDropReceive;
    String m_rowDragSend;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridDragDropSequence(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        // create unique drag-drop-type - so that there is no drag drop between identical
        // pages. Each instance of the grid has its own, isolated drag-drop-type.
        // Otherwise: potencial problems if the page bean is e.g. created twice.
        m_dragDropType = "TYPE_" + UniqueIdCreator.createId();
        m_rowDragSend = m_dragDropType+":this";
        m_rowDropReceive = m_dragDropType +":verticalsplit";
        for (int i=0; i<50; i++)
        {
            GridItem gi = new GridItem();
            gi.setText("Text " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogriddragdropsequence.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridDragDropSequence}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getRowDropReceive() { return m_rowDropReceive; }
    public void setRowDropReceive(String value) { this.m_rowDropReceive = value; }

    public String getRowDragSend() { return m_rowDragSend; }
    public void setRowDragSend(String value) { this.m_rowDragSend = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
