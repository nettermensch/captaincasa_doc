The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Asynchronous Processing
  • One Long Polling for different Events

Instead of having many LONGPOLLING components that work in parallel (each one reserving one connection!) it makes sense to have some central LONGPOLLING component and do all the asynchronous eventing via this component.

In the example there are two different events (simulated by two threads firing server side events), that use one LONGPOLLING. The events are queued into one event queue (m_events), when passing an event then the LONGPOLLING is triggered in order to invoke the client.

Please carefully note the usage of "synchronized" within the processing the events in the vector "m_events"!

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Vector;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.jsfserver.polling.DefaultLongPolling;
import org.eclnt.jsfserver.polling.LongPollingMgr;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoLongPollingCentral}")

public class DemoLongPollingCentral
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class MyEvent
    {
        Object i_source;
        String i_text;
        public MyEvent(Object source, String text)
        {
            i_source = source;
            i_text = text;
        }
    }
    
    public class Event1Creator extends Thread
    {
        int i_counter = 0;
        @Override
        public void run()
        {
            while (m_active)
            {
                try { Thread.currentThread().sleep(1000); } catch (Throwable t) {}
                i_counter++;
                throwEvent(this,"Event1 - " + i_counter);
            }
        }
    }
    
    public class Event2Creator extends Thread
    {
        int i_counter = 0;
        @Override
        public void run()
        {
            while (m_active)
            {
                try { Thread.currentThread().sleep(3333); } catch (Throwable t) {}
                i_counter++;
                throwEvent(this,"Event2 - " + i_counter);
            }
        }
    }
    
    public class MyLongPolling extends DefaultLongPolling
    {
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Event1Creator m_event1Creator;
    Event2Creator m_event2Creator;
    
    Object m_syncher = new Object();
    
    boolean m_active = true;
    MyLongPolling m_longPolling;
    
    String m_protocol = "";
    
    Vector<MyEvent> m_events = new Vector<MyEvent>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLongPollingCentral(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_longPolling = new MyLongPolling();
        m_event1Creator = new Event1Creator();
        m_event2Creator = new Event2Creator();
        m_event1Creator.start();
        m_event2Creator.start();
        LongPollingMgr.add(m_longPolling);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnDestroyed()
            {
                super.reactOnDestroyed();
                m_active = false;
                LongPollingMgr.remove(m_longPolling);
                m_longPolling = null;
            }
        });
    }

    public String getPageName() { return "/workplace/demolongpollingcentral.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLongPollingCentral}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getLongPollingURL() { return m_longPolling.getURL(); }
    public String getProtocol() { return m_protocol; }

    public void onLongPollingAction(ActionEvent actionEvent)
    {
        MyEvent[] events;
        synchronized (m_syncher) 
        {
            events = new MyEvent[m_events.size()];
            m_events.toArray(events);
            m_events.clear();
        }
        for (MyEvent event: events)
        {
            executeEvent(event);
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void throwEvent(Object source, String text)
    {
        MyEvent event = new MyEvent(source,text);
        synchronized (m_syncher) 
        {
            m_events.add(event);
        }
        if (m_longPolling != null)
            m_longPolling.wakeup(true);
    }
    
    private void executeEvent(MyEvent event)
    {
        String line = "Event: " + event.i_source.getClass().getSimpleName() + ": " + event.i_text + "\n";
        m_protocol += line;
    }
}
