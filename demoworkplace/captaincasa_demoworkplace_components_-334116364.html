The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Content "flying in"
  • Master detail - with moving detail

This is an example how to create some interactive Master-Detail-view: simple selections in the grid trigger an update of a "preview"-area which is positioned on the right bottom or right top - depending on the current selection in the grid.

When "executing" (double clicking or hitting the return key) then the preview area is opened to cover the whole grid area.

Please check the keyboard behavior: when rushing with cursor up/down through the grid then the preview is updated - when hitting the return key, the detail area is fully sized, once again continuing with the cursor keys it is reduced again.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoMasterDetailWithSizing}")

public class DemoMasterDetailWithSizing
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataGridView2<Person> m_dataGrid = new CCDataGridView2(Person.class);
    public CCDataGridView2 getDataGrid() { return m_dataGrid; }
    DemoMasterDetailWithSizingDetail m_detailPage = new DemoMasterDetailWithSizingDetail();

    boolean m_detailIsShrinked = true;
    double m_detailScale = 0.6d;
    Trigger m_detailAnimationTrigger = new Trigger();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMasterDetailWithSizing(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        fillGrid();
    }


    public String getPageName() { return "/workplace/demomasterdetailwithsizing.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMasterDetailWithSizing}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DemoMasterDetailWithSizingDetail getDetailPage() { return m_detailPage; }
    
    public double getDetailScale() { return m_detailIsShrinked ? 0.6d:1d; }
    public String getDetailHeight() { return m_detailIsShrinked ? "350":"100%-100"; }
    public String getDetailWidth() { return m_detailIsShrinked ? "300":"100%-100"; }
    public String getDetailY() 
    { 
        if (isSelectionInBottomArea())
            return m_detailIsShrinked ? "100%-30":"100%-50";
        else
            return m_detailIsShrinked ? "30":"100%-50";
    }
    public String getDetailX() { return m_detailIsShrinked ? "100%-30":"100%-50"; }
    public String getDetailAnchor() 
    { 
        if (isSelectionInBottomArea())
            return m_detailIsShrinked ? "rightbottom":"rightbottom";
        else
            return m_detailIsShrinked ? "righttop":"rightbottom";
    }

    public Trigger getDetailAnimationTrigger() { return m_detailAnimationTrigger; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        List<Person> ps = new ArrayList<Person>();
        for (int i=0; i<100; i++)
            ps.add(RandomLogic.createPerson());
        m_dataGrid.prepare(ps,new CCDataGridView2.DefaultListener<Person>()
        {
            @Override
            public void reactOnGridSelect(Person object)
            { 
                m_detailPage.prepare(object,null);
                m_detailAnimationTrigger.trigger();
                m_detailIsShrinked = true;
            }
            @Override
            public void reactOnGridExecute(Person object)
            {
                m_detailPage.prepare(object,null);
                m_detailIsShrinked = false;
            }
        });
        m_dataGrid.getGrid().selectAndFocusItem(m_dataGrid.getGrid().getItems().get(0));
    }
    
    private boolean isSelectionInBottomArea()
    {
        try
        {
            Object o = m_dataGrid.getGrid().getSelectedItem();
            int index = m_dataGrid.getGrid().getItems().indexOf(o);
            int relIndex = index - m_dataGrid.getGrid().getSbvalue();
            int sbVisibleAmount = m_dataGrid.getGrid().getClientvisibleamount();
            if (relIndex < sbVisibleAmount / 2)
                return true;
        }
        catch (Throwable t)
        {
        }
        return false;
    }

}
