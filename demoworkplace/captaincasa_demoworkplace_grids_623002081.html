The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid data export
  • Extending PDF and Excel export

The default grid export to PDF and Excel can be fine-tuned.

1. By providing an implementation of interface "IFIXGRIDExportInfoProvider" formatting rules can be applied to the export. Most of the rules are considered both by the default PDF and by the default Excel export functions. - Please extend your implementation from class "DefaultFIXGRIDExportInfoProvider" in order to stay compatible for future extensions of the interface.

2. Both for PDF and for Excel processing the default export class can be extended. The classes is internally structured to allow extension - and provides corresponding extension points as protected methods that you can override.

You extended functions can be either set into the concrete grid instance - or you can define in general via system.xml that your extension class is to be chosen for all grids.

(Please note: in the demo code all classes are coded as inner classes in order to have a compact view on all the code that is involved. In reality you would write own, stand-alone classes for extending the PDF and Excel export functions.)

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import org.apache.poi.ss.usermodel.Cell;
import org.apache.poi.ss.usermodel.CellStyle;
import org.apache.poi.ss.usermodel.Row;
import org.apache.poi.ss.usermodel.Sheet;
import org.apache.poi.xssf.usermodel.XSSFFont;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.ColumnInfo;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.DefaultFIXGRIDExportInfoProvider;
import org.eclnt.jsfserver.elements.util.FIXGRIDExcelExporter;
import org.eclnt.jsfserver.elements.util.FIXGRIDPDFExporter;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;

import com.lowagie.text.Document;
import com.lowagie.text.DocumentException;
import com.lowagie.text.Font;
import com.lowagie.text.Paragraph;

@CCGenClass (expressionBase="#{d.DemoGridExcelExport}")

public class DemoGridExcelExport
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * Excel exporter with special features, here: additional head lines in front
     * of the normal header row.
     */
    public class MyExcelExporter extends FIXGRIDExcelExporter
    {
        
        @Override
        protected void createBeforeHeader(XSSFWorkbook wb, Sheet sheet, FIXGRIDBinding fixgrid)
        {
            super.createBeforeHeader(wb, sheet, fixgrid);
            // headline
            {
                Row row = createRow(sheet);
                XSSFFont xfont = createHeaderFont(wb);
                xfont.setFontHeightInPoints((short)40);
                Cell cell = row.createCell(0);
                CellStyle cellStyle = wb.createCellStyle();
                cellStyle.setFont(xfont);
                cell.setCellStyle(cellStyle);
                cell.setCellValue("This is some headline for the excel data.");
            }
            // sub headline
            {
                Row row = createRow(sheet);
                XSSFFont xfont = createHeaderFont(wb);
                Cell cell = row.createCell(0);
                CellStyle cellStyle = wb.createCellStyle();
                cellStyle.setFont(xfont);
                cell.setCellStyle(cellStyle);
                cell.setCellValue("This is some sub-headline for the excel data.");
            }
            // distance row
            {
                createRow(sheet);
            }
        }
    }
    
    /**
     * PDF exporter with special features, here: additional head lines in front
     * of the normal header row.
     */
    public class MyPDFExporter extends FIXGRIDPDFExporter
    {
        final Font MYFONT_HEADLINE = new Font(Font.HELVETICA,Font.DEFAULTSIZE+2,Font.BOLD);
        final Font MYFONT_SUBHEADLINE = new Font(Font.HELVETICA,Font.DEFAULTSIZE-2,Font.NORMAL);
        @Override
        protected void beforeGrid(Document document, FIXGRIDBinding fixgrid, List<ColumnInfo> columns,
                List<List<String>> data)
        {
            try
            {
                Paragraph p = createParagraph("This is some headline for the PDF data.",MYFONT_HEADLINE,null);
                p.setSpacingAfter(10);
                document.add(p);
                p = createParagraph("This is some sub-headline for the PDF data.",MYFONT_SUBHEADLINE,null);
                p.setSpacingAfter(10);
                document.add(p);
            }
            catch (DocumentException exc)
            {
                CLog.L.log(CLog.LL_ERR,"Problems when adding title to the PDF document",exc);
            }
        }
    }
    
    /**
     * Interface for individual cell formatting.
     */
    public class MyExportInfoProvider extends DefaultFIXGRIDExportInfoProvider
    {
        final static String FIRSTNAME3COLOR = "#FFC0C0";
        final static String LASTNAME5COLOR = "#C0FFC0";
        @Override
        public String findBackgroundColorForDataCell(int rowIndex, String dataCellValue, ColumnInfo column)
        {
            if (column.getSortreference().equals("firstName"))
            {
                if (dataCellValue.contains("3")) return "#FFC0C0";
            }
            if (column.getSortreference().equals("lastName"))
            {
                if (dataCellValue.contains("5")) return "#C0FFC0";
            }
            return super.findBackgroundColorForDataCell(rowIndex, dataCellValue, column);
        }
        @Override
        public String findTextColorForDataCell(int rowIndex, String dataCellValue, ColumnInfo column)
        {
            if (column.getSortreference().equals("lastName"))
            {
                if (dataCellValue.contains("7")) return "#800000";
            }
            return super.findTextColorForDataCell(rowIndex, dataCellValue, column);
        }
    }
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }


    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    MyPDFExporter m_pdfExporter = new MyPDFExporter();
    MyExcelExporter m_excelExporter = new MyExcelExporter();
    MyExportInfoProvider m_exportInfoProvider = new MyExportInfoProvider();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridExcelExport(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        // set file name for export
        m_grid.getExporter().setExportFilePrefix("SomeGreatGrid");
        // the overridden exporter can be either directly set in the individual
        // grid instance - or it can be set via system.xml for all grid instances
        m_grid.getExporter().setExcelExporter(m_excelExporter);
        m_grid.getExporter().setPDFExporter(m_pdfExporter);
        m_excelExporter.setExportInfoProvider(m_exportInfoProvider);
        m_pdfExporter.setExportInfoProvider(m_exportInfoProvider);
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First " + i);
            gi.setLastName("Last " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridexcelexport.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridExcelExport}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
}
