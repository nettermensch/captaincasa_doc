The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Style and Language Management
  • Client localization

The CLIENTCONFIG component provides two attribute COUNTRY and LANGUAGE, by which you can set the client localization.

The country is the one that defines the formatting of certain values. You may either choose a concrete country or you may use one of the "CCTYPE"-countries which represent some oftenly used formatting.

The language is the one that defines the text output of all client components (e.g. the CALENDARFIELD).

Countries and languages are defined via XML files within the eclnt/risc/i18n directory of your web application.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.jsfserver.util.ClientI18NReader;
import org.eclnt.jsfserver.util.HttpSessionAccess;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoClientLocale}")

public class DemoClientLocale
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_clientCountry;
    String m_clientLanguage;
    ValidValuesBinding m_languageVVS = new ValidValuesBinding();
    ValidValuesBinding m_countryVVS = new ValidValuesBinding();
    String m_country;
    String m_language;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoClientLocale(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        // get current localization of client
        m_language = HttpSessionAccess.getCurrentRequest().getHeader("eclnt-language");
        m_country = HttpSessionAccess.getCurrentRequest().getHeader("eclnt-country");
        // load combo boxes
        List<String> countries = new ArrayList<String>(ClientI18NReader.getCountries());
        Collections.sort(countries);
        List<String> languages = new ArrayList<String>(ClientI18NReader.getLanguages());
        Collections.sort(languages);
        for (String country: countries)
            m_countryVVS.addValidValue(country,country);
        for (String language: languages)
            m_languageVVS.addValidValue(language,language);
    }

    public String getPageName() { return "/workplace/democlientlocale.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoClientLocale}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getClientCountry() { return m_clientCountry; }
    public String getClientLanguage() { return m_clientLanguage; }

    public ValidValuesBinding getLanguageVVS() { return m_languageVVS; }
    public ValidValuesBinding getCountryVVS() { return m_countryVVS; }

    public String getCountry() { return m_country; }
    public void setCountry(String value) { this.m_country = value; }

    public String getLanguage() { return m_language; }
    public void setLanguage(String value) { this.m_language = value; }

    public void onSetLocaleAction(javax.faces.event.ActionEvent event) 
    {
        // transfer values from text fields into fields that
        // are referenced by client config
        m_clientLanguage = m_language;
        m_clientCountry = m_country;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
