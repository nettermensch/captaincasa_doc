The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Date Input
  • Add info - direct way

Both CALENDARFIELD and CALENDAR component provide an attribute EXTCALENDARINFOS by which you can pass information on certain days.

The information includes:

. a comment (tooltip)
. a background color
. a definition if the day is select-able
. a definition if a warning is shown to the user when selecting a date

As consequence you make the calendar more colorful - and add your information!

Please note: when using EXTCALENDARINFOS then all the day configuration is loaded to the client in one step. Do not use this attribute when e.g. setting up a general calendar definition - which might span years of information. Here we provide the possibility of defining a so called "extended calendar id" (EXTCALENDARID). The information about days will then be picked month by month, following the navigation of the user.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Calendar;
import java.util.Date;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.calendar.ExtCalendarInfos;
import org.eclnt.jsfserver.calendar.IExtCalendarMgr.ExtCalendarInfo;
import org.eclnt.jsfserver.elements.events.BaseActionEventFlush;
import org.eclnt.jsfserver.elements.impl.CALENDARComponent;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoExtCalendarInfos}")

public class DemoExtCalendarInfos
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Date m_dateValue = new Date();
    String m_extCalendarInfosNext7;
    
    Date m_dateTo;
    Date m_dateFrom;
    String m_extCalendarInfosFromTo;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoExtCalendarInfos(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        buildNext7();
        buildFromTo();
    }

    public String getPageName() { return "/workplace/demoextcalendarinfos.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoExtCalendarInfos}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getExtCalendarInfosNext7() { return m_extCalendarInfosNext7; }

    public Date getDateValue() { return m_dateValue; }
    public void setDateValue(Date value) { this.m_dateValue = value; }
    public void onDateValueAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventFlush)
        {
            buildNext7();
        }
    }

    public Date getDateTo() { return m_dateTo; }
    public void setDateTo(Date value) { this.m_dateTo = value; }

    public Date getDateFrom() { return m_dateFrom; }
    public void setDateFrom(Date value) { this.m_dateFrom = value; }

    public String getExtCalendarInfosFromTo() { return m_extCalendarInfosFromTo; }

    public void onDateToAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventFlush)
        {
            if (m_dateTo != null)
            {
                if (m_dateFrom == null || m_dateFrom.getTime() > m_dateTo.getTime())
                {
                    m_dateFrom = m_dateTo;
                }
            }
        }
        buildFromTo();
    }

    public void onDateFromAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventFlush)
        {
            if (m_dateFrom != null)
            {
                if (m_dateTo == null || m_dateTo.getTime() < m_dateFrom.getTime())
                {
                    m_dateTo = m_dateFrom;
                }
            }
        }
        buildFromTo();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildFromTo()
    {
        m_extCalendarInfosFromTo = null;
        if (m_dateFrom == null) return;
        if (m_dateTo == null) return;
        ExtCalendarInfos ecis = new ExtCalendarInfos();
        Calendar cal = Calendar.getInstance();
        cal.setTime(m_dateFrom);
        cal.set(Calendar.HOUR_OF_DAY,0);
        cal.set(Calendar.MINUTE,0);
        cal.set(Calendar.SECOND,0);
        cal.set(Calendar.MILLISECOND,0);
        while (true)
        {
            long l = cal.getTimeInMillis();
            if (l <= m_dateTo.getTime())
            {
                Date iDate = new Date(l);
                ExtCalendarInfo eci = new ExtCalendarInfo(iDate);
                if (l == m_dateFrom.getTime() || l == m_dateTo.getTime())
                    eci.setBackground("#00800060");
                else
                    eci.setBackground("#00800030");
                eci.setComment("From - to");
                ecis.addExtCalendarInfo(eci);
                cal.add(Calendar.DAY_OF_YEAR,1);
            }
            else
            {
                break;
            }
        }
        m_extCalendarInfosFromTo = ecis.toXML();
    }
    
    private void buildNext7()
    {
        Date now = m_dateValue;
        m_extCalendarInfosNext7 = null;
        String[] backgrounds = new String[] 
        {
            "#FF000070","#FF000060","#FF000050","#FF000040","#FF000030","#FF000020","#FF000010"
        };
        if (now == null) return;
        Calendar cal = Calendar.getInstance();
        cal.setTime(now);
        cal.set(Calendar.HOUR_OF_DAY,0);
        cal.set(Calendar.MINUTE,0);
        cal.set(Calendar.SECOND,0);
        cal.set(Calendar.MILLISECOND,0);
        ExtCalendarInfos ecis = new ExtCalendarInfos();
        for (int i=0; i<7; i++)
        {
            Date iDate = new Date(cal.getTimeInMillis());
            ExtCalendarInfo eci = new ExtCalendarInfo(iDate);
            eci.setBackground(backgrounds[i]);
            eci.setComment("Next 7");
            ecis.addExtCalendarInfo(eci);
            cal.add(Calendar.DAY_OF_YEAR,1);
        }
        m_extCalendarInfosNext7 = ecis.toXML();
    }
}
