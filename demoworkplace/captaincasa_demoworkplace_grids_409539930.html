The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Drag & Drop
  • Single Selection Grid

Drag and drop is available on most of the components.

In the example you can drag items from the list on the left and drop them onto the corresponding top element on the right. Drop operations are only possible if the nature of the item on the left fits to the nature of the item on the right.

What's the principle behind?

The objects on the left specify their DRAGSEND attribute by expressing what they represent. The expression is done in the way "<type>:<id>", for example: "car:Ford". (It is also possible that one object is expressing more representations, then you concatenate the definitions: "<type>:<id>;<type>:<id>;<type>:<id>").

The components on the right specify their DROPRECEIVE attribtues by expressing what types they are able to process. Example the car node on the right has the DROPRECEIVE attribute "car". A component can process multiple types by concatenating these types: "<type>;<type>;<type>".

When the user drops a component onto the other component then the receiving component gets passed the DRAGSEND information of the dragged component into the corresponding action listener. It can parse this information and react accordingly.

This is the Java code:

package workplace;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.events.BaseActionEventDropCopy;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoDragDrop extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class LeftRow extends FIXGRIDItem
    {
        public LeftRow(String text, String type)
        {
            m_text = text;
            m_draginfo = type+":"+text;
            m_type = type;
        }
        String m_text;
        String m_draginfo;
        String m_type;
        public void setText(String value) { m_text = value; }
        public String getText() { return m_text; }
        public void setDraginfo(String value) { m_draginfo = value; }
        public String getDraginfo() { return m_draginfo; }
        public String getImage() { return "../images/icons/"+m_type+".png"; }
    }
    
    /**
     * Overridden FIXGRIDTreeBinding with specialized drop management
     * in onGridAction().
     */
    public class MyRightGrid extends FIXGRIDTreeBinding<RightRow>
    {
        public void onGridAction(ActionEvent event)
        {
            super.onGridAction(event);
            Statusbar.outputSuccess("Event was processed: " + event.getClass().getName());
            if (event instanceof BaseActionEventDrop || 
                event instanceof BaseActionEventDropCopy) 
            {
                String droppedValue = ((BaseActionEventDrop)event).getDragInfo();
                if (droppedValue.startsWith("car:")) m_carTop.processTREENDOEAction(event);
                else if (droppedValue.startsWith("person:")) m_personTop.processTREENDOEAction(event);
                else if (droppedValue.startsWith("computer:")) m_computerTop.processTREENDOEAction(event);
            }
            BaseActionEvent bae = (BaseActionEvent)event;
        }
    }
    
    
    public class RightRow extends FIXGRIDTreeItem
    {
        String m_dropReceive;
        String m_type;
        public RightRow(FIXGRIDTreeItem parentNode, String text, String dropReceive, String type)
        {
            super(parentNode);
            setText(text);
            m_dropReceive = dropReceive;
            m_type = type;
        }
        public String getDropReceive() { return m_dropReceive; }
        public void processTREENDOEAction(ActionEvent event)
        {
            super.processTREENDOEAction(event);
            if (event instanceof BaseActionEventDrop)
            {
                BaseActionEventDrop bae = (BaseActionEventDrop)event;
                int index = bae.getParams()[0].indexOf(':');
                String value = bae.getParams()[0].substring(index+1);
                RightRow rr = new RightRow(this,value,null,m_type);
                rr.setStatus(rr.STATUS_ENDNODE);
                // remove from left 
                for (int i=0; i<m_leftRows.getItems().size(); i++)
                {
                    if (m_leftRows.getItems().get(i).getText().equals(value))
                    {
                        m_leftRows.getItems().remove(i);
                        break;
                    }
                }
            }
            else if (event instanceof BaseActionEventDropCopy)
            {
                BaseActionEventDrop bae = (BaseActionEventDropCopy)event;
                int index = bae.getParams()[0].indexOf(':');
                String value = bae.getParams()[0].substring(index+1);
                RightRow rr = new RightRow(this,value,null,m_type);
                rr.setStatus(rr.STATUS_ENDNODE);
            }
        }
        public String getImage() { return "../images/icons/"+m_type+".png"; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    FIXGRIDListBinding<LeftRow> m_leftRows = new FIXGRIDListBinding<LeftRow>();
    FIXGRIDTreeBinding<RightRow> m_rightRows = new MyRightGrid();
    RightRow m_carTop;
    RightRow m_personTop;
    RightRow m_computerTop;
    
    String m_textAreaContent = "";
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoDragDrop(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // left side
        m_leftRows.getItems().add(new LeftRow("Harry","person"));
        m_leftRows.getItems().add(new LeftRow("Ford","car"));
        m_leftRows.getItems().add(new LeftRow("Laptop","computer"));
        m_leftRows.getItems().add(new LeftRow("Desktop Computer","computer"));
        m_leftRows.getItems().add(new LeftRow("Audi","car"));
        m_leftRows.getItems().add(new LeftRow("Mercedes","car"));
        m_leftRows.getItems().add(new LeftRow("Claude","person"));
        m_leftRows.getItems().add(new LeftRow("Toyota","car"));
        m_leftRows.getItems().add(new LeftRow("Host","computer"));
        m_leftRows.getItems().add(new LeftRow("Joan","person"));
        m_leftRows.getItems().add(new LeftRow("Chrysler","car"));
        m_leftRows.getItems().add(new LeftRow("Volkswagen","car"));
        // right side
        m_personTop = new RightRow(m_rightRows.getRootNode(),"Persons","person","person");
        m_carTop = new RightRow(m_rightRows.getRootNode(),"Cars","car","car");
        m_computerTop = new RightRow(m_rightRows.getRootNode(),"Computers","computer","computer");
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<LeftRow> getLeftRows() { return m_leftRows; }
    public FIXGRIDTreeBinding<RightRow> getRightRows() { return m_rightRows; }
    
    public void setTextAreaContent(String value) { m_textAreaContent = value; }
    public String getTextAreaContent() { return m_textAreaContent; }
    
    public void onTextAreaAction(ActionEvent ae)
    {
        if (ae instanceof BaseActionEventDrop)
        {
            BaseActionEventDrop baed = (BaseActionEventDrop)ae;
            m_textAreaContent = "Dropped Info: " + baed.getDragInfo();
        }
    }
    
}
