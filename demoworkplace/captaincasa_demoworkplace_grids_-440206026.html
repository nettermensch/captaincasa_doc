The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid Header Cell
  • Abbreviated column header texts

The components LABEL and GRIDCOL provide an attribute TEXTABBREVIATION. If defined then this text will be used instead of the normal TEXT-definition, if the screen space is not sufficient.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoLabelAbbreviation}")

public class DemoLabelAbbreviation
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_desription = new String();
        String i_articleId;
        public String getDesription() { return i_desription; }
        public String getArticleId() { return i_articleId; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLabelAbbreviation(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.i_articleId = "47276"+i;
            gi.i_desription = "Description for article " + gi.i_articleId;
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demolabelabbreviation.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLabelAbbreviation}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
