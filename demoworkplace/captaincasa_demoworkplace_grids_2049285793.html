The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Sorting
  • Column sorting

Click onto the column headers - the content will be sorted.

The grid processing provides a default framework for sorting. You normally just have to specify the SORTREFERENCE on GRIDCOL level - the grid will automatically provide the function to sort the contained rows by the referenced property, once the user clicks onto the title column. The automatic sorting is "data-type-aware", i.e. Strings are sorted in a different way than integer numbers.

You can also embed your own sorting. To do so you need to override the FIXGRIDListBinding class and override the method "sortGrid(...)". In the example the second grid has an explicitly overriden sort algorithm.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Calendar;
import java.util.Date;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.managedbean.IDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridSort}")

public class DemoGridSort
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // top grid
    // ------------------------------------------------------------------------
    
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected Date m_birthdate;
        public Date getBirthdate() { return m_birthdate; }
        public void setBirthdate(Date value) { m_birthdate = value; }

        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }

    }

    // ------------------------------------------------------------------------
    // bottom grid with explicit sorting ==> override method sortGrid
    // ------------------------------------------------------------------------
    
    public class MyRow2 extends FIXGRIDItem implements java.io.Serializable
    {
        int m_intValue;
        
        protected String m_value;
        public String getValue() { return m_value; }
        public void setValue(String value) { m_value = value; }
    }

    public class MyRow2ListBinding extends FIXGRIDListBinding<MyRow2>
    {
        @Override
        protected void sortGrid(String sortReference,
                                String objectBindingString, 
                                boolean ascending)
        {
            System.out.println("Sort called: " + sortReference + ", direction " + ascending);
            // sort "value" by integer, but by text; use the most
            // sophisticated algorithm for sorting: bubble sort ;-)
            while (true)
            {
                boolean updatedSequence = false;
                for (int i=0; i<m_rows2.getItems().size()-1; i++)
                {
                    if (ascending && m_rows2.getItems().get(i).m_intValue > m_rows2.getItems().get(i+1).m_intValue ||
                        !ascending && m_rows2.getItems().get(i).m_intValue < m_rows2.getItems().get(i+1).m_intValue)
                    {
                        updatedSequence = true;
                        MyRow2 temp = m_rows2.getItems().get(i);
                        m_rows2.getItems().set(i,m_rows2.getItems().get(i+1));
                        m_rows2.getItems().set(i+1,temp);
                    }
                }
                if (updatedSequence == false)
                    break;
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // constructor ==> creation of items
    // ------------------------------------------------------------------------
    
    public DemoGridSort(IDispatcher dispatcher)
    {
        super(dispatcher);
        int maxi = 100;
        for (int i=0; i<maxi; i++)
        {
            // firs grid
            MyRow r = new MyRow();
            r.setFirstName("First Name " + i);
            r.setLastName("Last Name " + (maxi-i-1));
            Calendar c = Calendar.getInstance();
            c.set(Calendar.YEAR,1970);
            c.add(Calendar.DAY_OF_YEAR,i * 3);
            r.setBirthdate(c.getTime());
            m_rows.getItems().add(r);
            // second grid
            MyRow2 r2 = new MyRow2();
            r2.setValue("Value " + i);
            r2.m_intValue = i;
            m_rows2.getItems().add(r2);
        }
    }
    
    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    
    protected MyRow2ListBinding m_rows2 = new MyRow2ListBinding();
    public MyRow2ListBinding getRows2() { return m_rows2; }

}
