The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Date Input
  • Add info - factory calendar

The CALENDARFIELD provides the attribute EXTCALENDARID. By defining this attribute you automatically link the date input to a server side definition of a calendar.

The definition of the server side calendar includes the following information per day:

1. comment - the comment is shown as tooltip
2. selectable - if selectable is set to false then the day cannot be selected
3. background - a backgroundcolor
4. warningonselection - the user can select the field but a warning showing the comment will be shown after selection

The server side interface is "IExtCalendarMgr", there is a default implementation available - which is typically overridden by your own implementation.
The CALENDARFIELD component communicates to the server side when the user selects a certain date, or when the user navigates through the calendar.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.calendar.ExtCalendarMgr;
import org.eclnt.jsfserver.calendar.IExtCalendarMgr;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoExtCalendar}")

public class DemoExtCalendar
    extends PageBean
    implements Serializable
{
    protected ValidValuesBinding m_calendarIdVVS = new ValidValuesBinding();
    public ValidValuesBinding getCalendarIdVVS() { return m_calendarIdVVS; }
    public void setCalendarIdVVS(ValidValuesBinding value) { m_calendarIdVVS = value; }

    protected String m_calendarId;
    public String getCalendarId() { return m_calendarId; }
    public void setCalendarId(String value) { m_calendarId = value; }

    protected String m_testMonthInfo;
    public String getTestMonthInfo() { return m_testMonthInfo; }
    public void setTestMonthInfo(String value) { m_testMonthInfo = value; }

    protected Date m_testDay = new Date();
    public Date getTestDay() { return m_testDay; }
    public void setTestDay(Date value) { m_testDay = value; }

 
    public DemoExtCalendar(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        List<String> ids = ExtCalendarMgr.getInstance().getAvailableCalendarIds();
        for (String id: ids)
            m_calendarIdVVS.addValidValue(id,id);
    }
    
    public void onTest(ActionEvent event) 
    {
        if (m_calendarId == null)
        {
            Statusbar.outputError("No calendar id defined.");
            return;
        }
        IExtCalendarMgr ecm = ExtCalendarMgr.getInstance();
        IExtCalendarMgr.ExtCalendarInfo ri = ecm.getCalendarInfoForDay(m_calendarId,m_testDay,"UTC");
        if (ri == null)
            Statusbar.outputSuccess("This is a normal working day.");
        else
            Statusbar.outputAlert("This is NOT a working day: " + ri.getComment());
        m_testMonthInfo = "Month Info:\n";
        List<IExtCalendarMgr.ExtCalendarInfo> ecis = ecm.getCalendarInfoForMonth(m_calendarId,m_testDay,"UTC");
        for (IExtCalendarMgr.ExtCalendarInfo eci: ecis)
        {
            m_testMonthInfo += "" + eci.getDate().toString() + " // " + eci.getComment() + "\n";
        }
    }

}
