The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Structure Components
  • Key Selector

The KEYSELECTOR shows a list of characters, each one can be pressed and triggers a corresponding event on server side.

The configuration of the component includes:
(.) the definition of the keys that are shown - both as part of the page definition and by using a runtime API
(.) the ability to decide at runtime which key is active and which not

The server side representation of the KEYSELECTOR component is an instance of class KEYSELECTORBinding.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoKeySelector}" >
<t:row id="g_2" >
<t:keyselector id="g_3" actionListener="#{d.DemoKeySelector.onKeySelectorAction}" keys="ABCDEFGHIJKLMNOPQRSTUVWXYZ" objectbinding="#{d.DemoKeySelector.selector}" />

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventKeySelected;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.KEYSELECTORBinding;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoKeySelector}")

public class DemoKeySelector
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_name;
        public String getName() { return m_name; }
        public void setName(String value) { this.m_name = value; }

    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    KEYSELECTORBinding m_selector = new KEYSELECTORBinding();
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoKeySelector(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        super.initRefComponent("keyselector");
        m_selector.addKeyInfosAZ();
        for (int i=0; i<26; i++)
        {
            String s = (char)((int)'A'+i) + "_xxxxxx";
            GridItem gi = new GridItem();
            gi.setName(s);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demokeyselector.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoKeySelector}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public KEYSELECTORBinding getSelector() { return m_selector; }

    public void onKeySelectorAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventKeySelected)
        {
            BaseActionEventKeySelected e = (BaseActionEventKeySelected)event;
            positionGrid(e.getSelectedKey());
            Statusbar.outputSuccess("Key was pressed: " + e.getSelectedKey());
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void positionGrid(String key)
    {
        m_grid.deselectCurrentSelection();
        for (GridItem gi: m_grid.getItems())
        {
            if (gi.getName().startsWith(key))
            {
                m_grid.selectAndFocusItem(gi);
                break;
            }
        }
    }
}
