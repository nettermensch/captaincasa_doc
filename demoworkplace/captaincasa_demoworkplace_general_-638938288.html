  • General
  • Workplace Management
  • Multiple workplaces
  • Base functions

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.MultiWorkplaceConnector;
import org.eclnt.workplace.MultiWorkplaceConnector.WorkplaceConfiguration;
import org.eclnt.workplace.MultiWorkplaceWorkpageDefaultProcessingEventListener;
import org.eclnt.workplace.WorkpageProcessingEvent;
import org.eclnt.workplace.WorkpageStartInfo;

@CCGenClass (expressionBase="#{d.DemoMultiWorkplace}")

public class DemoMultiWorkplace
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_startURL;
    Trigger m_startURLTrigger = new Trigger();
    String m_firstName = "Captain";
    String m_lastName = "Casa";
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMultiWorkplace(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        getWorkpage().addWorkpageProcessingEventListener(new MultiWorkplaceWorkpageDefaultProcessingEventListener()
        {
            @Override
            public boolean checkIfEventIsRelevant(WorkpageProcessingEvent event)
            {
                if (event instanceof DemoMultiWorkplaceTest.MyEvent)
                    return true;
                else
                    return false;
            }
            @Override
            public void processEvent(WorkpageProcessingEvent event)
            {
                if (event instanceof DemoMultiWorkplaceTest.MyEvent)
                {
                    DemoMultiWorkplaceTest.MyEvent e = (DemoMultiWorkplaceTest.MyEvent)event;
                    Statusbar.outputSuccess("Received event: " + e.getEventMessage());
                }
            }
        });
    }

    public String getPageName() { return "/workplace/demomultiworkplace.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMultiWorkplace}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public Trigger getStartURLTrigger() { return m_startURLTrigger; }
    public String getStartURL() { return m_startURL; }

    public String getFirstName() { return m_firstName; } 
    public void setFirstName(String firstName) { m_firstName = firstName; }
    
    public String getLastName() { return m_lastName; }
    public void setLastName(String lastName) { m_lastName = lastName; }

    public void onStart2ndWorkplaceAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            WorkplaceConfiguration wpconfig = new WorkplaceConfiguration();
            wpconfig.setWindowFeatures("left=100,top=100,width=1000,height=800");
            wpconfig.setWorkplaceURL("/workplace.workplaceMWPBlank.risc");
            MultiWorkplaceConnector.instance().startWorkplace
            (
                getOwningDispatcher(),
                "SECOND",
                wpconfig,
                null
            );
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }

    public void onStartPageAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setPageBeanName(DemoHelloWorld.class.getSimpleName());
            wpsi.setText("Hello!");
            MultiWorkplaceConnector.instance().startWorkpage(getOwningDispatcher(),"SECOND",null,wpsi,false);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }
    
    public void onStartPageAndPassDataAction(ActionEvent event)
    {
        try
        {
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setPageBeanName(DemoWpPassData.class.getSimpleName());
            wpsi.setOpenMultipleInstances(false);
            wpsi.setText("Pass data");
            wpsi.setParam("firstName",m_firstName);
            wpsi.setParam("lastName",m_lastName);
            wpsi.setId(DemoWpPassData.class.getName());
            MultiWorkplaceConnector.instance().startWorkpage(getOwningDispatcher(),"SECOND",null,wpsi,false);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }

    public void onStartTestPageAction(ActionEvent event)
    {
        try
        {
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setPageBeanName(DemoMultiWorkplaceTest.class.getSimpleName());
            wpsi.setOpenMultipleInstances(false);
            wpsi.setText("Test");
            wpsi.setId(DemoMultiWorkplaceTest.class.getName());
            MultiWorkplaceConnector.instance().startWorkpage(getOwningDispatcher(),"SECOND",null,wpsi,false);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
