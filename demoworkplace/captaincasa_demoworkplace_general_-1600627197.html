The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Change Style Variant at Runtime

Select a style variant by using the combo box and the field below will be updated accordingly.

From update 20131014 on style variants can be assigned dynamically - and can be changed at runtime.

This means a significant improvement when using style variants. Typically a style variant represents a certain semantics, e.g. "key fields" are rendered differently to "normal fields". If the semantics of a component now changes, the style variant can follow accordingly.

When defining corresponding style variants (Swing: style.xml, FX: fx.css and style.xml) please note that component attributes that are marked as "FIX" in the layout editor cannot be changed at runtime due to assigning a different style variant.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoStyleVariantChange}")

public class DemoStyleVariantChange
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_text;
        public String getText() { return m_text; }
        public void setText(String value) { this.m_text = value; }

        String m_styleVariant;
        public String getStyleVariant() { return m_styleVariant; }
        public void setStyleVariant(String value) { this.m_styleVariant = value; }

    }

    ValidValuesBinding m_buttonStyleVariantVVS = new ValidValuesBinding();
    String m_buttonStyleVariant = "RED";
    String[] VARIANTS = new String[] {"RED","GREEN","BLUE"};

    public DemoStyleVariantChange(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_buttonStyleVariantVVS.addValidValue("RED","RED");
        m_buttonStyleVariantVVS.addValidValue("GREEN","GREEN");
        m_buttonStyleVariantVVS.addValidValue("BLUE","BLUE");
        m_buttonStyleVariantVVS.addValidValue(null,"null (default)");
        for (int i=0; i<100; i++)
        {
            GridItem g = new GridItem();
            String styleVariant = VARIANTS[i%VARIANTS.length];
            g.setText("Text " + i + " / " + styleVariant);
            g.setStyleVariant(styleVariant);
            m_grid.getItems().add(g);
        }
    }

    public String getPageName() { return "/workplace/demostylevariantchange.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoStyleVariantChange}"; }

    public ValidValuesBinding getButtonStyleVariantVVS() { return m_buttonStyleVariantVVS; }
    
    public String getButtonStyleVariant() { return m_buttonStyleVariant; }
    public void setButtonStyleVariant(String value) { this.m_buttonStyleVariant = value; }

}
