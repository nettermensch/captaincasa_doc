This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.util.IAcceleratedPropertyAccess;

@CCGenClass (expressionBase="#{d.DemoPerformance2}")

public class DemoPerformance2 implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class PropData implements IAcceleratedPropertyAccess
    {
        public Object getPropertyValue(String property)
        {
            if ("value".equals(property)) return getValue();
            if ("enabled".equals(property)) return i_enabled; 
            if ("bgpaint".equals(property)) return i_bgpaint; 
            if ("flush".equals(property)) return true;
            if ("requestfocus".equals(property)) return 0;
            if ("maxlength".equals(property)) return 10;
            return IAcceleratedPropertyAccess.NOT_AVAILABLE;
        }
        String i_value = s_random.nextInt(10000)+"";
        boolean i_enabled = (s_random.nextInt()%2 == 0);
        String i_bgpaint = "#FF000030";
        public String getValue()
        {
            i_value = s_random.nextInt(10000)+"";
            return i_value;
        }
        public void setValue(String value)
        {
            i_value = value;
        }
        public boolean isEnabled()
        {
            return i_enabled;
        }
        public String getBgpaint()
        {
            return i_bgpaint;
        }
        public String getFlush()
        {
            return "true";
        }
        public String getRequestfocus()
        {
            return "0";
        }
        public String getMaxlength()
        {
            return "10";
        }
    }
    
    public class GridItem extends FIXGRIDItem
    {
        Map<String,PropData> i_propmap = new HashMap<String,PropData>();
        public Map<String, PropData> getPropmap()
        {
            return i_propmap;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static int COLCOUNT = 20;
    static boolean CHANGEINDEX = true;
    static Random s_random = new Random();
    
    protected ROWDYNAMICCONTENTBinding m_dyn1 = new ROWDYNAMICCONTENTBinding();
    protected ROWDYNAMICCONTENTBinding m_dyn2 = new ROWDYNAMICCONTENTBinding();
    protected ROWDYNAMICCONTENTBinding m_dyn3 = new ROWDYNAMICCONTENTBinding();
    protected ROWDYNAMICCONTENTBinding m_dyn4 = new ROWDYNAMICCONTENTBinding();
    protected ROWDYNAMICCONTENTBinding m_dyn5 = new ROWDYNAMICCONTENTBinding();
    
    FIXGRIDListBinding<GridItem> m_grid1 = new FIXGRIDListBinding<GridItem>(CHANGEINDEX);
    FIXGRIDListBinding<GridItem> m_grid2 = new FIXGRIDListBinding<GridItem>(CHANGEINDEX);
    FIXGRIDListBinding<GridItem> m_grid3 = new FIXGRIDListBinding<GridItem>(CHANGEINDEX);
    FIXGRIDListBinding<GridItem> m_grid4 = new FIXGRIDListBinding<GridItem>(CHANGEINDEX);
    FIXGRIDListBinding<GridItem> m_grid5 = new FIXGRIDListBinding<GridItem>(CHANGEINDEX);

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoPerformance2()
    {
        super();
        m_dyn1.setContentXml(initDyn(1));
        m_dyn2.setContentXml(initDyn(2));
        m_dyn3.setContentXml(initDyn(3));
        m_dyn4.setContentXml(initDyn(4));
        m_dyn5.setContentXml(initDyn(5));
        initGridData(m_grid1);
        initGridData(m_grid2);
        initGridData(m_grid3);
        initGridData(m_grid4);
        initGridData(m_grid5);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public ROWDYNAMICCONTENTBinding getDyn1() { return m_dyn1; }
    public ROWDYNAMICCONTENTBinding getDyn2() { return m_dyn2; }
    public ROWDYNAMICCONTENTBinding getDyn3() { return m_dyn3; }
    public ROWDYNAMICCONTENTBinding getDyn4() { return m_dyn4; }
    public ROWDYNAMICCONTENTBinding getDyn5() { return m_dyn5; }
    
    public FIXGRIDListBinding<GridItem> getGrid1() { return m_grid1; }
    public FIXGRIDListBinding<GridItem> getGrid2() { return m_grid2; }
    public FIXGRIDListBinding<GridItem> getGrid3() { return m_grid3; }
    public FIXGRIDListBinding<GridItem> getGrid4() { return m_grid4; }
    public FIXGRIDListBinding<GridItem> getGrid5() { return m_grid5; }

    private String initDyn(int dyni)
    {
        StringBuffer sb = new StringBuffer();
        sb.append("<t:fixgrid objectbinding='#{d.DemoPerformance2.grid"+dyni+"}' sbvisibleamount='10'" +
                            " rowheight='16' bordercolor='#00000030'" +
                            " borderheight='1' borderwidth='1'" +
                            " width='100%'>\n");
        for (int coli=0; coli<COLCOUNT; coli++)
        {
            sb.append("<t:gridcol text='Column "+coli+"'>\n");
            sb.append("<t:field text='.{propmap.column"+coli+".value}'" +
//                              " enabled='.{propmap.column"+coli+".enabled}'" +
                              " bgpaint='.{propmap.column"+coli+".bgpaint}'" +
                              " flush='.{propmap.column"+coli+".flush}'" +
                              " maxlength='.{propmap.column"+coli+".maxlength}'" +
                              " requestfocus='.{propmap.column"+coli+".requestfocus}'/>\n");
            sb.append("</t:gridcol>\n");
        }
        sb.append("</t:fixgrid>\n");
        return sb.toString();
    }
    
    private void initGridData(FIXGRIDListBinding<GridItem> grid)
    {
        for (int rowi=0; rowi<50; rowi++)
        {
            GridItem gi = new GridItem();
            grid.getItems().add(gi);
            for (int coli=0; coli<COLCOUNT; coli++)
            {
                String propName = "column"+coli;
                PropData propData = new PropData();
                gi.i_propmap.put(propName,propData);
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    


    
}
