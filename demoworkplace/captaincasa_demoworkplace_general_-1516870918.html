  • General Issues
  • #{rr.literals['tiSwitchLangauge']}

This demo shows how to switch between different languages/country settings - both on server side and on client side.

(.) The server side switch ("application level") is done through the view root of the FacesContext.
(.) The client side switch ("client level") is done through the CLIENTCONFIG.

The CLIENTCONFIG component is not contained within this .jps definition - but is contained in the out-most jsp-page (workplace/workplace.jsp). In general the CLIENTCONFIG definition should be "as outmost as possible".

The reload is required in this case, in which you want to re-render all existing components on client side. Otherwise changes are only applied to newly created components.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Locale;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.util.HttpSessionAccess;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoSwitchLocale}")

/*
 * The CLIENTCONFIG should be defined in the "outmost" page. In the demo workplace
 * this is the workplace.jsp page. The data for passing the language and country settings
 * are kept on central dispatcher level.
 */
public class DemoSwitchLocale
    extends PageBean 
    implements Serializable
{
    String m_country;
    String m_language;

    public String getCountry() { return m_country; } 
    public void setCountry(String country) { m_country = country; }

    public String getLanguage() { return m_language; } 
    public void setLanguage(String language) { m_language = language; }

    public DemoSwitchLocale(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        updateFieldsFromServerLocale();
    }

    public void onSwitchToEnglish(ActionEvent event) 
    {
        Locale l = new Locale("en","US");
        switchServerAndClientLocale(l);
    }

    public void onSwitchToGerman(ActionEvent event) 
    {
        Locale l = new Locale("de","DE");
        switchServerAndClientLocale(l);
    }

    public void onSwitch(javax.faces.event.ActionEvent event) 
    {
        Locale l = new Locale(m_language,m_country);
        switchServerAndClientLocale(l);
    }
    
    private void switchServerAndClientLocale(Locale l)
    {
        // switch the server side language
        HttpSessionAccess.setCurrentLocale(l);
        // in addition (optional): switch client side locale
        ((Dispatcher)getOwningDispatcher()).defineClientConfigLocale(l);
        getWorkpageContainer().getFunctionsManager().importWorkplaceFunctionTreeInfoNode("default");
        HttpSessionAccess.reloadClient();
        updateFieldsFromServerLocale();
    }

    private void updateFieldsFromServerLocale()
    {
        Locale l = HttpSessionAccess.getCurrentLocale();
        m_country = l.getCountry();
        m_language = l.getLanguage();
    }

}
