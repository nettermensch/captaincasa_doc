The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Images
  • Lazy loading of PDF thumbnails

This example handles a quite complex case: each line of the grid represents some PDF document, which is internally available as resource.

For each line of the grid a thumb nail image is produced. The thumb nail image is loaded in a separate thread - assuming that the production of the image is an expensive activity which should not be executed when loading the grid - but which should in the backgrond.

The images are loaded by implementing a BufferedContent class and creating instances at runtime.

It is crucially important to add the file name to the buffered content - and in particular to specify the dimension of the image in this file name (here: "_100x100"). If not doing so then the first thing the client will do, is to calculate the size of the image - and for this reason it requires to access the image data! If not defining the size as part of the image name, then the result of the demo will be a very long duration that the client waits for the images to be loaded.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Hashtable;
import java.util.List;
import java.util.Map;

import org.eclnt.ccee.pdf.PDF2ImageConverter;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.BufferedContentMgr;
import org.eclnt.jsfserver.bufferedcontent.DefaultBufferedContent;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.ThreadData;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.polling.LongPollingMgr;
import org.eclnt.jsfserver.polling.websocket.DefaultLongPollingWebSocket;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoPDFThumbNails}")

public class DemoPDFThumbNails
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class ThumbNailBufferedContent extends DefaultBufferedContent
    {
        String i_name;
        public ThumbNailBufferedContent(String name)
        { 
            i_name = name; 
        }
        @Override
        public byte[] getContent()
        {
            try
            {
                int counter = 0;
                while (counter < 25)
                {
                    counter++;
                    byte[] image = m_thumbNails.get(i_name);
                    if (image != null)
                        return image;
                    Thread.sleep(1000);
                    System.out.println("No image available yet: " + image);
                }
                throw new Exception("Counter exceeded 25 " + i_name);
            }
            catch (Throwable t)
            {
                CLog.L.log(CLog.LL_ERR,"Problem resolving thumbNail: " + i_name,t);
                throw new Error("Problem resolving thumbNail: " + i_name,t);
            }
        }
        @Override
        public String getContentType()
        {
            return null; 
        }
    }
    
    public class ThumbnailLoaderThread extends Thread
    {
        @Override
        public void run()
        {
            // create a "clone" so that looping through the list is done
            // in own instance which independent from oriignal list (which may
            // be manipulated in e.g. UI thread processing)
            ClassloaderReader clr = new ClassloaderReader(true);
            List<GridItem> l = new ArrayList<>(m_grid.getItems());
            for (GridItem gi: l)
            {
                try 
                { 
                    Thread.sleep(2500); // slow down to simulate heavy activity
                    byte[] pdf = clr.readFile(gi.getName(),false);
                    byte[] image = PDF2ImageConverter.convertPDFIntoThumbImagePNG(pdf,200,200);
                    m_thumbNails.put(gi.getName(),image);
                    System.out.println("PROCESSED IMAGE " + gi.getName());
                }
                catch (Throwable t) 
                {
                    CLog.L.log(CLog.LL_ERR,"Problem when loading thumbnail: " + gi.getName(),t);
                }
            }
        }
    }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_name;
        String i_image;
        ThumbNailBufferedContent i_bufferedContent;
        public GridItem(String name)
        { 
            i_name = name;
            i_bufferedContent = new ThumbNailBufferedContent(i_name);
            i_bufferedContent.setFileName("thumbNail_200x200.png");
            BufferedContentMgr.add(i_bufferedContent);
        }
        public String getImage() { return i_bufferedContent.getURL(); }
        public String getName() { return i_name; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    DefaultLongPollingWebSocket m_polling;
    Map<String,byte[]> m_thumbNails = new Hashtable<String,byte[]>(); // thread-safe map!
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPDFThumbNails(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_polling = new DefaultLongPollingWebSocket();
        LongPollingMgr.add(m_polling);
        loadGrid();
        startThumbnailLoader();
        // registrations to BufferedContentMgr are removed when closing a dialog session 
        // automatically. You may do this earlier: in this case it is assumed that the
        // page is started within a workpage - and all registrations are removed when
        // the workpage is removed from the workplace.
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnDestroyed()
            {
                super.reactOnDestroyed();
                destroy();
            }
        });
    }

    public String getRootExpressionUsedInPage() { return "#{d.DemoPDFThumbNails}"; }
    public String getLayoutPageName() { return "/worplace/demopdfthumbnails.xml"; }

    public String getPollingURL() { return m_polling.getURL(); }

    public void onPollingAction(javax.faces.event.ActionEvent event) 
    {
        Statusbar.outputMessage("HALLO " + System.currentTimeMillis());
    }
    
    public void destroy()
    {
        cleanBufferedContentRegistrations();
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    private void cleanBufferedContentRegistrations()
    {
        for (GridItem gi: m_grid.getItems())
        {
            BufferedContentMgr.remove(gi.i_bufferedContent);
        }
    }
    
    private void loadGrid()
    {
        m_grid.getItems().clear();
        for (int i=1; i<=5; i++)
        {
            GridItem gi = new GridItem("/workplace/resources/pdf0"+i+".pdf");
            m_grid.getItems().add(gi);
        }
    }

    private void startThumbnailLoader()
    {
        ThumbnailLoaderThread tlt = new ThumbnailLoaderThread();
        ThreadData.injectExtThread(tlt);
        tlt.start();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
