The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Field with remove icon

By setting attribute WITHREMOVEICON an explicit remove icons is a added to the field control.

The attribute is available for all field controls, including: FIELD, PASSWORD, COMBOFIELD, COMBOBOX, LONGTEXTFIELD.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFieldWithRemoveIcon}")

public class DemoFieldWithRemoveIcon
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_someInt = 4711;
    String m_someText = "Test";
    String m_someDepartment = "SALES";
    ValidValuesBinding m_costCenterVVS = new ValidValuesBinding();
    String m_someCostCenter = "CC_1";
    String m_someLongText = "Some long text";
    int m_someSpinnerInt = 10;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFieldWithRemoveIcon(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<20; i++)
            m_costCenterVVS.addValidValue("CC_"+i,"Cost center " + i);
    }

    public String getPageName() { return "/workplace/demofieldwithremoveicon.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFieldWithRemoveIcon}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getCostCenterVVS() { return m_costCenterVVS; }

    public String getSomeCostCenter() { return m_someCostCenter; }
    public void setSomeCostCenter(String value) { this.m_someCostCenter = value; }

    public int getSomeInt() { return m_someInt; }
    public void setSomeInt(int value) { this.m_someInt = value; }

    public String getSomeText() { return m_someText; }
    public void setSomeText(String value) { this.m_someText = value; }

    public String getSomeDepartment() { return m_someDepartment; }
    public void setSomeDepartment(String value) { this.m_someDepartment = value; }

    public void onDepartmentAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            IdTextSelection idts = IdTextSelection.createInstance();
            idts.addLine("SALES","Sales");
            idts.addLine("PROD","Production");
            idts.addLine("HR","Human Resources");
            idts.setCallBack(new ISetId()
            {
                @Override
                public void setId(String id)
                {
                    m_someDepartment = id;
                }
            });
        }
    }

    public String getSomeLongText() { return m_someLongText; }
    public void setSomeLongText(String value) { this.m_someLongText = value; }

    public int getSomeSpinnerInt() { return m_someSpinnerInt; }
    public void setSomeSpinnerInt(int value) { this.m_someSpinnerInt = value; }

}
