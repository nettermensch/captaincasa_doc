The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Tab Pane
  • Place components at begin/end of tabbed line

The component TABBEDPANE provides an area at the beginning (TABBEDAREABEGIN) and at the end (TABBEAREAEND) of the tabbed line, in which you can place one own component. (Of course you can e.g. also add a PANE component and several components inside the PANE component.)

Typical use case: you want to add some more additional tab instances by some "add"-icon.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoTabbedArea}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoTabbedArea}" >
<t:row id="g_49" >
<t:tabbedpane id="g_59" value="#{d.DemoTabbedArea.tabValue}" width="100%" >
<t:tabbedareabegin id="g_82" >
<t:icon id="g_75" actionListener="#{d.DemoTabbedArea.onAddNewCarBeforeAction}" image="/images/iconssvg/add_16x16.svg" />
</t:tabbedareabegin>
<t:tabbedareaend id="g_6" >

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTabbedArea}")

public class DemoTabbedArea
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class CarListItem extends Object implements java.io.Serializable
    {
        String i_id;
        int i_maxSpeed;
        int i_power;
        String i_manufacturer;
        String i_type;
        public int getMaxSpeed() { return i_maxSpeed; }
        public void setMaxSpeed(int value) { this.i_maxSpeed = value; }
        public int getPower() { return i_power; }
        public void setPower(int value) { this.i_power = value; }
        public String getManufacturer() { return i_manufacturer; }
        public void setManufacturer(String value) { this.i_manufacturer = value; }
        public String getId() { return i_id; }
        public void setId(String value) { this.i_id = value; }
        public String getType() { return i_type; }
        public void setType(String value) { this.i_type = value; }
    }


    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<CarListItem> m_carList = new ArrayList<CarListItem>();
    int m_tabValue;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTabbedArea(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        CarListItem car = new CarListItem();
        car.setId("HD-CC 6999");
        car.setManufacturer("Mercedes");
        car.setType("Vito Tourer");
        m_carList.add(car);
    }

    public String getPageName() { return "/workplace/demotabbedarea.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTabbedArea}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<CarListItem> getCarList() { return m_carList; }

    public void onAddNewCarBeforeAction(javax.faces.event.ActionEvent event) 
    {
        CarListItem car = new CarListItem();
        car.setId("...new Id...");
        m_carList.add(0,car);
        m_tabValue = 0;
    }

    public void onAddNewCarAction(javax.faces.event.ActionEvent event) 
    {
        CarListItem car = new CarListItem();
        car.setId("...new Id...");
        m_carList.add(car);
        m_tabValue = m_carList.size()-1;
    }

    public int getTabValue() { return m_tabValue; }
    public void setTabValue(int value) { this.m_tabValue = value; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
