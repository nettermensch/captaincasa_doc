The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Positioning in free style HTML
  • Example: order letter

This example shows the usage of TEXTWITHCONTROLS in order to define some kind of "order letter".

The text is read from a server side template file. The template file is updated on server side first (title and last name are added to the text).

Please observe on of the disadvantages: when updating the text on server side (e.g. by changing the last name), then the whole template is reprocessed: the focus gets lost and you see a fast flickering.

Use the TEXTWITHCONTROLS if you really want to provide some templating as part of your application or if you want to provide some layout to the user which is close to some print layout.

Do not use it for "normal layouting" of your components!

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.time.LocalDate;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoOrderLetter}")

public class DemoOrderLetter
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        BigDecimal i_price;
        public BigDecimal getPrice() { return i_price; }
        public void setPrice(BigDecimal value) { this.i_price = value; }

        int i_quantity;
        public int getQuantity() { return i_quantity; }
        public void setQuantity(int value) { this.i_quantity = value; }

        String i_article;
        public String getArticle() { return i_article; }
        public void setArticle(String value) { this.i_article = value; }

        public void onRemoveAction(ActionEvent event)
        {
            m_grid.getItems().remove(this);
        }
        
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    LocalDate m_paymentDate;
    String m_template;
    String m_text;
    String m_country;
    String m_city;
    String m_street;
    String m_zipCode;
    String m_firstName;
    String m_lastName;
    String m_title;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    BigDecimal m_total;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoOrderLetter(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_template = new ClassloaderReader(true).readUTF8File("/workplace/DemoOrderLetter.html",true);
        updateText();
    }

    public String getPageName() { return "/workplace/demoorderletter.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoOrderLetter}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getText() { return m_text; } 
    
    public String getCountry() { return m_country; }
    public void setCountry(String value) { this.m_country = value; }

    public String getCity() { return m_city; }
    public void setCity(String value) { this.m_city = value; }

    public String getStreet() { return m_street; }
    public void setStreet(String value) { this.m_street = value; }

    public String getZipCode() { return m_zipCode; }
    public void setZipCode(String value) { this.m_zipCode = value; }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { this.m_lastName = value; updateText(); }

    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { this.m_firstName = value; }

    public String getTitle() { return m_title; }
    public void setTitle(String value) { this.m_title = value; updateText(); }

    public LocalDate getPaymentDate() { return m_paymentDate; }
    public void setPaymentDate(LocalDate value) { this.m_paymentDate = value; }

    public BigDecimal getTotal()
    {
        m_total = new BigDecimal(0);
        for (GridItem gi: m_grid.getItems())
        {
            m_total = m_total.add(gi.getPrice());
        }
        return m_total;
    }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onNewItemAction(javax.faces.event.ActionEvent event) 
    {
        GridItem gi = new GridItem();
        gi.setArticle("...please define article...");
        gi.setQuantity(1);
        gi.setPrice(new BigDecimal("10.99"));
        m_grid.getItems().add(gi);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void updateText()
    {
        // the template text contains placeholders which are resolved within the
        // text (title and last name)
        m_text = m_template;
        String title = getTitle();
        if (title == null) title = "(title unknown)";
        m_text = m_text.replace("||title||",title);
        String lastName = getLastName();
        if (lastName == null) lastName = "(last name unknown)";
        m_text = m_text.replace("||lastName||",lastName);
    }
}
