The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Content Lifecycle

Input some data into the two fields. The close the page. A message box will come up and warn that data might get lost.

A page that is started via the workplace management is associated with an IWorkpage object instance. This IWorkpage instance is the one to register for notifications that are thrown when the user (or an inside function) wants to close a page.

By implementing the interfaec "IWorkpageLifecycleListener" you get notified in two ways:

(1.) the "close()" method allows your application to intervent and return false.
(2.) the "closeForced()" method is called when all listeners have agreed to close, or if the closeForced method is called itself.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Date;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoWpLifecycle}")

public class DemoWpLifecycle 
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    class MyLifecycleListener extends WorkpageDefaultLifecycleListener
    {
        public boolean close()
        {
            protocol("close()");
            if (m_dataWasChanged == false)
                return true;
            else
            {
                YESNOPopup.createInstance("UnsavedData",
                                          "Data was changed. Do you really want to exit?",
                                          new YESNOPopup.IYesNoListener()
                                          {
                                              public void reactOnNo()
                                              {
                                              }
                                              public void reactOnYes()
                                              {
                                                  getWorkpageContainer().closeWorkpageForced(getWorkpage());
                                                  if (getCloseContinueOperation() != null)
                                                      getCloseContinueOperation().run();
                                              }
                                          });
                return false;
            }
        }
        public void closeForced()
        {
            protocol("closeForced()");
        }
        public void reactOnDestroyed()
        {
            protocol("reactOnDestroyed()");
        }
        @Override
        public void reactOnShownInContentArea()
        {
            System.out.println("*******************************************");
            System.out.println("* "+createTimeStamp()+" reactOnShownInContentArea()");
            System.out.println("*******************************************");
            protocol("reactOnShownInContentArea()");
        }
        @Override
        public void reactOnHiddenInContentArea()
        {
            System.out.println("*******************************************");
            System.out.println("* "+createTimeStamp()+" reactOnHiddenInContentArea()");
            System.out.println("*******************************************");
            protocol("reactOnHiddenInContentArea()");
        }
        @Override
        public void reactOnShownInPopup()
        {
            protocol("reactOnShownInPopup()");
        }
        
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected boolean m_dataWasChanged = false;
    protected String m_lastName = "Last";
    protected String m_firstName = "First";
    protected String m_protocol = "Protocol:";
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoWpLifecycle(IDispatcher dispatcher)
    {
        super(dispatcher);
        getWorkpage().addLifecycleListener(new MyLifecycleListener());
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { m_firstName = value; m_dataWasChanged = true; }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { m_lastName = value; m_dataWasChanged = true; }

    public void setProtocol(String value) { m_protocol = value; }
    public String getProtocol() { return m_protocol; }
    
    public void onSave(ActionEvent event) 
    {
        Statusbar.outputSuccess("Data was saved.");
        m_dataWasChanged = false;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void protocol(String s)
    {
        m_protocol += "\n" + createTimeStamp() + " - " + s;
        CLog.L.log(CLog.LL_INF,"##### " + s);
    }
    
    private String createTimeStamp()
    {
        DateFormat df = new SimpleDateFormat("yyyy/MM/dd - HH.mm.ss.SSS");
        return df.format(new Date());
    }
}
