The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Avoid selection of certain lines

All items containing a "3" or a "5" in the last name are not select-able.

Take a look into the code of the "onRowSelect" implementation of the Java class.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridAvoidSelection}")

public class DemoGridAvoidSelection
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_firstName;
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }

        String i_lastName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }

        public void onRowSelect()
        {
            if (i_firstName.contains("3") || i_firstName.contains("5"))
            {
                m_grid.deselectItem(this);
            }
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridAvoidSelection(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);       
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setLastName("Last " + i);
            gi.setFirstName("First " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridavoidselection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridAvoidSelection}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public String getSummary() 
    {
        StringBuffer sb = new StringBuffer();
        for (GridItem gi: m_grid.getItems())
        {
            if (m_grid.getSelectedItems().contains(gi))
            {
                if (sb.length() > 0)
                    sb.append(", ");
                sb.append(gi.i_lastName);
            }
        }
        return sb.toString();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
