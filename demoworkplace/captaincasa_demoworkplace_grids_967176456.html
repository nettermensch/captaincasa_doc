The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid data export
  • Accessing client side value strings

Sometimes you want to get the client side text representation on server side as well - e.g. when exporting information.

In this case you can use the method "getItemColumnClientStringValue(...)" which is available with the server side grid object.

The method internally uses the class ValueManager with its methods "convertObject2DisplayString..." and passes the column configuration of the grid into the parameters of this method.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.ColumnInfo;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.READTYPE;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoGridGetClientTexts}")

public class DemoGridGetClientTexts
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        double i_weight;
        public double getWeight() { return i_weight; }
        public void setWeight(double value) { this.i_weight = value; }

        Date i_birthDate;
        public Date getBirthDate() { return i_birthDate; }
        public void setBirthDate(Date value) { this.i_birthDate = value; }

        String i_name;
        public String getName() { return i_name; }
        public void setName(String value) { this.i_name = value; }

        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    String m_protcol;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridGetClientTexts(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<50; i++)
        {
            GridItem gi = new GridItem();
            gi.setName(RandomLogic.getLastName() + ", " + RandomLogic.getFirstName());
            gi.setBirthDate(new Date());
            gi.setWeight((float)RandomLogic.getHeight());
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridgetclienttexts.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridGetClientTexts}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onOutputTextAction(javax.faces.event.ActionEvent event) 
    {
        StringBuffer sb = new StringBuffer();
        List<ColumnInfo> cols = m_grid.getVisibleColumns();
        for (GridItem gi: m_grid.getItems())
        {
            int counter = 0;
            for (ColumnInfo col: cols)
            {
                if (counter != 0) sb.append(", ");
                String cv = m_grid.getItemColumnClientStringValue(gi,col,READTYPE.EXPORT);
                sb.append(cv);
                counter++;
            }
            sb.append("\n");
        }
        m_protcol = sb.toString();
    }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public String getProtcol() { return m_protcol; }
    public void setProtcol(String value) { this.m_protcol = value; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
