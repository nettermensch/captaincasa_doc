The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Sorting
  • Tree Sorting

By default the tree-sorting strictly sorts the contained items by their content (e.g. by their text in case of sorting on the TREENODE column).

But: by setting property FIXGRIDTREEBinding.setFoldersAlwaysSortedToTop(...) you can enforce that folder nodes are always sorted to the top.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoTreeFolderSorting}")

public class DemoTreeFolderSorting
    extends PageBean 
    implements Serializable
{
    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        public TreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        
    }

    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();
    static String[] TEXTS = new String[] {"A","B","C","D","E"};

    public DemoTreeFolderSorting(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<TEXTS.length; i++)
        {
            TreeNode top = new TreeNode(m_tree.getRootNode());
            top.setText(TEXTS[i]);
            if (i%2 == 0)
            {
                top.setStatus(FIXGRIDTreeItem.STATUS_CLOSED);
                for (int j=0; j<TEXTS.length; j++)
                {
                    TreeNode sub = new TreeNode(top);
                    sub.setText(TEXTS[j]);
                    if (j%2 == 0)
                    {
                        sub.setStatus(FIXGRIDTreeItem.STATUS_CLOSED);
                        for (int k=0; k<TEXTS.length; k++)
                        {
                            TreeNode subsub = new TreeNode(sub);
                            subsub.setText(TEXTS[j]);
                            subsub.setStatus(FIXGRIDTreeItem.STATUS_ENDNODE);
                        }
                    }
                    else
                    {
                        sub.setStatus(FIXGRIDTreeItem.STATUS_ENDNODE);
                    }
                }
            }
            else
            {
                top.setStatus(FIXGRIDTreeItem.STATUS_ENDNODE);
            }
        }
    }

    public FIXGRIDTreeBinding<TreeNode> getTree() { return m_tree; }
}
