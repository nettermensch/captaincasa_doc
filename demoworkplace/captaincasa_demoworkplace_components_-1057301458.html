The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Text Input/Output
  • Simple text editor
  • Caret info - Popup Menu

This example shows how to insert text (here: HTML statements) into a document during editing.

The editor component informs about the current caret location by the attribute TEXTSELECTIONINFO - which provides a semicolon separated integer sequence. Index 0 holds holds the carte position. In case of a selection in client side index 1 holds the start of the selection and index 2 the end of the selection.

When actively setting the caret on server side then you need to trigger the client explicitly using attribute TRIGGERTEXTSELECTION.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoSimpleTextEditorCaretMgmt}")

public class DemoSimpleTextEditorCaretMgmt
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static String NBSP = "\u202F";
    
    String m_textSelectionInfo;
    String m_text = RandomLogic.getLoremIpsumShort();
    Trigger m_textSelectionInfoTrigger = new Trigger();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoSimpleTextEditorCaretMgmt(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demosimpletexteditorcaretmgmt.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSimpleTextEditorCaretMgmt}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getTextSelectionInfo() { return m_textSelectionInfo; }
    public void setTextSelectionInfo(String value) { m_textSelectionInfo = value; }
    public Trigger getTextSelectionInfoTrigger() { return m_textSelectionInfoTrigger; }

    public String getText() { return m_text; }
    public void setText(String value) 
    { 
        m_text = value;
        // &nbsp; is treated as one character in browser! but would be treated as
        // 5 characters in java processing
        m_text = m_text.replace("&nbsp;",NBSP);
    }
    

    public void onAddImageAction(javax.faces.event.ActionEvent event) 
    {
        updateText("<img src=\"images/iconssvg/add_16x16.svg\">");
    }

    public void onAddSpanAction(javax.faces.event.ActionEvent event) 
    {
        updateText("<span contenteditable=\"false\" style=\"text-weight:bold;color:blue\">SOME FIX TEXT</span>");
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void updateText(String sForInsert)
    {
        int insertPosition = findInsertPosition();
        m_text = m_text.substring(0,insertPosition) + sForInsert + m_text.substring(insertPosition);
        int newCaretPosition = insertPosition + sForInsert.length();
        m_textSelectionInfo = newCaretPosition + ";" + newCaretPosition + ";" + newCaretPosition;
        m_textSelectionInfoTrigger.trigger();
    }

    private int findInsertPosition()
    {
        int insertPosition = m_text.length();
        try
        {
            int[] tsiValues = ValueManager.decodeStraightIntCSV(m_textSelectionInfo);
            if (tsiValues[0] >= 0)
                insertPosition = tsiValues[0];
        }
        catch (Throwable t) { }
        return insertPosition;
    }

}
