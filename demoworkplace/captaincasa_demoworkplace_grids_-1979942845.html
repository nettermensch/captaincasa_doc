The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Fix columns
  • Fix columns - by configuration

When setting attribute FIXGRID-WITHFIXCOLUMNSBYUSERCONFIGURATION to "true", then the dialog for configuring the selection and sequence of columns will also show a field in which you can define the number of fix columns.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoGridFixColumnsDirectConfig}")

public class DemoGridFixColumnsDirectConfig
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        Person i_person;
        public GridItem(Person person)
        {
            i_person = person;
        }
        public void onRowSelect() { }
        public void onRowExecute() { }
        public Person getPerson() { return i_person; } 
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridFixColumnsDirectConfig(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillGrid();
    }

    public String getPageName() { return "/workplace/demogridfixcolumnsdirectconfig.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridFixColumnsDirectConfig}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
 
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        for (int i=0; i<100; i++)
            m_grid.getItems().add(new GridItem(RandomLogic.createPerson()));
    }
}
