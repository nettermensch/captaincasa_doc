The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Button Popup Menu

The component BUTTONPOPUPMENU is a simple button that opens its popup menu when the user clicks the button.

Main purpose: for touch screens (phone, tablet, ...) the starting of a popup menu is not such obvious - there is no right mouse button and alternative gestures (such as "long touch") are not really user-friendly.

So you can add a popup menu for direct clicking - and the popup menu management is the same as with normal popup menus.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuLoad;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoButtonPopupMenu}")

public class DemoButtonPopupMenu
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DYNAMICCONTENTBinding m_dynMenuContent = new DYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoButtonPopupMenu(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demobuttonpopupmenu.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoButtonPopupMenu}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynMenuContent() { return m_dynMenuContent; }

    public void onButton1Action(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            Statusbar.outputSuccess("Menut item selected: " + e.getCommand());
        }
    }

    public void onButton2Action(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuLoad)
        {
            int maxi = new Random().nextInt(10) + 2;
            List<ComponentNode> nodes = new ArrayList<ComponentNode>();
            for (int i=0; i<maxi; i++)
            {
                MENUITEMNode min = new MENUITEMNode()
                        .setText("Dyn Demo " + i)
                        .setCommand("DYNDEMO"+i);
                nodes.add(min);
            }
            m_dynMenuContent.setContentNodes(nodes);
            Statusbar.outputMessage("Number of menu items created: " + maxi);
        }
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            Statusbar.outputSuccess("Menut item selected: " + e.getCommand());
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
