The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Chained popups

Select one or more items from the grid (hold down the ctrl- or shift-key for multiple selection). Then remove the selected items. A popup will show up for each item, letting the user confirm if the items really is to be removed.

The example shows how to use a sequence of popups. Be aware about the separation between UI client and server side processing. The processing is not managed within a loop, as you may expect when coming from fat client applications. But the processing is managed by a chained server side algorithm.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.managedbean.IDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupChain}")

public class DemoPopupChain
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }
        
        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    protected MyRow m_selectedRow = null;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoPopupChain(IDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<100; i++)
        {
            MyRow r = new MyRow();
            r.setFirstName("First " + i);
            r.setLastName("Last " + i);
            m_rows.getItems().add(r);
        }
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    
    public void onRemoveSelected(ActionEvent event) 
    {
        removeNextSelectedItem();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void removeNextSelectedItem()
    {
        // check if there is a selected item. Do not use m_gird.getSelectedItems()
        // because this returns a set. In this processing we want the items 
        // to be processed in the sequence they appear in the grid.
        m_selectedRow = null;
        for (MyRow row: m_rows.getItems()) 
        {
            if (row.getSelected() == true)
            {
                m_selectedRow = row;
                break;
            }
        }
        if (m_selectedRow == null)
            return;
        YESNOPopup ynp = YESNOPopup.createInstance(m_selectedRow.m_lastName + ", " + m_selectedRow.m_firstName,
                                                   "Do you really want to remove this item?",
                                                   new YESNOPopup.IYesNoListener()
        {
            public void reactOnNo()
            {
                // do not remove ==> deselect so that is is not processed with next
                // removeNextSelectedItem()
                m_rows.deselectItem(m_selectedRow);
                removeNextSelectedItem();
            }
            public void reactOnYes()
            {
                // remove
                m_rows.getItems().remove(m_selectedRow);
                removeNextSelectedItem();
            }
        });
        if (getWorkpage().getModelessPopupOfWorkpage() != null)
            ynp.getModalPopup().setOwner(getWorkpage().getModelessPopupOfWorkpage());
        else
            ynp.getModalPopup().setStartfromrootwindow(true);
        ynp.getModalPopup().setLeftTopReferenceCentered();
    }

}
