The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Mobile Periphery
  • Touch / Tile tree (PBC)

The page bean component "CCTreeByTiles" renders a normal FIXGRIDTreeBinding-instance as tree that is made out of tiles.

The first line of the component shows the tree node hierarchy that is currently opened, the line(s) below shows the node of the currently selected level.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPBCCCTreeByTiles}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPBCCCTreeByTiles}" >
<t:row id="g_6" >
<t:pagebeancomponent id="g_8" pagebeanbinding="#{d.DemoPBCCCTreeByTiles.treeUI}" pagebeanclass="org.eclnt.ccaddons.pbc.CCTreeByTiles" pagebeaninitdata="tileBorder:0;lineColor:#00000030;tileBackgroundFolderNode:#00000018" shownullcontent="true" />
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Random;

import org.eclnt.ccaddons.pbc.CCTreeByTiles;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCTreeByTiles}")

public class DemoPBCCCTreeByTiles
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        boolean i_childrenRead = false;
        public TreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        /*
         * This method is NOT called as part of the tile tree processing. 
         */
        public void onRowSelect()
        {
        }
        /*
         * This method is called when clicking onto an end node of the tile tree. 
         */
        public void onRowExecute()
        {
            Statusbar.outputSuccess("Row execute called with item: " + getText());
        }
        public void onToggle()
        {
            // In this example: dynamic loading of child nodes according to user
            // navigation within the tree
            if (i_childrenRead == false)
            {
                i_childrenRead = true;
                String textPrefix = "";
                for (int j=0; j<(getLevelInt()+1); j++)
                    textPrefix += "Sub ";
                int maxi = new Random().nextInt(10) + 1;
                for (int i=0; i<maxi; i++)
                {
                    TreeNode tn = new TreeNode(this);
                    tn.setText(textPrefix + "folder " + i);
                    tn.setStatus(TreeNode.STATUS_CLOSED);
                    tn.setImage("/images.iconssvg.folder_16x16._ffc000.16x16.ccsvg");
                }
                maxi = new Random().nextInt(15) + 1;
                for (int i=0; i<maxi; i++)
                {
                    TreeNode tn = new TreeNode(this);
                    tn.setText(textPrefix + "node " + i);
                    tn.setStatus(TreeNode.STATUS_ENDNODE);
                    tn.setImage("/images.iconssvg.endnode_16x16._808080.16x16.ccsvg");
                }
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCTreeByTiles m_treeUI = new CCTreeByTiles();
    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCTreeByTiles(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        buildTree();
        m_treeUI.prepare(m_tree,new CCTreeByTiles.DefaultListener()
        {
        });
    }

    public String getPageName() { return "/workplace/demopbccctreebytiles.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCTreeByTiles}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCTreeByTiles getTreeUI() { return m_treeUI; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildTree()
    {
        for (int i=0; i<10; i++)
        {
            TreeNode tn = new TreeNode(m_tree.getRootNode());
            tn.setText("Top folder " + i);
            tn.setStatus(TreeNode.STATUS_CLOSED);
            tn.setImage("/images.iconssvg.folder_16x16._ffc000.16x16.ccsvg");
        }
    }

}
