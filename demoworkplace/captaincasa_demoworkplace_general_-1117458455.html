The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Hide close/maximize icon

By calling ModalPopup/ModelessPopup.hideCloseIcon() you can influence the title bar of the decorated dialog.

In addition to the optical removing of the icon, the user also cannot close the dialog by pressing "escape" on the keyboard.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPopupHideCloseIcon}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPopupHideCloseIcon}" rowdistance="10" >
<t:row id="g_6" >

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupHideCloseIcon}")

public class DemoPopupHideCloseIcon
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupHideCloseIcon(IWorkpageDispatcher workpageDispatcher) 
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demopopuphidecloseicon.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupHideCloseIcon}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onPopupDefaultAction(javax.faces.event.ActionEvent event) 
    {
        // here the popup is used by using a default popup...
        YESNOPopup ynp = YESNOPopup.createInstance("Default closing","By default users can close a dialog by a close icon - and, in case of modal dialogs, by pressing the escape-key.",new YESNOPopup.IYesNoListener()
        {
            @Override
            public void reactOnYes() { }
            @Override
            public void reactOnNo() { }
        });
    }

    public void onPopupWithoutCloseIconAction(javax.faces.event.ActionEvent event) 
    {
        // here the popup is used by using a default popup...
        YESNOPopup ynp = YESNOPopup.createInstance("No closing!","There is no closing by some close icon and no closing by pressing escape on the keyboard! You have to either press Yes or No.",new YESNOPopup.IYesNoListener()
        {
            @Override
            public void reactOnYes() { }
            @Override
            public void reactOnNo() { }
        });
        ynp.getModalPopup().hideCloseIcon();
    }

    public void onPopupWithoutMaximizeIconAction(javax.faces.event.ActionEvent event) 
    {
        // here the popup is used by using a default popup...
        YESNOPopup ynp = YESNOPopup.createInstance("No maximizing!","There is no maximize icon inside the title bar of this popup.",new YESNOPopup.IYesNoListener()
        {
            @Override
            public void reactOnYes() { }
            @Override
            public void reactOnNo() { }
        });
        ynp.getModalPopup().hideMaximizeIcon();
    }

    public void onPopupWithoutIconAction(javax.faces.event.ActionEvent event) 
    {
        // here the popup is used by using a default popup...
        YESNOPopup ynp = YESNOPopup.createInstance("No icon at all!","There is no icon inside the title bar of this popup.",new YESNOPopup.IYesNoListener()
        {
            @Override
            public void reactOnYes() { }
            @Override
            public void reactOnNo() { }
        });
        ynp.getModalPopup().hideCloseIcon();
        ynp.getModalPopup().hideMaximizeIcon();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
