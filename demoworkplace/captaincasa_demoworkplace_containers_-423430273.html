The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Adaptive Containers - Controls
  • Adaptive side area with button

The page bean components CCAdaptiveSideContentArea and CCAdaptiveSideContentAreaButton are two content areas:

(.) The left side content area is either shown as own area or is shown when pressing a button, if the screen space is not sufficient
(.) The right main content area is always shown

This is the XML layout definition:

<t:rowheader id="g_3" coldistance="5" >

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCAdaptiveSideContentArea;
import org.eclnt.ccaddons.pbc.CCAdaptiveSideContentAreaButton;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.workplace.IWorkpageDispatcher;

import appstock.ArticleSearchUI;
import appstock.ArticleStockUI;
import appstock.EmptyContentUI;
import appstock.logic.Article;

@CCGenClass (expressionBase="#{d.DemoPBCCCAdaptiveSide}")

public class DemoPBCCCAdaptiveSide
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCAdaptiveSideContentAreaButton m_button;
    CCAdaptiveSideContentArea m_contentArea;
    
    ArticleSearchUI m_articleSearch;
    EmptyContentUI m_emptyContent;
    ArticleStockUI m_articleStock;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCAdaptiveSide(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        init();
    }

    public String getPageName() { return "/workplace/demopbcccadaptiveside.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCAdaptiveSide}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCAdaptiveSideContentAreaButton getButton() { return m_button; }
    public CCAdaptiveSideContentArea getContentArea() { return m_contentArea; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void init()
    {
        // whole area and button
        m_contentArea = new CCAdaptiveSideContentArea();
        m_contentArea.setSideBeanBorder("right:1;color:#00000010");
        m_button = m_contentArea.createOpenerButton();
        // content taken from appstock demo...
        m_articleSearch = new ArticleSearchUI();
        m_articleSearch.prepare(new ArticleSearchUI.IListener()
        {
            @Override
            public void reactOnArticleSelected(Article a)
            {
                m_articleStock = new ArticleStockUI();
                m_articleStock.prepare(a,null);
                showStockInContentArea(a);
                m_button.closeSideBeanPopup();
            }
            @Override
            public void reactOnArticleDetailsShown(Article a, ModelessPopup p)
            {
            }
        });
        m_articleSearch.setWithArticlePopup(false);
        m_articleStock = null;
        m_emptyContent = new EmptyContentUI();
        m_contentArea.prepare(m_articleSearch,m_emptyContent,"300",800,true,new CCAdaptiveSideContentArea.IListener()
        {
            @Override
            public void reactOnSideContentShown()
            {
            }
            @Override
            public void reactOnSideContentHidden()
            {
            }
        });
    }

    private void showStockInContentArea(Article a)
    {
        if (m_articleStock == null)
        {
            m_articleStock = new ArticleStockUI();
        }
        m_articleStock.prepare(a,null);
        m_contentArea.setContentBean(m_articleStock);
        m_contentArea.setShowSideBeanAtSide(false);
    }
    
}
