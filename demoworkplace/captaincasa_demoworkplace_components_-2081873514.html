The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Performance considerations
  • Avoiding large 1:n relations

In this EXAMPLE an explicit own PAINTAREA is arranged around 100 fields.

This means that the number of children in parent-child relationships both on server and on client side is not reaching critical levels.

For better visibility each PAINTAREA that is internally generated is colored with some background color.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPaintAreaManyFields2}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPaintAreaManyFields2}" rowdistance="#{ccstylevalue[&#39;@defaultrowdistance@&#39;]}" >
<t:row id="g_7" coldistance="#{ccstylevalue[&#39;@defaultcoldistance@&#39;]}" >
<t:formattedfield id="g_8" format="int" labeltext="Number of fields" value="#{d.DemoPaintAreaManyFields2.numberOfFields}" width="200" />
<t:button id="g_9" actionListener="#{d.DemoPaintAreaManyFields2.onRenderAction}" text="Render" width="100+" />
</t:row>
<t:rowdynamiccontent id="g_6" contentbinding="#{d.DemoPaintAreaManyFields2.dynContent}" />
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREANode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.componentnodes.SCROLLPANENode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPaintAreaManyFields2}")

public class DemoPaintAreaManyFields2
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private Map<String,Object> m_data = new HashMap<String,Object>();
    int m_numberOfFields = 2000;
    ROWDYNAMICCONTENTBinding m_dynContent = new ROWDYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPaintAreaManyFields2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demopaintareamanyfields2.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPaintAreaManyFields2}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public Map<String, Object> getData() { return m_data; } 

    public void onRenderAction(javax.faces.event.ActionEvent event) { render(); }

    public int getNumberOfFields() { return m_numberOfFields; }
    public void setNumberOfFields(int value) { this.m_numberOfFields = value; }

    public ROWDYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void render()
    {
        int x = 0;
        int y = -30;
        SCROLLPANENode sp = new SCROLLPANENode()
            .setHeight("100%")
            .setWidth("100%");
        PAINTAREANode pa = null;
        List<ComponentNode> nodes = new ArrayList<ComponentNode>();
        boolean paintAreabackgroundToggle = false;
        for (int i=0; i<m_numberOfFields; i++)
        {
            x += 100;
            if (i%100 == 0)
            {
                ROWNode r = new ROWNode();
                sp.addSubNode(r);
                pa = new PAINTAREANode()
                    .setWidth("100%")
                    .setHeight(300);
                if (paintAreabackgroundToggle == true)
                    pa.setBackground("#00000010");
                r.addSubNode(pa);
                y = -30;
                paintAreabackgroundToggle = !paintAreabackgroundToggle;
            }
            if (i%10 == 0)
            {
                x = 0;
                y += 30;
            }
            PAINTAREAITEMNode pi = new PAINTAREAITEMNode()
                .setBounds(x+";"+y+";90;25");
            pa.addSubNode(pi);
            FIELDNode f = new FIELDNode()
                .setText(pbx("data['D"+i+"']"));
            m_data.put("D"+i,"Text "+i);
            pi.addSubNode(f);
        }
        m_dynContent.setContentNode(sp);
    }

}
