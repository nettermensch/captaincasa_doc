The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Auto Completion
  • Double Line Items

This demo shows how to provide double line information within the item popup, presenting the auto-complete proposals to the user.

You need to...
(.) include the comment information on server side and
(.) set AUTOCOMPLETE-WITHVALUECOMMENT to "true"

In this example the AUTOCOMPLETE component is used by passing id-information also. You can also use the double line information without passing explicit ids, too.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.util.AutoCompleteMgr;
import org.eclnt.jsfserver.util.DefaultAutoComplete2Provider;
import org.eclnt.jsfserver.util.IAutoCompleteProvider2.AutoCompleteItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoAutoCompleteComment}")

public class DemoAutoCompleteComment
    extends PageBean
    implements Serializable
{
      // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    class MyAutoCompleteProvider extends DefaultAutoComplete2Provider
    {
        public List<AutoCompleteItem> getProposals(String value)
        {
            List<AutoCompleteItem> result = new ArrayList<AutoCompleteItem>();
            if ("ABC".equals(value))
            {
                result.add(new AutoCompleteItem("ABC Text 1","1","This is comment 1"));
                result.add(new AutoCompleteItem("ABC Text 2","2","This is comment 2"));
                result.add(new AutoCompleteItem("ABC Text 3","3","This is comment 3"));
                result.add(new AutoCompleteItem("ABC Text 4","4","This is comment 4"));
                result.add(new AutoCompleteItem("ABC Text 5","5","This is comment 5"));
                result.add(new AutoCompleteItem("ABC Text 6","6","This is comment 6"));
                result.add(new AutoCompleteItem("ABC Text 7","7","This is comment 7"));
                result.add(new AutoCompleteItem("ABC Text 8","8","This is comment 8"));
                result.add(new AutoCompleteItem("ABC Text 9","9","This is comment 9"));
            }
            else
            {
                result.add(new AutoCompleteItem("No information found","",""));
            }
            return result;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyAutoCompleteProvider m_acp;
    protected String m_value;
    protected String m_valueId;
    
  // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoAutoCompleteComment(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("autocomplete"); 
        m_acp = new MyAutoCompleteProvider();
        AutoCompleteMgr.add(m_acp);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener() 
        {
            public void reactOnDestroyed()
            {
                super.reactOnDestroyed();
                AutoCompleteMgr.remove(m_acp);
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getValueURL() { return m_acp.getURL(); }

    public String getValue() { return m_value; }
    public void setValue(String value) { m_value = value; }

    public String getValueId() { return m_valueId; }
    public void setValueId(String value) { m_valueId = value; }

    public void onValueFlush(ActionEvent event) 
    {
        Statusbar.outputMessage("Passed value: " + m_value + " / Id: " + m_valueId);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
