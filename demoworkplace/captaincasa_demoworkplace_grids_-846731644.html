The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Flexible row heights
  • Avoid row stretching

By default the grid (FIXGRID, ARRAYGRID, ...) processing always streteches its contained rows so that no row is cut within the grid.

This strategy is fine for "normal" grids - and ensures that the grid always is "properly filled" with rows. - For grids with dynamic row height sizing the strategy is not optimal, especially if rows might occupy some significant height.

By setting the attribute AVOIDROWSTRETCHING the grid rows are output with the height they require - the bottom content row which is shown in a grid may as consequence only be partially visible.

The grid management ensures that you always can scroll to the last row so that it becomes fully visible.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoGridNoRowStretching}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoGridNoRowStretching}" >
<t:row id="g_7" >
<t:fixgrid id="g_8" avoidrowstretching="true" dynamicheightsizing="true" height="100%" objectbinding="#{d.DemoGridNoRowStretching.grid}" sbvisibleamount="15" selectorcolumn="1" width="100%" >
<t:gridfooter id="g_12" >
<t:label id="g_3" stylevariant="ccemphasized" text="SOME FOOTER" />
</t:gridfooter>
<t:gridcol id="g_9" text="Content" width="100%;100" >
<t:pane id="g_10" >
<t:rowdynamiccontent id="g_6" contentbinding=".{rowContent}" />
</t:pane>
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.componentnodes.SCROLLPANENode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoGridNoRowStretching}")

public class DemoGridNoRowStretching
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        ROWDYNAMICCONTENTBinding i_rowContent = new ROWDYNAMICCONTENTBinding();
        public GridItem()
        {
            render();
        }
        /**
         * Creates some random number of rows inside in order to create
         * content with different row height per row.
         */
        private void render()
        {
            int maxi = new Random().nextInt(10) + 1;
            SCROLLPANENode p = new SCROLLPANENode()
                .setPadding("10")
                .setWidth("100%")
                .setHeight("100%;;300")
                .setRowdistance(7);
            {
                ROWNode r = new ROWNode()
                    .setColdistance(7);
                p.addSubNode(r);
                LABELNode l = new LABELNode()
                    .setText(""+maxi + " persons:")
                    .setFont("weight:bold");
                r.addSubNode(l);
            }
            for (int i=0; i<maxi; i++)
            {
                ROWNode r = new ROWNode()
                    .setColdistance(7);
                p.addSubNode(r);
                {
                    LABELNode l = new LABELNode()
                        .setText(""+(i+1))
                        .setAlign("left")
                        .setWidth("30");
                    r.addSubNode(l);
                }
                {
                    LABELNode l = new LABELNode()
                        .setText(RandomLogic.getLastName())
                        .setWidth("100%;150");
                    r.addSubNode(l);
                }
                {
                    LABELNode l = new LABELNode()
                        .setText(RandomLogic.getFirstName())
                        .setWidth("100%;150");
                    r.addSubNode(l);
                }
            }
            i_rowContent.setContentNode(p);
        }
        public ROWDYNAMICCONTENTBinding getRowContent() { return i_rowContent; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridNoRowStretching(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);   
        fillGrid();
    }

    public String getPageName() { return "/workplace/demogridnorowstretching.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridNoRowStretching}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        for (int i=0; i<100; i++)
            m_grid.getItems().add(new GridItem());
    }

}
