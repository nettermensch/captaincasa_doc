The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Server Roundtrips
  • By area

The logon button is only visible if all three fields are defined.

There is an attribute FLUSHAREA available on all container components. This triggers a server round trip in case of value changes inside the area.

The ENABLED status of the button is controlled by the server side - and only is enabled if all properties behind the fields are specified.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFlushArea}")

public class DemoFlushArea
    extends PageBean
    implements Serializable
{
    public DemoFlushArea(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }
    
    public boolean getLogonEnabled() 
    { 
        if (m_systemId == null || m_systemId.length() == 0) return false;
        if (m_password == null || m_password.length() == 0) return false;
        if (m_userName == null || m_userName.length() == 0) return false;
        return true;
    }

    String m_systemId;
    public String getSystemId() { return m_systemId; }
    public void setSystemId(String value) { this.m_systemId = value; }

    String m_password;
    public String getPassword() { return m_password; }
    public void setPassword(String value) { this.m_password = value; }

    String m_userName;
    public String getUserName() { return m_userName; }
    public void setUserName(String value) { this.m_userName = value; }

}
