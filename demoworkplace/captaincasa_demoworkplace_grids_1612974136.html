The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Fine granular Focus Management

This demo shows how to position the cursor in a cell directly.

At runtime a grid binding (FIXGRIDListBinding instance) is connected to its corresponding component representation - and can ask the component for its column/ cell configuration.
In the demo each cell component is holding corresponding reference information within the REFERENCE attribute - which is the base for positioning the cursor at the right place.

Consequence: in the example the column structure is completely managed within the layout (.jsp page) - the logical processing (.java) is not affected, if e.g. the column sequence is updated within the layout.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseComponent;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.ColumnInfo;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.session.RequestFocusManager;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoGridSelectAndEdit}")

public class DemoGridSelectAndEdit
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        Map<String,Long> i_focusValues = new HashMap<String,Long>();
        public Map<String,Long> getFocusValues() { return i_focusValues; }

        protected String m_street;
        public String getStreet() { return m_street; }
        public void setStreet(String value) { m_street = value; }

        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
    }
    
    public class Grid extends FIXGRIDListBinding<GridItem>
    {
        @Override
        public void initialize()
        {
            super.initialize();
            // Set the initial selection and focus to "1,1".
            // This must not be done within the constructor of the grid - because then
            // the grid is not bound to its server side component yet
            m_columnIndex = 1;
            m_rowIndex = 1;
            selectAndFocus();
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    Grid m_grid = new Grid();
    int m_columnIndex = 0;
    int m_rowIndex = 0;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGridSelectAndEdit(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        for (int i=0; i<5; i++)
        {
            GridItem gi = new GridItem();
            m_grid.getItems().add(gi);
        }
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public Grid getGrid() { return m_grid; }

    public int getColumnIndex() { return m_columnIndex; }
    public void setColumnIndex(int columnIndex) { m_columnIndex = columnIndex; }

    public int getRowIndex() { return m_rowIndex; }
    public void setRowIndex(int rowIndex) { m_rowIndex = rowIndex; }
    
    public void onSelectAndFocus(ActionEvent event)
    {
        selectAndFocus();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void selectAndFocus()
    {
        try
        {
            // the "reference" attribute of the cell component contains the information
            // about the contained property
            ColumnInfo ci = m_grid.getColumnInfos().get(m_columnIndex);
            BaseComponent bc = (BaseComponent)ci.getGridcontrolComponent();
            String reference = bc.getAttributeValueAsString("reference");
            m_grid.deselectCurrentSelection();
            GridItem gi = m_grid.getItems().get(m_rowIndex);
            m_grid.selectItem(gi);
            gi.getFocusValues().put(reference,RequestFocusManager.getNewRequestFocusCounter());
        }
        catch (Throwable t)
        {
            Statusbar.outputError(t.toString());
        }
    }
    
}
