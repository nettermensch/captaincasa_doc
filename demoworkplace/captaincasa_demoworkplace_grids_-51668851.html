The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Flexible row heights
  • For content rows - by server definition

This example shows a grid containing items with individual row heights. The height is defeind by the server side program by explicitly setting the height per item.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridWithIndividualRowHeight}")

public class DemoGridWithIndividualRowHeight
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_comment;
        String i_firstName;
        String i_name;
        public String getComment() { return i_comment; }
        public String getFirstName() { return i_firstName; }
        public String getName() { return i_name; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
            Statusbar.outputSuccess("Execute on: " + i_name);
        }
        @Override
        public int getRowHeight()
        {
            if (i_comment != null)
                return 60;
            else
                return super.getRowHeight();
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridWithIndividualRowHeight(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            m_grid.getItems().add(gi);
            gi.i_name = "Last " + i;
            gi.i_firstName = "First " + i;
            if (i%5 == 2)
            {
                gi.i_comment = "This is a comment which may occcupy some more pixels to show up.";
            }
        }
    }

    public String getPageName() { return "/workplace/demogridwithindividualrowheight.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridWithIndividualRowHeight}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
