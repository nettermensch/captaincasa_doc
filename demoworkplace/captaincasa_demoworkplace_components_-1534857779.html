The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Reporting, PDF, SVG
  • RGraph - Dynamic creation

The RGRAPH component includes the charting library "rchart". The library provides a rich set of charts. Please check for details via http://www.rgraph.net.

The configuration is done by passing a server side JSON string into the client side chart configuration. For further detail on the exact configuration object format, take a look into the documentation at http://www.rgraph.net.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventFinishedRendering;
import org.eclnt.jsfserver.elements.events.BaseActionEventRGraphPointSelection;
import org.eclnt.workplace.IWorkpageDispatcher;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;

@CCGenClass (expressionBase="#{d.DemoRGraph}")

/**
 * Please check the documentation of http://www.rgraph.net <br>
 * In the server side processing you create the chat configuration by creating
 * a corresponding JSON-String.<br>
 * The only extra attribute that you have to set (compared to the JavaScript
 * interface) is to in addition pass the info, what type of chart you want to
 * create (rgraphType).
 */
public class DemoRGraph
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    static class JO extends HashMap<String,Object> {}
    static class JOs extends ArrayList<JO> {}
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_config;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRGraph(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        createGaugeChart();
    }

    public String getPageName() { return "/workplace/demorgraph.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRGraph}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public void onCreateChartGauge(ActionEvent event) { createGaugeChart(); }
    public void onCreateChartFuel(ActionEvent event) { createChartFuel(); }
    public void onCreateChartMeter(ActionEvent event) { createChartMeter(); }
    public void onCreateChartOdometer(ActionEvent event) { createChartOdometer(); }
    public void onCreateChartBar(ActionEvent event) { createChartBar(); }
    public void onCreateChartBars(ActionEvent event) { createChartBars(); }
    public void onCreateChartDonut(ActionEvent event) { createChartDonut(); }
    
    public String getConfig() { return m_config; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    public void onRGraphAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventRGraphPointSelection)
        {
            BaseActionEventRGraphPointSelection e = (BaseActionEventRGraphPointSelection)event;
            Statusbar.outputSuccess("Selection in RGraph, Dataset: " + e.getDatasetIndex() + ", Point: " + e.getPointIndex());
        }
    }

    private void createChartDonut()
    {
        JO chart = new JO();
        chart.put("rgraphType","Pie");
        List<BigDecimal> data = new ArrayList<BigDecimal>();
        chart.put("data",data);
        data.add(new BigDecimal(41));
        data.add(new BigDecimal(37));
        data.add(new BigDecimal(16));
        data.add(new BigDecimal(3));
        data.add(new BigDecimal(3));
        JO options = new JO();
        chart.put("options",options);
        options.put("variant","donut");
        options.put("linewidth",5);
        options.put("exploded",5);
        options.put("strokestyle","rgba(0,0,0,0)");
        options.put("title","Browser share");
        List<String> tooltips = new ArrayList<String>();
        options.put("tooltips",tooltips);
        tooltips.add("MSIE 7");
        tooltips.add("MSIE 6");
        tooltips.add("Firefox");
        tooltips.add("Safari");
        tooltips.add("Other");
        List<String> key = new ArrayList<String>();
        options.put("key",key);
        key.add("MSIE 7");
        key.add("MSIE 6");
        key.add("Firefox");
        key.add("Safari");
        key.add("Other");
        options.put("keyShadow",true);
        options.put("keyShadowOffsetx",1);
        options.put("keyShadowOffsety",1);
        options.put("keyShadowBlur",3);
        options.put("keyShadowColor","#ccc");
        options.put("keyAlign","right");
        options.put("align","left");
        options.put("shadowOffsetx",2);
        options.put("shadowOffsety",2);
        options.put("shadowBlur",3);
        // options.put("centerx",185);
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        m_config = gson.toJson(chart);
    }

    private void createChartBars()
    {
        Random r = new Random();
        JO chart = new JO();
        chart.put("rgraphType","Bar");
        List<List<BigDecimal>> data = new ArrayList<List<BigDecimal>>();
        chart.put("data",data);
        for (int i=0; i<4; i++)
        {
            List<BigDecimal> col = new ArrayList<BigDecimal>();
            data.add(col);
            for (int j=0; j<4; j++)
            {
                col.add(new BigDecimal(r.nextInt(1000)));
            }
        }
        JO options = new JO();
        chart.put("options",options);
        List<String> labels = new ArrayList<String>();
        options.put("labels",labels);
        labels.add("2008");
        labels.add("2009");
        labels.add("2010");
        labels.add("2011");
        List<String> colors = new ArrayList<String>();
        options.put("colors",colors);
        colors.add("#C7CFC7");
        colors.add("#B2C8B2");
        colors.add("#D9E0DE");
        colors.add("#CDDED1");
        options.put("hmargin",15);
        options.put("hmarginGrouped",3);
        options.put("gutterLeft",45);
        options.put("gutterBottom",45);
        options.put("backgroundGridVlines",false);
        options.put("backgroundGridBorder",false);
        options.put("textSize",12);
        options.put("strokestyle","rgba(0,0,0,0)");
        options.put("shadow",false);
        options.put("textAccessible",true);
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        m_config = gson.toJson(chart);
    }
