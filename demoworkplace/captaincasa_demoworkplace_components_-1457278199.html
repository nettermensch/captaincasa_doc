The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Machine Overview

The status of machines is visualized by color and text. The boxes are arranged in a matrix.

By clicking onto a box, detail information is shown.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoMachineOverview}")

public class DemoMachineOverview
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Item
    {
        Date i_date = new Date();
        String i_image;
        String i_text2;
        String i_text3;
        String i_background;
                
        public Item()
        {
            i_image = "/industry/images/machine" + m_random.nextInt(4) + ".png";
            update();
        }
        public void update()
        {
            i_text2 = STATUSTEXTS[m_random.nextInt(STATUSTEXTS.length)];
            i_text3 = MENGETEXTS[m_random.nextInt(MENGETEXTS.length)];
            i_background = COLORS[m_random.nextInt(COLORS.length)];
        }
        public String getBackground()
        {
            return i_background;
        }
        public String getText1()
        {
            return "4.4";
        }
        public String getText2()
        {
            return i_text2;
        }
        public String getText3()
        {
            return i_text3;
        }
        public Date getDate()
        {
            return i_date;
        }
        public String getImage()
        {
            return i_image;
        }
        public void onAction(ActionEvent event)
        {
            final DemoMachineDetail dui = new DemoMachineDetail();
            dui.prepare(null,this);
            ModelessPopup p = openModelessPopup(dui,"",0,0,new ModelessPopup.IModelessPopupListener()
            {
                public void reactOnPopupClosedByUser()
                {
                    closePopup(dui);
                }
            });
            p.setUndecorated(true);
            p.setCloseonclickoutside(true);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static String[] POSITIONS = new String[]
    {
        "170;300",
        "290;390",
        "465;300",
        "460;390",
        "50;100",
        "280;100",
        "535;150",
        "120;425",
        "210;165",
        
    };
    DYNAMICCONTENTBinding m_content = new DYNAMICCONTENTBinding();
    List<Item> m_items = new ArrayList<Item>();
    Random m_random = new Random();
    
    static String[] COLORS = new String[] 
    {
        "#FFFFA0",
        "#FFA0FF",
        "#A0FFFF",
    };
    
    static String[] STATUSTEXTS = new String[]
    {
        "Maschine arbeitet",
        "Maschine nicht belegt",
        "St�rung",
        "In Wartung",
    };
    
    static String[] MENGETEXTS = new String[]
    {
        "Menge:",
        "Menge: 108 St",
        "Menge: 224 St",
        "Menge: 2826 St",
        "Menge: 18 St",
        "Menge: 34 St"
    };
    
    int m_sliderScale = 100;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMachineOverview(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        render();
    }

    public String getPageName() { return "/workplace/demomachineoverview.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMachineOverview}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getContent() { return m_content; }
    public List<Item> getItems() { return m_items; }
    
    public void onRefreshAction(ActionEvent event) 
    {
        for (Item item: m_items)
            item.update();
    }

    public double getPaintAreaScale() { return (float)m_sliderScale/100f; }

    public int getSliderScale() { return m_sliderScale; }
    public void setSliderScale(int value) { this.m_sliderScale = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void render()
    {
        List<ComponentNode> nodes = new ArrayList<ComponentNode>();
        for (int i=0; i<POSITIONS.length; i++)
        {
            nodes.add(renderMachine(i));
        }
        m_content.setContentNodes(nodes);
    }
    
    private ComponentNode renderMachine(int index)
    {
        Item item = new Item();
        m_items.add(item);
        String baseExpression = "items["+index+"].";
        PAINTAREAITEMNode result = new PAINTAREAITEMNode()
            .setBackground(pbx(baseExpression+"background"))
            .setBorder("#808080")
            .setBounds(POSITIONS[index]+";130;70;2")
            .setInvokeevent("click")
            .setActionListener(pbx(baseExpression+"onAction"))
            .setMovetofrontonmouseover(false);
        PANENode pane = new PANENode()
            .setRowdistance(0);
        result.addSubNode(pane);
        {
            ROWNode r = new ROWNode();
            pane.addSubNode(r);
            LABELNode l = new LABELNode()
                .setStyleseq("risclabel_nopadding")
                .setText(pbx(baseExpression+"text1"))
                .setForeground("#000000")
                .setFont("weight:bold");
            r.addSubNode(l);
        }
        {
            ROWNode r = new ROWNode();
            pane.addSubNode(r);
            LABELNode l = new LABELNode()
                .setStyleseq("risclabel_nopadding")
                .setForeground("#000000")
                .setText(pbx(baseExpression+"text2"))
                .setFont("size:10");
            r.addSubNode(l);
        }
        {
            ROWDISTANCENode r = new ROWDISTANCENode()
                .setHeight("100%");
            pane.addSubNode(r);
        }
        {
            ROWNode r = new ROWNode();
            pane.addSubNode(r);
            LABELNode l = new LABELNode()
                .setStyleseq("risclabel_nopadding")
                .setForeground("#000000")
                .setText(pbx(baseExpression+"text3"));
            r.addSubNode(l);
        }
        {
            ROWNode r = new ROWNode();
            pane.addSubNode(r);
            LABELNode l = new LABELNode()
                .setStyleseq("risclabel_nopadding")
                .setForeground("#000000")
                .setText(pbx(baseExpression+"date"))
                .setFormat("datetime");
            r.addSubNode(l);
        }
        return result;
    }
}
