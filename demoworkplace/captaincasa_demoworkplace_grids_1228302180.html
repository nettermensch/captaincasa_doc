The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Tree
  • Select vs. Toggle

In the example the selected object does not change when the user is just toggling a certain node. It only changes if there is some real selection, that is not associated with a node toggling, is behind.

When the user clicks into a tree node then - as always within the grid processing - a selection is triggered.

Sometimes you want to differentiate on server side, if the click is associated with a toggling or if it is a "real" click. In the example "toggle clicks" do not automatically select the corresponding line - only "non toggle clicks" really cause a selection.

For this purpose you can access the events that are processed within a roundtrip - and can check if there is a toggle event part of the request processing.

When separating the line selection in the way shown, then you can easily show the user the selected node by adding some bgpaint command.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import javax.faces.event.FacesEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.ThreadData;
import org.eclnt.jsfserver.elements.events.BaseActionEventToggle;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoTreeSingleClickExecute}")

public class DemoTreeSelectVersusToggle
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        public TreeNode(FIXGRIDTreeItem parentNode, String text, int status)
        {
            super(parentNode);
            setText(text);
            setStatus(status);
        }
        @Override
        public void onRowSelect()
        {
            // find out if a toggle comes as event, too
            List<FacesEvent> requestEvents = ThreadData.getInstance().getRequestEvents();
            boolean foundToggle = false;
            for (FacesEvent requestEvent: requestEvents)
            {
                if (requestEvent instanceof BaseActionEventToggle) 
                {
                    foundToggle = true;
                    break;
                }
            }
            if (foundToggle == false)
            {
                m_treeNodeObserved = this;
                Statusbar.outputSuccess("This was a real selection...");
            }
            else
            {
                Statusbar.outputMessage("This was just a toggle...");
            }
        }
        public String getTreeNodeBgpaint()
        {
            if (this == m_treeNodeObserved)
                return "rectangle(0,0,100%,100%,#00FF0050,#00FF0020,vertical)";
            else
                return null;
        }
    }

    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();
    TreeNode m_treeNodeObserved = null;
    
    public DemoTreeSelectVersusToggle(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        {
            TreeNode top = new TreeNode(m_tree.getRootNode(),"Germany",TreeNode.STATUS_OPENED);
            new TreeNode(top,"Bammental",TreeNode.STATUS_ENDNODE);
            new TreeNode(top,"Heidelberg",TreeNode.STATUS_ENDNODE);
            new TreeNode(top,"Unterhaching",TreeNode.STATUS_ENDNODE);
        }
        {
            TreeNode top = new TreeNode(m_tree.getRootNode(),"Austria",TreeNode.STATUS_OPENED);
            new TreeNode(top,"Kufstein",TreeNode.STATUS_ENDNODE);
            new TreeNode(top,"Salzburg",TreeNode.STATUS_ENDNODE);
        }
    }
    
    public FIXGRIDTreeBinding<TreeNode> getTree() { return m_tree; }
    
    public String getObservedTreeNodeText()
    {
        if (m_treeNodeObserved == null)
            return null;
        else
            return m_treeNodeObserved.getText();
    }
}
