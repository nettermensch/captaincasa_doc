The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Flexible row heights
  • Row heights by content

The height of the lines in this grid are automatically driven by the content of their cell components.

The parameters to set are:

1. FIXGRID-DYNAMICHEIGHTSIZING to be set to "true"
2. FIXGRID-ROWHEIGHT is the minimum row height that is used of any row

When using: please check performance! In a "normal" gird the height of its cells is not calculated but is driven by the FIXGRID-ROWHEIGHT definition. Now the grid sizing needs to analyze the dimens of its cells with every activity in the grid (e.g. scrolling).

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import workplace.DemoGridDynamicHeightSizing.ListItem;

@CCGenClass (expressionBase="#{d.DemoGridDynamicHeightSizing}")

public class DemoGridDynamicHeightSizing
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class ListItem extends Object implements java.io.Serializable
    {    
        GridItem i_gridItem;
        public ListItem(GridItem gridItem)
        {
            i_gridItem = gridItem;
        }
        public void onRemoveAction(javax.faces.event.ActionEvent event) 
        {   
            i_gridItem.i_list.remove(this);
        }
        String i_text;
        public String getText() { return i_text; } 
        public void setText(String text) { i_text = text; } 
    }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_text;
        List<ListItem> i_list = new ArrayList<ListItem>();
        public String getText() { return i_text; }
        public void setText(String value) { this.i_text = value; }
        public List<ListItem> getList() { return i_list; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        public void onAddItemAction(javax.faces.event.ActionEvent event) 
        {
            ListItem li = new ListItem(this);
            li.setText("New item");
            i_list.add(li);
        }
        public boolean getAvailableAddItem() 
        { 
            return i_list.size() < 5; 
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridDynamicHeightSizing(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            if (i%5 == 0)
                gi.setText("A\nB\nC");
            else if (i%6 == 3)
                gi.setText("This is some longer text that may be broken into several lines. This is some longer text that may be broken into several lines. This is some longer text that may be broken into several lines. This is some longer text that may be broken into several lines. ");
            else
                gi.setText("A B C");
            int maxj = 1;
            if (i%7 == 2)
                maxj = 3;
            for (int j=0; j<maxj; j++)
            {
                ListItem li = new ListItem(gi);
                li.setText("List item " + j);
                gi.getList().add(li);
            }
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogriddynamicheightsizing.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridDynamicHeightSizing}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
