The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • One column - two controls

Typically column components within a grid are defined by column. But: you can also dynamically define the component of a column per row.

In the example the RENDERED attribute is used to switch between the components that are placed inside one column cell.

Note that the RENDERED attribute is interpreted on server side. This means that even when placing certain components into the layout for one grid column, then only this one component (per row) is sent to the client, for which the RENDERED binding tells "true".

This is the XML layout definition:

<t:rowdemobodypane id="g_1" objectbinding="#{d.demoGridflexcols}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#D0D0D0" borderheight="1" borderwidth="1" objectbinding="#{d.demoGridflexcols.rows}" rowbgpaint="background(#FFFFFF)" sbvisibleamount="10" width="100%" >

</t:gridcol>

<t:pane id="g_7" bgpaint="null!" >

<t:field id="g_9" background="#FFFFFF00" bgpaint="null!" border="top:0;left:0;bottom:0;right:0" height="100%" rendered=".{fieldRendered}" text=".{firstName}" width="100%" />

This is the Java code:

package workplace;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.FIXGRIDPivotListBinding;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoGridflexcols extends WorkpageDispatchedPageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class MyRow extends FIXGRIDItem
    {
        String m_firstName;
        int m_controlType = 0; // 0 => FIELD, 1 ==> COMBOFIELD
        public int getControlType() { return m_controlType; }
        public void setControlType(int controlType) { m_controlType = controlType; }
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public boolean getFieldRendered()
        {
            if (m_controlType == 0)
                return true;
            else
                return false;
        }
        public boolean getComboFieldRendered()
        {
            if (m_controlType == 1)
                return true;
            else
                return false;
        }
        public ValidValuesBinding getRowControlTypes()
        {
            return getControlTypes();
        }
        public void onComboFieldAction(ActionEvent ae)
        {
            IdTextSelection idts = IdTextSelection.createInstance();
            idts.addLine("Bj�rn","Bj�rn - This is a nice name!");
            idts.addLine("Tom","Tom - OK, as well!");
            idts.setCallBack(new ISetId()
            {
                public void setId(String id)
                {
                    m_firstName = id;
                }
            });
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoGridflexcols(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<7; i++)
        {
            MyRow row = new MyRow();
            row.setFirstName("First Name " + i);
            m_rows.getItems().add(row);
            if (i%2 == 0)
                row.setControlType(1);
        }
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    
    public ValidValuesBinding getControlTypes()
    {
        ValidValuesBinding result = new ValidValuesBinding();
        result.addValidValue("0","Field (FIELD)");
        result.addValidValue("1","Combo Field (COMBOFIELD)");
        return result;
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
