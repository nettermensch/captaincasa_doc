The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Single / multiple files

By using the attribute MULTISELECT you can decide if the user can upload one or multiple files. The client file names and the client file data are passed at server side via corresponding methods of BaseActionEventUpload, in which you pass the number of the file that is uploaded as parameter.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventUpload;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFileMultiSingle}")

public class DemoFileSingleMulti
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    String m_protocol;
    public void setProtocol(String value) { m_protocol = value; }
    public String getProtocol() { return m_protocol; }
    
    public DemoFileSingleMulti(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }
    
    public void onUpload(ActionEvent event)
    {
        if (event instanceof BaseActionEventUpload)
        {
            BaseActionEventUpload baeu = (BaseActionEventUpload)event;
            StringBuffer sb = new StringBuffer();
            sb.append("Number of uploaded files: " + baeu.getNumberOfUploadedFiles() + "\n\n");
            for (int i=0; i<baeu.getNumberOfUploadedFiles(); i++)
            {
                sb.append(i + ".) " + baeu.getClientFileName(i) + "\n");
            }
            m_protocol = sb.toString();
        }
    }
}
