The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Download multiple files

This demo shows how 5 dynamically created files are downloaded from the server side to the client side.

The browser only allows to download one file. As consequence you need to chain the downloads, so that the actual download of one file directly triggers the download of the next file.

The FILEDOWNLOAD component provides an attribute WITHCALLBACK which is set to "true". As consequence the client notifies the server when a download of content happens.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.tempfile.TempFileManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFileDownloadMultiple}")

public class DemoFileDownloadMultiple
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_downloadCounter;
    String m_downloadUrl;
    Trigger m_downloadTrigger = new Trigger();
    String m_protocol = "";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFileDownloadMultiple(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demofiledownloadmultiple.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFileDownloadMultiple}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    public void onDownloadAction(javax.faces.event.ActionEvent event) 
    {
        m_protocol = "Download started.";
        m_downloadCounter = 5;
        onNextDownload(null);
    }

    public void onNextDownload(javax.faces.event.ActionEvent event) 
    {
        try
        {
            m_downloadCounter--;
            if (m_downloadCounter >= 0)
            {
                String fileContent = "This is the new file number " + m_downloadCounter;
                // content-type "application/octet" ==> browser always downloads; otherwise "txt" are shown in own browser window
                m_downloadUrl = TempFileManager.saveTempFile("file"+m_downloadCounter+".txt","application/octet",fileContent.getBytes("UTF-8"));
                m_protocol += "\nDownload URL: " + m_downloadUrl;
                m_downloadTrigger.trigger();
            }
            else
            {
                m_protocol += "\nDownload ended.";
            }
        }
        catch (Throwable t)
        {
            m_protocol += "\n" + t;
        }
    }

    public String getDownloadUrl() { return m_downloadUrl; }
    public void setDownloadUrl(String value) { this.m_downloadUrl = value; }

    public Trigger getDownloadTrigger() { return m_downloadTrigger; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
