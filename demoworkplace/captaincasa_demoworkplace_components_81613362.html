The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Lines moving with shapes

This demo produces a number of PAINTAREAITEM instances - with connecting PAINTAREALINEITEMs in between.

When moving a PAINTAREAITEM instance then the lines connected with the instance are following automatically. This is done by assigning a LINEID to each PAINTAREALINEITEM instance and by referencing this instance by PAINTAREAITEM-LINEIDSSTARTING and LINEIDSENDING.

Client side redrawing of lines is done using the Canvas component. The performance of the Canvas implementation is quite different between different browsers. So please use this demo to check the performance for your scenarios.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREALINEITEMNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoManyShapes}")

public class DemoManyShapes
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class ShapeInfo
    {
        String i_bounds;
        int i_x;
        int i_y;
        public String getBounds() { return i_bounds; }
        public void setBounds(String value) 
        { 
            i_bounds = value;
            int[] vs = ValueManager.decodeStraightIntCSV(i_bounds);
            i_x = vs[0];
            i_y = vs[1];
        }
        public void onShapeAction(ActionEvent event)
        {
            // called when shape was moved (flush set to true!)
            // no action required here: all coordinates are calculated within
            // the getter-implementations...
        }
    }
    
    public class LineInfo
    {
        ShapeInfo i_fromShape;
        ShapeInfo i_toShape;
        public String getBounds()
        {
            int x = i_fromShape.i_x + 30;
            int y = i_fromShape.i_y + 10;
            int w = i_toShape.i_x - i_fromShape.i_x;
            int h = i_toShape.i_y - i_fromShape.i_y;
            return x+";"+y+";"+w+";"+h;
        }
        public void setBounds(String value)
        {
            // no need for implementation - the lines are always
            // correctly calulated in the getter
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Random R = new Random();
    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();
    int m_numberOfShapes = 50;
    List<ShapeInfo> m_sis = new ArrayList<ShapeInfo>();
    List<LineInfo> m_lis = new ArrayList<LineInfo>();
    boolean m_automLineMoving = true;
    int m_rangeY = 2000;
    int m_rangeX = 2000;
    boolean m_withDrawingOfLines = true;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoManyShapes(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);      
        renderContent();
    }

    public String getPageName() { return "/workplace/demomanyshapes.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoManyShapes}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }
    public List<ShapeInfo> getSis() { return m_sis; }
    public List<LineInfo> getLis() { return m_lis; }

    public int getNumberOfShapes() { return m_numberOfShapes; }
    public void setNumberOfShapes(int value) { this.m_numberOfShapes = value; }

    public void onCreateShapesAction(javax.faces.event.ActionEvent event) 
    {
        renderContent();
    }

    public boolean getAutomLineMoving() { return m_automLineMoving; }
    public void setAutomLineMoving(boolean value) { this.m_automLineMoving = value; }

    public int getRangeY() { return m_rangeY; }
    public void setRangeY(int value) { this.m_rangeY = value; }

    public int getRangeX() { return m_rangeX; }
    public void setRangeX(int value) { this.m_rangeX = value; }

    public boolean getWithDrawingOfLines() { return m_withDrawingOfLines; }
    public void setWithDrawingOfLines(boolean value) { this.m_withDrawingOfLines = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderContent()
    {
        m_sis.clear();
        m_lis.clear();
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (int i=0; i<m_numberOfShapes; i++)
        {
            ShapeInfo si = new ShapeInfo();
            si.setBounds(R.nextInt(m_rangeX) + ";" + R.nextInt(m_rangeY) + ";" + "60;20");
            m_sis.add(si);
            PAINTAREAITEMNode pin = new PAINTAREAITEMNode()
                    .setBounds(pbx("sis["+i+"].bounds"))
                    .setBackground("#FFFFFF")
                    .setBorder("#c0c0c0")
                    .setMovingenabled(true)
                    .setActionListener(pbx("sis["+i+"].onShapeAction"))
                    .setFlush(true);
            // define the lines ending and starting with this item
            if (m_withDrawingOfLines == true && m_automLineMoving == true)
            {
                if (i != m_numberOfShapes-1) pin.setLineidsstarting("L_"+i);
                if (i != 0) pin.setLineidsending("L_"+(i-1));
            }
            l.add(pin);
        }
        if (m_withDrawingOfLines)
        {
            for (int i=0; i<m_numberOfShapes-1; i++)
            {
                LineInfo li = new LineInfo();
                li.i_fromShape = m_sis.get(i);
                li.i_toShape = m_sis.get(i+1);
                m_lis.add(li);
                PAINTAREALINEITEMNode lin = new PAINTAREALINEITEMNode()
                        .setBounds(pbx("lis["+i+"].bounds"))
                        .setLineid("L_"+i);
                l.add(lin);
            }
        }
        m_dynContent.setContentNodes(l);
    }
}
