The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Drag & Drop
  • Multi Selection Grid

This example shows how to drag & drop multiple items from the grid into the text area below.

The drag-send information is not statically defined with a grid row, but is updated with every selection. The drag info is a string starting with "indices:" and then following a semicolon separated list of values. The drag-send-string itself is built in the way "type:value;type:value" - this is the reason why the semicolon separated must be properly built using the functions of the ValueManager helper class.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDragMulti}")

public class DemoDragMulti
    extends PageBean
    implements Serializable
{
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_name;
        public String getName() { return m_name; }
        public void setName(String value) { m_name = value; }
    }

    protected String m_selectedNames;
    public String getSelectedNames() { return m_selectedNames; }
    public void setSelectedNames(String value) { m_selectedNames = value; }

    protected String m_droppedValue;
    public String getDroppedValue() { return m_droppedValue; }
    public void setDroppedValue(String value) { m_droppedValue = value; }

    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    public void setRows(FIXGRIDListBinding<MyRow> value) { m_rows = value; }
    
    public DemoDragMulti(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<20; i++)
        {
            MyRow r = new MyRow();
            r.setName("NAME"+i);
            m_rows.getItems().add(r);
        }
    }
    
    public void onDropAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventDrop)
        {
            m_droppedValue = ((BaseActionEventDrop)event).getDragInfo();
            String indicesCSV = m_droppedValue.substring("indices:".length());
            String[] indices = ValueManager.decodeCSV(indicesCSV);
            StringBuffer sb = new StringBuffer("Selected Names:\n");
            for (int i=0; i<indices.length; i++)
            {
                int index = (new Integer(indices[i])).intValue();
                sb.append(m_rows.getItems().get(index).getName());
                sb.append("\n");
            }
            m_selectedNames = sb.toString();
        }
    }

    /**
     * Returns a comma separated list of all selected lines.
     */
    public String getDragSend()
    {
        List<String> indices = new ArrayList<String>();
        for (int i=0; i<m_rows.getItems().size(); i++)
        {
            MyRow r = m_rows.getItems().get(i);
            if (r.getSelected() == true)
                indices.add(""+i);
        }
        String indicesString = ValueManager.encodeCSV(indices); // forms 1;2;3
        String encodedString = ValueManager.encodeCSV(new String[] {indicesString});
        String result = "indices:"+encodedString;
        return result;
    }

}
