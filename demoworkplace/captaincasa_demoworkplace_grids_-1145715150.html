The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Bean data grid (PBC)
  • With own controls
  • Own columns - by layout definition

Within the data grid component you can define individual controls by coplumn - if you want to override the default control that is selected by the grid processing.

Take a look into columns "Name", "Married" and into the last column: here explicit controls are assigned.

When defining explicit controls, then these controls also have to match corresponding data and methods within the server side processing. Take a look into the code in order to see how to add new properties and functions to the data grid processing.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoPBCCCDatagridview2WithControls1}")

public class DemoPBCCCDatagridview2WithControls1
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public enum ENUMGender
    {
        MALE,
        FEMALE,
        DIVERSE
    }
    
    /** 
     * The bean... This normally is an own class. In the demo it
     * is an inner class so that all code is kept within the
     * demo class.
     */
    public static class TestPerson
    {
        String i_firstName;
        String i_lastName;
        int i_height;
        LocalDate i_birthDate;
        boolean i_married;
        ENUMGender i_gender;
        public String getFirstName() { return i_firstName; }
        public String getLastName() { return i_lastName; }
        public int getHeight() { return i_height; }
        public LocalDate getBirthDate() { return i_birthDate; }
        public boolean isMarried() { return i_married; }
        public void setFirstName(String firstName) { i_firstName = firstName; }
        public void setLastName(String lastName) { i_lastName = lastName; }
        public void setHeight(int height) { i_height = height; }
        public void setBirthDate(LocalDate birthDate) { i_birthDate = birthDate; }
        public void setMarried(boolean married) { i_married = married; }
        public ENUMGender getGender() { return i_gender; }
        public void setGender(ENUMGender gender) { i_gender = gender; } 
    }

    /**
     * Listener to the data grid view's listener so that in order to 
     * react on events of the grid processing.
     */
    public class MyDGVListener extends CCDataGridView2.DefaultListener<TestPerson>
    {
        @Override
        public void reactOnGridSelect(TestPerson object) { reactOnItemSelection(object); }
    }

    /**
     * Overridden CCDataGridView2 with new properties/methods on item level.
     */
    public class MyDataGridView extends CCDataGridView2<TestPerson>
    {
        /**
         * Overridden inner class MyGridItem with new properties/methods.
         */
        public class MyGridItem extends GridItem
        {
            public MyGridItem(TestPerson object)
            {
                super(object);
            }
            public String getMarriedText()
            {
                if (getObject().isMarried())
                    return "Married";
                else
                    return "Not married";
            }
            public void onRemoveAction(javax.faces.event.ActionEvent event) 
            {
                m_dataView.removeItem(getObject());
            }
            public String getFullName() 
            { 
                return getObject().getLastName() + ", " + getObject().getFirstName(); 
            }
        }
        public MyDataGridView()
        {
            super(TestPerson.class);
        }
        @Override
        /**
         * Important: this is the "factory method" for grid items, so you
         * have to override and create the own instance.
         */
        protected GridItem createGridItem(TestPerson object)
        {
            return new MyGridItem(object);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyDataGridView m_dataView = new MyDataGridView();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDatagridview2WithControls1(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillData();
        initialSort();
    }

    public String getPageName() { return "/workplace/demopbcccdatagridview2simple.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCDataGridView2Simple}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2<TestPerson> getDataView() { return m_dataView; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void reactOnItemSelection(TestPerson object)
    {
        Statusbar.outputSuccess("Selection of item: " + object.getLastName() + ", " + object.getFirstName());
    }

    private void fillData()
    {
        List<TestPerson> ps = new ArrayList<TestPerson>();
        for (int i=0; i<100; i++)
        {
            TestPerson p = new TestPerson();
            p.setFirstName(RandomLogic.getFirstName());
            p.setLastName(RandomLogic.getLastName());
            p.setHeight(RandomLogic.getHeight());
            p.setMarried(i%2 == 0);
            p.setBirthDate(RandomLogic.getBirthDate());
            p.setGender(ENUMGender.FEMALE);
            if (i%2 == 0) p.setGender(ENUMGender.MALE);
            if (i%9 == 4) p.setGender(ENUMGender.DIVERSE);
            ps.add(p);
        }
        m_dataView.prepare(ps,new MyDGVListener());
    }

    private void initialSort()
    {
        m_dataView.getGrid().sort("object.firstName",true);
    }

    
}
