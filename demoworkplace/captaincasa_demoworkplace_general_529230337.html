The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Opening of value help dialogs by code

When selecting a value in the popup for department thent directly the value popup for the employee is opened.

By default a dialog is opened at the point of last user activity - so in this case you want to explicitly position the employee popup to be located at the employee field - though not being opened from there. That's the reason why explicit component ids are used an referenced when opening the employee popup.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoComboFieldPopupSequence}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >
<t:row id="g_2" >
<t:textpane id="g_3" text="When selecting the department then the dialog for the employee will be opened automatically." width="100%" />
</t:row>
<t:rowdistance id="g_4" height="20" />
<t:row id="g_5" >
<t:label id="g_6" text="Department" width="100" />
<t:combofield id="CFDEPARTMENT" actionListener="#{d.DemoComboFieldPopupSequence.onDepartmentAction}" text="#{d.DemoComboFieldPopupSequence.department}" width="150" />
</t:row>
<t:row id="g_7" >
<t:label id="g_8" text="Employee" width="100" />
<t:combofield id="CFEMPLOYEE" actionListener="#{d.DemoComboFieldPopupSequence.onEmployeeAction}" text="#{d.DemoComboFieldPopupSequence.employee}" width="150" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoComboFieldPopupSequence}")

public class DemoComboFieldPopupSequence
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    String m_employee;
    String m_department;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------


    public DemoComboFieldPopupSequence(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }
    
    public String getPageName() { return "/workplace/democombofieldpopupsequence.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoComboFieldPopupSequence}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getEmployee() { return m_employee; }
    public void setEmployee(String value) { this.m_employee = value; }

    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    public void onEmployeeAction(javax.faces.event.ActionEvent event) 
    {
        IdTextSelection idts = IdTextSelection.createInstance();
        for (int i=0; i<100; i++)
            idts.addLine("EMPL_00"+i,"Employee 00"+i);
        idts.setCallBack(new ISetId()
        {
            public void setId(String id)
            {
                m_employee = id;
            }
        });
        idts.getPopup().setLefTopReferenceComponentIdBottom("CFEMPLOYEE");
    }

    public void onDepartmentAction(javax.faces.event.ActionEvent event) 
    {
        IdTextSelection idts = IdTextSelection.createInstance();
        for (int i=0; i<10; i++)
            idts.addLine("DEP_"+i,"Department "+i);
        idts.setCallBack(new ISetId()
        {
            public void setId(String id)
            {
                m_department = id;
                onEmployeeAction(null); // maybe not the nicest way to call...
            }
        });
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
