The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Scale Pane
  • Scale explicitly set

The SCALEPANE allows to scale its content.

You may position any container into a SCALEPANE instance.
Please pay attention: the SCALEPANE must only hold exactly one container (which itself can contain any content of course!).

By setting WITHUSERSCALING to "true" you can allow to the scale the pane on his/her own (ctrl-mouse-wheel, ctrl-+/-/0 shortcuts).

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoScalePane}")

public class DemoScalePane
    extends PageBean 
    implements Serializable
{
    public void onSliderScaleAction(ActionEvent event) 
    {
        m_scale = new BigDecimal(m_sliderScale).divide(new BigDecimal(100));
    }

    int m_sliderScale = 100;
    public int getSliderScale() { return m_sliderScale; }
    public void setSliderScale(int value) { this.m_sliderScale = value; }

    BigDecimal m_scale = new BigDecimal(1);
    public BigDecimal getScale() { return m_scale; }
    public void setScale(BigDecimal value) 
    { 
        this.m_scale = value;
        this.m_sliderScale = Math.round(this.m_scale.floatValue() * 100f);
    }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoScalePane(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demoscalepane.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoScalePane}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
