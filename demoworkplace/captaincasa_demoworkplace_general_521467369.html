The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Server Side Value Binding
  • Property Change Notification - Version 2

Change some data and press "OK". In the protocol area you see, what changes are registered on server side by implementing interface IPropertyResolverAware2.

The interface IPropertyResolverAware2 may be implemented on any managed bean level. Via the interface you get notified whenever values are changed inside your bean.

There are two levels:
(.) you get notified about data changes that directly affect properties of your current object
(.) you get notified about data changed that happen within sub objects of your current object

Please pay attention: the notification is executed within the data-transfer phase ("set phase"). You should only do "minimal actions" within this phase and push all processing into the invoke phase. Use class PhaseManager in order to shift processing into the invoke phase, if explicitly required.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.util.IPropertyResolverAware2;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoPropertyResolverAware2}")

public class DemoPropertyResolverAware2
    extends WorkpageDispatchedPageBean 
    implements Serializable, IPropertyResolverAware2
{
    public class Address
    {
        public Address(String street, String zipCode, String town)
        {
            i_street = street;
            i_zipCode = zipCode;
            i_town = town;
        }
        String i_street;
        String i_zipCode;
        String i_town;
        public void setStreet(String value) { i_street = value; }
        public String getStreet() { return i_street; }
        public void setZipCode(String value) { i_zipCode = value; }
        public String getZipCode() { return i_zipCode; }
        public void setTown(String value) { i_town = value; }
        public String getTown() { return i_town; }
    }
    
    public class Car
    {
        String i_id;
        String i_type;
        public Car(String id, String type)
        {
            i_id = id;
            i_type = type;
        }
        public void setId(String value) { i_id = value; }
        public String getId() { return i_id; }
        public void setType(String value) { i_type = value; }
        public String getType() { return i_type; }
    }
    
    public DemoPropertyResolverAware2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);       
        m_firstName = "Captain";
        m_lastName = "Casa";
        m_data.put("title","Company");
    }

    // mapped data
    Map<String,String> m_data = new HashMap<String, String>();
    // directly available data
    String m_firstName;
    String m_lastName;
    // linked data
    Address m_address = new Address("Main Street","347623","Nice Town");
    protected String m_protocol = "";
    // array of data
    Car[] m_cars = new Car[] {new Car("K-AA 827","Audi A6"), new Car("M-HH","BMW 5")};
    // array of simple data
    String[] m_comments = new String[] {"Comment 1","Comment 2"};
    
    public void setFirstName(String value) { m_firstName = value; }
    public String getFirstName() { return m_firstName; }
    
    public void setLastName(String value) { m_lastName = value; }
    public String getLastName() { return m_lastName; }

    public Map<String,String> getData() { return m_data; }
    public Address getAddress() { return m_address; }
    public Car[] getCars() { return m_cars; }
    public String[] getComments() { return m_comments; }

    public String getProtocol() { return m_protocol; }


    public void reactOnSetValue(String completeExpression, String propertyName, Object value)
    {
        m_protocol += "\nreactOnSetValue: " + completeExpression + " / " + propertyName + " / " + value;
    }

    public void reactOnSetValue(String completeExpression, int index, Object value)
    {
        m_protocol += "\nreactOnSetValue: " + completeExpression + " / " + index + " / " + value;
    }

    public void reactOnSetValueInSubObject(String completeExpression, Object value)
    {
        m_protocol += "\nreactOnSetValueInSubObject: " + completeExpression + " / " + value;
    }

    public void onClearProtocol(ActionEvent event) 
    {
        m_protocol = "";
    }

    public void onOK(ActionEvent event) 
    {
    }   

}
