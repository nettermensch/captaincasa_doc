The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Field Value Help - Details
  • Changing the Icon

Use the attributes IMAGE, IMAGEROLLOVER, IMAGEPRESSED in order to update the image "on the right" of the combo field.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoComboFieldImage}")

public class DemoComboFieldImage
    extends PageBean
    implements Serializable
{
    protected String m_costCenter;

    public DemoComboFieldImage(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("combofield"); 
    }
    
    public String getCostCenter() { return m_costCenter; }
    public void setCostCenter(String value) { m_costCenter = value; }

    public void onCostCenterAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            IdTextSelection idts = IdTextSelection.createInstance();
            idts.addLine("BOARD","Executive Board");
            idts.addLine("DEV","Development");
            idts.addLine("PROD","Production");
            idts.addLine("SALES","Sales");
            idts.setCallBack(new ISetId() 
            {
                public void setId(String id)
                {
                    m_costCenter = id;
                }
            });
        }
    }

}
