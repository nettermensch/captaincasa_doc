The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Issues
  • Fix Columns

By arranging two grids one to another you can achieve the effect of "fix columns".

You need to pay attention to that the left grid is always drawn with the same sizing as the right grid:
(.) same number of rows (SBVISIBLEAMOUNT)
(.) same horizontal scrollbar (HORIZONTALSCROLLMODE)
(.) same row height (ROWHEIGHT)

Typically the left grid is defined without explicit scrollbar on its own. Use attribute VERTICALSCROLLMODE to define so.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Collection;
import java.util.Set;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.util.ArrayListWithChangeIndex;
import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.jsfserver.util.IClientLogAware;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridFixColumns}")

public class DemoGridFixColumns
    extends PageBean
    implements Serializable, IClientLogAware
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_id;
        String m_lastName;
        String m_firstName;
        String m_street;
        String m_zipCode;
        String m_town;
        public void setId(String value) { m_id = value; }
        public String getId() { return m_id; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public void setStreet(String value) { m_street = value; }
        public String getStreet() { return m_street; }
        public void setZipCode(String value) { m_zipCode = value; }
        public String getZipCode() { return m_zipCode; }
        public void setTown(String value) { m_town = value; }
        public String getTown() { return m_town; }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            if ("CMD_REMOVE".equals(event.getCommand()))
            {
                removeSelectedItems();
            }
        }
        @Override
        public void onRowSelect()
        {
            Statusbar.outputSuccess("Select: " + m_gridleft.isCurrentlyProcessingGridAction() + "/" + m_grid.isCurrentlyProcessingGridAction());
            // example: an item does not accept selection
            if ("23".equals(m_id))
            {
                m_grid.deselectItem(this);
            }
        }
        @Override
        public void onRowExecute()
        {
            Statusbar.outputSuccess("Execute: " + m_gridleft.isCurrentlyProcessingGridAction() + "/" + m_grid.isCurrentlyProcessingGridAction());
        }
        
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    ArrayListWithChangeIndex<GridItem> m_items = new ArrayListWithChangeIndex<GridItem>();
    
    protected FIXGRIDListBinding<GridItem> m_gridleft = new FIXGRIDListBinding<GridItem>();
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructor
    // ------------------------------------------------------------------------

    public DemoGridFixColumns(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // there are two grid instances - both sharing the same list
        // of items
        m_gridleft.setItems(m_items);
        m_grid.setItems(m_items);
        // both grids are connected in means of scrolling and selection
        m_gridleft.connectWithBinding(m_grid);
        fillGrid();
    }

    // ------------------------------------------------------------------------
    // methods
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGridleft() { return m_gridleft; }
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public int getNofSelectedItemsLeft() { return m_gridleft.getSelectedItems().size(); }
    public int getNofSelectedItems() { return m_grid.getSelectedItems().size(); }
    
    public void onClearSelection(ActionEvent event)
    {
        m_gridleft.deselectCurrentSelection();
    }
    
    public void onClearAndRefill(ActionEvent event)
    {
        m_items.clear();
        fillGrid();
    }
    
    public void onExplicitlySort(ActionEvent event)
    {
        m_grid.sort("firstName",true);
    }
    
    public boolean onCreateClientTestLogMessage()
    {
        StringBuffer sb = new StringBuffer();
        sb.append(m_grid.getSbvalue()+"//");
        sb.append(m_gridleft.getSbvalue()+"//");
        Collection<GridItem> gis = m_grid.getSelectedItemsAsSequence();
        for (GridItem gi: gis)
            sb.append(gi.getId()+"/");
        sb.append("////");
        gis = m_gridleft.getSelectedItemsAsSequence();
        for (GridItem gi: gis)
            sb.append(gi.getId()+"/");
        ClientTestLog.addClientTestLogMessage(sb.toString());
        return false;
    }
    
    private void removeSelectedItems()
    {
        Set<GridItem> gis = m_grid.getSelectedItems();
        for (GridItem gi: gis)
        {
            m_items.remove(gi);
        }
    }
    
    private void fillGrid()
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setId(""+i);
            gi.setFirstName("First " + i);
            gi.setLastName("Last " + i);
            gi.setStreet("Street " + i);
            gi.setZipCode("Zipcode " + i);
            gi.setTown("Town " + i);
            m_items.add(gi);
        }
    }
}
