This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Set;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection.IdTextLine;
import org.eclnt.jsfserver.elements.adapter.IComponentAdapterBinding;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.session.RequestFocusManager;
import org.eclnt.util.log.CLog;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFormAdapterBinding}")

public class DemoFormAdapterBinding
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    static Set<String> FIELDFIXATTRIBUTES = new HashSet<String>();
    
    static Set<String> FIELDDYNATTRIBUTES = new HashSet<String>();
    static
    {
        FIELDDYNATTRIBUTES.add("text");
        FIELDDYNATTRIBUTES.add("bgpaint");
        FIELDDYNATTRIBUTES.add("enabled");
        FIELDDYNATTRIBUTES.add("requestfocus");
    }
    
    /**
     * Adapter binding that is behind one form field.
     */
    public class FIELDAdapterBinding implements IComponentAdapterBinding
    {
        String m_text;
        String m_propertyName;
        boolean m_enabled = true;
        long m_requestfocus = -1;
        boolean m_hasError = false;
        boolean m_isMandatory = false;
        public FIELDAdapterBinding(boolean isMandatory, String propertyName)
        {
            m_isMandatory = isMandatory;
            m_propertyName = propertyName;
        }
        public Class getAttibuteType(String attributeName) { return String.class; }
        public Object getAttributeValue(String attributeName)
        {
            if ("text".equals(attributeName)) return m_text;
            if ("bgpaint".equals(attributeName)) return getBgpaint();
            if ("enabled".equals(attributeName)) return m_enabled;
            if ("requestfocus".equals(attributeName)) return m_requestfocus;
            throw new Error("Should never happen!");
        }
        public Set<String> getDynamicAttributeNames()
        {
            return FIELDDYNATTRIBUTES;
        }
        public Set<String> getFixAttributeNames()
        {
            return FIELDFIXATTRIBUTES;
        }
        public void onAction(ActionEvent event)
        {
            if (event instanceof BaseActionEventValueHelp)
            {
                openValidValueHelp(this);
            }
        }
        public void setAttributeValue(String attributeName, Object value)
        {
            if ("text".equals(attributeName)) m_text = (String)value;
        }
        private String getBgpaint()
        {
            if (m_hasError == true) return "error()";
            if (m_isMandatory == true) return "mandatory()";
            return null;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Map<String,FIELDAdapterBinding> m_fields = new HashMap<String,FIELDAdapterBinding>();

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoFormAdapterBinding(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        registerField("firstName",true);
        registerField("lastName",true);
        registerField("street",false);
        registerField("city",false);
        registerField("department",true);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public Map<String,FIELDAdapterBinding> getFields() { return m_fields; }
    
    public void onCheck(ActionEvent ae)
    {
        boolean foundError = false;
        // reset error information
        for (FIELDAdapterBinding field: m_fields.values())
            field.m_hasError = false;
        // check for mandatory
        for (FIELDAdapterBinding field: m_fields.values())
        {
            if (field.m_isMandatory == true && (field.m_text == null || field.m_text.length() == 0))
            {
                field.m_hasError = true;
                Statusbar.outputError("Mandatory input is missing!");
            }
        }
        // special rules
        FIELDAdapterBinding cityField = m_fields.get("city");
        if ("Paul".equals(cityField.m_text))
        {
            cityField.m_hasError = true;
            cityField.m_requestfocus = RequestFocusManager.getNewRequestFocusCounter();
            Statusbar.outputError("Paul ist heute nicht in der City.");
        }
    }
    
    public void onFocusStreet(ActionEvent ae)
    {
        m_fields.get("street").m_requestfocus = RequestFocusManager.getNewRequestFocusCounter();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void registerField(String propertyName, boolean isMandatory)
    {
        m_fields.put(propertyName,new FIELDAdapterBinding(isMandatory,propertyName));
    }
    
    private void openValidValueHelp(final FIELDAdapterBinding field)
    {
        CLog.L.log(CLog.LL_INF,"Opening valid value helpt for property " + field.m_propertyName);
        IdTextSelection idts = IdTextSelection.createInstance();
        for (int i=0; i<100; i++)
            idts.addLine(field.m_propertyName+"_" + i,field.m_propertyName+"_" + i);
        idts.setCallBack(new ISetId()
        {
            public void setId(String id)
            {
                field.m_text = id;
            }
        });
    }
    
}
