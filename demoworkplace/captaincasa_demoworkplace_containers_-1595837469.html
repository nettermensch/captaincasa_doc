The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Layering and absolute postioning
  • Fields on top of form

This demo shows absolutely positioned FIELDs on top of some real form.

The demo includes some editor: you can create new FIELD instances by using the right-mouse-button popup-menu and you can move and resize the FIELD instances.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.streamstore.StreamStore;
import org.eclnt.util.file.ClassloaderReader;
import org.eclnt.util.log.CLog;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFormInput}")
public class DemoFormInput
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class FieldsItem extends Object implements java.io.Serializable
    {
        String i_value;
        int i_height;
        int i_width;
        int i_y;
        int i_x;
        public String getValue() { return i_value; }
        public void setValue(String value) { this.i_value = value; }
        public int getHeight() { return i_height; }
        public int getWidth() { return i_width; }
        public int getY() { return i_y; }
        public int getX() { return i_x; }
        public String getBounds() { return i_x+";"+i_y+";"+i_width+";"+i_height+";2"; }
        public void setBounds(String value)
        {
            String[] values = ValueManager.decodeCSV(value);
            setBounds(values);
        }
        private void setBounds(String[] values)
        {
            i_x = ValueManager.decodeInt(values[0],0);
            i_y = ValueManager.decodeInt(values[1],0);
            i_width = ValueManager.decodeInt(values[2],100);
            i_height = ValueManager.decodeInt(values[3],20);
        }
        public void onItemAction(ActionEvent event)
        {
            if (event instanceof BaseActionEventPopupMenuItem)
            {
                BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
                if  ("REMOVEFIELD".equals(e.getCommand()))
                {
                    m_fields.remove(this);
                }
            }
        }
        private String toCSV()
        {
            return ValueManager.encodeCSV(new String[] {""+i_x,""+i_y,""+i_width,""+i_height,i_value});
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<FieldsItem> m_fields = new ArrayList<FieldsItem>();
    int m_scaleInt = 100;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFormInput(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        readFields();
    }

    public String getPageName() { return "/workplace/demoforminput.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFormInput}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<FieldsItem> getFields() { return m_fields; }
    public int getScaleInt() { return m_scaleInt; }
    public void setScaleInt(int value) { this.m_scaleInt = value; }
    public float getScale() { return ((float)m_scaleInt)/100f; }

    public void onCreateNewFieldAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            FieldsItem fi = new FieldsItem();
            fi.i_x = e.getPixelX();
            fi.i_y = e.getPixelY();
            fi.i_width = 100;
            fi.i_height = 25;
            m_fields.add(fi);
        }
    }

    public void onSaveAction(javax.faces.event.ActionEvent event) 
    {
        saveFields();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void saveFields()
    {
        StringBuffer all = new StringBuffer();
        for (FieldsItem fi: m_fields)
            all.append(fi.toCSV()+"\n");
        StreamStore.getInstance().writeUTF8("/ccdemos/DemoFormInput.csv",all.toString(),true);
    }
    
    private void readFields()
    {
        String txt = StreamStore.getInstance().readUTF8("/ccdemos/DemoFormInput.csv",false);
        if (txt == null)
        {
            txt = new ClassloaderReader(true).readUTF8File("/workplace/DemoFormInput.csv",true);
            txt = txt.replace("\r","");
        }
        String[] csvLines = ValueManager.tokenize(txt,"\n");
        for (String csv: csvLines)
        {
            try
            {
                FieldsItem fi = new FieldsItem();
                String[] values = ValueManager.decodeCSV(csv);
                fi.setBounds(values);
                fi.i_value = values[4];
                m_fields.add(fi);
            }
            catch (Throwable t)
            {
                CLog.L.log(CLog.LL_INF,"Could not parse csv: " + csv);
            }
        }
    }
}
