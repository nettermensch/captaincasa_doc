The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grouping column headers
  • Header colspan definition

In order to build trees with structured headers there are two features to pay attention to:

1. in the components within the GRIDHEADER (same with GRIDFOOTER) you may define a COLSPAN.

2. in the GRIDHEADER there is a new attribute BEFOREDEFAULTHEADER which arranges the corresponding header line correspondingly.

The COLSPAN management only correctly works if the grid column sequence is not modifiable by the user: set FIXGRID-COLUMNDRAGDROPENABLED to "false" as consequence.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridHeaderSpan}")

public class DemoGridHeaderSpan
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_comment;
        public String getComment() { return i_comment; }
        public void setComment(String value) { this.i_comment = value; }

        String i_kostenstelle;
        public String getKostenstelle() { return i_kostenstelle; }
        public void setKostenstelle(String value) { this.i_kostenstelle = value; }

        String i_persNummer;
        public String getPersNummer() { return i_persNummer; }
        public void setPersNummer(String value) { this.i_persNummer = value; }

        String i_town;
        public String getTown() { return i_town; }
        public void setTown(String value) { this.i_town = value; }

        String i_street;
        public String getStreet() { return i_street; }
        public void setStreet(String value) { this.i_street = value; }

        String i_zipCode;
        public String getZipCode() { return i_zipCode; }
        public void setZipCode(String value) { this.i_zipCode = value; }

        String i_title;
        public String getTitle() { return i_title; }
        public void setTitle(String value) { this.i_title = value; }

        String i_lastName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }

        String i_firstName;
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }

        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridHeaderSpan(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
          GridItem gi = new GridItem();
          if (i%2 == 0)
              gi.setTitle("Dr.");
          gi.setFirstName("First " + i);
          gi.setLastName("Last " + i);
          gi.setZipCode(""+(i+88888));
          gi.setStreet("Street " + i);
          gi.setTown("Town " + i);
          gi.setPersNummer("DE03"+i);
          gi.setKostenstelle("4711");
          gi.setComment("Comment " + i);
          m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridheaderspan.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridHeaderSpan}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
