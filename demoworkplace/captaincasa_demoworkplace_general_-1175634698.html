The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Style and Language Management
  • Session style & language

Change the style or the language by pressing the corresponding button. Then open another page and you will see according changes.

There are only very few pages translated at the moment. For checking the language open the page "Arranging Pages - Modal Dialogs" and open the default dialogs (e.g. Yes/No dialog).

Both style and language management are documented within the Developers' Guide. Please read details there.

This is the Java code:

package workplace;

import java.util.Locale;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.session.SessionInfo;
import org.eclnt.jsfserver.util.HttpSessionAccess;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoSwitchstyle extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    String m_country;
    String m_language;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoSwitchstyle(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getCountry() { return m_country; } 
    public void setCountry(String country) { m_country = country; }
    
    public String getLanguage() { return m_language; } 
    public void setLanguage(String language) { m_language = language; }

    public void onStyleAction(ActionEvent ae)
    {
        String styleName = ((BaseActionEvent)ae).getSourceReference();
        if (styleName == null) styleName = SessionInfo.getSessionDefaultStyleForRISCClient();
        HttpSessionAccess.setCurrentStyle(styleName);
    }
    
    public void onLanguageEnglish(ActionEvent ae)
    {
        HttpSessionAccess.setCurrentLocale(new Locale("en","US"));
        m_country = "US";
        m_language = "en";
    }
    
    public void onLanguageGerman(ActionEvent ae)
    {
        HttpSessionAccess.setCurrentLocale(new Locale("de","DE"));
        m_country = "DE";
        m_language = "de";
    }
    
    public void onLocaleChange(ActionEvent ae)
    {
        if (m_country == null || m_language == null)
        {
            Statusbar.outputAlert("Please define both country and language.");
            return;
        }
        HttpSessionAccess.setCurrentLocale(new Locale(m_language,m_country));
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------    

}
