The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Scroll Pane - Extended Functions
  • Dynamic loading of content

The SCROLLPANE fires a BaseActionEventScroll-Event when reaching the bottom scroll position - and when having set WITHSCROLLEVENTBOTTOM to true.

In the example the result of the server-side event processing is that additional 10 items are appended to the dynamic content within the SCROLLPANE content.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoScrollNotifier}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >
<t:row id="g_2" >
<t:label id="g_3" text="Scroll to the end of the following SCROLLPANE:" />

</t:row>

</t:rowdemobodypane>

</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.IMAGENode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventScroll;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoScrollNotifier}")

public class DemoScrollNotifier
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    PANENode m_paneNode = new PANENode();
    int m_renderIndex = 0;
    ROWDYNAMICCONTENTBinding m_content = new ROWDYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoScrollNotifier(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);       
        m_paneNode.addAttribute("id","PANE");
        m_paneNode.setRowdistance(5);
        renderAppend();
    }

    public String getPageName() { return "/workplace/demoscrollnotifier.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoScrollNotifier}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getContent() { return m_content; }

    public void onScrollAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventScroll)
        {
            BaseActionEventScroll e = (BaseActionEventScroll)event;
            if ("bottom".equals(e.getPosition()) && e.checkIfPositionWasReached() == true)
            {
                renderAppend();
            }
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderAppend()
    {
        // Pleae note:
        // Ids are explicitly assigned! Otherwise the client interprets all
        // controls as new controls, and as consequence re-renders the full dynamic
        // content.
        // with passing the ids (and keeping them constant!) the client
        // know that some content already was available before any only renders
        // the new content.
        for (int i=0; i<10; i++)
        {
            ROWNode rn = new ROWNode();
            m_paneNode.addSubNode(rn);
            rn.addAttribute("id","ROW"+m_renderIndex);
            rn.setColdistance(5); 
            IMAGENode in = new IMAGENode();
            rn.addSubNode(in);
            in.addAttribute("id","IM"+m_renderIndex);
            in.setImage("/images/pictures/photo"+i+".gif");
            in.setWidth(50);
            in.setHeight(50);
            in.setKeepratio(false);
            LABELNode ln = new LABELNode();
            rn.addSubNode(ln);
            ln.addAttribute("id","LA"+m_renderIndex);
            ln.setText("Text of item " + m_renderIndex);
            m_renderIndex++;
        }
        m_content.setContentNode(m_paneNode);
    }
}
