The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Tab Pane
  • Create new tab by dummy tab

This demo shows how to use an "end-tab" with some image as kind of button to manage tabs before.

The functions used are:

1. instead of using counted indices the TABBEDPANETAB instances use a defined SELECTIONID. These selection ids are stable - if not defined, you would have to work with an counted index and would have to recalculate the index of tab permanently.
2. the TABBEDPANE is transferring the selected value directly when the user selects a tab - the FLUSH is defined to "true".
3. On server side the last tab is used to open other tabs, and to immediately move the content-value to the new tab.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTabbedPaneWithPlus}")

public class DemoTabbedPaneWithPlus
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static final String TAB_2_SELID = "SECOND";
    static final String TAB_3_SELID = "THIRD";
    static final String TAB_4_SELID = "FOURTH";
    static final String TAB_PLUS_SELID = "PLUS";
    
    String m_tabValue;
    boolean m_rendered2 = false;
    boolean m_rendered3 = false;
    boolean m_rendered4 = false;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTabbedPaneWithPlus(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demotabbedpanewithplus.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTabbedPaneWithPlus}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getTabValue() { return m_tabValue; }
    public void setTabValue(String value) { this.m_tabValue = value; }

    public boolean getRendered2() { return m_rendered2; }
    public boolean getRendered3() { return m_rendered3; }
    public boolean getRendered4() { return m_rendered4; }
    public boolean getRenderedPlus() 
    { 
        return m_rendered2 == false || m_rendered3 == false || m_rendered4 == false; 
    }

    public void onTabClose2(javax.faces.event.ActionEvent event) { m_rendered2 = false; }
    public void onTabClose3(javax.faces.event.ActionEvent event) { m_rendered3 = false; }
    public void onTabClose4(javax.faces.event.ActionEvent event) { m_rendered4 = false; }

    public void onTabbedPaneAction(javax.faces.event.ActionEvent event) 
    {
        if (TAB_PLUS_SELID.equals(m_tabValue))
        {
            if (m_rendered2 == false) 
            {
                m_rendered2 = true;
                m_tabValue = TAB_2_SELID;
            }
            else if (m_rendered3 == false)
            {
                m_rendered3 = true;
                m_tabValue = TAB_3_SELID;
            }
            else if (m_rendered4 == false)
            {
                m_rendered4 = true;
                m_tabValue = TAB_4_SELID;
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
