The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Delayed execution

Use AUTOEXECUTE to define that the grid will send an execute event after not changing the selection for a certain duration.

This means: you can quickly navigate through the grid, without performing the execution of expansive activities item by item.

Instead you move the expansive activity into the execute-event-handler which is then called after certain duration of not changing the selection.

The duration can be defined by attribute AUTOEXECUTEDURATION. The default is 1000ms.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;

@CCGenClass (expressionBase="#{d.DemoGridAutoExecute}")

public class DemoGridAutoExecute
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_id;
        String i_firstName;
        String i_lastName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public String getId() { return i_id; }
        public void setId(String value) { this.i_id = value; }
        public String getBackground()
        {
            if (m_currentlyShownItem == this)
                return "#FFd080";
            else
                return null;
        }
        @Override
        public void onRowSelect()
        {
            // no activity here - otherwise the grid processing gets slower
            // and there is a lot of flickering on the screen
        }
        @Override
        public void onRowExecute()
        {
            // this method is called after explicit selection by the user
            // (hitting the return key or double clicking) ...and it
            // is called when not changing the selection for a while.
            createChart(this);
        }
    }

    static class JO extends HashMap<String,Object> {}
    static class JOs extends ArrayList<JO> {}

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_chartTitle;
    String m_chart;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    GridItem m_currentlyShownItem;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridAutoExecute(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setId(4711+i+"");
            gi.setLastName("Last " + gi.getId());
            gi.setFirstName("First " + gi.getId());
            m_grid.getItems().add(gi);
        }
        // select first person
        createChart(m_grid.getItems().get(0));
    }

    @Override
    public String getPageName() { return "/workplace/demogridautoexecute.jsp"; }
    @Override
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridAutoExecute}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getChartTitle() { return m_chartTitle; }

    public String getChart() { return m_chart; }
    public void setChart(String value) { this.m_chart = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    static final String[] REGIONS = new String[] {"North","West","South","East"};
    static final String[] COLORS = new String[] {"#FFC0C0","#C0FFC0","#C0C0FF","#C0C0C0"};

    static final String[] CATEGORIES = new String[] {"Hardware","Software","Other"};

    private void createChart(GridItem gi)
    {
        m_currentlyShownItem = gi;
        m_chartTitle = "Sales of " + gi.getLastName() + ", " + gi.getFirstName();
        Random r = new Random();
        JO chart = new JO();
        chart.put("type","bar");
        JO data = new JO();
        chart.put("data",data);
        {
            List<String> labels = new ArrayList<String>();
            for (String category: CATEGORIES) labels.add(category);
            data.put("labels",labels);
            JOs datasets = new JOs();
            data.put("datasets",datasets);
            for (int i=0; i<REGIONS.length; i++)
            {
                JO dataset = new JO();
                datasets.add(dataset);
                List<BigDecimal> datasetData = new ArrayList<BigDecimal>();

                for (int j=0; j<CATEGORIES.length; j++)
                    datasetData.add(new BigDecimal(r.nextInt(45000)));
                dataset.put("label",REGIONS[i]);
                dataset.put("data",datasetData);
                dataset.put("fill",true);
                dataset.put("backgroundColor",COLORS[i]);
            }
        }
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        m_chart = gson.toJson(chart);
    }


}
