The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Popup positioning

By default a popup is opened below the component that triggered the popup processing.

But you can fine tune by using one of the various setLeftTopReference*-methods.

You even can open a popup at a specific component (that is not the one triggering the popup processing) - and then again decide where to open this, relative to this component.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupLefttopreference}")

public class DemoPopupLefttopreference
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    public DemoPopupLefttopreference(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public void onOpenAtAAAAA(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLefTopReferenceComponentId("AAAAA");
    }
    
    public void onOpenAtAAAAABottom(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLefTopReferenceComponentIdBottom("AAAAA");
    }
    
    public void onOpenLeft(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceLeft();
    }
    
    public void onOpenRight(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceRight();
    }
    
    public void onOpenTop(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceTop();
    }
    
    public void onOpenRightShifted(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceRight();
        popup.setTop(-100);
    }
    
    public void onOpenAtAAAAARight(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceComponentIdRight("AAAAA");
    }
    
    public void onOpenAtAAAAARightShifted(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceComponentIdRight("AAAAA");
        popup.setTop(-100);
    }
    
    public void onOpenAtBBBBB(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLefTopReferenceComponentId("BBBBB");
    }

    public void onOpenCentered(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopReferenceCentered();
    }

    public void onOpenWithoutReference(ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
    }

    public void onOpenScreenBottom(javax.faces.event.ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopScreenBottom(true);
    }

    public void onOpenScreenTop(javax.faces.event.ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopScreenTop(true);
    }

    public void onOpenScreenLeft(javax.faces.event.ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopScreenLeft(true);
    }

    public void onOpenScreenRight(javax.faces.event.ActionEvent event) 
    {
        ModalPopup popup = openModalPopup();
        popup.setLeftTopScreenRight(true);
    }

    // ------------------------------------------------------------------------
    
    private ModalPopup openModalPopup()
    {
        final DemoHelloWorld bean = new DemoHelloWorld(getOwningDispatcher());
        ModalPopup result = openModalPopup(bean,"Hello world!",400,320,new ModalPopup.IModalPopupListener()
        {
            public void reactOnPopupClosedByUser()
            {
                closePopup(bean);
            }
        });
        return result;
    }

}
