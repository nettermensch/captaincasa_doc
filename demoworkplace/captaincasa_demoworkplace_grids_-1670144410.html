The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Selection en/disabling

The FIXGRID component provide an ENABLED attribute, too. If set to false, then the grid itself will not send any type of selection events (onRowSelect, onRowExecute) to the server side.

Please note, that the enabled-status of the grid has nothing to do with the enabled-status of its contained items!

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridWholeEnabled}")

public class DemoGridWholeEnabled extends PageBean
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }
        
        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
        
        @Override
        public void onRowSelect()
        {
            m_protocol += "Select processing on item: " + m_lastName + ", " + m_firstName + "\n";
        }
        @Override
        public void onRowExecute()
        {
            m_protocol += "Execute processing on item: " + m_lastName + ", " + m_firstName + "\n";
        }
    }
    
    protected boolean m_wholeGridEnabled = true;
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    protected String m_protocol = "";
    
    public DemoGridWholeEnabled(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<20; i++)
        {
            GridItem gi = new GridItem();
            gi.setLastName("LN " + i);
            gi.setFirstName("FN " + i);
            m_grid.getItems().add(gi);
        }
    }
    
    public boolean getWholeGridEnabled() { return m_wholeGridEnabled; }
    public void setWholeGridEnabled(boolean value) { m_wholeGridEnabled = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void setProtocol(String value) { m_protocol = value; }
    public String getProtocol() { return m_protocol; }
}
