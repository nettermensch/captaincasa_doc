This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.componentnodes.CHECKBOXNode;
import org.eclnt.jsfserver.elements.componentnodes.COMBOFIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.FORMATTEDFIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.componentnodes.SIMPLELABELNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.IDatabagProvider;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.jsfserver.util.IAcceleratedPropertyAccess;
import org.eclnt.util.valuemgmt.ValueManager;

@CCGenClass (expressionBase="#{d.DemoGridPerformance}")

public class DemoGridPerformance
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem 
        extends FIXGRIDItem 
        implements java.io.Serializable
    {
        List<String> i_data = new ArrayList<String>();
        public List<String> getData() { return i_data; }
        public void onAction(ActionEvent event)
        {
            if (event instanceof BaseActionEventValueHelp)
            {
                IdTextSelection idts = IdTextSelection.createInstance();
                idts.addLine("Harry","Harry");
                idts.addLine("Jo","Jo");
                idts.addLine("Karl","Karl");
            }
        }
    }

    public class GridItemWithDatabag 
        extends GridItem
        implements IDatabagProvider
    {
        String i_databag = null;
        public String getDatabag()
        {
            if (i_databag == null)
            {
                Map<String,String> m = new HashMap<String,String>();
                for (int i=0; i<i_data.size(); i++)
                {
                    m.put(""+i,i_data.get(i));
                }
                i_databag = ValueManager.encodeComplexValue(m);
            }
            return i_databag;
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    ROWDYNAMICCONTENTBinding m_content = new ROWDYNAMICCONTENTBinding();
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>(true);
    
    int m_controlType = 0;
    int m_rowCount = 10;
    int m_colCount = 20;
    int m_roundtripLatence = 0;
    String m_creationHint = "toggle";
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridPerformance()
    {
    }

    public String getPageName() { return "/workplace/demogridperformance.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridPerformance}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public int getControlType() { return m_controlType; }
    public void setControlType(int value) { this.m_controlType = value; }

    public int getRowCount() { return m_rowCount; }
    public void setRowCount(int value) { this.m_rowCount = value; }

    public int getColCount() { return m_colCount; }
    public void setColCount(int value) { this.m_colCount = value; }

    public ROWDYNAMICCONTENTBinding getContent() { return m_content; }
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public void onRender(ActionEvent event) 
    {
        FIXGRIDNode fg = new FIXGRIDNode();
        fg.setObjectbinding("#{d.DemoGridPerformance.grid}");
        fg.setSbvisibleamount(m_rowCount);
        fg.setWidth("100%");
        fg.setHeight("100%");
//        fg.setHeight("100%");
        fg.setBordercolor("#00000030");
        fg.setBorderheight(1);
        fg.setBorderwidth(1);
        for (int i=0; i<m_colCount; i++)
        {
            GRIDCOLNode gn = new GRIDCOLNode();
            fg.addSubNode(gn);
            gn.setText(""+i);
            gn.setWidth(60);
            gn.setSortreference(".{data["+i+"]}"); // explicit definition ownly required for "4"
            if (m_controlType == 0)
            {
                LABELNode ln = new LABELNode();
                gn.addSubNode(ln);
                ln.setFormat("bigdecimal");
                ln.setFormatmask("dec2");
                ln.setText(".{data["+i+"]}");
            }
            else if (m_controlType == 1)
            {
                FIELDNode fn = new FIELDNode();
                gn.addSubNode(fn);
                fn.setText(".{data["+i+"]}");
                fn.addAttribute("creationhint",m_creationHint);
            }
            else if (m_controlType == 2)
            {
                FORMATTEDFIELDNode fn = new FORMATTEDFIELDNode();
                gn.addSubNode(fn);
                fn.setValue(".{data["+i+"]}");
                fn.setFormat("float");
                fn.addAttribute("creationhint",m_creationHint);
            }
            else if (m_controlType == 3)
            {
                SIMPLELABELNode ln = new SIMPLELABELNode();
                gn.addSubNode(ln);
                ln.setFormat("bigdecimal");
                ln.setFormatmask("dec2");
                ln.setText(".{data["+i+"]}");
            }
            else if (m_controlType == 4)
            {
                SIMPLELABELNode ln = new SIMPLELABELNode();
                gn.addSubNode(ln);
                ln.setText("@cc_db:"+i+"@");
            }
            else if (m_controlType == 5)
            {
                COMBOFIELDNode ln = new COMBOFIELDNode();
                gn.addSubNode(ln);
                ln.setText(".{data["+i+"]}");
                ln.setActionListener(".{onAction}");
                ln.addAttribute("creationhint",m_creationHint);
            }
            else if (m_controlType == 6)
            {
                CHECKBOXNode cn = new CHECKBOXNode();
                cn.setSelected(".{data["+i+"]}");
                gn.addSubNode(cn);
            }
        }
        m_content.setContentNode(fg);
        m_grid.getItems().clear();
        for (int i=0; i<100; i++)
        {
            GridItem gi = null;
            switch (m_controlType) 
            {
                case 0:
                case 1:
                case 2:
                case 3:
                case 5:
                case 6:
                    gi = new GridItem();
                    break;
                case 4:
                    gi = new GridItemWithDatabag();
                    break;
            }
            for (int j=0; j<m_colCount; j++)
            {
                if (m_controlType != 6)
                {
                    gi.getData().add(i + "." + j);
                }
                else
                {
                    int r = new Random().nextInt(10);
                    if (r%2 == 0)
                        gi.getData().add("true");
                    else
                        gi.getData().add("false");
                }
            }
            m_grid.getItems().add(gi);
        }
    }

    public void onBeforeRenderAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            if (m_roundtripLatence > 0)
                Thread.currentThread().sleep(m_roundtripLatence);
        }
        catch (Throwable t)
        {
        }
    }

    public int getRoundtripLatence() { return m_roundtripLatence; }
    public void setRoundtripLatence(int value) { this.m_roundtripLatence = value; }

    public String getCreationHint() { return m_creationHint; }
    public void setCreationHint(String value) { this.m_creationHint = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
