The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Issues
  • Mouse wheel delta

By setting attribute FIXGRID-MOUSEWHEELDELTA you can define the number of rows that are scrolled per mouse wheel movement.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoGridMouseWheelDelta}" rowdistance="20" >
<t:row id="g_2" >
<t:label id="g_3" text="Mouse wheel delta" width="120" />
<t:combobox id="g_4" editable="false" flush="true" value="#{d.DemoGridMouseWheelDelta.mouseWheelDelta}" width="50" >
<t:comboboxitem id="g_5" text="1" value="1" />

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridMouseWheelDelta}")

public class DemoGridMouseWheelDelta
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }

        String i_firstName;
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }

    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_mouseWheelDelta = 1;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridMouseWheelDelta(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<1000; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First " + i);
            gi.setLastName("Last " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridmousewheeldelta.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridMouseWheelDelta}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public int getMouseWheelDelta() { return m_mouseWheelDelta; }
    public void setMouseWheelDelta(int value) { this.m_mouseWheelDelta = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
