The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Modify Function Tree at Runtime

This demo shows how to update the function tree of the workplace using the corresponding Java API.

The workplace receives an object (WorkPlaceFunctionTreeInfoNode hierarchy). The top hierarchy layer is the one to create the outtlookbar-items. The follow on layers are directly transferred into the corresponding tree.

The XML configuration of the workplace is a 1:1 mapping of the object passed via the API into XML using JAX-B serialization.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.ThreadData;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageContainer;
import org.eclnt.workplace.WorkpageStartInfo;
import org.eclnt.workplace.WorkplaceFunctionTreeInfoManager;
import org.eclnt.workplace.WorkplaceFunctionTreeInfoNode;

@CCGenClass (expressionBase="#{d.DemoWpSetFunctionTree}")

public class DemoWpSetFunctionTree
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // constructor
    // ------------------------------------------------------------------------

    public DemoWpSetFunctionTree(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /*
     * Update of the function tree with some dynamic structure.
     */
    public void onShowDynamicallyCreatedTree(ActionEvent event) 
    {
        WorkplaceFunctionTreeInfoNode n1 = new WorkplaceFunctionTreeInfoNode();
        for (int i=0; i<5; i++)
        {
            WorkplaceFunctionTreeInfoNode n2 = new WorkplaceFunctionTreeInfoNode();
            n2.setText("Node Level 1 / Instance " + i);
            n1.getSubNodes().add(n2);
            for (int j=0; j<5; j++)
            {
                WorkplaceFunctionTreeInfoNode n3 = new WorkplaceFunctionTreeInfoNode();
                n3.setText("Node Level 2 / Instance " + j);
                for (int k=0; k<5; k++)
                {
                    WorkplaceFunctionTreeInfoNode n4 = new WorkplaceFunctionTreeInfoNode();
                    n4.setText("Node Level 3 / Instance " + k);
                    WorkpageStartInfo wpsi = new WorkpageStartInfo();
                    n4.setWorkpageStartInfo(wpsi);
                    wpsi.setJspPage("/workplace/demohelloworld.jsp");
                    n3.getSubNodes().add(n4);
                }
                n2.getSubNodes().add(n3);
            }
        }
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().importWorkplaceFunctionTreeInfoNode(n1);
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().setObIndex(0);
        // the following line is only required if using performance optimization (isoaltion of update areas)
        ThreadData.getInstance().registerChangeUpdatingAllAreas();
    }

    /*
     * Update of the function tree by using the XML definition contained in the "components"-file.
     * The file is epxected to be located in the stream store in /ccworkplace/functiontrees/components.xml.
     * The XML of the file is parsed using JAX-B and the result of this parsing is padded into the workplace. 
     */
    public void onShowComponentsXml(ActionEvent event)
    {
        WorkplaceFunctionTreeInfoNode node = WorkplaceFunctionTreeInfoManager.loadFunctionTreeInfo("components");
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().importWorkplaceFunctionTreeInfoNode(node);
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().setObIndex(0);
        // the following line is only required if using performance optimization (isoaltion of update areas)
        ThreadData.getInstance().registerChangeUpdatingAllAreas();
    }
    
    public void onShowDefaultXml(ActionEvent event)
    {
        WorkplaceFunctionTreeInfoNode node = WorkplaceFunctionTreeInfoManager.loadFunctionTreeInfo("default");
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().importWorkplaceFunctionTreeInfoNode(node);
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().setObIndex(0);
        // the following line is only required if using performance optimization (isoaltion of update areas)
        ThreadData.getInstance().registerChangeUpdatingAllAreas();
    }
    
    public void onShowModifiedDefaultXml(ActionEvent event)
    {
        WorkplaceFunctionTreeInfoNode node = WorkplaceFunctionTreeInfoManager.loadFunctionTreeInfo("default",true);
        modifyNodes(node);
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().importWorkplaceFunctionTreeInfoNode(node);
        ((WorkpageContainer)getWorkpageContainer()).getFunctionsManager().setObIndex(0);
        // the following line is only required if using performance optimization (isoaltion of update areas)
        ThreadData.getInstance().registerChangeUpdatingAllAreas();
    }
    
    public void onShowDefaultForUser(ActionEvent event)
    {
        getWorkpageContainer().prepareWorkplaceForCurrentUser();
        // the following line is only required if using performance optimization (isoaltion of update areas)
        ThreadData.getInstance().registerChangeUpdatingAllAreas();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    /**
     * Just some "stupid" modification as demo: all nodes starting with "A", "H", "F"
     * are taken out of the function tree.
     */
    private void modifyNodes(WorkplaceFunctionTreeInfoNode node)
    {
        List<WorkplaceFunctionTreeInfoNode> subNodes = node.getSubNodes();
        for (int i=subNodes.size()-1; i>=0; i--)
        {
            WorkplaceFunctionTreeInfoNode subNode = subNodes.get(i);
            String text = subNode.getText();
            if (text != null && (text.startsWith("A") || text.startsWith("H") || text.startsWith("F")))
            {
                subNodes.remove(i);
            }
            else
            {
                modifyNodes(subNode);
            }
        }
    }

    
}
