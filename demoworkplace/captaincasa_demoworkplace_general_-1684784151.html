The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Dynamic Layout
  • Tuning - Stable Ids
  • Example: Schedule grid

This demos shows a grid with a lot of schedule items, that are dynamically rendered into the grid content.

Please observe: depending on the button you press, the right schedule items are flickering or not. The reason for flickering is that by default new components replaces the old ones: the old schedule items are removed on client side, the new ones are created.

But: you may influence this by explicitly assigning a stable id to the items. By this id the UI remembers the component, and the UI does no remove/create the optical representation but uses the existing one. Result: faste rendering and no flickering.

Please pay attention: stable ids have to be adde on every level of the dynamic node tree that you create. You see that in the demo both the SCHEDULEITEM and the contained DRAWAREA receive a stabled id (otherwise the DRAWAREA would flicker within the stable SCHEDULEITEM).

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoGridScheduleMass}" rowdistance="10" >
<t:row id="g_2" coldistance="5" >
<t:button id="g_3" actionListener="#{d.DemoGridScheduleMass.onPlusAction}" text="Plus" />
<t:button id="g_4" actionListener="#{d.DemoGridScheduleMass.onPlusActionNoBuffer}" text="Plus (no buffer)" />
</t:row>
<t:row id="g_5" >
<t:fixgrid id="g_6" height="100%" objectbinding="#{d.DemoGridScheduleMass.grid}" sbvisibleamount="20" width="100%" >
<t:gridcol id="g_7" text="Machine" width="100" >
<t:label id="g_8" text=".{machineId}" />
</t:gridcol>

</t:row>

</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.DRAWAREANode;
import org.eclnt.jsfserver.elements.componentnodes.SCHEDULEITEMNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridScheduleMass}")

public class DemoGridScheduleMass
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Order
    {
        String i_id;
        long i_from;
        long i_duration;
        public String getId()
        {
            return i_id;
        }
        public long getFrom()
        {
            return m_base + i_from;
        }
        public long getDuration()
        {
            return i_duration;
        }
        
    }
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_machineId;
        public String getMachineId() { return i_machineId; }
        public void setMachineId(String value) { this.i_machineId = value; }

        DYNAMICCONTENTBinding i_dynContent = new DYNAMICCONTENTBinding();
        public DYNAMICCONTENTBinding getDynContent() {  return i_dynContent; }
        
        List<Order> i_orders = new ArrayList<DemoGridScheduleMass.Order>();
        public List<Order> getOrders() { return i_orders; }
        
        private void renderDynContent(boolean withBuffering)
        {
            List<ComponentNode> l = new ArrayList<ROWDYNAMICCONTENTBinding.ComponentNode>();
            int orderCounter = -1;
            for (Order order: i_orders)
            {
                orderCounter++;
                SCHEDULEITEMNode sin = new SCHEDULEITEMNode()
                     .setScheduleleft(".{orders["+orderCounter +"].from}")
                     .setSchedulewidth(".{orders["+orderCounter +"].duration}")
                     .setResizingenabled(false)
                     .setBackground("#C0C0C0");
                if (withBuffering)
                    sin.addAttribute("id","SCHED_"+order.i_id);
                l.add(sin);
                DRAWAREANode da = new DRAWAREANode()
                        .setDrawcommands("rectangle(0,0,10%,100%,#800000);rectangle(10%,0,90%,100%,#008000);");
                if (withBuffering)
                    da.addAttribute("id","DRAW_"+order.i_id);
                sin.addSubNode(da);
            }
            i_dynContent.setContentNodes(l);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    private long m_base = 200;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridScheduleMass(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        int orderCounter = -1;
        for (int i=0; i<50; i++)
        {
            String machineId = "" + i;
            GridItem gi = new GridItem();
            gi.i_machineId = machineId;
            m_grid.getItems().add(gi);
            for (int j=0; j<20; j++)
            {
                orderCounter++;
                Order o = new Order();
                o.i_id = ""+orderCounter;
                o.i_from = j*50;
                o.i_duration = 40;
                gi.i_orders.add(o);
            }
        }
        rerender(true);
    }

    public String getPageName() { return "/workplace/demogridschedulemass.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridScheduleMass}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public void rerender(boolean withBuffering)
    {
        for (GridItem gi: m_grid.getItems())
            gi.renderDynContent(withBuffering);
    }

    public void onPlusAction(javax.faces.event.ActionEvent event) 
    {
        m_base += 10;
        rerender(true);
    }

    public void onPlusActionNoBuffer(javax.faces.event.ActionEvent event) 
    {
        m_base += 10;
        rerender(false);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
