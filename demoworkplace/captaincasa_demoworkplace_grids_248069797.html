The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Issues
  • Explicit paging bar (PBC)

The page bean component CCGridPager allows to scroll through grid data using a "web style" navigation bar.

The page bean component is part of the addong package (eclnt_pbc).

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPBCCCGridPager}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPBCCCGridPager}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >
<t:row id="g_6" >
<t:fixgrid id="g_2" height="100%" objectbinding="#{d.DemoPBCCCGridPager.grid}" selectorcolumn="2" selectorcolumnwidth="60" width="100%" >
<t:gridcol id="g_3" text="First name" width="100%" >
<t:label id="g_8" text=".{firstName}" />
</t:gridcol>
<t:gridcol id="g_9" text="Last name" width="100%" >
<t:label id="g_10" text=".{lastName}" />
</t:gridcol>
</t:fixgrid>
</t:row>
<t:row id="g_5" >
<t:pagebeancomponent id="g_12" pagebeanbinding="#{d.DemoPBCCCGridPager.pager}" pagebeanclass="org.eclnt.ccaddons.pbc.CCGridPager" pagebeaninitdata="pagesAfter:4;pagesBefore:4" shownullcontent="true" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCGridPager;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPBCCCGridPager}")

public class DemoPBCCCGridPager
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCGridPager m_pager = new CCGridPager();
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCGridPager(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        m_pager.prepare(m_grid,null);
        for (int i=0; i<125; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First name " + (i+1));
            gi.setLastName("Last name " + (i+1));
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demopbcccgridpager.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCGridPager}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCGridPager getPager() { return m_pager; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
