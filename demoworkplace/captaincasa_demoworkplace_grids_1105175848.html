The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Repeat Processing
  • REPEAT Component

The REPEAT in this example binds to some simple List on server side. For each item of the list, the content below the REPEAT component is repeated.

In the controls below the REPEAT component, you reference the properties and methods of the item by using the ".{...}" expression notation in order to bind to the corresponding item.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoRepeat}" >
<t:row id="g_2" >
<t:foldablepane id="g_3" text="Some repeated rows..." >
<t:repeat id="g_4" listbinding="#{d.DemoRepeat.items}" >
<t:rowdistance id="g_5" height="5" />
<t:row id="g_6" coldistance="5" >

<t:field id="g_10" text=".{lastName}" width="100" />
<t:button id="g_11" actionListener=".{onClear}" text="Clear" />
</t:row>
<t:rowdistance id="g_12" height="5" />
</t:repeat>
</t:foldablepane>
</t:row>
<t:rowdistance id="g_13" height="10" />
<t:row id="g_14" coldistance="5" >
<t:button id="g_15" actionListener="#{d.DemoRepeat.onAddItemAction}" text="Add one item" />
<t:button id="g_16" actionListener="#{d.DemoRepeat.onRemoveItemsAction}" text="Remove selected items" />

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoRepeat}")

public class DemoRepeat
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class Item
    {    
        boolean m_selected = false;
        String i_firstName;
        String i_lastName;
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String firstName) { i_firstName = firstName; }
        public String getLastName() { return i_lastName; }
        public void setLastName(String lastName) { i_lastName = lastName; }
        public boolean getSelected() { return m_selected; }
        public void setSelected(boolean value) { this.m_selected = value; }
        public void onClear(ActionEvent event) 
        {
            i_firstName = null;
            i_lastName = null;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<Item> m_items = new ArrayList<Item>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRepeat(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<5; i++)
        {
            Item it = new Item();
            it.i_firstName = "First " + i;
            it.i_lastName = "Last " + i;
            m_items.add(it);
        }
    }

    public String getPageName() { return "/workplace/demorepeat.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRepeat}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<Item> getItems() { return m_items; }
    
    public void onAddItemAction(ActionEvent event) 
    {
        Item it = new Item();
        it.i_firstName = "New item";
        it.i_lastName = "New item";
        m_items.add(it);
    }

    public void onRemoveItemsAction(ActionEvent event) 
    {
        for (int i=m_items.size()-1; i>=0; i--)
        {
            Item item = m_items.get(i);
            if (item.getSelected() == true)
            {
                m_items.remove(i);
            }
        }
    }

    public void onUpdateItemsAction(ActionEvent event) 
    {
        for (Item item: m_items)
        {
            item.i_firstName += "#";
            item.i_lastName += "#";
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
