The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Lightweight side popups

This examples shows some lightweight popups that are attached to some side area of the dialog.

The attributes of the popup are defined in the following way:

1. setUndecorated(true)
2. setAlwaysOnTop(true)
3. setLeftTopScreenTop/Right(true)

The size is passed as width=0 and height=0 so that the size is automatically calculated on frontend side.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.elements.util.DefaultModelessPopupListener;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoSidePopups}")

public class DemoSidePopups
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    DemoSidePopupsPopup m_topBottomUI; 
    DemoSidePopupsSideIcons m_sideUI;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoSidePopups(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener() 
        {
            @Override
            public void closeForced()
            {
                if (m_topBottomUI != null) closePopup(m_topBottomUI);
                if (m_sideUI != null) closePopup(m_sideUI);
                m_sideUI = null;
                m_topBottomUI = null;
                super.closeForced();
            }
        });
    }

    public String getPageName() { return "/workplace/demosidepopups.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSidePopups}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onOpenTopPopupAction(javax.faces.event.ActionEvent event) 
    {
        openTopBottomPopup(true);
    }
    
    public void onOpenBottomPopupAction(javax.faces.event.ActionEvent event) 
    {
        openTopBottomPopup(false);
    }
    
    private void openTopBottomPopup(boolean top)
    {
        if (m_topBottomUI != null)
        {
            closePopup(m_topBottomUI);
        }
        m_topBottomUI = new DemoSidePopupsPopup();
        m_topBottomUI.prepare(14,new DemoSidePopupsPopup.IListener()
        {
            @Override
            public void reactOnClosed()
            {
                if (m_topBottomUI != null) closePopup(m_topBottomUI);
                m_topBottomUI = null;
            }
        });
        ModelessPopup p = openModelessPopup(m_topBottomUI,"",0,0,new ModelessPopup.IModelessPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser()
            {
                if (m_topBottomUI != null) closePopup(m_topBottomUI);
                m_topBottomUI = null;
            }
        });
        if (top == true)
            p.setLeftTopScreenTop(true);
        else
            p.setLeftTopScreenBottom(true);
        p.setUndecorated(true);
        p.setAlwaysOnTop(true);
    }

    
    public void onOpenLeftSidePopupAction(javax.faces.event.ActionEvent event) 
    {
        openSidePopup(true);
    }
    
    public void onOpenSidePopupAction(javax.faces.event.ActionEvent event) 
    {
        openSidePopup(false);
    }

    private void openSidePopup(boolean left)
    {
        if (m_sideUI != null)
        {
            closePopup(m_sideUI);
        }
        m_sideUI = new DemoSidePopupsSideIcons();
        m_sideUI.prepare(new DemoSidePopupsSideIcons.IListener()
        {
            @Override
            public void reactOnClosed()
            {
                closePopup(m_sideUI);
                m_sideUI = null;
            }
        });
        ModelessPopup p = openModelessPopup(m_sideUI,"",0,0,new ModelessPopup.IModelessPopupListener()
        {
            @Override
            public void reactOnPopupClosedByUser()
            {
                if (m_sideUI != null) closePopup(m_sideUI);
                m_sideUI = null;
            }
        });
        if (left == true)
            p.setLeftTopScreenLeft(true);
        else
            p.setLeftTopScreenRight(true);
        p.setUndecorated(true);
        p.setAlwaysOnTop(true);
    }
}
