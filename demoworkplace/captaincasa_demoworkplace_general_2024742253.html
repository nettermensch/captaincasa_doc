The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Long Operations
  • Startup and progress info

Before starting some long operation on server side it is a good idea to inform the user about what happens inside this long operation.

A nice component to use for this purpose is the component CALLSERVERONTRIGGER.

The component gets triggered by some server side Trigger-instance - and then calls the server exactly one time. The actual starting of the long operation is done in two steps as consequence:

1. showing the user what will be done
2. actual processing of the long operation

Step 2 is started "through the client".

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.BlockerInfo;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoLongOperationWithInfo}")

public class DemoLongOperationWithInfo
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Trigger m_executeTrigger = new Trigger();
    DemoLongOperationWithInfoPopup m_infoPopup = null;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLongOperationWithInfo(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demolongoperationwithinfo.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLongOperationWithInfo}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onButton1Action(javax.faces.event.ActionEvent event) 
    {
        processItems();
    }

    public Trigger getExecuteTrigger() { return m_executeTrigger; }

    /*
     * This is the method that is called by the button: it opens
     * a popup and triggers the next processing through the client
     */
    public void onButton2Action(javax.faces.event.ActionEvent event) 
    {
        m_infoPopup = new DemoLongOperationWithInfoPopup();
        m_infoPopup.setNumberOfItems(100);
        ModalPopup p = openModalPopup(m_infoPopup,"Processing items",0,0,null);
        p.setLeftTopReferenceCentered();
        p.setTop(-200);
        p.setUndecorated(true);
        m_executeTrigger.trigger();
    }

    /**
     * This is the method that is called by the CALLSERVERONTRIGGER component.
     * This method starts the actual processing - and closes the info popup
     * afterwards.
     */
    public void onButton2ExecuteAction(javax.faces.event.ActionEvent event) 
    {
        processItems();
        closePopup(m_infoPopup);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void processItems()
    {
        for (int i=0; i<100; i++)
        {
            try { Thread.sleep(40); } catch (Throwable t) {}
            BlockerInfo.sendProgressToClient("Processing item " + i,(i+1));
        }
        Statusbar.outputSuccess("Finished!");
    }
}
