The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Dynamic Layout
  • Direct binding of objects
  • Hello world - with inner classes

The complete content of the dialog is rendered dynamically - by using component ROWDYNAMICCONTENT.

On server side the dynamic attributes of the field and the action listener of the button are implemented using the bind-methods that are available for each component attribute.

Here you can directly pass an implementation of interface IValueDelegation - best by sub-classing from one of the *Delegation classes of package "org.eclnt.jsfserver.util.valuemgmt.impl".

So you see: for dynamic content you either can use the set* methods in order to pass fix values or expression values. Or you can use the bind* method to directly implement dynamic attributes.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoDynamicContentbinding}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoDynamicContentbinding}" >
<t:rowdynamiccontent id="g_2" contentbinding="#{d.DemoDynamicContentbinding.dynamicContent}" />
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.COLDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FOLDABLEPANENode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.util.valuemgmt.IActionListenerDelegation;
import org.eclnt.jsfserver.util.valuemgmt.impl.StringDelegation;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDynamicContentbinding}")

public class DemoDynamicContentbinding
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ROWDYNAMICCONTENTBinding m_dynamicContent = new ROWDYNAMICCONTENTBinding();
    String m_name;
    String m_result;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDynamicContentbinding(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        renderContent();
    }

    public String getPageName() { return "/workplace/demodynamiccontentbinding.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDynamicContentbinding}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getDynamicContent() { return m_dynamicContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderContent()
    {
        FOLDABLEPANENode p = new FOLDABLEPANENode()
                .setText("Hello world - Value/action binding by inner classes")
                .setWidth("100%")
                .setRowdistance(10);
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            FIELDNode fn = new FIELDNode()
                    .setWidth(300)
                    .setAlign("center")
                    .bindText(new StringDelegation()
                    {
                        @Override
                        public String getValue()
                        {
                            return m_name;
                        }
                        @Override
                        public void setValue(String value)
                        {
                            m_name = value;
                        }
                    });
            r.addSubNode(fn);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }        
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            BUTTONNode bu = new BUTTONNode()
                    .setText("Say hello!")
                    .setWidth("200")
                    .bindActionListener(new IActionListenerDelegation()
                    {
                        @Override
                        public void onAction(ActionEvent event)
                        {
                            if (m_name == null)
                                m_result = "No name defined.";
                            else
                                m_result = "Hello world, " + m_name;
                        }
                    });
            r.addSubNode(bu);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }        
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            LABELNode l = new LABELNode()
                    .setWidth("100%")
                    .setAlign("center")
                    .bindText(new StringDelegation()
                    {
                        @Override
                        public String getValue()
                        {
                            return m_result;
                        }
                    });
            r.addSubNode(l);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }    
        m_dynamicContent.setContentNode(p);
    }

}
