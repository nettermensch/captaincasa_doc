The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Asynchronous Processing
  • Timer

The TIMER component allows to regularly trigger a server side action from client side. You can define the duration in milliseconds that is used as delay between timer calls.

Well, we think we do not have to mention that you can mis-use a timer by setting the duration to a very low value. Any duration that is defined lower than 10 seconds should be treated as critical.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.managedbean.IDispatcher;

@CCGenClass (expressionBase="#{d.DemoTimer}")

public class DemoTimer
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    protected Trigger m_resetTrigger = new Trigger();
    protected String m_durationType = "regular";
    String m_protocol;
    String m_firstName = "";
    String m_lastName = "";
    int m_timerDuration = 3000;
    
    public DemoTimer(IDispatcher dispatcher)
    {
        super(dispatcher);
        onTimer(null);
    }

    public Trigger getResetTrigger() { return m_resetTrigger; }
    public void setResetTrigger(Trigger value) { m_resetTrigger = value; }

    public String getDurationType() { return m_durationType; }
    public void setDurationType(String value) { m_durationType = value; }

    public void setFirstName(String value) { m_firstName = value; }
    public String getFirstName() { return m_firstName; }
    
    public void setLastName(String value) { m_lastName = value; }
    public String getLastName() { return m_lastName; }
    
    public String getProtocol() { return m_protocol; }

    public int getTimerDuration() { return m_timerDuration; }

    public void onTimer(ActionEvent event)
    {
        m_protocol = "Current time: " + (new Date()).getTime() + "\r\n" +
                     "First name: " + m_firstName + "\r\n" +
                     "Last name: " + m_lastName;
        m_resetTrigger.trigger();
    }
    
    public void onTimerDuration0(ActionEvent event)
    {
        m_timerDuration = 0;
    }
    
    public void onTimerDuration3000(ActionEvent event)
    {
        m_timerDuration = 3000;
    }
    
    public void onTimerDuration10000(ActionEvent event)
    {
        m_timerDuration = 10000;
    }
    
    public void onCauseRoundtrip(ActionEvent event) 
    {
        Statusbar.outputMessage("Roundtrip was caused: " + System.currentTimeMillis());
    }

}
