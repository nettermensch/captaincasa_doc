The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Update workplace layout

By pressing one of the buttons the workplace layout is updated accordingly.

The workplace layout either is defined as XML configuration (in /eclntjsfserver/config/ccworkplace/perspectives) or it is passed as configuration object at runtime.

The API "WorkplaceTileManager.importWorkplaceTileInfo(...)" provides two parameters:

(.) the first one to pass the workplace layout configuration
(.) the second one that decides if exsiting pages are close before switiching, of if they are left open

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageStartInfo;
import org.eclnt.workplace.WorkplaceTileInfo;
import org.eclnt.workplace.WorkplaceTileInfoPageContainer;
import org.eclnt.workplace.WorkplaceTileInfoSplitPane;

@CCGenClass (expressionBase="#{d.DemoWorkplaceTileLayout}")

public class DemoWorkplaceTileLayout
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoWorkplaceTileLayout(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demoworkplacetilelayout.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoWorkplaceTileLayout}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onOneContentAreaAction(javax.faces.event.ActionEvent event) 
    {
        getWorkpageContainer().getTileManager().importWorkplaceTileInfo("OnlyContent",false);
        getWorkpageContainer().switchToWorkpage(getWorkpage());
    }
    
    public void onNavigationLeftAction(javax.faces.event.ActionEvent event) 
    {
        getWorkpageContainer().getTileManager().importWorkplaceTileInfo("DemoWorkplace",false);
        getWorkpageContainer().switchToWorkpage(getWorkpage());
    }
    
    public void onSearchRightAction(javax.faces.event.ActionEvent event) 
    {
        WorkplaceTileInfo tileInfo = buildWorkplaceLayout();
        getWorkpageContainer().getTileManager().importWorkplaceTileInfo(tileInfo,false);
        getWorkpageContainer().switchToWorkpage(getWorkpage());
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private WorkplaceTileInfo buildWorkplaceLayout()
    {
        WorkplaceTileInfo result = new WorkplaceTileInfo();
        {
            // outest area is a split pane, horizontally divided
            WorkplaceTileInfoSplitPane splitOutest = new WorkplaceTileInfoSplitPane();
            result.setSplit(splitOutest);
            splitOutest.setOrientation("horizontal");
            splitOutest.setDividerLocation("300");
            {
                // left split area is a split pane, vertically divided
                WorkplaceTileInfoSplitPane splitLeft = new WorkplaceTileInfoSplitPane();
                splitOutest.setSubSplit1(splitLeft);
                splitLeft.setOrientation("vertical");
                splitLeft.setDividerLocation("50%");
                {
                    // left split is providing two content areas
                    {
                        WorkplaceTileInfoPageContainer functionContent = new WorkplaceTileInfoPageContainer();
                        functionContent.setId("FUNCTIONS");
                        splitLeft.setSubContainer1(functionContent);
                    }
                    {
                        WorkplaceTileInfoPageContainer functionContent = new WorkplaceTileInfoPageContainer();
                        functionContent.setId("SEARCH");
                        splitLeft.setSubContainer2(functionContent);
                    }
                }
                // right area is the content area
                WorkplaceTileInfoPageContainer mainContent = new WorkplaceTileInfoPageContainer();
                splitOutest.setSubContainer2(mainContent);
            }
            // start default pages
            {
                WorkpageStartInfo wpsi = new WorkpageStartInfo();
                wpsi.setJspPage("/workplace/wpfunctiontrees.jsp");
                wpsi.setId("wpfunctionbar");
                wpsi.setDecorated(false);
                wpsi.setCloseSupported(false);
                wpsi.setStartSubWorkpageContainerId("FUNCTIONS");
                result.getStartViews().add(wpsi);
            }
            {
                WorkpageStartInfo wpsi = new WorkpageStartInfo();
                wpsi.setJspPage("/workplace/wpfunctionsearch.jsp");
                wpsi.setId("/workplace/wpfunctionsearch.jsp");
                wpsi.setDecorated(false);
                wpsi.setCloseSupported(false);
                wpsi.setStartSubWorkpageContainerId("SEARCH");
                result.getStartViews().add(wpsi);
            }
        }
        return result;
    }
}
