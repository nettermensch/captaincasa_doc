The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Invokers
  • Button with long click

When setting attribute BUTTON-WITHLONGCLICK to "true" then the user may press the button for a while. Triggered by some internal timer an invoke-roundtrip will be started automatically.

In addition you my set BUTTON-WITHLONGCLICKENDEVENT to "true". Then you get some additional notification when the user actually finished the long clicking and releases the mouse button.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventInvoke;
import org.eclnt.jsfserver.elements.events.BaseActionEventLongClickEnd;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoButtonLongClick}")

public class DemoButtonLongClick
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_protocol = "";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoButtonLongClick(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demobuttonlongclick.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoButtonLongClickUI}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    public void onLongClickButtonAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventInvoke)
        {
            BaseActionEventInvoke e = (BaseActionEventInvoke)event;
            m_protocol += "Long Click Button Action: long click = " + e.isLongClick() + "\n";
        }
        else if (event instanceof BaseActionEventLongClickEnd)
        {
            m_protocol += "Long Click Button Action: long click ended\n";
        }
    }

    public void onNormalButtonAction(javax.faces.event.ActionEvent event) 
    {
        BaseActionEventInvoke e = (BaseActionEventInvoke)event;
        m_protocol += "Normal Button Action: long click = " + e.isLongClick() + "\n";
    }

    public void onClearProtocolAction(javax.faces.event.ActionEvent event) 
    {
        m_protocol = "";
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
