The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Repeat Processing
  • Nesting REPEAT Components

It is possible to nest REPEAT components into other REPEAT components.

In this example the outside REPEAT component binds to a list of departments. Each department item itself provides a list of employees.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoRepeatRepeat}")

public class DemoRepeatRepeat
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Employee
    {
        String i_id;
        String i_name;
        public String getId() { return i_id; }
        public Employee setId(String id) { i_id = id; return this; }
        public String getName() { return i_name; }
        public Employee setName(String name) { i_name = name; return this; }
    }
    
    public class Department
    {
        String i_id;
        String i_text;
        List<Employee> i_employees = new ArrayList<Employee>();
        public String getId() { return i_id; }
        public Department setId(String id) { i_id = id; return this; }
        public String getText() { return i_text; }
        public Department setText(String text) { i_text = text; return this; }
        public Department addEmployee(Employee e) { i_employees.add(e); return this; }
        public List<Employee> getEmployees() { return i_employees; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    List<Department> m_departments = new ArrayList<Department>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRepeatRepeat(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_departments.add
        (
            new Department()
            .setId("001")
            .setText("Human Resources")
            .addEmployee(new Employee().setId("0001").setName("John Wayne"))
            .addEmployee(new Employee().setId("0002").setName("Mark Twain"))
        );
        m_departments.add
        (
            new Department()
            .setId("002")
            .setText("Sales")
            .addEmployee(new Employee().setId("0003").setName("Hugo Egon Balder"))
            .addEmployee(new Employee().setId("0004").setName("Wolfgang Ambross"))
            .addEmployee(new Employee().setId("0005").setName("Loriot"))
        );
        m_departments.add
        (
            new Department()
            .setId("003")
            .setText("Production")
            .addEmployee(new Employee().setId("0006").setName("Henry Ford"))
            .addEmployee(new Employee().setId("0007").setName("Robert Bosch"))
        );
    }

    public String getPageName() { return "/workplace/demorepeatrepeat.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRepeatRepeat}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<Department> getDepartments() { return m_departments; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
