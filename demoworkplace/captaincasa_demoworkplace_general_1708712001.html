The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Client Info

Each request that is sent from the CaptainCasa client contains certain http header parameters that may be useful for your application processing on server side.

A special parameter is the so called client-id ("eclnt-id"). It contains an id that is assigned to each client, so that the server can keep data dependent from the client. The client-id is generated by default, but can also be explicitly assigned. Please view the Developer's Guide (chapter "Client-Id Management") for more information.

Pay attention: the client-id is only available if calling pages through ".ccapplet" or ".ccwebstart".

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Enumeration;

import javax.servlet.http.HttpServletRequest;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.HttpSessionAccess;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoClientId}")

public class DemoClientId
    extends PageBean
    implements Serializable
{
    String m_newClientId;
    Trigger m_refreshTrigger = new Trigger();
    
    public DemoClientId(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public String getCurrentClientType() 
    { 
        if (HttpSessionAccess.checkIfCurrentClientTypeIsRisc()) return "RISC-HTML Client";
        if (HttpSessionAccess.checkIfCurrentClientJavaTypeIsSwing()) return "Java-Swing Client";
        if (HttpSessionAccess.checkIfCurrentClientJavaTypeIsFX()) return "Java-FX Client";
        return "Undefined Client";
    }
    
    public String getCurrentClientId() 
    { 
        return HttpSessionAccess.getCurrentClientId(); 
    }
    
    public String getCurrentClientIp() 
    { 
        return HttpSessionAccess.getCurrentRequest().getRemoteAddr(); 
    }
    
    public String getCurrentClientHost() 
    { 
        return HttpSessionAccess.getCurrentRequest().getRemoteHost(); 
    }
    
    public String getCurrentClientLanguage() 
    { 
        return HttpSessionAccess.getCurrentRequest().getHeader("eclnt-language"); 
    }
    
    public String getCurrentClientCountry() 
    { 
        return HttpSessionAccess.getCurrentRequest().getHeader("eclnt-country"); 
    }
    
    public String getNewClientId() { return m_newClientId; }
    public void setNewClientId(String value) { this.m_newClientId = value; }

    public void onSetAction(javax.faces.event.ActionEvent event) 
    {
        if (m_newClientId == null || m_newClientId.trim().length() == 0)
        {
            Statusbar.outputError("Please define a valid new client id");
            return;
        }
        HttpSessionAccess.setCurrentClientId(m_newClientId);
        m_refreshTrigger.trigger();
    }

    public Trigger getRefreshTrigger() { return m_refreshTrigger; }

    public void onRefreshAction(javax.faces.event.ActionEvent event) {}

    public String getRequestHeaderParams() 
    { 
        StringBuffer sb = new StringBuffer();
        HttpServletRequest req = HttpSessionAccess.getCurrentRequest();
        if (req != null)
        {
            Enumeration<String> names = req.getHeaderNames();
            while (names.hasMoreElements())
            {
                String name = names.nextElement();
                String value = req.getHeader(name);
                sb.append(name + ": " + value + "\n");
            }
        }
        return sb.toString();
    }

}
