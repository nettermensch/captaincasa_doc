The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Radio Button in Grid (II)

This example shows how to implement a radio button scenario spanning multiple rows.

The steps are:
1. The radio button(s) must refer to one central value that is managed outside the grid processing.
2. The radio button must flush its value to the server side when clicked by the user.
3. Each radio button must bind to some unique value

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" objectbinding="#{d.DemoGridRadioButton}">
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" borderwidth="1" objectbinding="#{d.DemoGridRadioButton.grid}" sbvisibleamount="10" width="100%" >
<t:gridcol id="g_4" text="Person" width="100%" >
<t:label id="g_5" text=".{person}" />
</t:gridcol>
<t:gridcol id="g_6" text="Column" width="100" >
<t:radiobutton id="g_7" flush="true" group="BOSS" refvalue=".{rbIndex}" value="#{d.DemoGridRadioButton.rbValue}" />
</t:gridcol>
</t:fixgrid>
</t:row>
<t:row id="g_8" >
<t:label id="g_9" text="Radio button value" width="150" />
<t:label id="g_10" text="#{d.DemoGridRadioButton.rbValue}" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridRadioButton}")

public class DemoGridRadioButton
    extends PageBean
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_rbIndex;
        public String getRbIndex() { return m_rbIndex; }
        public void setRbIndex(String value) { this.m_rbIndex = value; }
        
        String m_person;
        public String getPerson() { return m_person; }
        public void setPerson(String value) { this.m_person = value; }
        
    }
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    String m_rbValue = "0";
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public String getRbValue() { return m_rbValue; }
    public void setRbValue(String value) { this.m_rbValue = value; }

    public DemoGridRadioButton(IWorkpageDispatcher d)
    {
        super(d);
        for (int i=0; i<20; i++)
        {
            GridItem gi = new GridItem();
            gi.setPerson("Person " + i);
            gi.setRbIndex(i+"");
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/deomgridradiobutton.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridRadioButton}"; }

}
