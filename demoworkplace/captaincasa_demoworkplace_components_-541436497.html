The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Drag/drop files into full area

The FILEUPLOADAREAASYNCHRONOUS is an optical area in which you can drag & drop files from the file system so that their content is uploaded to the server side.

You can place any component(s) into the is area.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.DefaultUploadContent;
import org.eclnt.jsfserver.bufferedcontent.UploadContentMgr;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventUploadAsynchronousFinished;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoFileUploadArea}")

public class DemoFileUploadArea
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyUploadContent extends DefaultUploadContent
    {
        String i_fileName;
        byte[] i_content;
        @Override
        public void beginPassing()
        {
        }
        @Override
        public void passClientFile(String fileName, byte[] content)
        {
            i_fileName = fileName;
            i_content = content;
        }
        @Override
        public void endPassing()
        {
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyUploadContent m_uploadContent = new MyUploadContent();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFileUploadArea(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        UploadContentMgr.add(m_uploadContent);
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnDestroyed()
            {
                UploadContentMgr.remove(m_uploadContent);
            }
        });
    }

    public String getPageName() { return "/workplace/demofileuploadarea.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFileUploadArea}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getUploadUrl() { return m_uploadContent.getURL(); }
    
    public void onAreaAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventUploadAsynchronousFinished)
        {
            Statusbar.outputSuccess("Uploaded: " + m_uploadContent.i_fileName + ", size: " + m_uploadContent.i_content.length);
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
