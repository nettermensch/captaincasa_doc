The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Types of Binding
  • Large output lists

For output-only scenarios there is a special grid - the ARRAYGRID. In the ARRAYGRID all layout information about the columns is passed through array parameters. The mandatory parameters on grid level are the columns[] and widths[] array - the mandatory parameter on item level is the values[] array.

Aside the dynamic configuration there is one important aspect: ARRAYGRID components are not "multiplied out" into their cell components at runtime. This makes them usable for large grids that normally, with normal FIXGRID rendering, would occupy quite a lot of memory for holding the corresponding component tree.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDItem;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoArrayGridBinding}")

/**
 * Demo of dynamic grid.
 * The first column (index 0) is a special one in this demo - by assigning
 * the text value ARRAYGRIDListBinding.TEXT_GRIDROWSELECTOR or TEXT_GRIDROWINDEX,
 * a grid row selector is rendered as column content. Otherwise (with normal)
 * text a normal LABEL is rendered.
 */
public class DemoArrayGridBinding
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyARRAYGRIDItem extends ARRAYGRIDItem
    {
        @Override
        public void onRowExecute()
        {
            Statusbar.outputSuccess("onRowExecute was called");
        }
        @Override
        public void onRowSelect()
        {
            Statusbar.outputSuccess("onRowSelect was called");
        }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            super.onRowPopupMenuItem(event);
            if ("remove".equals(event.getCommand()))
            {
                m_grid.getItems().remove(this);                
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static Random s_random = new Random();
    
    ARRAYGRIDListBinding<MyARRAYGRIDItem> m_grid = new ARRAYGRIDListBinding<MyARRAYGRIDItem>();
    int m_columnCount = 25;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoArrayGridBinding(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initGrid();
        for (int i=0; i<100; i++)
            onAddItem(null);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public ARRAYGRIDListBinding getGrid() { return m_grid; }
    
    public void setColumnCount(int value) 
    { 
        m_columnCount = value; 
        if (m_columnCount <= 0)
            m_columnCount = 1;
    }
    public int getColumnCount() { return m_columnCount; }
    
    public void onAddItem(ActionEvent event)
    {
        Random r = new Random();
        MyARRAYGRIDItem agi = new MyARRAYGRIDItem();
        String[] values = new String[m_columnCount];
        String[] backgrounds = new String[m_columnCount];
        String[] fonts = new String[m_columnCount];
        for (int j=0; j<m_columnCount; j++)
        {
            if (j == 2 && m_grid.getItems().size()%2 == 0)
                continue;
            values[j] = j+"_"+s_random.nextInt(99999);
        }
        for (int j=0; j<m_columnCount; j++) 
        {
            int randomInt = r.nextInt(10);
            if (randomInt == 5)
            {
                backgrounds[j] = "#ffc0c040";
                fonts[j] = "weight:bold";
            }
            else if (j % 2 == 0)
                backgrounds[j] = "#ffffc040";
            else if (j % 7 == 0)
                backgrounds[j] = "#c0ffc040";
            else
                backgrounds[j] = "#c0ffff40";
        }
        String[] images = new String[m_columnCount];
        images[0] = "/images/icons/add.png";
        images[1] = "/images/icons/add.png";
        images[2] = "/images/icons/add.png";
        agi.setValues(values);
        agi.setBackgrounds(backgrounds);
        agi.setImages(images);
        agi.setFonts(fonts);
        m_grid.getItems().add(agi);
        if (event != null)
            m_grid.ensureItemToBeDisplayed(agi);
    }
    
    public void onNewColumns(ActionEvent event)
    {
        initGrid();
    }
    
    public void onTimer(ActionEvent event)
    {
        Statusbar.outputSuccess("Timer: " + System.currentTimeMillis());
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void initGrid()
    {
        m_grid.getItems().clear();
        String[] titles = new String[m_columnCount];
        String[] widths = new String[m_columnCount];
        String[] aligns = new String[m_columnCount];
        for (int i=0; i<m_columnCount; i++) titles[i] = "Title " + i;
        for (int i=0; i<m_columnCount; i++)
        {
            widths[i] = "100";
            if (i%5 == 4) widths[i] = "150";
        }
        for (int i=0; i<m_columnCount; i++)
        {
            aligns[i] = "left";
            if (i%5 == 4) aligns[i] = "right";
        }
        m_grid.setTitles(titles);
        m_grid.setWidths(widths);
        m_grid.setAligns(aligns);
    }
    
}
