The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Bean data grid (PBC)
  • Discrete values

The column "gender" is an integer data type. By default it is formatted as number and the filter dialog is a from-to-dialog - as you can see in the "height" column.

You can update the default behavior in two ways:

1. Set initialization parameter "withDisplayValueConversion" within the column configurartion to "true". Now the value that is rendered inside the grid is picked through method "calculateDisplayText", which is part of CCDataGridView2.IListener.

2. Use method "createColumnFilter" of CCDataGridView2.IListener and pass a filter component that is better suitable.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.ccaddons.pbc.datagridview2.ConfigurationColumn;
import org.eclnt.ccaddons.pbc.datagridview2.DGVColumnFilterExplicitValues;
import org.eclnt.ccaddons.pbc.datagridview2.IDGVColumnFilter;
import org.eclnt.ccaddons.pbc.datagridview2.PropertyFormatData;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup.IYesNoListener;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass(expressionBase = "#{d.DemoPBCCCDataGridView2ExplicitValuesFilter}")

public class DemoPBCCCDataGridView2ExplicitValuesFilter extends PageBean implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * The bean... This normally is an own class. In the demo it is an inner class
     * so that all code is kept within the demo class.
     */
    public static class TestPerson
    {
        String i_firstName;
        String i_lastName;
        int i_height;
        LocalDate i_birthDate;
        boolean i_married;
        int i_gender; // 0 male, 1 female, 2 divers
        public String getFirstName() { return i_firstName; } 
        public String getLastName() { return i_lastName; } 
        public int getHeight() { return i_height; } 
        public LocalDate getBirthDate() { return i_birthDate; } 
        public boolean isMarried() { return i_married; } 
        public void setFirstName(String firstName) { i_firstName = firstName; } 
        public void setLastName(String lastName) { i_lastName = lastName; } 
        public void setHeight(int height) { i_height = height; } 
        public void setBirthDate(LocalDate birthDate) { i_birthDate = birthDate; } 
        public void setMarried(boolean married) { i_married = married; }
        public int getGender() { return i_gender; }
        public void setGender(int gender) { i_gender = gender; }
    }

    /**
     * Listener to the data grid view's listener so that in order to react on events
     * of the grid processing.
     */
    public class MyDGVListener extends CCDataGridView2.DefaultListener<TestPerson>
    {
        @Override
        public void reactOnGridSelect(TestPerson object)
        {
            reactOnItemSelection(object);
        }
        @Override
        public String calculateDisplayText(TestPerson object, String propertyName, Object propertyValue)
        {
            // of course: the coding of "calculateDisplayText" should be better linked
            // to the coding of "createColumnFilter"...! But we wanted to make it
            // as simple as possible.
            if ("gender".equals(propertyName) && propertyValue != null)
            {
                if (propertyValue.equals(0)) return "male";
                if (propertyValue.equals(1)) return "female";
                if (propertyValue.equals(2)) return "diverse";
            }
            return super.calculateDisplayText(object, propertyName, propertyValue);
        }
        @Override
        public IDGVColumnFilter createColumnFilter(ConfigurationColumn cc, PropertyFormatData format)
        {
            if ("gender".equals(cc.getPropertyName()))
            {
                // of course: the coding of "calculateDisplayText" should be better linked
                // to the coding of "createColumnFilter"...! But we wanted to make it
                // as simple as possible.
                DGVColumnFilterExplicitValues filter = new DGVColumnFilterExplicitValues();
                filter.addValue("0","male");
                filter.addValue("1","female");
                filter.addValue("2","diverse");
                return filter;
            }
            return super.createColumnFilter(cc, format);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataGridView2<TestPerson> m_dataView = new CCDataGridView2<TestPerson>(TestPerson.class);
    List<TestPerson> m_persons = new ArrayList<TestPerson>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDataGridView2ExplicitValuesFilter(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        fillData();
    }

    public String getPageName()
    {
        return "/workplace/demopbcccdatagridview2explicitvaluesfilter.xml";
    }

    public String getRootExpressionUsedInPage()
    {
        return "#{d.DemoPBCCCDataGridView2ExplicitValuesFilter}";
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2<TestPerson> getDataView()
    {
        return m_dataView;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void deleteItem(final TestPerson object)
    {
        YESNOPopup.createInstance("Deletion of item","Do you really want to remove the selected item?",new IYesNoListener()
        {
            @Override
            public void reactOnYes() { deleteItemExecute(object); }
            @Override
            public void reactOnNo() {}
        });
    }

    private void deleteItemExecute(TestPerson object)
    {
        m_persons.remove(object);
        m_dataView.refreshData(m_persons);
    }

    private void reactOnItemSelection(TestPerson object)
    {
        Statusbar.outputSuccess("Selection of item: " + object.getLastName() + ", " + object.getFirstName());
    }

    private void fillData()
    {
        m_persons.clear();
        ;
        for (int i = 0; i < 100; i++)
        {
            TestPerson p = new TestPerson();
            p.setFirstName(RandomLogic.getFirstName());
            p.setLastName(RandomLogic.getLastName());
            p.setHeight(RandomLogic.getHeight());
            p.setMarried(i % 2 == 0);
            p.setBirthDate(RandomLogic.getBirthDate());
            p.setGender(i%3);
            m_persons.add(p);
        }
        m_dataView.prepare(m_persons, new MyDGVListener());
    }

}
