The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Statusbar and Messages
  • Message toasts

So called "message toasts" are a common way to notify the user about certain information. The information is shown as popup dialog together with some title and some image. The dialog will close automatically after some duration of time.

You may pass a type (see constant definitions in MessageToast class). This type is associated with a style class modifier, so that each type may have its own color style.

Message toasts can be directly used, using class MessageToast. You do not have to embed some explicit component inside your layout definition.

The duration of time that each message waits for being closed is centrally configure-able by calling MessageToast.initDuration(..).

<t:row id="g_13" >

<t:row id="g_17" >

</t:row>
<t:row id="g_3" >

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.MessageToast;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoMessageToast}")

public class DemoMessageToast
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_messageType = MessageToast.TYPE_INFO;
    String m_messageTitle = "This is the title";
    String m_messageText = "Lorem ipsum dolor sit amet, consetetur sadipscing elitr, sed diam nonumy eirmod tempor.";
    ValidValuesBinding m_messagePositionVVS = new ValidValuesBinding();
    String m_messagePosition = MessageToast.POSITION_rightbottom;
    int m_messageDuration = MessageToast.s_duration;

    ValidValuesBinding m_messageTypeVVS = new ValidValuesBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMessageToast(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_messageTypeVVS.addValidValue(MessageToast.TYPE_INFO,MessageToast.TYPE_INFO);
        m_messageTypeVVS.addValidValue(MessageToast.TYPE_SUCCESS,MessageToast.TYPE_SUCCESS);
        m_messageTypeVVS.addValidValue(MessageToast.TYPE_WARNING,MessageToast.TYPE_WARNING);
        m_messageTypeVVS.addValidValue(MessageToast.TYPE_ERROR,MessageToast.TYPE_ERROR);
        for (String position: MessageToast.POSITIONS_all)
            m_messagePositionVVS.addValidValue(position,position);
        // optional: you may explicitly set the duration that messages show up on client side
        // pay attention: this setting is done globally for all users!
        // MessageToast.initDuration(5000);
        // MessageToast.initPosition(MessageToast.POSITION_leftmiddle);
    }

    public String getPageName() { return "/workplace/demomessagetoast.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMessageToast}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getMessageTitle() { return m_messageTitle; }
    public void setMessageTitle(String value) { this.m_messageTitle = value; }

    public String getMessageText() { return m_messageText; }
    public void setMessageText(String value) { this.m_messageText = value; }
    
    public String getMessageType() { return m_messageType; } 
    public void setMessageType(String messageType) { m_messageType = messageType; }

    public ValidValuesBinding getMessageTypeVVS() { return m_messageTypeVVS; }

    public ValidValuesBinding getMessagePositionVVS() { return m_messagePositionVVS; }

    public String getMessagePosition() { return m_messagePosition; }
    public void setMessagePosition(String value) { this.m_messagePosition = value; }

    public int getMessageDuration() { return m_messageDuration; }
    public void setMessageDuration(int value) { this.m_messageDuration = value; }

    public void onCreateMessageAction(javax.faces.event.ActionEvent event) 
    {
        MessageToast.showMessage(m_messageType,"/images.iconssvg.save._606060.16x16.ccsvg",m_messageTitle,m_messageText,m_messagePosition,m_messageDuration);
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
