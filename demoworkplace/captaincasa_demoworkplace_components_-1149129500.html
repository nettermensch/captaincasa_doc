The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Cascading popup menu

Check the content of the popup menu on various components of the screen by right clicking them - first with "cascading" being switched on, then with "cascading" being switched off. You will see that the menus contain more items if switched on.

Popup menus can be defined on any component level.

When clicking the right mouse button then by default only this popup menu shown that is defined on the lowest level.

By setting POPUPMENU-CASCADEPARENTMENU to "true" you can define that a popup menu will also embed the parent level's popup menu.

This is especially useful in a grid where you have popup menu definitions on item level and you have popup menu definitions on the grid level. If not using the cascade-feature then it's difficult for the user to reach the whole grid's popup menu once the grid is filled with items.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupMenuCascade}")

public class DemoPopupMenuCascade
    extends WorkpageDispatchedPageBean     
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            if ("DELETE".equals(event.getCommand()))
            {
                m_grid.getItems().remove(this);
            }
        }
    }
    
    public class MyGrid extends FIXGRIDListBinding<GridItem>
    {
        @Override
        protected void onPopupMenuItem(BaseActionEventPopupMenuItem popupMenuItemEvent)
        {
            if ("ADDNEW".equals(popupMenuItemEvent.getCommand()))
            {
                GridItem gi = new GridItem();
                m_grid.getItems().add(0,gi);
                m_grid.deselectCurrentSelection();
                m_grid.selectAndFocusItem(gi);
            }
            else if ("SELECTALL".equals(popupMenuItemEvent.getCommand()))
            {
                m_grid.deselectCurrentSelection();
                for (GridItem gi: m_grid.getItems())
                    m_grid.selectItem(gi);
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyGrid m_grid = new MyGrid();
    String m_field2 = "Test";
    String m_field1 = "Test";
    boolean m_withCascading = true;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupMenuCascade(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);      
        for (int i=0; i<3; i++)
        {
            GridItem gi = new GridItem();
            gi.setLastName("Last " + i);
            gi.setFirstName("First " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demopopupmenucascade.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupMenuCascade}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getField2() { return m_field2; }
    public void setField2(String value) { this.m_field2 = value; }

    public String getField1() { return m_field1; }
    public void setField1(String value) { this.m_field1 = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onBodyAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("SAVE".equals(e.getCommand()))
            {
                Statusbar.outputSuccess("Data was saved");
            }
        }
    }

    public void onAreaAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("CLEARAREA".equals(e.getCommand()))
            {
                m_field1 = null;
                m_field2 = null;
            }
        }
    }

    public void onField2Action(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("CLEAR".equals(e.getCommand()))
            {
                m_field2 = "";
            }
            else if ("DOUBLE".equals(e.getCommand()))
            {
                m_field2 += m_field2;
            }
        }
    }

    public void onField1Action(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("CLEAR".equals(e.getCommand()))
            {
                m_field1 = "";
            }
            else if ("DOUBLE".equals(e.getCommand()))
            {
                m_field1 += m_field1;
            }
        }
    }

    public boolean getWithCascading() { return m_withCascading; }
    public void setWithCascading(boolean value) { this.m_withCascading = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
