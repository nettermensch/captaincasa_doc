The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Dynamic Layout
  • Direct binding of objects
  • Directly binding a grid

The Java implementation is the interesting part of this demo - the whole layout is created dynamically.

"Normally" you are used to using node-components just as direct representation of the corresponding component instances, that you normally define within the XML layout definition. This means you either set a fix value or you a bind a value by assigning an expression.

"Now" you see that you can directly bind objects and processing to the component node by using corresponding "bind" methods. When doing so then the expression for accessing the corresponding object is managed internally.

This simplifies the creation of dynamic scenarios significantly because you do not have to think about how to calculate expressions anymore.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoRowDynamicContentBinding}" >
<t:rowdynamiccontent id="g_2" contentbinding="#{d.DemoRowDynamicContentBinding.dynContent}" />
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.util.valuemgmt.IActionListenerDelegation;
import org.eclnt.jsfserver.util.valuemgmt.impl.BooleanDelegation;
import org.eclnt.jsfserver.util.valuemgmt.impl.BooleanValue;
import org.eclnt.util.valuemgmt.UniqueIdCreator;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoRowDynamicContentBinding}")

public class DemoRowDynamicContentBinding
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // entity object - could be any pojo object
    public static class TestPerson
    {
        String m_id = UniqueIdCreator.createUUID();
        String m_firstName = "not set";
        String m_lastName = "not set";
        String m_town = "not set";
        public String getId() { return m_id; }
        public void setId(String id) { m_id = id; }
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String firstName) { m_firstName = firstName; }
        public String getLastName() { return m_lastName; }
        public void setLastName(String lastName) { m_lastName = lastName; }
        public String getTown() { return m_town; }
        public void setTown(String town) { m_town = town; }
    }
    
    public class GridItem extends FIXGRIDItem
    {
        Object i_instance;
        public GridItem(Object instance)
        {
            i_instance = instance;
        }
        public Object getInstance() { return i_instance; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ROWDYNAMICCONTENTBinding m_dynContent = new ROWDYNAMICCONTENTBinding();
    String[] m_propertySequence;
    Class m_class;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRowDynamicContentBinding(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        prepare(TestPerson.class, new String[] {"id","firstName","lastName","town"});
    }

    public String getPageName() { return "/workplace/demorowdynamiccontentbinding.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRowDynamicContentBinding}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void prepare(Class<TestPerson> clazz, String[] propertySequence)
    {
        m_class = clazz;
        m_propertySequence = propertySequence;
        render();
    }

    public ROWDYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void render()
    {
        // outest pane
        PANENode outestPane = new PANENode()
                .setWidth("100%")
                .setRowdistance(5);
        // grid
        {
            m_grid = new FIXGRIDListBinding<GridItem>(); // create the object which is bound
            ROWNode pr = new ROWNode();
            outestPane.addSubNode(pr);
            FIXGRIDNode fg = new FIXGRIDNode()
                    .bindObjectbinding(m_grid) // <== assigns the value AND automatically binds it via expression
                    .setWidth("100%")
                    .setSbvisibleamount(10);
            pr.addSubNode(fg);
            for (String property: m_propertySequence)
            {
                GRIDCOLNode gc = new GRIDCOLNode()
                        .setText(property)
                        .setWidth("100%;100");
                fg.addSubNode(gc);
                FIELDNode f = new FIELDNode()
                        .setText(".{instance."+property+"}"); // <== normal binding to expression
                gc.addSubNode(f);
            }
        }
        // buttons
        {
            ROWNode buttonRow = new ROWNode()
                    .setColdistance(5);
            outestPane.addSubNode(buttonRow);
            {
                BUTTONNode b = new BUTTONNode()
                        .setText("Create new item")
                        .setWidth("100+")
                        .bindActionListener(new IActionListenerDelegation()
                        {
                            @Override
                            public void onAction(ActionEvent event)
                            {
                                createNewItem();
                            }
                        });
                buttonRow.addSubNode(b);
            }
            {
                BUTTONNode b = new BUTTONNode()
                        .setText("Remove selected item")
                        .setWidth("100+")
                        .bindEnabled(new BooleanDelegation()
                        {
                            @Override
                            public void setValue(Boolean value) {}
                            @Override
                            public Boolean getValue()
                            {
                                return m_grid.getSelectedItem() != null;
                            }
                        })
                        .bindActionListener(new IActionListenerDelegation()
                        {
                            @Override
                            public void onAction(ActionEvent event)
                            {
                                removeSelectedItem();
                            }
                        });
                buttonRow.addSubNode(b);
            }
            {
                ROWDISTANCENode rd = new ROWDISTANCENode()
                        .setHeight(20);
                outestPane.addSubNode(rd);
            }
        }
        m_dynContent.setContentNode(outestPane);
    }
    
    private void createNewItem()
    {
        try
        {
            Object o = m_class.newInstance();
            m_grid.getItems().add(new GridItem(o));
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }
    
    private void removeSelectedItem()
    {
        GridItem selItem = m_grid.getSelectedItem();
        if (selItem != null)
        {
            m_grid.getItems().remove(selItem);
        }
    }
}
