The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • Column Content rendering

This example demonstrates how to build scenarios in which the content of columns is rendered dependent on some other information.

In the example the check boxes "married" and "children" are only rendered when the check box "Pers. Details" is activated.

There is one important aspect to pay attention to: all dynamic behaviour (such as rendering or hiding the check boxes) must NOT happen on the component level which is located directly below the GRIDCOL component. This level must be defined in a fix way. What happens below this level is then up to you!

This is the reason why the check boxes are surrounded by a PANE / ROW combination.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.demoGriddyncbs}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" borderwidth="1" objectbinding="#{d.demoGriddyncbs.rows}" rowbgpaint="background(#FFFFFF)" sbvisibleamount="10" width="100%" >
<t:gridcol id="g_4" text="Name" width="100%" >
<t:field id="g_5" text=".{name}" />
</t:gridcol>
<t:gridcol id="g_6" text="Pers. Details" width="100" >
<t:checkbox id="g_7" flush="true" selected=".{withPersonalDetails}" />
</t:gridcol>
<t:gridcol id="g_8" text="Children" width="60" >
<t:pane id="g_9" >
<t:row id="g_10" >
<t:checkbox id="g_11" rendered=".{withPersonalDetails}" selected=".{withChildren}" />
</t:row>
</t:pane>
</t:gridcol>
<t:gridcol id="g_12" text="Married" width="60" >
<t:pane id="g_13" >
<t:row id="g_14" >
<t:checkbox id="g_15" rendered=".{withPersonalDetails}" selected=".{married}" />
</t:row>
</t:pane>
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass(expressionBase="#{d.demoGriddyncbs}")

public class DemoGriddyncbs extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyRow extends FIXGRIDItem
    {
        String i_name;
        boolean i_married;
        boolean i_withChildren;
        boolean i_withPersonalDetails;
        public String getName()
        {
            return i_name;
        }
        public void setName(String name)
        {
            i_name = name;
        }
        public boolean isMarried()
        {
            return i_married;
        }
        public void setMarried(boolean married)
        {
            i_married = married;
        }
        public boolean isWithChildren()
        {
            return i_withChildren;
        }
        public void setWithChildren(boolean withChildren)
        {
            i_withChildren = withChildren;
        }
        public boolean isWithPersonalDetails()
        {
            return i_withPersonalDetails;
        }
        public void setWithPersonalDetails(boolean withPersonalDetails)
        {
            i_withPersonalDetails = withPersonalDetails;
        }
        public boolean isWithoutPersonalDetails()
        {
            return !i_withPersonalDetails;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGriddyncbs(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        MyRow r;
        r = new MyRow(); r.setName("George"); m_rows.getItems().add(r);
        r = new MyRow(); r.setName("John"); m_rows.getItems().add(r);
        r = new MyRow(); r.setName("Paul"); m_rows.getItems().add(r);
        r = new MyRow(); r.setName("Ringo"); m_rows.getItems().add(r);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
}
