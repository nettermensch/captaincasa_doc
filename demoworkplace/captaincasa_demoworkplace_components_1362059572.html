The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Invokers
  • Button with number information (1)

You may place buttons inside an OVERLAYAREA and as result may add additional information on top.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoButtonWithNumber}")

public class DemoButtonWithNumber
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_refreshItems = 1;
    int m_filterItems = 1;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoButtonWithNumber(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demobuttonwithnumber.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoButtonWithNumber}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public int getRefreshItems() { return m_refreshItems; }
    public void setRefreshItems(int value) { this.m_refreshItems = value; }

    public int getFilterItems() { return m_filterItems; }
    public void setFilterItems(int value) { this.m_filterItems = value; }

    public void onFilterAction(javax.faces.event.ActionEvent event) 
    {
        m_filterItems++;
    }

    public void onRefreshAction(javax.faces.event.ActionEvent event) 
    {
        m_refreshItems++;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
