The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Invokers
  • Button with number information (2)

This example shows the usage of OVERLAYAREA components to render label information on top of some icon component.

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoOverlayButton}")

public class DemoOverlayButton
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    public DemoOverlayButton(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demooverlaybutton.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoOverlayButton}"; }

}
