The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Label Control
  • Label flipping to editable content

The component LABELFLIP represents a label which replaces its content by a defined sub-content when clicked. The sub-content is shown until the user places the focus outside (e.g. by clicking or by navigating with the keyboard).

Please pay attention: the content components within the LABELFLIP component typically use their FLUSH attribute to synchronize their server value with the server side - so that the TEXT value behind the LABELFLIP component gets updated.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoLabelFlip}")

public class DemoLabelFlip
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_zipCode = "69245";
    String m_town = "Bammental";
    String m_street = "Hindemithweg";
    String m_firstName = "Harry";
    String m_department = "SALES";
    String m_name = "Miller";

    ValidValuesBinding m_departmentVVS = new ValidValuesBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLabelFlip(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillDepartmentVVS();
    }

    public String getPageName() { return "/workplace/demolabelflip.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLabelFlip}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getZipCode() { return m_zipCode; }
    public void setZipCode(String value) { this.m_zipCode = value; }

    public String getTown() { return m_town; }
    public void setTown(String value) { this.m_town = value; }

    public String getStreet() { return m_street; }
    public void setStreet(String value) { this.m_street = value; }

    public String getAddress() { return m_zipCode + " " + m_town + ", " + m_street; }

    public String getFullName() 
    { 
        return getFirstName() + " " + getName(); 
    }
 
    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { this.m_firstName = value; }
    
    public ValidValuesBinding getDepartmentVVS() { return m_departmentVVS; }

    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }
    public String getDepartmentText() { return m_department == null ? "Department not set" : m_departmentVVS.getValidValueByValue(m_department).getText(); }

    public String getName() { return m_name; }
    public void setName(String value) { this.m_name = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillDepartmentVVS()
    {
        m_departmentVVS.addValidValue("FIN","Financials");
        m_departmentVVS.addValidValue("HR","Human Resources");
        m_departmentVVS.addValidValue("PROD","Production");
        m_departmentVVS.addValidValue("SALES","Sales");
    }
}
