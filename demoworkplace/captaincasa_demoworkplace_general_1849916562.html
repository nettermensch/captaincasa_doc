The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Close all others on open

This example shows how to remove "all other" work pages by page that is opened in the workplace environment.

Please take a look into the Java Code for more information.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpage;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoWpRemoveOthersOnOpen}")

public class DemoWpRemoveOthersOnOpen
    extends PageBean 
    implements Serializable
{
    protected int m_numberOfRemovedWorkpages;
    public int getNumberOfRemovedWorkpages() { return m_numberOfRemovedWorkpages; }

    public DemoWpRemoveOthersOnOpen(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        removeOthers();
    }
    
    private void removeOthers()
    {
        List<IWorkpage> wps = getWorkpageContainer().getAllWorkpagesIncludingSubWorkpageContainers();
        for (IWorkpage wp: wps)
        {
            if (wp == getWorkpage()) continue; // do not remove myself...
            if (wp.isCloseSupported() == false) continue; // do not remove the ones that are fix
            getWorkpageContainer().closeWorkpageForced(wp);
            m_numberOfRemovedWorkpages++;
        }
    }

}
