The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Image in front of text

By using the attribute TEXTIMAGE you can arrange an image in front of the text.

The image can be used for explanation purpose and/or (esp. in case of COMBOFIED) for showing the image of the selected object.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFieldWithImage}")

public class DemoFieldWithImage
    extends PageBean 
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_name = "Hello";
        public String getName() { return m_name; }
        public void setName(String value) { m_name = value; }

    }

    public DemoFieldWithImage(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<10; i++)
            m_grid.getItems().add(new GridItem());
    }

    protected String m_imageName = "/images/icons/accept.png";
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public String getImageName() { return m_imageName; }

    public void onSetAddImage(ActionEvent event) 
    {
        m_imageName = "/images/icons/add.png";
    }
    
    public void onSetAcceptImage(ActionEvent event) 
    {
        m_imageName = "/images/icons/accept.png";
    }
    
    public void onClearImage(ActionEvent event) 
    {
        m_imageName = null;
    }
    
}
