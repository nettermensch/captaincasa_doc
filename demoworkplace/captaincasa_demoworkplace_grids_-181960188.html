The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid of Bean Objects

This demo shows that beans (e.g. coming from the logical layer) can directly be used within grid items. The "." notation of expression is used to navigate into the bean's properties.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.LogicFacade.Department;

@CCGenClass (expressionBase="#{d.DemoGridWithBeans}")

public class DemoGridWithBeans
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        Department i_dep;
        public GridItem(Department dep)
        {
            super();
            i_dep = dep;
        }
        public Department getDep() 
        { 
            return i_dep; 
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGridWithBeans(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        List<Department> deps = LogicFacade.getDepartments();
        for (Department dep: deps)
            m_grid.getItems().add(new GridItem(dep));
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
   
}
