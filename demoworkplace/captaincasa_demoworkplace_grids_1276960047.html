The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid in Grid
  • Grid Sequence

Press the button at the end of the grid - and sub items will be shown.

This example shows a grid in which two different object types are contained.

Technically the grid cell is a dynamic content area (ROWDYNAMICCONTENT within a PANE) that is rendered differently per object type.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_2" objectbinding="#{d.DemoGridInGrid2}" >
<t:row id="g_3" >
<t:fixgrid id="g_4" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" height="100%" objectbinding="#{d.DemoGridInGrid2.rows}" rowheight="23" sbvisibleamount="30" suppressheadline="true" width="100%;450" >
<t:gridcol id="g_5" text="Whole Row" width="100%" >
<t:pane id="g_6" >
<t:rowdynamiccontent id="g_1" contentbinding=".{dynamicContent}" />
</t:pane>
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        ROWDYNAMICCONTENTBinding m_dynamicContent = new ROWDYNAMICCONTENTBinding();
        protected PANEComponent m_pane = new PANEComponent();
        protected IRenderLine m_object;
        protected boolean m_asHeadline;
        boolean m_showCars = false;
        public MyRow(IRenderLine object)
        {
            m_object = object;
            m_object.renderLine(m_dynamicContent);
        }
        public MyRow(IRenderLine object, boolean asHeadline)
        {
            m_object = object;
            m_asHeadline = asHeadline;
            if (asHeadline == false)
                m_object.renderLine(m_dynamicContent);
            else
                m_object.renderHeadLine(m_dynamicContent);
        }
        public ROWDYNAMICCONTENTBinding getDynamicContent() { return m_dynamicContent; }
        public Object getO()
        {
            return m_object;
        }
        public void onShowCars(ActionEvent event)
        {
            if (m_showCars == false)
            {
                m_showCars = true;
                MyPerson person = (MyPerson)m_object;
                if (person.m_cars.size() == 0)
                    return;
                int counter = 1;
                int myIndex = m_rows.getItems().indexOf(this);
                MyRow mr = new MyRow(person.m_cars.get(0),true);
                m_rows.getItems().add(myIndex+counter,mr);
                counter++;
                for (int i=0; i<person.m_cars.size(); i++)
                {
                    mr = new MyRow(person.m_cars.get(i));
                    m_rows.getItems().add(myIndex+counter,mr);
                    counter++;
                }
                mr = new MyRow(m_object,true);
                m_rows.getItems().add(myIndex+counter,mr);
                counter++;
            }
            else
            {
                m_showCars = false;
                MyPerson person = (MyPerson)m_object;
                if (person.m_cars.size() == 0)
                    return;
                int myIndex = m_rows.getItems().indexOf(this);
                for (int i=0; i<(person.m_cars.size()+2); i++)
                    m_rows.getItems().remove(myIndex+1);
            }
        }
    }

    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    public void setRows(FIXGRIDListBinding<MyRow> value) { m_rows = value; }
    protected DemoGridInGrid2 m_this = this;

    public DemoGridInGrid2(IDispatcher dispatcher)
    {
        super(dispatcher);
        Random r = new Random();
        for (int i=0; i<100; i++)
        {
            MyPerson p = new MyPerson();
            p.setFirstName("FN"+i);
            p.setLastName("LN"+i);
            p.setStreet("ST"+i);
            if (i == 0)
                m_rows.getItems().add(new MyRow(p,true));
            m_rows.getItems().add(new MyRow(p));
            // create cars
            int jMax = 2 + r.nextInt(4);
            for (int j=0; j<jMax; j++)
            {
                MyCar c = new MyCar();
                c.setId("ID"+i+"/"+j);
                c.setType("TYPE"+i+"/"+j);
                p.m_cars.add(c);
            }
        }
    }
