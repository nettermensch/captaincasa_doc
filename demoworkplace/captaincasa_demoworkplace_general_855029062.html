The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Dynamic Layout
  • Individual Grid Column Layout

This grid may look a bit confusing... - but this is by intention! You see that there are two types of grid rows combined within one grid - the one with 4 fields, the other one with 2 fields.

Actually, if you have a look inside, then each grid row is a wide column pane, in which the content is rendered dynamically. The dynamic rendering is done using the ROWDYNAMICCONTENT component. This is the easiest and recommended way to do dynamic rendering. (You also can directly manipulate the JSF component tree, there are also some examples within this demo workplace, but especially in case of a grid rendering this is quite complex).

When looking inside the server side code, then you see that each grid item knows itself how it is rendered by just defining the corresponding XML definition.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_01" objectbinding="#{d.DemoGridRowDynamic}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" borderwidth="1" height="100%" objectbinding="#{d.DemoGridRowDynamic.grid}" rowheight="27" sbvisibleamount="25" width="100%" >
<t:gridcol id="g_4" width="30" >
<t:gridrowselector id="g_5" />
</t:gridcol>
<t:gridcol id="g_6" text="Column" width="100%" >
<t:pane id="g_7" >
<t:rowdynamiccontent id="g_8" contentbinding=".{content}" />
</t:pane>
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.awt.event.ActionListener;
import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridRowDynamic}")

/**
 * In this example there are two types of grid lines: persons and cars. Each 
 * item specifies its dynamic content rendering by returning the corresponding
 * content, that is defined statically.
 * <br><br>
 * In more complex scenarios you can, of course, build up an individual content
 * binding instance per line.
 */
public class DemoGridRowDynamic 
    extends WorkpageDispatchedPageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public abstract class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        public abstract ROWDYNAMICCONTENTBinding getContent();
        public void onRemove(ActionEvent event)
        {
            m_grid.getItems().remove(this);
        }
    }
    
    public class PersonGridItem extends GridItem
    {
        String m_firstName;
        String m_lastName;
        String m_street;
        String m_town;
        public ROWDYNAMICCONTENTBinding getContent() { return CB_PERSON; }
        public void setFirstName(String value) { m_firstName = value; }
        public String getFirstName() { return m_firstName; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
        public void setStreet(String value) { m_street = value; }
        public String getStreet() { return m_street; }
        public void setTown(String value) { m_town = value; }
        public String getTown() { return m_town; }
    }

    public class CarGridItem extends GridItem
    {
        String m_id;
        String m_producer;
        public ROWDYNAMICCONTENTBinding getContent() { return CB_CAR; }
        public String getId() { return m_id; }
        public void setId(String id) { m_id = id; }
        public String getProducer() { return m_producer; }
        public void setProducer(String producer) { m_producer = producer; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static String XML_PERSON = "<t:field text='.{firstName}' width='25%' height='100%' border='top:0' background='#FFFFFF80'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:field text='.{lastName}' width='25%' height='100%' border='top:0' background='#FFFFFF80'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:button actionListener='.{onRemove}' text='Rem.' width='50' height='100%' border='top:0'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:field text='.{street}' width='25%' height='100%' border='top:0' background='#FFFFFF80'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:field text='.{town}' width='25%' height='100%' border='top:0' background='#FFFFFF80'/>"; 
    static String XML_CAR = "<t:field text='.{id}' width='60%' height='100%' border='top:0' background='#FFFFFF80'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:button actionListener='.{onRemove}' text='Rem.' width='50' height='100%' border='top:0'/>"
        +"<t:colline background='#00000030' height='100%'/>"
        +"<t:field text='.{producer}' width='40%' height='100%' border='top:0' background='#FFFFFF80'/>";
    private static ROWDYNAMICCONTENTBinding CB_PERSON = new ROWDYNAMICCONTENTBinding();
    private static ROWDYNAMICCONTENTBinding CB_CAR = new ROWDYNAMICCONTENTBinding();
    
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    static
    {
        CB_PERSON.setContentXml(XML_PERSON);
        CB_CAR.setContentXml(XML_CAR);
    }
    
    public DemoGridRowDynamic(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<20;i++)
        {
            if (i%3 == 0 || i%3 == 1)
            {
                PersonGridItem gi = new PersonGridItem();
                gi.setFirstName("FN"+i);
                gi.setLastName("LN"+i);
                gi.setStreet("ST"+i);
                gi.setTown("TO"+i);
                m_grid.getItems().add(gi);
            }
            else
            {
                CarGridItem gi = new CarGridItem();
                gi.setId("ID"+i);
                gi.setProducer("PR"+i);
                m_grid.getItems().add(gi);
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

}
