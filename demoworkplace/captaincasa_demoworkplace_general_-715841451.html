The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Many images in ROWFLEXLINE

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.apache.batik.gvt.TextNode;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.IMAGENode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoFlexImageList}")

public class DemoFlexImageList
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    int m_style = 0;
    public int getStyle() { return m_style; }
    public void setStyle(int value) { this.m_style = value; }

    public void onClearAction(javax.faces.event.ActionEvent event) 
    {
        PANENode p = new PANENode();
        m_dynContent.setContentNode(p);
    }

    public void onRenderAction(javax.faces.event.ActionEvent event) 
    {
        renderContent();
    }

    int m_numberOfItems = 500;
    public int getNumberOfItems() { return m_numberOfItems; }
    public void setNumberOfItems(int value) { this.m_numberOfItems = value; }

    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();
    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFlexImageList(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        renderContent();
    }

    public String getPageName() { return "/workplace/demofleximagelist.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFlexImageList}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderContent()
    {
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (int i=0; i<m_numberOfItems; i++)
        {
            if (m_style == 0)
            {
                IMAGENode in = new IMAGENode()
                    .setImage("/images/icons/bigspeaker.png");
                l.add(in);
            }
            else
            {
                PANENode p = new PANENode()
                    .setPadding("3")
                    .setBackground("#00000010");
                {
                    ROWNode r = new ROWNode();
                    p.addSubNode(r);
                    IMAGENode in = new IMAGENode()
                        .setImage("/images/icons/bigspeaker.png");
                    r.addSubNode(in);
                }
                {
                    ROWNode r = new ROWNode();
                    p.addSubNode(r);
                    LABELNode t = new LABELNode()
                        .setText("" + i)
                        .setWidth("100%")
                        .setAlign("center");
                    r.addSubNode(t);
                }
                l.add(p);
            }
        }
        m_dynContent.setContentNodes(l);
    }
}
