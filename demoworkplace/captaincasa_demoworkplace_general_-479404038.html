The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • RISC Client Integration
  • Combination: Scanner <=> Field input

This demo shows a combination out of "scanned input" and "keyboard input".

By default the KEYBOARDSCANNER component is shown - reqeusting the focus regularly and waiting for scanner input (which is a fast sequence of keyboard events).

The user may change to manual keyboard input: in this case the KEYBOARDSCANNER component is not rendered anymore (and thus also not requesting the focus), so the user can use a field to key in the value.

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventKeySequence;
import org.eclnt.jsfserver.elements.events.BaseActionEventScan;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoKeyboardScannerWithField}")

public class DemoKeyboardScannerWithField
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_fieldValue;
    String m_scannedValue;
    boolean m_renderedScanner = true;
    boolean m_renderedField = false;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoKeyboardScannerWithField(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demokeyboardscannerwithfield.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoKeyboardScannerWithField}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getFieldValue() { return m_fieldValue; }
    public void setFieldValue(String value) { this.m_fieldValue = value; }

    public void onScanAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventScan)
        {
            BaseActionEventScan e = (BaseActionEventScan)event;
            m_scannedValue = e.getScannedValue();
        }
    }

    public String getScannedValue() { return m_scannedValue; }
    public void setScannedValue(String value) { this.m_scannedValue = value; }

    public void onFieldCancelAction(javax.faces.event.ActionEvent event) 
    {
        m_renderedScanner = true;
        m_renderedField = false;
        m_fieldValue = null;
    }

    public void onFieldOKAction(javax.faces.event.ActionEvent event) 
    {
        m_renderedScanner = true;
        m_renderedField = false;
        m_scannedValue = m_fieldValue;
        m_fieldValue = null;
    }

    public void onManualInputAction(javax.faces.event.ActionEvent event) 
    {
        m_renderedScanner = false;
        m_renderedField = true;
    }

    public boolean getRenderedScanner() { return m_renderedScanner; }
    public boolean getRenderedField() { return m_renderedField; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
