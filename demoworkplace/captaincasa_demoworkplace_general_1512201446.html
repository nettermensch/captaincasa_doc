The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Animation during opening

There are two types of animations that can be applied to a popup dialog when opening up:

1. the "size animation" is the most visible one: this one controls how the dialog finds its position and size, e.g. you may define some rotation or scaling.

2. the "content animation" is some animation effect that is executed on top of the content are of the dialog: here you have different animation types, e.g. you may define some shading that disappears into a certain direction.

Please pay attention: there is a global definition (CLIENTCONFIG-ANIMATE and/or URL-client-parameter "?ccanimate=false") that you might use to switch off all animations. This switch overrides individual definitions, that you apply for popup dialogs.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.elements.util.DefaultModalPopupListener;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemopopupsizeanimationUI}")

public class DemoPopupSizeAnimation
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_contentAnimationType;
    String m_sizeAnimationType;
    ValidValuesBinding m_contentAnimationTypeVVS = new ValidValuesBinding();
    ValidValuesBinding m_sizeAnimationTypeVVS = new ValidValuesBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupSizeAnimation(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_sizeAnimationTypeVVS.addValidValue("scalein","scalein");
        m_sizeAnimationTypeVVS.addValidValue("topdown","topdown");
        m_sizeAnimationTypeVVS.addValidValue("bottomup","bottomup");
        m_sizeAnimationTypeVVS.addValidValue("leftright","leftright");
        m_sizeAnimationTypeVVS.addValidValue("rightleft","rightleft");
        m_sizeAnimationTypeVVS.addValidValue("rotatehorizontally","rotatehorizontally");
        m_sizeAnimationTypeVVS.addValidValue("rotatevertically","rotatevertically");
        m_sizeAnimationTypeVVS.addValidValue("noanimation","noanimation");
        m_contentAnimationTypeVVS.addValidValue("hidetotop","hidetotop");
        m_contentAnimationTypeVVS.addValidValue("hidetobottom","hidetobottom");
        m_contentAnimationTypeVVS.addValidValue("hidetoleft","hidetoleft");
        m_contentAnimationTypeVVS.addValidValue("hidetoright","hidetoright");
        m_contentAnimationTypeVVS.addValidValue("shrinktocenter","shrinktocenter");
        m_contentAnimationTypeVVS.addValidValue("foglight","foglight");
        m_contentAnimationTypeVVS.addValidValue("fogdark","fogdark");
        m_contentAnimationTypeVVS.addValidValue("fogred","fogred");
        m_contentAnimationTypeVVS.addValidValue("shake","shake");
        m_contentAnimationTypeVVS.addValidValue("noanimation","noanimation");
    }

    public String getPageName() { return "/workplace/demopopupsizeanimation.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemopopupsizeanimationUI}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getContentAnimationTypeVVS() { return m_contentAnimationTypeVVS; }
    public ValidValuesBinding getSizeAnimationTypeVVS() { return m_sizeAnimationTypeVVS; }

    public String getContentAnimationType() { return m_contentAnimationType; }
    public void setContentAnimationType(String value) { this.m_contentAnimationType = value; }

    public String getSizeAnimationType() { return m_sizeAnimationType; }
    public void setSizeAnimationType(String value) { this.m_sizeAnimationType = value; }

    public void onOpenPopupAction(javax.faces.event.ActionEvent event) 
    {
        final DemoHelloWorld ui = new DemoHelloWorld(getOwningDispatcher());
        ModalPopup p = openModalPopup(ui,"Hello world",600,400,new DefaultModalPopupListener(this,ui));
        p.setAnimationtype(m_contentAnimationType);
        p.setSizeAnimationtype(m_sizeAnimationType);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
