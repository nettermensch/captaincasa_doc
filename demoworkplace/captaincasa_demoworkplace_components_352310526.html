The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Multi Value Selection (PBC)

The page bean component CCMultiValueSelection renders a grid of values, from which the user can select/deselect by using checkboxes.

There is a popup menu (right mouse button) providing functions to clear the selection or to select all items.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;

import org.eclnt.ccaddons.pbc.CCMultiValueSelection;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.session.RequestFocusManager;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCMultiValueSelection}")

public class DemoPBCCCMultiValueSelection
    extends PageBean 
    implements Serializable
{
    boolean m_selecteValuesAtTop;
    public boolean getSelecteValuesAtTop() { return m_selecteValuesAtTop; }
    public void setSelecteValuesAtTop(boolean value) { this.m_selecteValuesAtTop = value; }

    CCMultiValueSelection m_mvs = new CCMultiValueSelection();
    
    // ------------------------------------------------------------------------
    // constructor
    // ------------------------------------------------------------------------

    public DemoPBCCCMultiValueSelection(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_buildingMap.put("0001","Bammental");
        m_buildingMap.put("0002","Bayrischzell");
        m_buildingMap.put("0003","Sofia");
        m_buildingMap.put("0004","Karlsruhe");
        m_buildingMap.put("0005","Mï¿½nchen");
        m_buildingMap.put("0006","Stuttgart");
        m_buildingMap.put("0007","Heidelberg");
        m_buildingMap.put("0008","Mannheim");
        m_buildingMap.put("0009","Lingental");
        m_buildingMap.put("0010","Peking");
        fillSelection(m_mvs);
        m_mvs.prepare(new CCMultiValueSelection.IListener()
        {
            @Override
            public void reactOnSelectionUpdate(Set<String> selectedIds)
            {
                m_selectedIds = selectedIds;
            }
        });
        m_mvs.setAvailableButtonBar(true);
        m_mvs.setAvailableFilterBar(true);
    }

    public String getPageName() { return "/workplace/demopbcccmultivalueselection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCMultiValueSelection}"; }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    Map<String,String> m_buildingMap = new HashMap();
    Set<String> m_selectedIds = new HashSet<String>();
    
    public String getBuildings() 
    { 
        List<String> values = new ArrayList<String>();
        for (String key: m_selectedIds)
        {
            String text = m_buildingMap.get(key);
            values.add(text);
        }
        return ValueManager.encodeCSV(values);
    }

    public void onBuildingsAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            openValueSelection();
        }
    }

    public CCMultiValueSelection getMvs() { return m_mvs; }

    private void openValueSelection()
    {
        final CCMultiValueSelection mvs = CCMultiValueSelection.openAsPopup(this,250,300);
        fillSelection(mvs);
        mvs.setAvailableFilterBar(true);
        mvs.setFilterFieldRequestFocus(RequestFocusManager.REQFOCUS_CREATION_LONG);
        mvs.prepare(new CCMultiValueSelection.IListener()
        {
            @Override
            public void reactOnSelectionUpdate(Set<String> selectedIds)
            {
                m_selectedIds = selectedIds;
            }
        });
    }
    
    private void fillSelection(CCMultiValueSelection mvs)
    {
        List<String> keys = new ArrayList<>(m_buildingMap.keySet());
        Collections.sort(keys);
        for (String key: keys)
            mvs.addValue(key,m_buildingMap.get(key));
        mvs.setSelectedIds(m_selectedIds);
    }

}
