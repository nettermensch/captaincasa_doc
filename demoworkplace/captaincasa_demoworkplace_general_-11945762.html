The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Multiple workplaces
  • Isolate parts of a page

This example shows how to isolate a certain page bean of an existing screen and start it in some own browser window.

The detail window is connected to the starting window by using the "DialogMessageBus" with its feature of cross client messaging. The detail screen gets notified when the user selected a person from the list - and vice versa the list screen gets notified when the detail is closed.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.messages.CrossDialogMessage;
import org.eclnt.jsfserver.messages.DefaultDialogMessageBusListener;
import org.eclnt.jsfserver.messages.DialogMessageBus;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.MultiWorkplaceConnector;
import org.eclnt.workplace.MultiWorkplaceConnector.WorkplaceConfiguration;
import org.eclnt.workplace.WorkpageStartInfo;
import org.eclnt.workplace.eximport.WorkplacePageBeanStateExtractor;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoWorkplaceIsolateDetail}")

public class DemoWorkplaceIsolateDetail
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    public static final String PARAM_NOTIFYONCLOSE = "notifyOnClose_" + DemoWorkplaceIsolateDetail.class.getName();
    
    public static final String MESSAGETYPE_DETAILCLOSED = "detailClosed_" + DemoWorkplaceIsolateDetail.class.getName();
    public static final String MESSAGETYPE_UPDATEDETAIL = "updateDetail_" + DemoWorkplaceIsolateDetail.class.getName();
    
    CCDataGridView2<Person> m_dataGrid = new CCDataGridView2<Person>(Person.class);
    DemoWorkplaceIsolateDetailDetail m_detailUI;
    boolean m_detailIsIsolated = false;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoWorkplaceIsolateDetail(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        List<Person> ps = new ArrayList<Person>(RandomLogic.getAllPersons().values());
        m_dataGrid.prepare(ps,new CCDataGridView2.DefaultListener<Person>() 
        {
            @Override
            public void reactOnGridSelect(Person object)
            {
                showPersonDetail(object.getId());
            }
        });
        m_detailUI = new DemoWorkplaceIsolateDetailDetail(workpageDispatcher);
        DialogMessageBus.instance().addListener(new DefaultDialogMessageBusListener()
        {
            @Override
            public void reactOnMessage(Object o)
            {
                if (CrossDialogMessage.checkIfMessageMatches(o,MESSAGETYPE_DETAILCLOSED))
                {
                    m_detailIsIsolated = false;
                }
            }
        });
    }

    public String getPageName() { return "/workplace/demoworkplaceisolatedetail.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoWorkplaceIsolateDetail}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2 getDataGrid() { return m_dataGrid; }
    public DemoWorkplaceIsolateDetailDetail getDetailUI() { return m_detailUI; }

    public boolean getRenderedDetail() 
    { 
        return m_dataGrid.getSelectedObject() != null && m_detailIsIsolated == false; 
    }
    
    public void onIsolateAction(javax.faces.event.ActionEvent event) 
    {
        WorkpageStartInfo wpsi = WorkplacePageBeanStateExtractor.instance().extractWorkpageStartInfo(m_detailUI);
        wpsi.setText("Person Detail");
        wpsi.setCloseSupported(false);
        wpsi.setParam(PARAM_NOTIFYONCLOSE,"true");
        wpsi.setDecorated(false);
        WorkplaceConfiguration wpconfig = new WorkplaceConfiguration();
        wpconfig.setWorkplaceURL("/workplace.workplaceMWPBlank.risc");
        wpconfig.setWindowFeatures("left=250,top=250,width=500,height=800,toolbar=no,menubar=no,resizable=yes");
        MultiWorkplaceConnector.instance().startWorkpage
        (
            Dispatcher.getDialogSessionInstance(),
            "PERSONDETAIL",
            wpconfig,
            wpsi, 
            true
        );
        m_detailIsIsolated = true;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void showPersonDetail(String personId)
    {
        CrossDialogMessage cdm = new CrossDialogMessage(MESSAGETYPE_UPDATEDETAIL,personId);
        DialogMessageBus.instance().publishMessageToAllClientDialogs(cdm);
    }
}
