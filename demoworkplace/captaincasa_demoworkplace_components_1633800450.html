The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Touch Screen
  • Field with touch popup

The normal FIELD and FORMATTEDFIELD component provides touch support as well. When defining attribute TOUCHSUPPORT as true, then a corresponding touch pad is opended when the user clicks onto the field component.

The window that is popped up can either be closed by explicitly pressing the OK button, or by clicking anywhere else onto the screen.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.jsfserver.util.IClientLogAware;

@CCGenClass (expressionBase="#{d.DemoTouchOnField}")

public class DemoTouchOnField 
    extends PageBean
    implements IClientLogAware
{
    protected String m_comment = "Some comment...";
    public String getComment() { return m_comment; }
    public void setComment(String value) { m_comment = value; }

    protected String m_password;
    public String getPassword() { return m_password; }
    public void setPassword(String value) { m_password = value; }

    public DemoTouchOnField(IDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("field");
    }

    protected float m_weight = 80.5f;
    public float getWeight() { return m_weight; }
    public void setWeight(float value) { m_weight = value; }

    protected int m_shoeSize = 40;
    public int getShoeSize() { return m_shoeSize; }
    public void setShoeSize(int value) { m_shoeSize = value; }

    protected String m_firstName  = "Captain";
    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { m_firstName = value; }
    
    protected String m_lastName  = "Casa";
    public void setLastName(String value) { m_lastName = value; }
    public String getLastName() { return m_lastName; }
    
    public void onOK(ActionEvent event)
    {
        String result = m_firstName + "/" +
                        m_lastName + "/" +
                        m_shoeSize + "/" +
                        m_password;
        Statusbar.outputSuccess(result);
    }
    
    public boolean onCreateClientTestLogMessage()
    {
        String result = m_firstName + "/" +
                        m_lastName + "/" +
                        m_shoeSize + "/" +
                        m_password;
        ClientTestLog.addClientTestLogMessage(result);
        return true;
    }
}
