The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Flexible row heights
  • Individual row heights in tree

Individual row height sizing can also be applied to trees, of course.

In the example the font of the tree nodes is individually set for each level. As consequence the height of the corresponding rows is different.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoTreeDynRowHeight}" />
<t:rowdemobodypane id="g_2" objectbinding="#{d.DemoTreeDynRowHeight}" >
<t:row id="g_3" >
<t:fixgrid id="g_4" border="#00000010" drawoddevenrows="false" dynamicheightsizing="true" height="100%" objectbinding="#{d.DemoTreeDynRowHeight.tree}" suppressheadline="true" width="100%" >
<t:gridcol id="g_5" text="Column" width="100%;200" >
<t:treenode id="g_6" font=".{font}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTreeDynRowHeight}")

public class DemoTreeDynRowHeight
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        String i_font;
        public TreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        public String getFont() { return i_font; }
        public void setFont(String value) { this.i_font = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        public void onToggle()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTreeDynRowHeight(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        buildTree();
    }

    public String getPageName() { return "/workplace/demotreedynrowheight.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTreeDynRowHeight}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDTreeBinding<TreeNode> getTree() { return m_tree; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildTree()
    {
        for (int i=0; i<10; i++)
        {
            TreeNode top = new TreeNode(m_tree.getRootNode());
            top.setText("Top chapter" + i);
            top.setFont("size:24;weight:bold");
            if (i != 0)
                top.setStatus(TreeNode.STATUS_CLOSED);
            else
                top.setStatus(TreeNode.STATUS_OPENED);
            for (int j=0; j<4; j++)
            {
                TreeNode sub = new TreeNode(top);
                sub.setText("Sub chapter " + i);
                sub.setFont("size:18");
                sub.setStatus(TreeNode.STATUS_CLOSED);
                if (j != 0)
                    sub.setStatus(TreeNode.STATUS_CLOSED);
                else
                    sub.setStatus(TreeNode.STATUS_OPENED);
                for (int k=0; k<3; k++)
                {
                    TreeNode subsub = new TreeNode(sub);
                    subsub.setText("Detail chapter " + i);
                    subsub.setFont(null);
                    subsub.setStatus(TreeNode.STATUS_ENDNODE);
                }
            }
        }
    }

}
