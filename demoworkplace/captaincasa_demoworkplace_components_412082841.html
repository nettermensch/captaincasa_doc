The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Line drawing
  • Line arrows

You may define arrows at three different positions of a line:

1. start ("from arrow")
2. end ("to arrow")
3. centered ("central")

There are a couple of pre-defined arrow types - each one represented by a number.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREALINEITEMNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPaintareaLineTypes}")

public class DemoPaintareaLineTypes
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPaintareaLineTypes(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        render();
    }
    
    public String getPageName() { return "/workplace/demopaintarealinetypes.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPaintareaLineTypes}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void render()
    {
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (int i=0; i<26; i++)
        {
            int y = i * 40 + 20;
            {
                PAINTAREALINEITEMNode ln = new PAINTAREALINEITEMNode()
                    .setBounds("20;"+y+";100;0")
                    .setArrowfrom(i)
                    .setText(i+"");
                l.add(ln);
            }
            {
                PAINTAREALINEITEMNode ln = new PAINTAREALINEITEMNode()
                    .setBounds("170;"+y+";100;0")
                    .setArrowto(i)
                    .setText(i+"");
                l.add(ln);
            }
            {
                PAINTAREALINEITEMNode ln = new PAINTAREALINEITEMNode()
                    .setBounds("320;"+y+";100;0")
                    .setArrowcentral(i)
                    .setText(i+"");
                l.add(ln);
            }
        }
        m_dynContent.setContentNodes(l);
    }
}
