The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Hot Keys
  • Test if hot key is available

With this page you can test the availability of hot key combinations within your browser environment.

Within the page buttons are generated for the followin hot keys:

(.) "ctrl-" a-z,0-9,F1-F12
(.) "ctrl-shift" a-z,0-9,F1-F12
(.) "ctrl" a-z,0-9,F1-F12
(.) "ctrl-alt" a-z,0-9,F1-F12

The button's HOTKEY attribute is set correspondingly.

Whenever the button reacts on the hotkey combination then a corresponding message is output in the status bar. If there is no output and/or some behavior that comes from the browser processing then the hot key is not usable.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Calendar;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.util.valuemgmt.IActionListenerDelegation;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.util.valuemgmt.ValueManager.HotKeyInfo;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoAllHotkeys}")

public class DemoAllHotkeys
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoAllHotkeys(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        createDynContent();
    }

    public String getPageName() { return "/workplace/demoallhotkeys.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoAllHotkeys}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void createDynContent()
    {
        List<int[]> ranges = new ArrayList<int[]>();
        ranges.add(new int[] {65,90});
        ranges.add(new int[] {48,57});
        ranges.add(new int[] {112,123});
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (int[] range: ranges)
        {
            for (int i=range[0]; i<=range[1]; i++)
            {
                {
                    HotKeyInfo hki = new HotKeyInfo();
                    hki.m_keyCode = i;
                    hki.m_isControlDown = true;
                    BUTTONNode b = createButton(hki);
                    l.add(b);
                }
                {
                    HotKeyInfo hki = new HotKeyInfo();
                    hki.m_keyCode = i;
                    hki.m_isAltDown = true;
                    BUTTONNode b = createButton(hki);
                    l.add(b);
                }
                {
                    HotKeyInfo hki = new HotKeyInfo();
                    hki.m_keyCode = i;
                    hki.m_isControlDown = true;
                    hki.m_isAltDown = true;
                    BUTTONNode b = createButton(hki);
                    l.add(b);
                }
                {
                    HotKeyInfo hki = new HotKeyInfo();
                    hki.m_keyCode = i;
                    hki.m_isControlDown = true;
                    hki.m_isShiftDown = true;
                    BUTTONNode b = createButton(hki);
                    l.add(b);
                }
            }
        }
        m_dynContent.setContentNodes(l);
    }

    private BUTTONNode createButton(HotKeyInfo hki)
    {
        String hotkey = ""+hki.m_keyCode;
        if (hki.m_isAltDown == true) hotkey = "alt-" + hotkey;
        if (hki.m_isShiftDown == true) hotkey = "shift-" + hotkey;
        if (hki.m_isControlDown == true) hotkey = "ctrl-" + hotkey;
        final String text = ValueManager.createHotKeyString(hki);
        BUTTONNode b = new BUTTONNode()
                .setText(text)
                .setWidth("150+")
                .setHotkey(hotkey)
                .bindActionListener(new IActionListenerDelegation()
                {
                    @Override
                    public void onAction(ActionEvent event)
                    {
                        Statusbar.outputSuccess("Hotkey pressed: " + text + ", Stamp: " + System.currentTimeMillis());
                    }
                });
        return b;
    }
}
