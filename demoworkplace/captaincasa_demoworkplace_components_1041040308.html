The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Organization Tree

This demo shows the usage of PAINTAREA, PAINTAREAITEM and PAINTAREALINEITEM to represent a hierarchy of some organization.

Click onto the red items in order to open up / collapse the corresponding part of the diagram.

The PAINTAREA content is dynamically defined using a DYNAMICCONTENT component.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREALINEITEMNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPaintAreaOrganization}")

public class DemoPaintAreaOrganization
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public static class OrgNode
    {
        List<OrgNode> i_children = new ArrayList<OrgNode>();
        String i_text;
        int i_level = 0;
        boolean i_collapsed = false;
        public OrgNode(String text)
        {
            i_text = text;
        }
        public void addChild(OrgNode child)
        {
            child.i_level = i_level + 1;
            i_children.add(child);
        }
    }
    
    public class OrgNodeItem
    {
        OrgNode i_orgNode;
        int i_x;
        int i_y;
        ComponentNode i_componentNode;
        List<ComponentNode> i_lineComponentNodes = new ArrayList<ComponentNode>();
        public OrgNodeItem(OrgNode orgNode)
        {
            i_orgNode = orgNode;
        }
        public void onAction(ActionEvent event)
        {
            Statusbar.outputMessage(i_orgNode.i_text);
            i_orgNode.i_collapsed = !i_orgNode.i_collapsed;
            render();
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    DYNAMICCONTENTBinding m_content = new DYNAMICCONTENTBinding();
    OrgNode m_topNode;
    List<OrgNodeItem> m_onis = new ArrayList<OrgNodeItem>(); 
    Trigger m_renderTrigger = new Trigger();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPaintAreaOrganization(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        Random r = new Random();
        m_topNode = new OrgNode("Company xyz");
        int maxi = 5 + r.nextInt(8);
        for (int i=0; i<maxi; i++)
        {
            OrgNode locationNode = new OrgNode("Location " + i);
            m_topNode.addChild(locationNode);
            int maxj = 3 + r.nextInt(10);
            for (int j=0; j<maxj; j++)
            {
                OrgNode depNode = new OrgNode("Department " + j);
                locationNode.addChild(depNode);
            }
            locationNode.i_collapsed = true;
        }
        render();
    }

    public String getPageName() { return "/workplace/demopaintareaorganization.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPaintAreaOrganization}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getContent() { return m_content; }
    public List<OrgNodeItem> getOnis() { return m_onis; }
    public Trigger getRenderTrigger() { return m_renderTrigger; }

    private void render()
    {
        m_onis.clear();
        renderOrgNode(m_topNode,0);
        List<ComponentNode> cns = new ArrayList<ComponentNode>();
        for (OrgNodeItem oni: m_onis)
        {
            cns.add(oni.i_componentNode);
            for (ComponentNode lcn: oni.i_lineComponentNodes)
                cns.add(lcn);
        }
        m_content.setContentNodes(cns);
        m_renderTrigger.trigger();
    }
    
    private class RenderResponse
    {
        int currentY;
        OrgNodeItem oni;
        public RenderResponse(int currentY, OrgNodeItem oni)
        {
            super();
            this.currentY = currentY;
            this.oni = oni;
        }
    }
    
    private RenderResponse renderOrgNode(OrgNode orgNode, int currentY)
    {
        int onisIndex = m_onis.size();
        OrgNodeItem oni = new OrgNodeItem(orgNode);
        String color = "#D0D0D0";
        if (orgNode.i_collapsed == true)
            color = "#FFC0C0";
        oni.i_x = 180*orgNode.i_level;
        oni.i_y = currentY;
        PAINTAREAITEMNode cn = new PAINTAREAITEMNode()
            .setBounds(oni.i_x+";"+oni.i_y+";100;30")
            .setBackground(color)
            .setBorder("#00000030");
        if (orgNode.i_children.size() > 0)
        {
            cn.setInvokeevent("leftclick")
              .setActionListener(pbx("onis["+onisIndex+"].onAction"));
        }
        oni.i_componentNode = cn;
        String text = orgNode.i_text;
        if (orgNode.i_collapsed == true) text = text + " (+" + orgNode.i_children.size()+")";
        LABELNode ln = new LABELNode()
            .setText(text)
            .setAlign("center");
        cn.addSubNode(ln);
        m_onis.add(oni);
        if (orgNode.i_children.size() == 0 || orgNode.i_collapsed == true)
            currentY += 35;
        if (orgNode.i_collapsed == false)
        {
            int counter = 0;
            for (OrgNode childOrgNode: orgNode.i_children)
            {
                RenderResponse rr = renderOrgNode(childOrgNode,currentY);
                currentY = rr.currentY;
                int fromX = (oni.i_x+100)+4;
                int fromY =  (oni.i_y+15);
                int toY = (rr.oni.i_y+15) - fromY;
                renderLine(fromX,fromY,70,toY,oni);
                counter++;
            }
        }
        return new RenderResponse(currentY,oni);
    }
    
    private void renderLine(int fromX, int fromY, int toX, int toY, OrgNodeItem oni)
    {
        int midX = toX/2;
        {
            PAINTAREALINEITEMNode lcn = new PAINTAREALINEITEMNode()
                .setBounds(fromX+";"+fromY+";"+midX+";0")
                .setLinecolor("#808080");
            oni.i_lineComponentNodes.add(lcn);
        }
        {
            PAINTAREALINEITEMNode lcn = new PAINTAREALINEITEMNode()
                .setBounds((fromX+midX)+";"+fromY+";0;"+toY)
                .setLinecolor("#808080");
            oni.i_lineComponentNodes.add(lcn);
        }
        {
            PAINTAREALINEITEMNode lcn = new PAINTAREALINEITEMNode()
                .setBounds((fromX+midX)+";"+(fromY+toY)+";"+(toX-midX)+";0")
                .setArrowto(1)
                .setLinecolor("#808080");
            oni.i_lineComponentNodes.add(lcn);
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
