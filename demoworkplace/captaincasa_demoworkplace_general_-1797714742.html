The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Opacity of Popups

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoPopupOpacity}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.OKPopup;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoPopupOpacity}")

public class DemoPopupOpacity
    extends PageBean 
    implements Serializable
{    
    public DemoPopupOpacity(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public void onOpacity90(ActionEvent event) 
    {
        openPopup(90);
    }

    public void onOpacity75(ActionEvent event) 
    {
        openPopup(75);
    }

    public void onOpacity50(ActionEvent event) 
    {
        openPopup(50);
    }

    public void openPopup(int opacity) 
    {
        ModalPopup mp = OKPopup.createInstance("Test Opacity","This popup is rendered with an opacity of "+opacity+".").getModalPopup();
        mp.setOpacity(opacity);
        mp.setUndecorated(true);
    }

}
