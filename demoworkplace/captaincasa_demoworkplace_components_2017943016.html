The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Enabled/ Disabled

Use the attribute ENABLED in order to activate/deactivate a component.

Note: there is an attribute RENDERED as well, that decides if a component is drawn at all.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.managedbean.IDispatcher;

@CCGenClass (expressionBase="#{d.DemoEnabled}")

public class DemoEnabled
    extends PageBean
    implements Serializable
{
    public DemoEnabled(IDispatcher dispatcher)
    {
        super(dispatcher);
    }
    
    public void onAction(ActionEvent event) 
    {
        Statusbar.outputSuccess("Action was invoked");
    }

    protected boolean m_enabled;
    public boolean getEnabled() { return m_enabled; }
    public void setEnabled(boolean value) { m_enabled = value; }

    protected String m_value = "1000";
    public String getValue() { return m_value; }
    public void setValue(String value) { m_value = value; }

    public void onChange(ActionEvent event) 
    {
        m_enabled = !m_enabled;
    }
    
    public void onComboFieldAction(ActionEvent event)
    {
        IdTextSelection idts = IdTextSelection.createInstance();
        for (int i=0; i<100; i++)
            idts.addLine("ID_"+i,"VALUE_"+i);
        idts.setCallBack(new ISetId() { public void setId(String id) { m_value = id; }});
    }

}
