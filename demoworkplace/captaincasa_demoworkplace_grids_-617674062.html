The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid data export
  • Export - default or own

There is a default dialog that opens up when calling the FIXGRIDBinding-method "onOpenGridExportFunctions".

You also can embed the export directly into your own processing:

(.) FIXGRIDBinding.getExporter() provides the methods to create the data
(.) The data is passed into the TempFileManager so that it is reach-able by URL.
(.) The URL together with a Trigger-property is bound by a FILEDOWNLOAD component

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.ServletUtil;
import org.eclnt.jsfserver.util.tempfile.TempFileManager;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoGridExportExcel}")

public class DemoGridExportExcel
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String lastName) { i_lastName = lastName; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String firstName) { i_firstName = firstName; }
        public void onRowSelect() { }
        public void onRowExecute() { }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Trigger m_downloadTrigger = new Trigger();
    String m_downloadURL;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridExportExcel(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);     
        fillGrid();
    }

    public String getPageName() { return "/workplace/demogridexportexcel.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridExportExcel}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public Trigger getDownloadTrigger() { return m_downloadTrigger; }

    public String getDownloadURL() { return m_downloadURL; }
    public void setDownloadURL(String value) { this.m_downloadURL = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public void onDownloadExcelAction(javax.faces.event.ActionEvent event) 
    {
        byte[] bytes = m_grid.getExporter().exportToXLSXWholeGrid(false);
        String contentType = "application/octet-stream"; // browser will/may download
        m_downloadURL = TempFileManager.saveTempFile("gridExport.xlsx",contentType,bytes);
        m_downloadTrigger.trigger();
    }

    public void onDownloadPdfAction(javax.faces.event.ActionEvent event) 
    {
        byte[] bytes = m_grid.getExporter().exportToPDFWholeGrid(false);
        String contentType = "application/octet-stream"; // browser will/may download
        m_downloadURL = TempFileManager.saveTempFile("gridExport.pdf",contentType,bytes);
        m_downloadTrigger.trigger();
    }

    public void onShowPdfAction(javax.faces.event.ActionEvent event) 
    {
        byte[] bytes = m_grid.getExporter().exportToPDFWholeGrid(false);
        String contentType = "application/pdf"; // browser will show URL and not download
        m_downloadURL = TempFileManager.saveTempFile("gridExport.pdf",contentType,bytes);
        m_downloadTrigger.trigger();
    }

    public void onDownloadFinished(javax.faces.event.ActionEvent event) 
    {
        Statusbar.outputSuccessWithPopup("Download was successfully performed.");
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName(RandomLogic.getFirstName());
            gi.setLastName(RandomLogic.getLastName());
            m_grid.getItems().add(gi);
        }
    }

}
