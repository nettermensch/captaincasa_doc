The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Header and Footer

A grid typically has one header row - the one containing the grid columns' titles. But: you can add any number of additional header and footer rows.

In the example there is one header row that allows the user to search through the data: type some search term into the search fields and wait a bit: a server side search will be automatically triggered. Background: the search fields contain two attributes: FLUSH and FLUSHTIMER - after 1000ms of non-activity the changed data will be flushed to the server.

And there is a header row to show the totals of the last column.

Header and footer rows can be switched on/off by just using the normal RENDERED attribute provided by each component.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridHeaderFooter}")

public class DemoGridHeaderFooter
    extends PageBean
    implements Serializable
{
    public class BO implements Serializable
    {
        public BO(String firstName, String lastName, int fee)
        {
            m_firstName = firstName;
            m_lastName = lastName;
            m_fee = fee;
        }
        String m_firstName;
        String m_lastName;
        int m_fee;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String firstName) { m_firstName = firstName; }
        public String getLastName() { return m_lastName; }
        public void setLastName(String lastName) { m_lastName = lastName; }
        public int getFee() { return m_fee; }
        public void setFee(int fee) { m_fee = fee; }
    }
    
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected BO m_bo;
        public MyRow(BO bo)
        {
            m_bo = bo;
        }
        public BO getBo() { return m_bo; }

    }

    protected int m_totalFee;
    protected String m_searchLastName = "";
    protected String m_searchFirstName = "";
    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    protected BO[] BOS = new BO[]
    {
        new BO("Adrian","Miller",50),        
        new BO("Harald","Smith",30),        
        new BO("Johnny","Carpenter",40),        
        new BO("Claudia","Pecunia",30),        
        new BO("Caroline","Mocca",40),        
        new BO("Thomas","Thomassi",30),        
        new BO("Gary","Cooper",40),        
        new BO("Joseph","Conqueror",30),        
        new BO("Michael","Johnson",40),        
        new BO("Robin","Vista",30),        
        new BO("Jane","Carter",40),        
        new BO("Hillary","Bush",30),        
        new BO("Mats","Wilander",40),        
        new BO("Niki","Lauderdale",30),        
    };
    protected boolean m_headerRendered = true;
    protected boolean m_footerRendered = true;

    public DemoGridHeaderFooter(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        fillList();
    }
    
    public boolean getFooterRendered() { return m_footerRendered; }
    public void setFooterRendered(boolean value) { m_footerRendered = value; }
    
    public boolean getHeaderRendered() { return m_headerRendered; }
    public void setHeaderRendered(boolean value) { m_headerRendered = value; }

    public int getTotalFee() { return m_totalFee; }
    public void setTotalFee(int value) { m_totalFee = value; }

    public String getSearchLastName() { return m_searchLastName; }
    public void setSearchLastName(String value) { m_searchLastName = value; }

    public String getSearchFirstName() { return m_searchFirstName; }
    public void setSearchFirstName(String value) { m_searchFirstName = value; }

    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    public void setRows(FIXGRIDListBinding<MyRow> value) { m_rows = value; }

    public void onSearch(ActionEvent event) 
    {
        fillList();
    }
    
    private void fillList()
    {
        m_rows.getItems().clear();
        m_totalFee = 0;
        for (BO bo: BOS) 
        {
            if (bo.getFirstName().toLowerCase().startsWith(m_searchFirstName.toLowerCase()) && 
                bo.getLastName().toLowerCase().startsWith(m_searchLastName.toLowerCase()) )
            {
                m_rows.getItems().add(new MyRow(bo));
                m_totalFee += bo.getFee();
            }
        }
    }
}
