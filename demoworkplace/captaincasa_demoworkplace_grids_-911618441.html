The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Usage variant management
  • Usage variants with "Grids and Trees"

By adding a simple component (USAGEVARIANTBUTTON) from the addon-package, you add a "configuration variant management" to the FIXGRID processing.

The variant management stores all grid configuration data (column sequence, width, sorting) for a user, so that the user can pick up the configuration later on.

The user can define multiple variants, each variant keeping one configuration. One variant can be selected as default variant which is executed when starting the grid.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;

import javax.faces.event.ActionEvent;

import org.eclnt.ccaddons.pbc.usagevariants.CCUsageVariantButton;
import org.eclnt.ccee.usagevariants.UsageVariantManager;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.FIXGRIDExtractApply;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCUsageVariant}")

public class DemoPBCCCUsageVariant
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem
    {
        String i_firstName;
        String i_lastName;
        boolean i_married;
        Date i_birthday;
        // --------------------------------------------------------------------
        public void setFirstName(String value) { i_firstName = value; }
        public String getFirstName() { return i_firstName; }
        public void setLastName(String value) { i_lastName = value; }
        public String getLastName() { return i_lastName; }
        public void setMarried(boolean value) { i_married = value; }
        public boolean getMarried() { return i_married; }
        public Date getBirthday() { return i_birthday; }
        public void setBirthday(Date birthday) { i_birthday = birthday; }
        // --------------------------------------------------------------------
        public void onRemove(ActionEvent ae)
        {
            YESNOPopup.createInstance
            (
                "Removing item",
                "Do you really want to remove the selected item?",
                new YESNOPopup.IYesNoListener()
                {
                    public void reactOnYes()
                    {
                        removeExecute();
                    }
                    public void reactOnNo()
                    {
                    }
                }
            );
        }
        private void removeExecute()
        {
            m_grid.getItems().remove(this);
        }
        @Override
        public void onRowExecute()
        {
            Statusbar.outputSuccess("Item was executed: " + getLastName() + "/" + m_grid.getColumnIndexOfLastSelection() + "/" + m_grid.getShiftPressedOfLastSelection() + "/" + m_grid.getCtrlPressedOfLastSelection());
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    boolean m_oddeven = false;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    CCUsageVariantButton m_usageVariantButton = new CCUsageVariantButton();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoPBCCCUsageVariant(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        fillGrid();
        // Usage variant
        // 1. create instance of UsageVariantManager, pass:
        //    a. type of configured issue (any String, your definition)
        //    b. concrete id of configured issue (any String, your definition)
        //    c. implementation of IExtractApply
        // 2. activate - pass "true" as parameter, if the default variant should be immediately applied
        // 3. pass information to CCUsageVariantButton instance that is referenced from the page
        UsageVariantManager uvm = new UsageVariantManager("FIXGRID","DemoPBCCCUsagetVariant",new FIXGRIDExtractApply(m_grid));
        uvm.activate(true);
        m_usageVariantButton.prepare(uvm,null);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void setOddeven(boolean value) { m_oddeven = value; }
    public boolean getOddeven() { return m_oddeven; }
    
    public void onAddRow(ActionEvent ae)    
    {
        // add the item
        GridItem item = new GridItem();
        item.setFirstName("new");
        item.setLastName("new");
        m_grid.getItems().add(item);
        // select the item and focus the item
        m_grid.deselectCurrentSelection();
        m_grid.selectAndFocusItem(item);
    }
    
    public void onClear(ActionEvent ae)
    {
        m_grid.getItems().clear();
    }

    public CCUsageVariantButton getUsageVariantButton() { return m_usageVariantButton; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void fillGrid()
    {
        GridItem row;
        row = new GridItem();
        row.setFirstName("Michael");
        row.setLastName("Schumacher");
        row.setMarried(true);
        row.setBirthday(new Date());
        m_grid.getItems().add(row);
        row = new GridItem();
        row.setFirstName("Alain");
        row.setLastName("Prost");
        row.setMarried(true);
        row.setBirthday(new Date());
        m_grid.getItems().add(row);
        row = new GridItem();
        row.setFirstName("Niki");
        row.setLastName("Lauda");
        row.setMarried(true);
        m_grid.getItems().add(row);
        row = new GridItem();
        row.setFirstName("Nik");
        row.setLastName("Heidfeld");
        row.setMarried(false);
        m_grid.getItems().add(row);
    }

}
