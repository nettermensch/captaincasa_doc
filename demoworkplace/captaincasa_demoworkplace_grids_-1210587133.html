The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Bean data grid (PBC)
  • With own controls
  • Own columns - By API

This example demonstrates how to extend the data grid view with own components.

Inside the layout you can position components of type DATAGRIDVIEW2COLUMNEXIT. With each of the components you define an EXITID.

In your implementation you extends the CCDataGridView2 class and implement corresponding renderExit* methods. There are two types of renderExit* methods - one for rendering single nodes and the one for a list of nodes that can be passed back. In the example the "single node"-type methods are implemented.

By default you need to also extend the data grid view's item class, so that your own components can access data or invoke functions correspondingly.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPBCCCDataGridView2OwnControls}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPBCCCDataGridView2OwnControls}" >
<t:row id="g_2" >
<tpbc:datagridview2 id="g_3" pagebeanbinding="#{d.DemoPBCCCDataGridView2OwnControls.dataView}" shownullcontent="true" >
<tpbc:datagridview2columnexit id="g_5" exitId="IMAGE" />
<tpbc:datagridview2column id="g_25" propertyName="firstName" title="First name" />
<tpbc:datagridview2column id="g_27" propertyName="lastName" title="Last  name" />
<tpbc:datagridview2column id="g_29" propertyName="height" title="Height" />
<tpbc:datagridview2column id="g_31" propertyName="birthDate" title="Date of birth" />
<tpbc:datagridview2column id="g_33" propertyName="married" title="Married" />
<tpbc:datagridview2columnexit id="g_6" exitId="MARRIED" />
</tpbc:datagridview2>
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.ccaddons.pbc.datagridview2.ConfigurationColumnExit;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup.IYesNoListener;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.IMAGENode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass(expressionBase = "#{d.DemoPBCCCDataGridView2OwnControls}")

public class DemoPBCCCDataGridView2OwnControls extends WorkpageDispatchedPageBean implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * The bean... This normally is an own class. In the demo it is an inner class
     * so that all code is kept within the demo class.
     */
    public static class TestPerson
    {
        String i_firstName;
        String i_lastName;
        int i_height;
        LocalDate i_birthDate;
        boolean i_married;
        public String getFirstName() { return i_firstName; } 
        public String getLastName() { return i_lastName; } 
        public int getHeight() { return i_height; } 
        public LocalDate getBirthDate() { return i_birthDate; } 
        public boolean isMarried() { return i_married; } 
        public void setFirstName(String firstName) { i_firstName = firstName; } 
        public void setLastName(String lastName) { i_lastName = lastName; } 
        public void setHeight(int height) { i_height = height; } 
        public void setBirthDate(LocalDate birthDate) { i_birthDate = birthDate; } 
        public void setMarried(boolean married) { i_married = married; }
    }

    /**
     * Extension of CCDataGridView2. In the extension the creation of components is
     * updated, so that own controls are added. There are corresponding after* and
     * before* methods that are called during the generation of the actual components.
     * 
     * Please note: there is the rendering of the filter header and the rendering of
     * the grid columns! Both have to be in synch: if adding a new column you have
     * to embed own components at both places!
     */
    public class MyDGV extends CCDataGridView2<TestPerson>
    {
        /**
         * Extended implementation of GridItem - providing the data and functions that
         * is required for the additional components. In the method "createGridItem"
         * (see code below) you need to create the correct GridItem.
         */
        public class MyDGVGridItem extends GridItem
        {
            String i_avatarImage;
            public MyDGVGridItem(TestPerson object)
            {
                super(object);
                i_avatarImage = RandomLogic.getAvatarImage(object.getLastName(), object.getFirstName());
            }
            public String getAvatarImage() { return i_avatarImage; }
            public void onRemoveAction(ActionEvent event)
            {
                deleteItem(getObject());
            }
        }
        public MyDGV()
        {
            super(TestPerson.class);
        }
        @Override
        protected GridItem createGridItem(TestPerson object)
        {
            // return the extended item - and not use the default one!
            return new MyDGVGridItem(object);
        }
        @Override
        protected GRIDCOLNode renderExitGRIDCOLNode(ConfigurationColumnExit columnExit)
        {
            if ("IMAGE".equals(columnExit.getExitId()))
            {
                GRIDCOLNode gc = new GRIDCOLNode()
                    .setText("")
                    .setWidth("40");
                IMAGENode in = new IMAGENode()
                    .setImage(".{avatarImage}")
                    .setAlign("center")
                    .setImageheight(30).
                    setImagewidth(30).
                    setKeepratiofitmode("cover");
                gc.addSubNode(in);
                return gc;
            }
            else if ("MARRIED".equals(columnExit.getExitId()))
            {
                GRIDCOLNode gc = new GRIDCOLNode()
                    .setText("")
                    .setWidth("80");
                BUTTONNode bt = new BUTTONNode()
                        .setText("Remove...")
                        .setActionListener(".{onRemoveAction}");
                gc.addSubNode(bt);
                return gc;
            }
            else
            {
                return super.renderExitGRIDCOLNode(columnExit);
            }
        }
        @Override
        protected ComponentNode renderExitHeaderColumnNode(ConfigurationColumnExit columnExit)
        {
            if ("IMAGE".equals(columnExit.getExitId()) || 
                "MARRIED".equals(columnExit.getExitId()))
            {
                PANENode p = new PANENode().setStylevariant("ccaddons_datagridview2_filter");
                return p;
            }
            else
            {
                return super.renderExitHeaderColumnNode(columnExit);
            }
        }
    }

    /**
     * Listener to the data grid view's listener so that in order to react on events
     * of the grid processing.
     */
    public class MyDGVListener extends CCDataGridView2.DefaultListener<TestPerson>
    {
        @Override
        public void reactOnGridSelect(TestPerson object)
        {
            reactOnItemSelection(object);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    MyDGV m_dataView = new MyDGV();
    List<TestPerson> m_persons = new ArrayList<TestPerson>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDataGridView2OwnControls(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        fillData();
    }

    public String getPageName()
    {
        return "/workplace/demopbcccdatagridview2owncontrols.xml";
    }

    public String getRootExpressionUsedInPage()
    {
        return "#{d.DemoPBCCCDataGridView2OwnControls}";
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2<TestPerson> getDataView()
    {
        return m_dataView;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void deleteItem(final TestPerson object)
    {
        YESNOPopup.createInstance("Deletion of item","Do you really want to remove the selected item?",new IYesNoListener()
        {
            @Override
            public void reactOnYes() { deleteItemExecute(object); }
            @Override
            public void reactOnNo() {}
        });
    }

    private void deleteItemExecute(TestPerson object)
    {
        m_persons.remove(object);
        m_dataView.refreshData(m_persons);
    }

    private void reactOnItemSelection(TestPerson object)
    {
        Statusbar.outputSuccess("Selection of item: " + object.getLastName() + ", " + object.getFirstName());
    }

    private void fillData()
    {
        m_persons.clear();
        ;
        for (int i = 0; i < 100; i++)
        {
            TestPerson p = new TestPerson();
            p.setFirstName(RandomLogic.getFirstName());
            p.setLastName(RandomLogic.getLastName());
            p.setHeight(RandomLogic.getHeight());
            p.setMarried(i % 2 == 0);
            p.setBirthDate(RandomLogic.getBirthDate());
            m_persons.add(p);
        }
        m_dataView.prepare(m_persons, new MyDGVListener());
    }

}
