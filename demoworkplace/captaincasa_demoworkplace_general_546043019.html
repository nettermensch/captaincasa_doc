The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • RISC Client Integration
  • Update browser title and image

The component CLIENTCONFIG contains an attribute TITLE and an attribute TITLEIMAGE by which you can update the information that is shown inside a "browser tab".

The component either can be directly embedded in your layout - or you can use its default representation which is automatically added - by using the "Client"-class.

The API is:

1. Client.instance().setTitle(...)
2. Client.instance().setTitleimage(...)

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >

<t:clientconfig id="g_3" title="#{d.DemoBrowserTitle.browserTitle}" titleimage="#{d.DemoBrowserTitle.browserTitleImage}" />
</t:beanprocessing>
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoBrowserTitle}" rowdistance="#{ccstylevalue[&#39;@defaultrowdistance@&#39;]}" >
<t:row id="g_17" >
<t:textpane id="g_18" text="Observe the &#34;title tab&#34; of your browser while updating the definitions below...:" width="100%" />
</t:row>
<t:row id="g_14" >
<t:image id="g_2" image="/images/pictures/browsertab.png" />
</t:row>
<t:rowdistance id="g_5" height="10" />

<t:button id="g_15" actionListener="#{d.DemoBrowserTitle.onChangeBrowserTitle}" text="Change browser  title" width="150+" />

<t:row id="g_10" coldistance="#{ccstylevalue[&#39;@defaultcoldistance@&#39;]}" >

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Client;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoBrowserTitle}")

public class DemoBrowserTitle
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_inputTitle = "Some new title";
    String m_inputImage = null;
    ValidValuesBinding m_inputImageVVS = new ValidValuesBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoBrowserTitle(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_inputImageVVS.addValidValue("/eclntjsfserver/images/iconssvg/calendar_16x16.svg","/eclntjsfserver/images/iconssvg/calendar_16x16.svg");
        m_inputImageVVS.addValidValue("/eclntjsfserver/images/iconssvg/color_16x16.svg","/eclntjsfserver/images/iconssvg/color_16x16.svg");
        m_inputImageVVS.addValidValue("/eclntjsfserver/images/iconssvg/bulb_16x16.svg","/eclntjsfserver/images/iconssvg/bulb_16x16.svg");
    }

    public String getPageName() { return "/workplace/demobrowsertitle.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoBrowserTitle}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getInputImageVVS() { return m_inputImageVVS; }

    public String getInputImage() { return m_inputImage; }
    public void setInputImage(String value) { this.m_inputImage = value; }

    public String getInputTitle() { return m_inputTitle; }
    public void setInputTitle(String value) { this.m_inputTitle = value; }

    public void onChangeBrowserTitle(javax.faces.event.ActionEvent event) 
    {
        Client.instance().setTitle(m_inputTitle);
    }

    public void onChangeBrowserImage(javax.faces.event.ActionEvent event) 
    {
        Client.instance().setTitleimage(m_inputImage);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
