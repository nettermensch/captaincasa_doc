The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • 3D Animations
  • "Flying in" updated content

Press the buttons one the left one after the other and view the animation.

Whenever you update the content of the FLYINCONTENTAREA then you may invoke a trigger.

As consequence the updated content of the area "flies" into its position by some animation.

The point to start the animation from can be defined by the definition in attribute STARTPOINT. There are two notations supported currently:

1. component(...id...) - from the center of the component
2. point(x,y) - from an absolute coordinate

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.pagebean.IPageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoFlyInContentArea1}")

public class DemoFlyInContentArea1
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_startPoint;
    IPageBean m_contentBean;
    Trigger m_trigger = new Trigger();
    
    Map<Class,IPageBean> m_beans = new HashMap<Class,IPageBean>(); 
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFlyInContentArea1(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        {
            DemoHelloWorld bean = new DemoHelloWorld(workpageDispatcher);
            bean.setDocuRendered(false);
            m_beans.put(bean.getClass(),bean);
        }
        {
            DemoBudget bean = new DemoBudget(workpageDispatcher);
            bean.setDocuRendered(false);
            m_beans.put(bean.getClass(),bean);
        }
        {
            DemoMinispread bean = new DemoMinispread(workpageDispatcher);
            bean.setDocuRendered(false);
            m_beans.put(bean.getClass(),bean);
        }
        onHelloWorld(null);
    }

    public String getPageName() { return "/workplace/demoflyincontentarea1.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFlyInContentArea1}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getStartPoint() { return m_startPoint; }
    public IPageBean getContentBean() { return m_contentBean; }

    public void onBudget(javax.faces.event.ActionEvent event) 
    {
        m_trigger.trigger();
        m_contentBean = m_beans.get(DemoBudget.class);
        m_startPoint = "component(BUDGET)"; // Id of button
    }

    public void onMiniSpread(javax.faces.event.ActionEvent event) 
    {
        m_trigger.trigger();
        m_contentBean = m_beans.get(DemoMinispread.class);
        m_startPoint = "component(MINISPREAD)"; // Id of button
    }

    public void onHelloWorld(javax.faces.event.ActionEvent event) 
    {
        m_trigger.trigger();
        m_contentBean = m_beans.get(DemoHelloWorld.class);
        m_startPoint = "component(HELLOWORLD)"; // Id of button
    }

    public Trigger getTrigger() { return m_trigger; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
