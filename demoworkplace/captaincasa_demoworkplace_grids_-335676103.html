The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Hide empty rows

Add new rows by pressing the "Add" button - up to the row count of 10 they will be added to the grid, then the grid starts scrolling.

You can control this behaviour using the attribute SHOWEMPTYROWS. By default empty rows are occupying space - but you can switch off with setting SHOWEMPTYROWS to "false". Pay attention: this attribute is only relevant when NOT specifying any height with the component!

Please also pay attention: the grid automatically adds vertical space for dynamically rendering a horizontal scrollbar if required. In the example this was switched off by setting HORIZONTALSCROLLMODE to "autowithresize".

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridShowEmptyRows}")

public class DemoGridShowEmptyRows
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }
        
        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
    }
    
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    protected FIXGRIDListBinding<GridItem> m_grid2 = new FIXGRIDListBinding<GridItem>();

    public DemoGridShowEmptyRows(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public FIXGRIDListBinding<GridItem> getGrid2() { return m_grid2; }
    
    public void onAddItem(ActionEvent event)
    {
        GridItem gi = new GridItem();
        gi.setFirstName("<new>");
        gi.setLastName("<new>");
        m_grid.getItems().add(gi);
    }

    public void onAddItem2(ActionEvent event)
    {
        GridItem gi = new GridItem();
        gi.setFirstName("<new>");
        gi.setLastName("<new>");
        m_grid2.getItems().add(gi);
    }

}
