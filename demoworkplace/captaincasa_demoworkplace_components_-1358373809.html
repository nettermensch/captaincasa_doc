The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Drag/drop files

This example demonstrates that the user can also drag/drop files directly from the file explorer onto the file upload button.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.DefaultUploadContent;
import org.eclnt.jsfserver.bufferedcontent.UploadContentMgr;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoFileUploadDrop}")

public class DemoFileUploadDrop
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyUploadContent extends DefaultUploadContent
    {
        public void beginPassing()
        {
            m_protocol += "---------------------- begin\n";
        }
        public void passClientFile(String fileName, byte[] content)
        {
            m_protocol += "File upload: " + fileName + ", " + content.length + " bytes\n";
        }
        public void endPassing()
        {
            m_protocol += "---------------------- end\n";
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_protocol = "";
    MyUploadContent m_uploadContent;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoFileUploadDrop(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_uploadContent = new MyUploadContent();
        UploadContentMgr.add(m_uploadContent);
    }

    public String getPageName() { return "/workplace/demofileuploaddrop.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoFileUploadDrop}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getUploadURL() { return m_uploadContent.getURL(); }
    
    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    public void onUploadAction(javax.faces.event.ActionEvent event) 
    {
        Statusbar.outputSuccess("File upload was executed.");
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
