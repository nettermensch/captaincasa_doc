The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Drag & Drop
  • Moving boxes

This example shows how to use...

1. drag & drop
2. dynamic content management
3. the component ROWFLEXLINECONTAINER

...in order to build a scenario in which the sequence of the boxes can be changed by the user.

For smart drag & drop the following features are used:

1. drop zone definition in DROPRECEIVE ("horizontalsplit")
2. interpretation of the drop position in the drop-event processing in order to position the dropped box either in front or behind.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoDragDropSequence}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoDragDropSequence}" rowdistance="10" >
<t:row id="g_3" >
<t:textpane id="g_9" text="Change the sequence of the boxes by dragging &amp; dropping them:" width="100%" />
</t:row>
<t:rowflexlinecontainer id="g_6" >
<t:dynamiccontent id="g_7" contentbinding="#{d.DemoDragDropSequence.dynContent}" />
</t:rowflexlinecontainer>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.BOXNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.util.StyleManager;
import org.eclnt.jsfserver.util.valuemgmt.IActionListenerDelegation;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDragDropSequence}")

public class DemoDragDropSequence
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class BoxInfo
    {
        String i_text;
        String i_color;
        public String getText() { return i_text; } 
        public void setText(String text) { i_text = text; }
        public String getColor() { return i_color; }
        public void setColor(String color) { i_color = color; } 
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private static final String[] COLORS = new String[] 
    {
        StyleManager.getStyleValue("@backgroundColorGreen@"),
        StyleManager.getStyleValue("@backgroundColorYellow@"),
        StyleManager.getStyleValue("@backgroundColorRed@"),
    };
    
    List<BoxInfo> m_boxes = new ArrayList<DemoDragDropSequence.BoxInfo>();
    DYNAMICCONTENTBinding m_dynContent = new DYNAMICCONTENTBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDragDropSequence(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        createBoxes();
        renderBoxes();
    }

    public String getPageName() { return "/workplace/demodragdropsequence.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDragDropSequence}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getDynContent() { return m_dynContent; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void createBoxes()
    {
        for (int i=0; i<10; i++)
        {
            BoxInfo bi = new BoxInfo();
            bi.setText("Box " + i);
            bi.setColor(COLORS[i%COLORS.length]);
            m_boxes.add(bi);
        }
    }

    private void renderBoxes()
    {
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        int counter = 0;
        for (final BoxInfo bi: m_boxes)
        {
            BOXNode box = new BOXNode()
                    .setWidth("150")
                    .setHeight("150")
                    .setStylevariant("ccexposed")
                    .setDragsend("box:"+counter)
                    .setDropreceive("box:horizontalsplit")
                    .setBackground(bi.getColor())
                    .bindActionListener(new IActionListenerDelegation()
                    {
                        @Override
                        public void onAction(ActionEvent event)
                        {
                            processBoxEvent(bi,event);
                        }
                    });
            l.add(box);
            ROWNode row = new ROWNode();
            box.addSubNode(row);
            LABELNode lab = new LABELNode()
                    .setText(bi.getText())
                    .setWidth("100%")
                    .setHeight("100%")
                    .setAlign("center")
                    .setValign("middle")
                    .setStylevariant("ccbig");
            row.addSubNode(lab);
            counter++;
        }
        m_dynContent.setContentNodes(l);
    }

    private void processBoxEvent(BoxInfo bi, ActionEvent event)
    {
        if (event instanceof BaseActionEventDrop)
        {
            String di = ((BaseActionEventDrop)event).getDragInfo();
            if (di.startsWith("box:"))
            {
                String dragIndexString = di.substring("box:".length());
                int dragIndex = ValueManager.decodeInt(dragIndexString,-1);
                // horizontal drop position defines if to moved before or after
                boolean dropBefore = ((BaseActionEventDrop)event).getPercentageHorizontal() < 50;
                moveBox(m_boxes.get(dragIndex),bi,dropBefore);
            }
        }
    }

    private void moveBox(BoxInfo movedBox, BoxInfo refBox, boolean before)
    {
        if (movedBox == refBox)
            return;
        m_boxes.remove(movedBox);
        int index = m_boxes.indexOf(refBox);
        int movedBoxIndex = index;
        if (before == false)
            movedBoxIndex = index+1;
        m_boxes.add(movedBoxIndex,movedBox);
        renderBoxes();
    }

}
