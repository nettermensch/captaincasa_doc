The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Scheduling
  • Time Range Selection

This demo shows a time selection.

Drag and drop the coloured icons on the top right of each area into the schedule area. In the schedule area you can manipulat the size of and you can drag & drop.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoTimeSelectionGrid}")

public class DemoTimeSelectionGrid
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        DemoTimeSelection i_ts = new DemoTimeSelection();
        public DemoTimeSelection getTs() { return i_ts; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoTimeSelectionGrid(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);   
        onAddTimeSelection(null);
    }
    
    public String getPageName() { return "/workplace/demotimeselectiongrid.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTimeSelectionGrid}"; }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onAddTimeSelection(ActionEvent event) 
    {
        GridItem gi = new GridItem();
        m_grid.getItems().add(gi);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    

}
