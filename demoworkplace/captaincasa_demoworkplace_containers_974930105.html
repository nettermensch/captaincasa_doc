The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Positioning in free style HTML
  • Embedding into normal forms

The TEXTWITHCONTROLS component can be embedded into normal form processing. If you only the defined the width of the component then the height will be calculated corresponding to the content.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoTextWithControls2}")

public class DemoTextWithControls2
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    LocalDate m_deliveryDate = LocalDate.now();
    String m_firstName = "Captain";
    String m_lastName = "Casa";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTextWithControls2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demotextwithcontrols2.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTextWithControls2}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onClearAction(javax.faces.event.ActionEvent event) 
    {
        m_lastName = "";
        m_firstName = "";
    }

    public LocalDate getDeliveryDate() { return m_deliveryDate; }
    public void setDeliveryDate(LocalDate value) { this.m_deliveryDate = value; }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { this.m_lastName = value; }

    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { this.m_firstName = value; }

}
