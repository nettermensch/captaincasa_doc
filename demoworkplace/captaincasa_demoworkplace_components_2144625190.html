The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Field Value Help - Details
  • Opening the value help dialog "from outside"

When you select the department by radio button, then immediately the selection of employees is opened.

Check the code for the following issues:

(.) The RADIOBUTTONGROUP flushes selections to the server by setting FLUSH to "true"
(.) On server side the method to open the selection is both called from the COMBOFIELD and from the RADIOBUTTONGROUP processing
(.) For correct positioning the popup dialog is always opened at the COMBOFIELD - which is done by explicitly assigning an id ("EMPLOYEE") to the component and by referencing this id when opening the dialog

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition
    xmlns:t="http://www.CaptainCasa.com/controllibrary/t">
    <t:beanprocessing id="g_1"
        beanbinding="#{d.DemoComboFieldOpenValueHelpFromOutside}" />
    <t:rowdemobodypane id="g_4"
        objectbinding="#{d.DemoComboFieldOpenValueHelpFromOutside}">
        <t:row id="g_6">
            <t:radiobuttongroup id="g_2"
                actionListener="#{d.DemoComboFieldOpenValueHelpFromOutside.onDepartmentAction}"
                flush="true" labeltext="Department"
                validvaluesbinding="#{d.DemoComboFieldOpenValueHelpFromOutside.departmentVVS}"
                value="#{d.DemoComboFieldOpenValueHelpFromOutside.department}" />
        </t:row>
        <t:row id="g_3">
            <t:combofield id="EMPLOYEE"
                actionListener="#{d.DemoComboFieldOpenValueHelpFromOutside.onEmployeeAction}"
                labeltext="Employee"
                text="#{d.DemoComboFieldOpenValueHelpFromOutside.employee}"
                width="250" />
        </t:row>
    </t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.events.BaseActionEventFlush;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoComboFieldOpenValueHelpFromOutside}")

public class DemoComboFieldOpenValueHelpFromOutside
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_employee;
    String m_department = "SALES";
    ValidValuesBinding m_departmentVVS = new ValidValuesBinding();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoComboFieldOpenValueHelpFromOutside(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        m_departmentVVS.addValidValue("SALES","Sales");
        m_departmentVVS.addValidValue("PROD","Production");
        m_departmentVVS.addValidValue("ADMIN","Administration");
    }

    public String getPageName() { return "/workplace/democombofieldopenvaluehelpfromoutside.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoComboFieldOpenValueHelpFromOutside}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------


    public String getEmployee() { return m_employee; }
    public void setEmployee(String value) { this.m_employee = value; }

    public ValidValuesBinding getDepartmentVVS() { return m_departmentVVS; }

    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    public void onEmployeeAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            selectEmployeeByPopup();
        }
    }

    public void onDepartmentAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventFlush)
        {
            selectEmployeeByPopup();
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void selectEmployeeByPopup()
    {
        IdTextSelection idts = IdTextSelection.createInstance();
        if ("SALES".equals(m_department))
        {
            idts.addLine("4711","Jo Mastersellor");
            idts.addLine("4712","Marianne Greatsales");
            idts.addLine("4713","Andy Alwaysgreat");
        }
        else if ("PROD".equals(m_department))
        {
            idts.addLine("4721","Lisa Hardwork");
            idts.addLine("4722","Hans Superdriller");
            idts.addLine("4723","Betty Alwaysworking");
            idts.addLine("4724","John Thechainsaw");
        }
        else if ("ADMIN".equals(m_department))
        {
            idts.addLine("4731","Hugo Superboss");
        }
        idts.setCallBack(new ISetId()
        {
            @Override
            public void setId(String id)
            {
                m_employee = id;
            }
        });
        // force the popup to always be opened at the EMPLOYEE combofield
        idts.getPopup().setLefTopReferenceComponentId("EMPLOYEE");
    }

    
}
