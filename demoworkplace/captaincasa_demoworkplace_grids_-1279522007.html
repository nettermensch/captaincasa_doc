The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Info about selected Column

Every time a selection event is passed from the client to the server side, there is the information available on which column of the grid the selection was executed, too.

By using the column index and by using further API functions of the grid (please have a look into the code), you can obtain more information about the column that was hit.

Please note: the column index always refers to the sequence of GRIDCOL-definitions as defined in your JSP/XML definition. This means: it is independent from the user re-arranging the columns on his/her own.

And, please note: the default behaviour of the grid is, that once a row was selected the next selection is not passed to the server side. You can override this default behaviour by using the attribute FIXGRID-RESELECTABLE. If you want your user to select different columns, one after the other, within the same grid, you need to set FIXGRID-RESELECTABLE to "true".

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.IFIXGRIDColumnInfo;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridColumnIndex}")

public class DemoGridColumnIndex
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_town;
        public String getTown() { return m_town; }
        public void setTown(String value) { m_town = value; }
        
        protected String m_street;
        public String getStreet() { return m_street; }
        public void setStreet(String value) { m_street = value; }
        
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }
        
        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
        
        @Override
        public void onRowSelect()
        {
            Statusbar.outputSuccess("Selection was registered: Row-Index " + m_grid.getItems().indexOf(this) + 
                                    "/ Column-Index " + m_grid.getColumnIndexOfLastSelection());
            IFIXGRIDColumnInfo colInfo = m_grid.getColumnInfos().get(m_grid.getColumnIndexOfLastSelection());
            m_columnText = colInfo.getText();
            m_columnReference = colInfo.getSortreference();
        }
        
        @Override
        public void onRowExecute()
        {
            Statusbar.outputSuccess("Execution was registered: Row-Index " + m_grid.getItems().indexOf(this) + 
                                    "/ Column-Index " + m_grid.getColumnIndexOfLastSelection());
            IFIXGRIDColumnInfo colInfo = m_grid.getColumnInfos().get(m_grid.getColumnIndexOfLastSelection());
            m_columnText = colInfo.getText();
            m_columnReference = colInfo.getSortreference();
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    protected String m_columnText;
    protected String m_columnReference;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGridColumnIndex(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<10; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("FN "+i);
            gi.setLastName("LN "+i);
            gi.setStreet("ST "+i);
            gi.setTown("TO "+i);
            m_grid.getItems().add(gi);
        }
    }
    
    // ------------------------------------------------------------------------
    // methods
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public int getColumnIndex() { return m_grid.getColumnIndexOfLastSelection(); }
    public String getColumnText() { return m_columnText; }
    public String getColumnReference() { return m_columnReference; }
    
}
