The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid Popup Menu
  • Dynamic definition per cell

This demo shows how to build a popup menu that is specific for each cell of each row - without defining the POPUPMENU on cell component level (here LABEL inside GRIDCOL).

The building blocks are:

1. usage of FIXGRID-ROWPOPUPMENU and FIXGRID-ROWPOPUPMENULOADROUNDTRIP: the central popup menu with the id "GRIDPOPUPMENU" is defined as popup menu for the grid row, and a serve round trip is caused when the user presses the right mouse button

2. in the round trip (onRowPopupMenuLoad-method) the popup menu is built via DYNAMICCONTENT.

3. the grid provides the information about the selected column by calling method "getColumnIndexOfLastSelection()". The index that is returned always applies to the sequence of columns as defined in the FIXGRID definition (i.e. it does not depend from the column arrangement at runtime, where the user can update the column order by drag/drop)

You also see, that the grid provides a method "getColumnInfos()", returning an list of ColumnInfo-objects which hold the essential meta data for each columns. Using this meta data you can build up corresponding dynamic scenarios.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuLoad;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDComponent.ColumnInfo;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridCellPopupMenu}")

public class DemoGridCellPopupMenu
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_firstName;
        String i_lastName;
        public String getFirstName() { return i_firstName; }
        public String getLastName() { return i_lastName; }
        public void onRowSelect()
        {
            outputColumnInfo();
        }
        public void onRowExecute()
        {
            outputColumnInfo();
        }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            outputColumnInfo();
            int selectedColumn = m_grid.getColumnIndexOfLastSelection();
            if ("CLEAR".equals(event.getCommand()))
                setTextOfColumn(selectedColumn,"");
            else if ("ADD+".equals(event.getCommand()))
                setTextOfColumn(selectedColumn,getTextOfColumn(selectedColumn)+"+");
            else if ("ADD++".equals(event.getCommand()))
                setTextOfColumn(selectedColumn,getTextOfColumn(selectedColumn)+"++");
        }
        @Override
        public void onRowPopupMenuLoad(BaseActionEventPopupMenuLoad event)
        {
            outputColumnInfo();
            int selectedColumn = m_grid.getColumnIndexOfLastSelection();
            List<ComponentNode> l = new ArrayList<ComponentNode>();
            {
                MENUITEMNode min = new MENUITEMNode()
                        .setCommand("CLEAR")
                        .setText("Clear");
                l.add(min);
            }
            {
                MENUITEMNode min = new MENUITEMNode()
                        .setCommand("ADD+")
                        .setText("Change to " + getTextOfColumn(selectedColumn) + "+");
                l.add(min);
            }
            {
                MENUITEMNode min = new MENUITEMNode()
                        .setCommand("ADD++")
                        .setText("Change to " + getTextOfColumn(selectedColumn) + "++");
                l.add(min);
            }
            m_menuContent.setContentNodes(l);
        }
        private void setTextOfColumn(int column, String text)
        {
            switch (column)
            {
                case 0: i_firstName = text; return;
                case 1: i_lastName = text; return;
            }
            throw new Error("This column is not known: " + column);
        }
        private String getTextOfColumn(int column)
        {
            switch (column)
            {
                case 0: return getFirstName();
                case 1: return getLastName();
            }
            throw new Error("This column is not known: " + column);
        }
        private void outputColumnInfo()
        {
            int selectedColumn = m_grid.getColumnIndexOfLastSelection();
            ColumnInfo ci = m_grid.getColumnInfos().get(selectedColumn);
            Statusbar.outputMessage("Column info: " + ci.getSortreference());
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    DYNAMICCONTENTBinding m_menuContent = new DYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridCellPopupMenu(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.i_firstName = "First " + i;
            gi.i_lastName = "Last " + i;
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridcellpopupmenu.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridCellPopupMenu}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public DYNAMICCONTENTBinding getMenuContent() { return m_menuContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
