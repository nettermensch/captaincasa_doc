The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Character Input

CaptainCasa is 100% based on unicode.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.BaseComponent;
import org.eclnt.jsfserver.elements.impl.LABELComponent;
import org.eclnt.jsfserver.elements.impl.LABELComponentTag;
import org.eclnt.jsfserver.elements.impl.PANEComponent;
import org.eclnt.jsfserver.elements.impl.ROWComponentTag;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.util.valuemgmt.ValueManager;

@CCGenClass (expressionBase="#{d.DemoChars}")

public class DemoChars
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected int m_startValue = 160;
    protected String m_input;
    protected String m_output;
    protected PANEComponent m_dynPane = new PANEComponent();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoChars(IDispatcher dispatcher)
    {
        super(dispatcher);
        showValues();
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getInput() { return m_input; }
    public void setInput(String value) { m_input = value; }
    
    public void setOutput(String value) { m_output = value; }
    public String getOutput() { return m_output; }
    
    public int getStartValue() { return m_startValue; }
    public void setStartValue(int value) { m_startValue = value; }

    public void setDynPane(PANEComponent value) 
    { 
        if (m_dynPane == value)
            return;
        m_dynPane = value; 
        showValues();
    }

    public void onShowInput(ActionEvent event) 
    {
        int inputInt = ValueManager.decodeInt(m_input,65);
        m_output = "" + (char)inputInt;
    }

    public void onShowValues(ActionEvent event) 
    {
        if (m_startValue < 0)
            m_startValue = 0;
        showValues();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void showValues()
    {
        if (m_dynPane == null)
            return;
        m_dynPane.getChildren().clear();
        int counter = m_startValue;
        for (int i=0; i<25; i++)
        {
            ROWComponentTag rt = new ROWComponentTag();
            BaseComponent rc = rt.createBaseComponent();
            m_dynPane.getChildren().add(rc);
            for (int j=0; j<4; j++)
            {
                LABELComponentTag lt;
                lt = new LABELComponentTag();
                lt.setText(counter+"");
                lt.setWidth("60");
                rc.getChildren().add(lt.createBaseComponent());
                lt = new LABELComponentTag();
                lt.setText(""+(char)counter);
                lt.setWidth("20");
                rc.getChildren().add(lt.createBaseComponent());
                counter++;
            }
        }
    }
    
}
