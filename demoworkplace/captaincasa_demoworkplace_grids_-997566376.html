The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Single click execute

There are two ways to react on grid selections:

By default:

(.) onRowSelect is called in the grid item when the user clicks an item - or navigates into it by keyboard
(.) onRowExecute is called when the user double-clicks an item (long-click on mobile devices) - or the user hits the return key within the grid item

By using attribute SINGLECLICKEXECUTE you can fine-tune this behavior. By setting the attribute to true, the "onRowExecute" is already called when the user single-clicks into the grid.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.OKPopup;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridSingleClickExecute}")

public class DemoGridSingleClickExecute
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_name;
        public String getName() { return i_name; }
        public void setName(String value) { this.i_name = value; }

        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
            OKPopup.createInstance("onRowExecute called","onRowExecute was called on item: " + i_name);
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    boolean m_singelClickExecute;
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridSingleClickExecute(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setName("Name " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridsingleclickexecute.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridSingleClickExecute}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public boolean getSingelClickExecute() { return m_singelClickExecute; }
    public void setSingelClickExecute(boolean value) { this.m_singelClickExecute = value; }

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
