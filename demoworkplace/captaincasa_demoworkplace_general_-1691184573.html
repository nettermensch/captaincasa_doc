The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • Popup sizing

When opening a popup you can explicitly pass the popup's position and size.

In addition the popup sends its actual position, which might be changed due to user operations, to the server.

The positions (left/top) are relative to the window's coordinates.

All size definitions (width/ height/ left/ top) are scaled automatically by the general scaling factor that is defined on client side (default 100%).

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupControl}")

public class DemoPopupControl
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    boolean m_openMaximized = false;
    protected int m_top = 0;
    protected int m_left = 0;
    protected int m_height = 800;
    protected int m_width = 600;
    ModalPopup m_popup;
    DemoHelloWorld m_bean;
    
    public DemoPopupControl(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public boolean getOpenMaximized() { return m_openMaximized; }
    public void setOpenMaximized(boolean value) { this.m_openMaximized = value; }

    public int getTop() { return m_top; }
    public void setTop(int value) { m_top = value; }

    public int getLeft() { return m_left; }
    public void setLeft(int value) { m_left = value; }

    public int getHeight() { return m_height; }
    public void setHeight(int value) { m_height = value; }

    public int getWidth() { return m_width; }
    public void setWidth(int value) { m_width = value; }

    public void onOpenPopup(ActionEvent event) 
    {
        if (m_popup == null)
        {
            m_bean = new DemoHelloWorld(getOwningDispatcher());
            m_popup = openModalPopup(m_bean,"Hello World!",m_width,m_height, new ModalPopup.IModalPopupListener()
            {
                public void reactOnPopupClosedByUser()
                {
                    onClosePopup(null);
                }
            });
            m_popup.setTop(m_top);
            m_popup.setLeft(m_left);
            if (m_openMaximized)
            {
                m_popup.maximize(true);
            }
        }
    }
    
    public void onClosePopup(ActionEvent event) 
    {
        if (m_popup != null)
        {
            closePopup(m_bean);
            m_popup = null;
            m_bean = null;
        }
    }

    public String getClientValues() 
    { 
        if (m_popup == null)
            return "Popup instance not available.";
        else
            return m_popup.getLeft() + ", " +
                   m_popup.getTop() + ", " +
                   m_popup.getWidth() + ", " +
                   m_popup.getHeight();
    }

}
