The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Log Configuration

The configuration of the CaptainCasa server side logging can be updated at runtime by using some API. Please check the Java code for details.

The client side logging is done into the console of the browser. In order to save performance it is default switched off. You can switch on by appending query parameter "cclogactive=true" to the .risc-URL that starts your dialog.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.logging.Level;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.util.log.CLog;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoLogUpdate}")

public class DemoLogUpdate
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    boolean m_logConsole;
    String m_logLevel;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLogUpdate(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_logConsole = CLog.getLogToConsole();
        m_logLevel = CLog.L.getLevel().getName();
    }

    public String getPageName() { return "/workplace/demologupdate.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLogUpdate}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public boolean getLogConsole() { return m_logConsole; }
    public void setLogConsole(boolean value) { this.m_logConsole = value; }
    
    public String getLogLevel() { return m_logLevel; }
    public void setLogLevel(String value) { this.m_logLevel = value; }

    public void onTakeOverAction(ActionEvent event) 
    {
        try
        {
            Level newLevel = Level.parse(m_logLevel);
            CLog.updateLogLevel(newLevel);
            if (m_logConsole == true)
                CLog.addConsoleHandler();
            else
                CLog.removeConsoleHandler();
            Statusbar.outputSuccess("Configuration of logging was updated.");
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString(),ValueManager.getStackstraceString(t));
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
