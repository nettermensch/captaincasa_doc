The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Adapter Binding
  • By annotation

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashSet;
import java.util.Set;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.adapter.ComponentAdapterByAnnotation;
import org.eclnt.jsfserver.elements.adapter.ComponentAttribute;
import org.eclnt.jsfserver.elements.adapter.ComponentAttributeType;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoCabReflection}")

public class DemoCabReflection
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * In the demo: inner class. In real scenarios: real class that is used
     * by many page beans.
     */
    public class FieldAdapter extends ComponentAdapterByAnnotation
    {
        String m_label;
        String m_value;
        boolean i_containsError;
        String i_errorText;
        public FieldAdapter(String label)
        {
            m_label = label;
        }
        @ComponentAttribute(type=ComponentAttributeType.FIX,attribute="labeltext")
        public String getLabel() { return m_label; }
        @ComponentAttribute(attribute="text")
        public String getValue() { return m_value; }
        public void setValue(String value) { m_value = value; }
        @ComponentAttribute
        public String getBgpaint()
        {
            if (i_containsError)
                return "error()";
            else
                return null;
        }
        @ComponentAttribute
        public String getTooltip()
        {
            if (i_containsError == true)
                return i_errorText;
            else
                return null;
        }
        public void indicateError(String text) 
        { 
            i_containsError = true;
            i_errorText = text;
        }
        public void resetError() 
        { 
            i_containsError = false;
            i_errorText = null;
        }
    }

    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Set<FieldAdapter> m_adapters = new HashSet<FieldAdapter>();
    
    FieldAdapter m_firstName;
    FieldAdapter m_lastName;
    FieldAdapter m_title;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCabReflection(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_firstName = createFieldAdapter("First name");
        m_lastName = createFieldAdapter("Last name");
        m_title = createFieldAdapter("Title");
        m_firstName.setValue("Captain");
        m_lastName.setValue("Casa");
    }

    public String getPageName() { return "/workplace/democabreflection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCabReflection}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FieldAdapter getFirstName() { return m_firstName; } 
    public FieldAdapter getLastName() { return m_lastName; }
    public FieldAdapter getTitle() { return m_title; }

    public void onCheckAction(ActionEvent event)
    {
        clearAdapterErrors();
        processChecks();
        if (checkIfError() == true)
            Statusbar.outputError("Person contains error - please check.");
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    /**
     * In demo: checks are done directly in page bean (which means: on view layer).
     * Better: checks to be done on logic layer!
     */
    private void processChecks()
    {
        if (m_firstName.getValue() == null || m_firstName.getValue().trim().length() == 0)
            m_firstName.indicateError("First name does not contain any value.");
        if (m_lastName.getValue() == null || m_lastName.getValue().trim().length() == 0)
            m_lastName.indicateError("Last name does not contain any value.");
        if (m_title.getValue() == null || m_title.getValue().trim().length() == 0)
            m_title.indicateError("Last name does not contain any value.");
    }
    
    /**
     * In real scenario: this method is part of some base class.
     */
    private void clearAdapterErrors()
    {
        for (FieldAdapter pa: m_adapters)
            pa.resetError();
    }

    /**
     * In real scenario: this method is part of some base class.
     */
    private FieldAdapter createFieldAdapter(String label)
    {
        FieldAdapter result = new FieldAdapter(label);
        m_adapters.add(result);
        return result;
    }
    
    /**
     * In real scenario: this method is part of some base class.
     */
    private boolean checkIfError()
    {
        for (FieldAdapter pa: m_adapters)
            if (pa.i_containsError == true)
                return true;
        return false;
    }
}
