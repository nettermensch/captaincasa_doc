The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Dynamic Popup Menu in Grid

Click with the right mouse button into different cells of the grid: for each cell there is an own popup menu.

The way, this is done, is:

(.) there is a POPUPMENU definition with id "GRIDPOPUP"
(.) the grid references the "GRIDPOPUP" menu in its FIXGRID-ROWPOPUPMENU definition
(.) in the POPUPMENU there is a DYNAMICCONTENT component
(.) in the onRowPopupMenuLoad()-method of the grid processing the dynamic content of the grid is built

And, to be explicitly pointed out:
(.) the FIXGRID-ROWPOPUPMENULOADROUNDTRIP is switched to "true". This forces the client to trigger a roundtrip to the server side when the user clicks the right mouse button.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoDynamicMenu}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" borderheight="1" borderwidth="1" height="100%" multiselect="true" objectbinding="#{d.DemoDynamicMenu.grid}" rowpopupmenu="GRIDPOPUP" rowpopupmenuloadroundtrip="true" width="100%" >
<t:gridcol id="g_4" text="First Name" width="50%" >
<t:label id="g_5" text=".{firstName}" />
</t:gridcol>
<t:gridcol id="g_6" text="Last Name" width="50%" >
<t:label id="g_7" text=".{lastName}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:popupmenu id="GRIDPOPUP" >
<t:dynamiccontent id="g_8" contentbinding="#{d.DemoDynamicMenu.popupContent}" />
</t:popupmenu>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuLoad;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDynamicMenu}")

public class DemoDynamicMenu
    extends PageBean 
    implements Serializable
{
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }

        @Override
        public void onRowSelect()
        {
        }
        
        @Override
        public void onRowExecute()
        {
            updatePopupMenu(this);
        }
        
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            Statusbar.outputSuccess("Popup menu item invoked. Command = " + event.getCommand() + ", Item = " + m_lastName);
        }
        @Override
        public void onRowPopupMenuLoad(BaseActionEventPopupMenuLoad event)
        {
            updatePopupMenu(this);
        }
    }

    protected FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    protected DYNAMICCONTENTBinding m_popupContent = new DYNAMICCONTENTBinding();
    int m_maxItemsCounter = 2;

    public DemoDynamicMenu(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);    
        for (int i=0; i<10; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First " + i);
            gi.setLastName("Last " + i);
            m_grid.getItems().add(gi);
        }
    }

    public DYNAMICCONTENTBinding getPopupContent() { return m_popupContent; }
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void updatePopupMenu(GridItem gi)
    {
        int colIndex = m_grid.getColumnIndexOfLastSelection();
        String colName = null;
        if (colIndex == 0) 
            colName = "Column: firstName";
        else if (colIndex == 1)
            colName = "Column: lastName";
        List<ComponentNode> cns = new ArrayList<ComponentNode>();
        m_maxItemsCounter++;
        if (m_maxItemsCounter > 10) m_maxItemsCounter = 2;
        for (int i=0; i<m_maxItemsCounter; i++)
        {
            MENUITEMNode min = new MENUITEMNode()
                    .setId(""+i)
                    .setText(gi.getFirstName() + " " + gi.getLastName() + " / " + colName + " / " + i)
                    .setCommand("COMMAND_"+i);
            cns.add(min);
        }
        m_popupContent.setContentNodes(cns);
    }
}
