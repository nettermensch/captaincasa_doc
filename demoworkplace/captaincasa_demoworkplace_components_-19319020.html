The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Date Input
  • Exact Time of Day

When entering a date/time - either via keyboard or via calendar dialog - then you can define, that the time is exactly set to a certain point of time of the corresponding day.

When setting the attribute EXACTTIME to "true" then the date, selected by the user, is interpreted as "00:00:00 (hh:mm:ss)" of the day.

By in addition defining EXACTTIMEHHMMSSMMM you can define the exact point of time to be used, e.h. "120000000" is defined for the noon of the day.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoCalendarFieldExactTime}")

public class DemoCalendarFieldExactTime
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    String m_outResult;
    public String getOutResult() { return m_outResult; }
    public void setOutResult(String value) { this.m_outResult = value; }

    Date m_date3;
    public Date getDate3() { return m_date3; }
    public void setDate3(Date value) { this.m_date3 = value; }

    Date m_date2;
    public Date getDate2() { return m_date2; }
    public void setDate2(Date value) { this.m_date2 = value; }

    Date m_date1;
    public Date getDate1() { return m_date1; }
    public void setDate1(Date value) { this.m_date1 = value; }

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCalendarFieldExactTime(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        initRefComponent("calendarfield"); 
    }

    public void onOutResult(ActionEvent event)
    {
        StringBuffer sb = new StringBuffer();
        append(sb,"Date1: ",m_date1);
        append(sb,"Date2: ",m_date2);
        append(sb,"Date3: ",m_date3);
        m_outResult = sb.toString();
    }
    
    private void append(StringBuffer sb, String prefix, Date d)
    {
        try
        {
            sb.append(prefix);
            if (d == null)
            {
                sb.append("(null)\n");
                return;
            }
            Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
            cal.setTime(d);
            sb.append(cal.get(Calendar.YEAR)); sb.append("/");
            sb.append(cal.get(Calendar.MONTH)+1); sb.append("/");
            sb.append(cal.get(Calendar.DAY_OF_MONTH)); sb.append(" ");
            sb.append(cal.get(Calendar.HOUR_OF_DAY)); sb.append(":");
            sb.append(cal.get(Calendar.MINUTE)); sb.append(":");
            sb.append(cal.get(Calendar.SECOND)); sb.append(":");
            sb.append(cal.get(Calendar.MILLISECOND)); 
            sb.append("\n");
        }
        catch (Throwable t)
        {
            sb.append(t.toString() + "\n");
        }
    }
    
}
