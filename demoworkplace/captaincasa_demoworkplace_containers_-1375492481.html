The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Tab Pane (server side)
  • Dynamically assembled

This example shows how to dynamically assemble a TABBEDLINECONTAINER - each tab showing a page bean.

There is "nothing special" to pay attention to:

(.) use ROWDYNAMICCONTENT to open up an area in which the content is dynamically created
(.) build the dynamic content on Java code side (here: directly called from the constructor)
(.) create a ROWBAGEBEANINCLUDE component insde each tab and bind it to a page bean

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoDynamicTabbedlineContainer}" />
<t:rowdemobodypane id="g_2" objectbinding="#{d.DemoDynamicTabbedlineContainer}" >
<t:rowdynamiccontent id="g_3" contentbinding="#{d.DemoDynamicTabbedlineContainer.dynContent}" />
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.ROWPAGEBEANINCLUDENode;
import org.eclnt.jsfserver.elements.componentnodes.TABBEDLINECONTAINERNode;
import org.eclnt.jsfserver.elements.componentnodes.TABBEDLINECONTAINERTABNode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoDynamicTabbedlineContainer}")

public class DemoDynamicTabbedlineContainer
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ROWDYNAMICCONTENTBinding m_dynContent = new ROWDYNAMICCONTENTBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDynamicTabbedlineContainer(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        renderContent();
    }

    public String getPageName() { return "/workplace/demodynamictabbedlinecontainer.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDynamicTabbedlineContainer}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getDynContent() { return m_dynContent; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderContent()
    {
        TABBEDLINECONTAINERNode tlc = new TABBEDLINECONTAINERNode()
                .setWidth("100%")
                .setHeight("100%");
        {
            TABBEDLINECONTAINERTABNode tlct = new TABBEDLINECONTAINERTABNode()
                    .setText("Hello world");
            tlc.addSubNode(tlct);
            ROWPAGEBEANINCLUDENode rin = new ROWPAGEBEANINCLUDENode()
                    .bindPagebeanbinding(new DemoHelloWorld(getOwningDispatcher()));
            tlct.addSubNode(rin);
        }
        {
            TABBEDLINECONTAINERTABNode tlct = new TABBEDLINECONTAINERTABNode()
                    .setText("Mini Spreadsheet");
            tlc.addSubNode(tlct);
            ROWPAGEBEANINCLUDENode rin = new ROWPAGEBEANINCLUDENode()
                    .bindPagebeanbinding(new DemoMinispread(getOwningDispatcher()));
            tlct.addSubNode(rin);
        }
        {
            TABBEDLINECONTAINERTABNode tlct = new TABBEDLINECONTAINERTABNode()
                    .setText("Budget calculation");
            tlc.addSubNode(tlct);
            ROWPAGEBEANINCLUDENode rin = new ROWPAGEBEANINCLUDENode()
                    .bindPagebeanbinding(new DemoBudget(getOwningDispatcher()));
            tlct.addSubNode(rin);
        }
        m_dynContent.setContentNode(tlc);
    }
}
