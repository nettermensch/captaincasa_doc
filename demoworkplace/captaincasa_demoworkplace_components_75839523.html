The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Field Value Help - Details
  • Simplified COMBOFIELD for id/text scenarios (PBC)

The page bean component CCComboFieldIdText simplifies the usage of the COMBOFIELD when dealing with objects, consisting out of an id and a text representation.

Internally the id is transferred in and out - whereas the user sees the text and uses the text for searching values.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCComboFieldIdText;
import org.eclnt.ccaddons.pbc.util.combofieldidtext.LogicByLoadedList;
import org.eclnt.ccaddons.pbc.util.combofieldidtext.LogicByResource;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCComboFieldIdText}")

public class DemoPBCCCComboFieldIdText
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    class DepartmentLogic extends LogicByLoadedList
    {
        public DepartmentLogic()
        {
            addObject("0001","Production");
            addObject("0002","Sales");
            addObject("0003","Maintenance");
            addObject("0004","Finance and Controlling");
            sortObjectsByText();
        }
        public String getId() { return m_departmentId; }
        public void setId(String id) { m_departmentId = id; }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_departmentId = "0001";
    String m_literalId;
    
    CCComboFieldIdText m_cfDepartment = new CCComboFieldIdText(new DepartmentLogic());
    CCComboFieldIdText m_cfLiteral = new CCComboFieldIdText(new LogicByResource("org.eclnt.jsfserver.i18n.resources.Literals")
    {
        public String getId() { return m_literalId; }
        public void setId(String id) { m_literalId = id; }
    });
    
    public CCComboFieldIdText getCfLiteral() { return m_cfLiteral; }
    public CCComboFieldIdText getCfDepartment() { return m_cfDepartment; }

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCComboFieldIdText(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
    }

    public String getPageName() { return "/workplace/demopbccccombofieldidtext.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCComboFieldIdText}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------


    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
