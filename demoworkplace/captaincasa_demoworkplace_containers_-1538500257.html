The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Scroll Pane - Extended Functions
  • Scroll to certain Position

Within the layout that is placed into a SCROLLPANE component, you can define SCROLLANCHOR components.

These SCROLLANCHOR components are invisible components. Their only purpose: they provide a trigger, so that you can make the SCROLLPANE scrolling to the anchor's position from the server side.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoScrollAnchor}")

public class DemoScrollAnchor
    extends PageBean 
    implements Serializable
{
    protected Trigger m_trigger1 = new Trigger();
    protected Trigger m_trigger2 = new Trigger();
    protected Trigger m_trigger3 = new Trigger();
    protected Trigger m_trigger4 = new Trigger();
    
    public DemoScrollAnchor(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);  
        m_trigger4.trigger();
    }

    public Trigger getTrigger1() { return m_trigger1; }
    public void setTrigger1(Trigger value) { m_trigger1 = value; }

    public Trigger getTrigger2() { return m_trigger2; }
    public void setTrigger2(Trigger value) { m_trigger2 = value; }
    
    public Trigger getTrigger3() { return m_trigger3; }
    public void setTrigger3(Trigger value) { m_trigger3 = value; }
    
    public Trigger getTrigger4() { return m_trigger4; }
    public void setTrigger4(Trigger value) { m_trigger4 = value; }

    public void onScroll1(ActionEvent event) { m_trigger1.trigger(); }
    public void onScroll2(ActionEvent event) { m_trigger2.trigger(); }
    public void onScroll3(ActionEvent event) { m_trigger3.trigger(); }
    public void onScroll4(ActionEvent event) { m_trigger4.trigger(); }

}
