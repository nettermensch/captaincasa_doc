The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Dynamic Layout
  • Direct binding of objects
  • Hello world - with lambdas

Again the dynamically created Hello-World-dialog - this time implementing the dynamic attributes using lambdas.

For each type of attribute there is a corresponding "lambda-aware" delegation class: StringDelegationLambda, FloatDelegationLambda, DoubleDelgationLambda etc.

When creating instance you have to pass one lambda implementation for the get-implementation of the attribute - and you may pass a second one for the set-implementation, if required by the component.

For the implementation of the action-listener you can directly use a lambda-implementation of IActionListenerDelegation.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoDynamicContentbindingLambda}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoDynamicContentbindingLambda}" >
<t:rowdynamiccontent id="g_2" contentbinding="#{d.DemoDynamicContentbindingLambda.dynamicContent}" />
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

    package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.COLDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FOLDABLEPANENode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.util.valuemgmt.impl.StringDelegationLambda;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDynamicContentBindingLambda}")

public class DemoDynamicContentbindingLambda
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ROWDYNAMICCONTENTBinding m_dynamicContent = new ROWDYNAMICCONTENTBinding();
    String m_name;
    String m_result;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDynamicContentbindingLambda(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);   
        renderContent();
    }

    public String getPageName() { return "/workplace/demodynamiccontentbindinglambda.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDynamicContentBindingLambda}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getDynamicContent() { return m_dynamicContent; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderContent()
    {
        FOLDABLEPANENode p = new FOLDABLEPANENode()
                .setText("Hello world - Value/action binding by lambdas")
                .setWidth("100%")
                .setRowdistance(10);
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            FIELDNode fn = new FIELDNode()
                    .setWidth(300)
                    .setAlign("center")
                    .bindText(new StringDelegationLambda
                     (
                         () -> {return m_name;},
                         (value) -> {m_name = value;}
                     ));
            r.addSubNode(fn);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }        
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            BUTTONNode bu = new BUTTONNode()
                    .setText("Say hello!")
                    .setWidth("200")
                    .bindActionListener((event) -> 
                        {
                            if (m_name == null)
                                m_result = "No name defined.";
                            else
                                m_result = "Hello world, " + m_name;
                        }
                    );
            r.addSubNode(bu);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }        
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
            LABELNode l = new LABELNode()
                    .setWidth("100%")
                    .setAlign("center")
                    .bindText(new StringDelegationLambda(() -> {return m_result;}));
            r.addSubNode(l);
            r.addSubNode(new COLDISTANCENode().setWidth("100%"));
        }    
        m_dynamicContent.setContentNode(p);
    }
}
