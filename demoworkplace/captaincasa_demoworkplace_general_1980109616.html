The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Many Columns...

This page demonstrates the performance of grids with many contained columns. The grid has 40 columns, each one being rendered as fully available field control (could also be combo box, radio button or any other control). The number of rows which are rendered as maximum is 20 (in this example).

On server side the number of rows is much higher. Only these rows are brought to client side which are visible to the user. We call this "server side scrolling".

This is the Java code:

package workplace;

import java.util.HashMap;
import java.util.Map;
import java.util.Random;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.util.IAcceleratedPropertyAccess;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoBigGrid extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class Row 
        extends FIXGRIDItem
        implements IAcceleratedPropertyAccess
    {
        Map<String,Integer> m_v = new HashMap<String,Integer>();
        public Map<String,Integer> getV() { return m_v; }
        public String getBackground() { return getBackgroundAll(); }
        public boolean getEnabled() { return true; }
        // IAcceleratedPropertyAccess
        public Object getPropertyValue(String property)
        {
            if ("enabled".equals(property)) return true;
            if ("background".equals(property)) return getBackgroundAll();
            return NOT_AVAILABLE;
        }
    }
    
    public class MyGrid extends FIXGRIDListBinding<Row>
    {
        public boolean checkIfChangeIndexIsSupported()
        {
            return true;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static Random s_random = new Random();
    static int NUM_OF_ROWS = 100;
    static int NUM_OF_COLS = 40;
    MyGrid m_rows = new MyGrid();
    
    String m_backgroundAll = "#FFFFFF";
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoBigGrid(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // create rows
        for (int i=0; i<NUM_OF_ROWS; i++)
        {
            Row r = new Row();
            m_rows.getItems().add(r);
        }
        updateRowData();
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public MyGrid getRows() { return m_rows; }
    
    public int getNumberOfFixColumns() { return m_rows.getNumberOfFixColumns(); } 
    public void setNumberOfFixColumns(int numberOfFixColumns) { m_rows.setNumberOfFixColumns(numberOfFixColumns); }

    public String getBackgroundAll() { return m_backgroundAll; }
    public void setBackgroundAll(String backgroundAll) { m_backgroundAll = backgroundAll; }

    public void onBackgroundAllAction(ActionEvent event)
    {
        for (Row r: m_rows.getItems())
            r.getChangeIndex().indicateChange();
    }
    
    public void onUpdateData(ActionEvent ae)
    {
        updateRowData();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void updateRowData()
    {
        for (int i=0; i<NUM_OF_ROWS; i++)
        {
            Row r = m_rows.getItems().get(i);
            for (int j=0; j<NUM_OF_COLS; j++)
            {
                r.m_v.put("c"+j,s_random.nextInt(99));
                r.getChangeIndex().indicateChange();
            }
        }
    }

}
