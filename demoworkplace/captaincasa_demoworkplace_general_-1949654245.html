The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Workplace functions as component (PBC)

Since update 20210118 the workplace function tree is available as proper page bean component within the addons.

The interface is nearly the same as the one that you might know from existing workplace functions, but now the page is available as real page bean. Which means: you can define multiple instances on one or several pages which run completely separated from one another.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCWorkplaceFunctions;
import org.eclnt.ccaddons.pbc.CCWorkplaceSearchViaPopup;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageStartInfo;
import org.eclnt.workplace.WorkplaceFunctionTreeInfoManager;
import org.eclnt.workplace.WorkplaceFunctionTreeInfoNode;

@CCGenClass(expressionBase = "#{d.DemoPBCCCWorkplaceFunctions}")

public class DemoPBCCCWorkplaceFunctions 
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCWorkplaceFunctions m_first;
    CCWorkplaceFunctions m_second;
    CCWorkplaceSearchViaPopup m_searchFirst;
    CCWorkplaceSearchViaPopup m_searchSecond;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCWorkplaceFunctions(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_first = new CCWorkplaceFunctions(getOwningDispatcher());
        m_second = new CCWorkplaceFunctions(getOwningDispatcher());
        m_searchFirst = new CCWorkplaceSearchViaPopup(getOwningDispatcher());
        m_searchSecond = new CCWorkplaceSearchViaPopup(getOwningDispatcher());
        {
            // read from configuration
            WorkplaceFunctionTreeInfoNode root = WorkplaceFunctionTreeInfoManager.loadFunctionTreeInfo("default");
            m_first.prepare(root,null);
            m_searchFirst.prepare(m_first,null);
        }
        {
            // assemble via API
            WorkplaceFunctionTreeInfoNode root = buildFunctionTree();
            m_second.prepare(root,null);
            m_searchSecond.prepare(m_second,null);
        }
    }

    public String getPageName()
    {
        return "/workplace/demopbcccworkplacefunctions.jsp";
    }

    public String getRootExpressionUsedInPage()
    {
        return "#{d.DemoPBCCCWorkplaceFunctions}";
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCWorkplaceFunctions getFirst() { return m_first; } 
    public CCWorkplaceFunctions getSecond() { return m_second; } 
    public CCWorkplaceSearchViaPopup getSearchFirst() { return m_searchFirst; }
    public CCWorkplaceSearchViaPopup getSearchSecond() { return m_searchSecond; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private WorkplaceFunctionTreeInfoNode buildFunctionTree()
    {
        WorkplaceFunctionTreeInfoNode root = new WorkplaceFunctionTreeInfoNode();
        for (int i=0; i<5; i++)
        {
            WorkplaceFunctionTreeInfoNode top = new WorkplaceFunctionTreeInfoNode();
            top.setText("Top " + i);
            root.getSubNodes().add(top);
            for (int j=0; j<5; j++)
            {
                WorkplaceFunctionTreeInfoNode sub = new WorkplaceFunctionTreeInfoNode();
                sub.setText("Sub " + i);
                top.getSubNodes().add(sub);
                for (int k=0; k<5; k++)
                {
                    WorkplaceFunctionTreeInfoNode node = new WorkplaceFunctionTreeInfoNode();
                    node.setText("Node " + i);
                    WorkpageStartInfo wpsi = new WorkpageStartInfo();
                    node.setWorkpageStartInfo(wpsi);
                    wpsi.setPageBeanName("DemoHelloWorld");
                    sub.getSubNodes().add(node);
                }
            }
        }
        return root;
    }

}
