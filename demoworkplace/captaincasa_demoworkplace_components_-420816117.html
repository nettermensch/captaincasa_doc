The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Shape Charts

Drag and drop items from the left to the right. You may draw arrows between the items by picking the small rectangle on the right of an item and dropping it onto another item.

This demo shows how to use PAINTAREA and PAINTAREAITEM in combination with dynamic content and drag/drop management.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREALINEITEMNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.events.BaseActionEventFlush;
import org.eclnt.jsfserver.elements.events.BaseActionEventInvoke;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoShape2}")

public class DemoShape2
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    int m_scale100 = 100;
    public int getScale100() { return m_scale100; }
    public void setScale100(int value) { this.m_scale100 = value; }

    boolean m_withUserScaling;
    public boolean getWithUserScaling() { return m_withUserScaling; }
    public void setWithUserScaling(boolean value) { this.m_withUserScaling = value; }

    public float getScale() { return (float)m_scale100 / 100f; }
    public void setScale(float value) 
    { 
        m_scale100 = (int)Math.round(value * 100d);
        m_scale100 = m_scale100 - m_scale100 % 5;
    }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class RightShape
    {
        int i_x;
        int i_y;
        int i_width = BOXWIDTH;
        int i_height = BOXHEIGHT;
        int i_type;
        String i_text = "<undefined>";
        public RightShape(int x, int y, int type)
        {
            i_x = x;
            i_y = y;
            i_type = type;
        }
        public int getX() { return i_x; }
        public void setX(int x) { i_x = x; }
        public int getY() { return i_y; }
        public void setY(int y) { i_y = y; }
        public String getBounds() { return i_x + ";" + i_y + ";" + i_width + ";" + i_height; }
        public void setBounds(String s)
        {
            int[] coords = ValueManager.decodeStraightIntCSV(s);
            i_x = coords[0];
            i_y = coords[1];
            i_width = coords[2];
            i_height = coords[3];
        }
        public String getBoundsLinkAnchor()
        {
            return (i_x+i_width) + ";" + (i_y + i_height/2 - 4) + ";8;8";
        }
        public String getText() { return i_text; }
        public void setText(String text) { i_text = text; }
        public void onAction(ActionEvent event)
        {
            if (event instanceof BaseActionEventFlush)
            {
                // update lines
                renderRightContent();
            }
            else if (event instanceof BaseActionEventDrop)
            {
                BaseActionEventDrop e = (BaseActionEventDrop)event;
                String dis = e.getDragInfo(); 
                if (dis.startsWith("RIGHTSHAPECONNECT:"))
                {
                    int index = ValueManager.decodeInt(dis.substring(18),0);
                    RightShape fromRS = m_rightShapes.get(index);
                    RightShape toRS = this;
                    addLine(fromRS,toRS);
                }
            }
        }
        public void onLabelAction(ActionEvent event)
        {
            final DemoShape2Popup b = new DemoShape2Popup();
            b.prepare(i_text,new DemoShape2Popup.IListener()
            {
                public void reactOnOK()
                {
                    closePopup(b);
                    i_text = b.getText();
                }
            });
            ModelessPopup p = openModelessPopup(b,"",0,0,new ModelessPopup.IModelessPopupListener()
            {
                public void reactOnPopupClosedByUser()
                {
                    closePopup(b);
                }
            });
            p.setUndecorated(true);
            p.setCloseonclickoutside(true);
        }
    }
    
    public class RightShapeLine
    {
        public RightShapeLine(RightShape from, RightShape to)
        {
            i_from = from;
            i_to = to;
        }
        RightShape i_from;
        RightShape i_to;
        public void onAction(ActionEvent event)
        {
            if (event instanceof BaseActionEventInvoke)
            {
                BaseActionEventInvoke e = (BaseActionEventInvoke)event;
                final DemoShape2PopupLine b = new DemoShape2PopupLine();
                b.prepare(new DemoShape2PopupLine.IListener()
                {
                    public void reactOnRemove()
                    {
                        removeLine();
                        closePopup(b);
                        renderRightContent();
                    }
                });
                ModelessPopup p = openModelessPopup(b,"",0,0,new ModelessPopup.IModelessPopupListener()
                {
                    public void reactOnPopupClosedByUser()
                    {
                        closePopup(b);
                    }
                });
                p.setUndecorated(true);
                p.setCloseonclickoutside(true);
                // p.setLefTopReferenceComponentId("RIGHTPAINTAREA");
                p.setTop(e.getY());
                p.setLeft(e.getX());
            }
        }
        public void removeLine()
        {
            m_rightShapeLines.remove(this);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static String[] COLORS = new String[] {"#FFFFFF","#FFC0C0","#FFFFC0","#C0C0FF"};
    static int BOXWIDTH = 130;
    static int BOXHEIGHT = 50;
    
    boolean m_scrollByDragDrop = false;
    DYNAMICCONTENTBinding m_leftContent = new DYNAMICCONTENTBinding();
    DYNAMICCONTENTBinding m_rightContent = new DYNAMICCONTENTBinding();
    List<RightShape> m_rightShapes = new ArrayList<RightShape>();
    List<RightShapeLine> m_rightShapeLines = new ArrayList<RightShapeLine>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoShape2(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        renderLeftContent();
        renderRightContent();
    }

    public String getPageName() { return "/workplace/demoshape2.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoShape2}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DYNAMICCONTENTBinding getLeftContent() { return m_leftContent; }
    public DYNAMICCONTENTBinding getRightContent() { return m_rightContent; }
    public List<RightShape> getRightShapes() { return m_rightShapes; }
    public List<RightShapeLine> getRightShapeLines() { return m_rightShapeLines; }

    public void onRightPaintAreaAction(ActionEvent event) 
    {
        if (event instanceof BaseActionEventDrop)
        {
            BaseActionEventDrop e = (BaseActionEventDrop)event;
            String dis = e.getDragInfo(); 
            if (dis.startsWith("LEFTSHAPE:"))
            {
                int type = ValueManager.decodeInt(dis.substring(10),0);
                int left = e.getPixelX() + e.getDeltaX();
                int top = e.getPixelY() + e.getDeltaY();
                RightShape rs = new RightShape(left,top,type);
                m_rightShapes.add(rs);
                renderRightContent();
            }
            if (dis.startsWith("RIGHTSHAPE:"))
            {
                int index = ValueManager.decodeInt(dis.substring(11),0);
                RightShape rs = m_rightShapes.get(index);
                rs.setX(e.getPixelX());
                rs.setY(e.getPixelY());
                renderRightContent();
            }
        }
    }

    public boolean getScrollByDragDrop() { return m_scrollByDragDrop; }
    public void setScrollByDragDrop(boolean value) { this.m_scrollByDragDrop = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderLeftContent()
    {
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        for (int i=0; i<COLORS.length; i++)
        {
            int y = 10 +(i * (BOXHEIGHT + 10));  
            PAINTAREAITEMNode n = new PAINTAREAITEMNode()
                .setBounds("5;" + y + ";"+BOXWIDTH+";"+BOXHEIGHT)
                .setBackground(COLORS[i])
                .setBorder("#a0a0a0")
                .setDragsend("LEFTSHAPE:"+i);
            l.add(n);
        }
        m_leftContent.setContentNodes(l);
    }
    
    private void renderRightContent()
    {
        List<ComponentNode> l = new ArrayList<ComponentNode>();
        renderRightContentShapes(l);
        renderRightContentLines(l);
        m_rightContent.setContentNodes(l);
    }
    
    private void renderRightContentShapes(List<ComponentNode> l)
    {
        int counter = 0;
        for (RightShape rs: m_rightShapes)
        {
            PAINTAREAITEMNode n = new PAINTAREAITEMNode()
                .setBounds(pbx("rightShapes["+counter+"].bounds"))
                //.setDragsend("RIGHTSHAPE:"+counter)
                .setDropreceive("RIGHTSHAPECONNECT")
                .setMovingenabled(true)
                .setMovingwithcrosshair(true)
                .setResizingenabled(true)
                .setResizingwithcrosshair(true)
                .setFlush(true)
                .setActionListener(pbx("rightShapes["+counter+"].onAction"))
                .setFocusable(true);
            l.add(n);
            PANENode p = new PANENode();
            n.addSubNode(p);
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            {
                LABELNode la = new LABELNode()
                    .setWidth("100%")
                    .setBackground(COLORS[rs.i_type])
                    .setHeight("100%")
                    .setBorder("#a0a0a0")
                    .setAlign("center")
                    .setText(pbx("rightShapes["+counter+"].text"))
                    // .setInvokeevent("leftclick")
                    .setActionListener(pbx("rightShapes["+counter+"].onLabelAction"));
                r.addSubNode(la);
            }
            PAINTAREAITEMNode n2 = new PAINTAREAITEMNode()
                .setBounds(pbx("rightShapes["+counter+"].boundsLinkAnchor"))
                .setBackground("#FFFFFF")
                .setBorder("left:0;top:1;right:1;bottom:1;color:#00000020")
                .setDragsend("RIGHTSHAPECONNECT:"+counter);
            l.add(n2);
            counter++;
        }
    }
    
    private void renderRightContentLines(List<ComponentNode> l)
    {
        int counter = 0;
        for (RightShapeLine rsl: m_rightShapeLines)
        {
            int fromX = rsl.i_from.i_x + rsl.i_from.i_width + 5 + 4;
            int fromY = rsl.i_from.i_y + rsl.i_from.i_height / 2;
            int toX = rsl.i_to.i_x - 5;
            int toY = rsl.i_to.i_y + rsl.i_to.i_height / 2;
            int deltaX = toX - fromX;
            int deltaY = toY - fromY;
            PAINTAREALINEITEMNode n = new PAINTAREALINEITEMNode()
                .setBounds(fromX+";" +fromY+ ";"+deltaX+";"+deltaY)
                .setArrowto(1)
                .setInvokeevent("leftclick")
                .setActionListener(pbx("rightShapeLines["+counter+"].onAction"));
            l.add(n);
            counter++;
        }
    }
    
    private void addLine(RightShape from, RightShape to)
    {
        if (from == to) return;
        // check if already connected
        for (RightShapeLine rsl: m_rightShapeLines)
        {
            if (rsl.i_from == from && rsl.i_to == to)
                return;
        }
        // new connection
        m_rightShapeLines.add(new RightShapeLine(from,to));
        renderRightContent();
    }
}
