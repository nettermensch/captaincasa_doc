The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Drag & Drop
  • Caret / text-cursor position

In addition to the x,y position you also receive the caret position (text cursor position) as part of the event data.

You may react accordingly and e.g. place text exactly at the point where the user dropped the information.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDragDropCaretPosition}")

public class DemoDragDropCaretPosition
    extends PageBean 
    implements Serializable
{
    public void onTextAreaAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventDrop)
        {
            BaseActionEventDrop e = (BaseActionEventDrop)event;
            String dragInfo = e.getDragInfo();
            if (dragInfo.startsWith("mytext:"))
            {
                String text = dragInfo.substring("mytext:".length());
                int caretPosition = e.getCaretPosition();
                if (caretPosition < 0)
                    caretPosition = text.length();
                m_text = m_text.substring(0,caretPosition) + text + m_text.substring(caretPosition);
            }
        }
    }

    String m_text = "Lorem ipsum dolor sit amet, consetetur sadipscing elitr, sed diam nonumy eirmod "
        + "tempor invidunt ut labore et dolore magna aliquyam erat, sed diam voluptua. "
        + "At vero eos et accusam et justo duo dolores et ea rebum. Stet clita "
        + "kasd gubergren, no sea takimata sanctus est Lorem ipsum dolor sit amet.";
    public String getText() { return m_text; }
    public void setText(String value) { this.m_text = value; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener extends Serializable
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDragDropCaretPosition(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demodragdropcaretposition.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDragDropCaretPosition}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
