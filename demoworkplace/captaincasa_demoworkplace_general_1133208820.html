The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Open/Close Pages

The workplace API allows to directly close or open pages from your code.

When opening pages you may either use the declarative API in which a "WorkpageStartInfo"-instance is passed, containing all the configuration parameters that are also used when e.g. starting a workpage from a workplace function tree.

Or you may use an explicit API - in this case the creation of the page bean is completely yours, so that you can use the Java API of the page bean to configure it.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.BaseComponent;
import org.eclnt.jsfserver.elements.impl.FIELDComponentTag;
import org.eclnt.jsfserver.elements.impl.LABELComponentTag;
import org.eclnt.jsfserver.managedbean.DefaultDispatchedBean;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.workplace.IWorkpage;
import org.eclnt.workplace.IWorkpageContainer;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.Workpage;
import org.eclnt.workplace.WorkpageByPageBean;
import org.eclnt.workplace.WorkpageStartInfo;
import org.eclnt.workplace.WorkpageStarterFactory;

@CCGenClass (expressionBase="#{d.DemoWpClose}")

public class DemoWpClose 
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static int s_counter = 0;
    BaseComponent m_row;
    
    String m_firstName;
    String m_lastName;
    
    ModalPopup m_popup;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoWpClose(IDispatcher dispatcher)
    {
        super(dispatcher);
    }

    // ------------------------------------------------------------------------
    // public methods
    // ------------------------------------------------------------------------

    public void setFirstName(String value) { m_firstName = value; }
    public String getFirstName() { return m_firstName; }
    
    public void setLastName(String value) { m_lastName = value; }
    public String getLastName() { return m_lastName; }
    
    /**
     * Close the current page and removes it from the workplace.
     */
    public void onClose(ActionEvent event) 
    {
        try
        {
            getWorkpageContainer().closeWorkpage(getWorkpage());
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when closing: " + t.toString());
        }
    }

    /**
     * Opens a new workpage containing the hello world application.
     */
    public void onAddNewPageBean(ActionEvent event)
    {
        try
        {
            IWorkpageDispatcher wpd = (IWorkpageDispatcher)getOwningDispatcher().getTopOwner();
            IWorkpageContainer wpc = getWorkpageContainer();
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setPageBeanName("DemoHelloWorld");
            wpsi.setText("Hello world!");
            wpsi.setOpenMultipleInstances(true);
            WorkpageStarterFactory.getWorkpageStarter().startWorkpage(wpd,wpc,wpsi);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when adding page: " + t.toString());
        }
    }

    
    /**
     * Opens a new workpage containing the page bean "DemoWpClose".
     */
    int m_openCounter = 0;
    public void onAddNewSwitchToPage(ActionEvent event)
    {
        try
        {
            // same code as "onAddNewPage" - but now:
            // 1. passing an id
            // if a page with this id is already opened in the workplace then
            // no new page will be opened, but the existing one is pushed
            // to the front
            // 2. setting openMultipleInstances to false
            IWorkpageDispatcher wpd = (IWorkpageDispatcher)getOwningDispatcher().getTopOwner();
            IWorkpageContainer wpc = getWorkpageContainer();
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setJspPage("/workplace/demowpopened.jsp");
            wpsi.setText("Page HARRY");
            wpsi.setOpenMultipleInstances(false);
            wpsi.setId("HARRY_PAGE");
            m_openCounter++;
            IWorkpage startedWorkpage = WorkpageStarterFactory.getWorkpageStarter().startWorkpage(wpd,wpc,wpsi);
            DemoWPOpened wpo = (DemoWPOpened)startedWorkpage.getDispatcher().getDispatchedBean(DemoWPOpened.class);
            wpo.setCounter(m_openCounter);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when adding page: " + t.toString());
        }
    }

    /**
     * Opens a new workpage containing the page "/workplace/demominispread.jsp". 
     */
    public void onAddNewPage(ActionEvent event)
    {
        try
        {
            IWorkpageDispatcher wpd = (IWorkpageDispatcher)getOwningDispatcher().getTopOwner();
            IWorkpageContainer wpc = getWorkpageContainer();
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setJspPage("/workplace/demominispread.jsp");
            wpsi.setText("Spreadsheet Demo");
            wpsi.setOpenMultipleInstances(true);
            WorkpageStarterFactory.getWorkpageStarter().startWorkpage(wpd,wpc,wpsi);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when adding page: " + t.toString());
        }
    }

    /**
     * Opens the hello-world-page as workpage, being started as a 
     * popup dialog.
     * 
     */
    public void onOpenHelloWorld(ActionEvent ae)
    {
        try
        {
            IWorkpageDispatcher wpd = (IWorkpageDispatcher)getOwningDispatcher().getTopOwner();
            IWorkpageContainer wpc = getWorkpageContainer();
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setPageBeanName("DemoHelloWorld");
            wpsi.setText("Hello world");
            wpsi.setOpenMultipleInstances(true);
            wpsi.setOpenAsPopupByDefault(true);
            WorkpageStarterFactory.getWorkpageStarter().startWorkpage(wpd,wpc,wpsi);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when adding page: " + t.toString());
        }
    }

    /**
     * In this example the instance of the bean is directly created - and then
     * passed as existing instance into the workplace management.
     */
    public void onCreateAndOpenPageBean(ActionEvent event)
    {
        // create page bean
        DemoHelloWorld dhw = new DemoHelloWorld(null);
        dhw.setName("Harry");
        // open new workpage
        WorkpageByPageBean wp = new WorkpageByPageBean(getOwningDispatcher(),dhw,"Harry","Hello Harry",null,true);
        getWorkpageContainer().addWorkpage(wp);
    }
    
    
    public void onTestPageWithPopup(ActionEvent event) 
    {
        try
        {
            IWorkpageDispatcher wpd = getOwningDispatcher();
            IWorkpageContainer wpc = getWorkpageContainer();
            WorkpageStartInfo wpsi = new WorkpageStartInfo();
            wpsi.setJspPage("/workplace/demowpdoublepopup.jsp");
            wpsi.setText("Page with Popup");
            wpsi.setOpenMultipleInstances(true);
            IWorkpage wp = WorkpageStarterFactory.getWorkpageStarter().startWorkpage(wpd,wpc,wpsi);
            // prepare managed bean
            DemoWpDoublePopup dp = (DemoWpDoublePopup)wp.getDispatcher().getDispatchedBean(DemoWpDoublePopup.class);
            dp.prepare();
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert("Error when adding page: " + t.toString());
        }
    }
    
    public void onCloseAllWorkpages(ActionEvent event)
    {
        List<IWorkpage> wps = getWorkpageContainer().getAllWorkpagesIncludingSubWorkpageContainers();
        for (IWorkpage wp: wps)
        {
            if (wp == getWorkpage()) continue; // do not close this workpage
            if (wp.isCloseSupported() == false) continue; // pages that cannot be closed are NOT closed
            getWorkpageContainer().closeWorkpage(wp);
        }
    }

    // ------------------------------------------------------------------------
    // private methods
    // ------------------------------------------------------------------------
    

}
