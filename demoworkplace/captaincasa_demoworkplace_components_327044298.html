The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • 3D Animations
  • Cube rotation - Filter-list-detail-scenario

This demo shows how to use the component CUBEROTATOR for providing a "reporting cube":

On the top side you define the filter criteria, the left side shows the list of found items - and the right side shows a selected item.

Rotation are either implicitly triggered - e.g. by button "Show List" - or explicitly triggered by the cube buttons on the bottom right.

Please note:

1. this is a graphical demo only - there are no serious functions behind the screens!
2. all parts of the screen are included in the layout - normally you would better separate each sub-page (filter, list, detail) in a separate page.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoCubeRotatorForLists}")

public class DemoCubeRotatorForLists
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    final String IMAGE_TOP = "/images/pictures/cubeTop.png"; 
    final String IMAGE_LEFT = "/images/pictures/cubeLeft.png"; 
    final String IMAGE_RIGHT = "/images/pictures/cubeRight.png"; 
    
    final int STATUS_FILTER = 0;
    final int STATUS_LIST = 1;
    final int STATUS_DETAIL = 2;
    
    CCDataGridView2<Person> m_grdiView = new CCDataGridView2<Person>();
    Person m_selectedPerson;

    int m_status = STATUS_FILTER;
    Trigger m_rotationTrigger = new Trigger();
    String m_rotationDirection;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCubeRotatorForLists(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
    }

    public String getPageName() { return "/workplace/democuberotatorforlists.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCubeRotatorForLists}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getCubeImage() 
    { 
        switch (m_status)
        {
            case STATUS_FILTER:
                return IMAGE_TOP;
            case STATUS_LIST:
                return IMAGE_LEFT;
            case STATUS_DETAIL:
                return IMAGE_RIGHT;
            default:
                return null;
        } 
    }

    public boolean getRenderedRight() { return m_status == STATUS_DETAIL; }
    public boolean getRenderedLeft() { return m_status == STATUS_LIST; }
    public boolean getRenderedTop() { return m_status == STATUS_FILTER; }

    public String getRotationDirection() { return m_rotationDirection; }
    public void setRotationDirection(String value) { this.m_rotationDirection = value; }

    public Trigger getRotationTrigger() { return m_rotationTrigger; }

    public CCDataGridView2<Person> getGrdiView() { return m_grdiView; }
    public Person getSelectedPerson() { return m_selectedPerson; }

    public void onRightAction(javax.faces.event.ActionEvent event) 
    {
        if (m_status == STATUS_FILTER)
            rotate("up");
        if (m_status == STATUS_LIST)
            rotate("left");
        m_status = STATUS_DETAIL;
    }

    public void onLeftAction(javax.faces.event.ActionEvent event) 
    {
        if (m_status == STATUS_FILTER)
            rotate("up");
        if (m_status == STATUS_DETAIL)
            rotate("right");
        m_status = STATUS_LIST;
    }

    public void onTopAction(javax.faces.event.ActionEvent event) 
    {
        rotate("down");
        m_status = STATUS_FILTER;
    }

    public void onShowListAction(javax.faces.event.ActionEvent event) 
    {
        rotate("up");
        fillGridView();
        m_status = STATUS_LIST;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGridView()
    {
        int size = new Random().nextInt(300);
        List<Person> ps = new ArrayList<Person>();
        for (int i=0; i<size; i++)
        {
            Person p = RandomLogic.createPerson();
            ps.add(p);
        }
        m_grdiView.prepare(ps,new CCDataGridView2.DefaultListener<Person>()
        {
            @Override
            public void reactOnGridExecute(Person object)
            {
                showDetailOfPerson(object);
            }
        });
    }

    protected void showDetailOfPerson(Person object)
    {
        rotate("left");
        m_status = STATUS_DETAIL;
        m_selectedPerson = object;
    }

    private void rotate(String direction)
    {
        m_rotationTrigger.trigger();
        m_rotationDirection = direction;
    }
    
}
