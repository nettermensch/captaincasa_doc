The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid Row Selector
  • Types

The component GRIDROWSELECTOR simplifies the selection of grid rows. Dependent on whether a grid row is selected or unselected, a corresponding icon will show up.

The component typically is placed at the very beginning of a grid.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridSelector}")

public class DemoGridSelector
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_name;
        public String getName() { return m_name; }
        public void setName(String value) { m_name = value; }

    }
    
    protected FIXGRIDListBinding<GridItem> m_grid1 = new FIXGRIDListBinding<GridItem>();
    protected FIXGRIDListBinding<GridItem> m_grid2 = new FIXGRIDListBinding<GridItem>();
    protected FIXGRIDListBinding<GridItem> m_grid3 = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoGridSelector(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<10; i++)
        {
            GridItem gi1 = new GridItem();
            gi1.setName("Name " + i);
            m_grid1.getItems().add(gi1);
            GridItem gi2 = new GridItem();
            gi2.setName("Name " + i);
            m_grid2.getItems().add(gi2);
            GridItem gi3 = new GridItem();
            gi3.setName("Name " + i);
            m_grid3.getItems().add(gi3);
        }
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid1() { return m_grid1; }
    public FIXGRIDListBinding<GridItem> getGrid2() { return m_grid2; }
    public FIXGRIDListBinding<GridItem> getGrid3() { return m_grid3; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
