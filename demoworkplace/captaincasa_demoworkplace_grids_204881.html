The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Bean data grid (PBC)
  • With popup menu

This example demonstrates how to add popup menu items to the data grid view.

1. set the initialization parameter "withPopupMenu" to true in the page bean component
2. implement the methods "reactOnPopupMenuLoad" and "reactOnPopupMenuLoadInvoke" within your implementation

Please not: any time you update the list of objects that is managed within the grid processing, then you have to explicit refresh the grid by calling method "refreshData".

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.ccee.util.CloneUtil;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup.IYesNoListener;
import org.eclnt.jsfserver.elements.componentnodes.MENUITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoPBCCCDataGridView2PopupMenu}")

public class DemoPBCCCDataGridView2PopupMenu
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /** 
     * The bean... This normally is an own class. In the demo it
     * is an inner class so that all code is kept within the
     * demo class.
     */
    public static class TestPerson implements Serializable
    {
        String i_firstName;
        String i_lastName;
        int i_height;
        LocalDate i_birthDate;
        boolean i_married;
        public String getFirstName() { return i_firstName; }
        public String getLastName() { return i_lastName; }
        public int getHeight() { return i_height; }
        public LocalDate getBirthDate() { return i_birthDate; }
        public boolean isMarried() { return i_married; }
        public void setFirstName(String firstName) { i_firstName = firstName; }
        public void setLastName(String lastName) { i_lastName = lastName; }
        public void setHeight(int height) { i_height = height; }
        public void setBirthDate(LocalDate birthDate) { i_birthDate = birthDate; }
        public void setMarried(boolean married) { i_married = married; }
    }

    /**
     * Listener to the data grid view's listener so that in order to 
     * react on events of the grid processing.
     */
    public class MyDGVListener extends CCDataGridView2.DefaultListener<TestPerson>
    {
        @Override
        public void reactOnGridSelect(TestPerson object)
        {
            reactOnItemSelection(object);
        }
        @Override
        public void reactOnPopupMenuLoad(TestPerson object, DYNAMICCONTENTBinding dynPopupMenuContent)
        {
            List<ComponentNode> l = new ArrayList<ComponentNode>();
            l.add(new MENUITEMNode()
                    .setText("Delete Item: " + object.getLastName() + ", " + object.getFirstName())
                    .setCommand("DELETEITEM"));
            l.add(new MENUITEMNode()
                    .setText("Duplicate Item: " + object.getLastName() + ", " + object.getFirstName())
                    .setCommand("DUPLICATEITEM"));
            dynPopupMenuContent.setContentNodes(l);
        }
        @Override
        public void reactOnPopupMenuItemInvoke(TestPerson object, BaseActionEventPopupMenuItem event)
        {
            if ("DELETEITEM".equals(event.getCommand()))
            {
                deleteItem(object);
            }
            else if ("DUPLICATEITEM".equals(event.getCommand()))
            {
                duplicateItem(object);
            }
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataGridView2<TestPerson> m_dataView = new CCDataGridView2<TestPerson>(TestPerson.class);
    List<TestPerson> m_persons = new ArrayList<TestPerson>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDataGridView2PopupMenu(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillData();
    }

    public String getPageName() { return "/workplace/demopbcccdatagridview2popupmenu.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCDataGridView2PopupMenu}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2<TestPerson> getDataView() { return m_dataView; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void reactOnItemSelection(TestPerson object)
    {
        Statusbar.outputSuccess("Selection of item: " + object.getLastName() + ", " + object.getFirstName());
    }

    private void deleteItem(final TestPerson object)
    {
        YESNOPopup.createInstance("Deletion of item","Do you really want to remove the selected item?",new IYesNoListener()
        {
            @Override
            public void reactOnYes() { deleteItemExecute(object); }
            @Override
            public void reactOnNo() {}
        });
    }

    private void deleteItemExecute(TestPerson object)
    {
        m_persons.remove(object);
        m_dataView.refreshData(m_persons);
    }

    private void duplicateItem(TestPerson object)
    {
        TestPerson clone = CloneUtil.createClone(object);
        int index = m_persons.indexOf(object);
        m_persons.add(index+1,clone);
        m_dataView.refreshData(m_persons);
    }

    private void fillData()
    {
        m_persons.clear();
        for (int i=0; i<100; i++)
        {
            TestPerson p = new TestPerson();
            p.setFirstName(RandomLogic.getFirstName());
            p.setLastName(RandomLogic.getLastName());
            p.setHeight(RandomLogic.getHeight());
            p.setMarried(i%2 == 0);
            p.setBirthDate(RandomLogic.getBirthDate());
            m_persons.add(p);
        }
        m_dataView.prepare(m_persons,new MyDGVListener());
    }

}
