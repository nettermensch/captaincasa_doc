The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Tree
  • Data Tree (PBC)

The Page Bean Component CCDataTree converts a flat data item collection into a tree.

You can easily set up the columns by which the tree is organized and you can easily set up the calculation of figures.

Please note: you must only add String-parameters or BigDecimal-parameters into the data item map!

The Page Bean Component CCDataTree is available within the pagebean-addons package (eclnt_pbc.jar).

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Random;

import org.eclnt.ccaddons.pbc.CCDataTree;
import org.eclnt.ccaddons.pbc.CCDataTree.DataColumnInfo;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCDataTree}")

public class DemoPBCCCDataTree
    extends DemoPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataTree m_dataTree = new CCDataTree();
    int m_tabIndex;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCDataTree(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        init();
        initView0();
    }

    public String getPageName() { return "/workplace/demopbcccdatatree.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCDataTree}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataTree getDataTree() { return m_dataTree; }

    public int getTabIndex() { return m_tabIndex; }
    public void setTabIndex(int value) { this.m_tabIndex = value; }

    public void onTabAction(javax.faces.event.ActionEvent event) 
    {
        switch (m_tabIndex) 
        {
            case 0: initView0(); break;
            case 1: initView1(); break;
            case 2: initView2(); break;
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void init()
    {
        List<Map<String,Object>> data = loadData();
        // load data into tree
        m_dataTree.prepareData(data,new CCDataTree.DefaultListener()
        {
            public void reactOnSelect(List<String> keyValues)
            {
            }
            public void reactOnExecute(List<String> keyValues)
            {
            }
        });
    }

    private void initView0()
    {
        List<String> treeColumns = new ArrayList<String>();
        treeColumns.add("COUNTRY");
        treeColumns.add("PRODUCTGROUP");
        treeColumns.add("CUSTOMERCATEGORY");
        initView(treeColumns,"Country / Product Group / Customer Category");
    }
    
    private void initView1()
    {
        List<String> treeColumns = new ArrayList<String>();
        treeColumns.add("PRODUCTGROUP");
        treeColumns.add("COUNTRY");
        treeColumns.add("CUSTOMERCATEGORY");
        initView(treeColumns,"Product Group / Country / Customer Category");
    }
    
    private void initView2()
    {
        List<String> treeColumns = new ArrayList<String>();
        treeColumns.add("CUSTOMERCATEGORY");
        treeColumns.add("COUNTRY");
        treeColumns.add("PRODUCTGROUP");
        initView(treeColumns,"Customer Category / Country / Product Group");
    }
    
    private void initView(List<String> treeColumns,String treeTitle)
    {
        List<CCDataTree.DataColumnInfo> dataColumns = new ArrayList<CCDataTree.DataColumnInfo>();
        dataColumns.add(new DataColumnInfo("REVENUE","Revenue",CCDataTree.EnumCalculation.TOTAL,"dec2"));
        dataColumns.add(new DataColumnInfo("PIECES","Pieces",CCDataTree.EnumCalculation.TOTAL,"dec0"));
        dataColumns.add(new DataColumnInfo("REVENUE","Average revenue",CCDataTree.EnumCalculation.AVERAGE,"dec0"));
        m_dataTree.prepareView(treeColumns,dataColumns,treeTitle);
        m_dataTree.sortByDataColumn(dataColumns.get(0),false);
    }
    
    private List<Map<String,Object>> loadData()
    {
        Random r = new Random();
        String[] countries = new String[] 
        {
            "Germany",
            "Austria",
            "United Kingdom",
            "United States",
            "Italy",
            "France"
        };
        String[] productGroups = new String[]
        {
             "Hardware",
             "Software",
             "Consumer Electronics",
             "Books",
             "Food",
             "Beverage",
             "Services"
        };
        String[] customerCategories = new String[]
        {
             "Private",
             "Retailer",
             "Business"
        };
        List<Map<String,Object>> result = new ArrayList<Map<String,Object>>();
        for (String country: countries)
        {
            for (String productGroup: productGroups)
            {
                for (String customerCategory: customerCategories)
                {
                    Map<String,Object> dataItem = new HashMap<String, Object>();
                    dataItem.put("COUNTRY",country);
                    dataItem.put("PRODUCTGROUP",productGroup);
                    dataItem.put("CUSTOMERCATEGORY",customerCategory);
                    BigDecimal revenue = new BigDecimal(r.nextInt(100000));
                    BigDecimal pieces = new BigDecimal(r.nextInt(1000));
                    dataItem.put("REVENUE",revenue);
                    dataItem.put("PIECES",pieces);
                    result.add(dataItem);
                }
            }
        }
        return result;
    }
}
