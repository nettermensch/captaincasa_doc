The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Slide Container
  • Basics

The SLIDECONTAINER is a container around (exactly!) one content component. If the component exceeds the size of the container then the user can slide the content by mouse (desktop) or by finger (touch device).

You may place any content inside the SLIDECONTAINER. In the example a REPEAT component is used to produce quite a lot of colored rectangles.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventSlideContainer;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoSlideContainer}")

public class DemoSlideContainer
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class ColorsItem extends Object implements java.io.Serializable
    {
        String m_color;
        public String getColor() { return m_color; }
        public void setColor(String value) { this.m_color = value; }

    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String[] hexs = new String[] {"FF","C0","80","40","00"};
    List<ColorsItem> m_colors = new ArrayList<ColorsItem>();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoSlideContainer(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // create colors
        for (int ir=0; ir<hexs.length; ir++)
        {
            for (int ig=0; ig<hexs.length; ig++)
            {
                for (int ib=0; ib<hexs.length; ib++)
                {
                    ColorsItem c = new ColorsItem();
                    c.m_color = "#" + hexs[ir] + hexs[ig] + hexs[ib];
                    m_colors.add(c);
                }
            }
        }
    }

    public String getPageName() { return "/workplace/demoslidecontainer.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSlideContainer}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<ColorsItem> getColors() { return m_colors; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
}
