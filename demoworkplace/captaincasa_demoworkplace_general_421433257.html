The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • General Issues
  • Complex expressions

An expression binds a component attribute to some Java property.

By default we recommend to use straight expressions - so that the full logic of a dialog is contained in the corresponding server side class.

For certain, limited scenarios we "allow" to use certain complex expressions as well. - Please strictly keep to the expressions that are listed in this example: CaptainCasa only guarantees these ones to properly work!

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoExpressions}")

public class DemoExpressions
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        int i_height;
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public int getHeight() { return i_height; }
        public void setHeight(int value) { this.i_height = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    public class PersonsItem extends Object implements java.io.Serializable
    {
        int i_height;
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public int getHeight() { return i_height; }
        public void setHeight(int value) { this.i_height = value; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_someNumber = 0;
    String m_someName = "Captain";
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    List<PersonsItem> m_persons = new ArrayList<PersonsItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoExpressions(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        initGridData();
        initListData();
    }

    public String getPageName() { return "/workplace/demoexpressions.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoExpressions}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public List<PersonsItem> getPersons() { return m_persons; }

    public String getSomeName() { return m_someName; }
    public void setSomeName(String value) { this.m_someName = value; }

    public int getSomeNumber() { return m_someNumber; }
    public void setSomeNumber(int value) { this.m_someNumber = value; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void initGridData()
    {
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName(RandomLogic.getFirstName());
            gi.setLastName(RandomLogic.getLastName());
            if (i % 5 == 0)
                gi.setHeight(80);
            m_grid.getItems().add(gi);
        }
    }
    
    private void initListData()
    {
        for (int i=0; i<10; i++)
        {
            PersonsItem gi = new PersonsItem();
            gi.setFirstName(RandomLogic.getFirstName());
            gi.setLastName(RandomLogic.getLastName());
            if (i % 5 == 0)
                gi.setHeight(80);
            m_persons.add(gi);
        }
    }

}
