The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Blockable Area
  • "Semi-modal" popup dialogs

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:row id="g_1" >
<t:blockablearea id="g_2" blocked="#{d.DemoBlockableAreaModelessPopup.blocked}" height="100%" width="100%" >
<t:scrollpane id="g_3" background="#ffffff" height="100%" padding="20" width="100%" >
<t:row id="g_4" >
<t:button id="g_5" actionListener="#{d.DemoBlockableAreaModelessPopup.onOpenPopupAction}" text="Open &#34;semi-modal&#34; popup" />
</t:row>
</t:scrollpane>
</t:blockablearea>
</t:row>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModelessPopup;
import org.eclnt.jsfserver.pagebean.IPageBean;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoBlockableAreaModelessPopup}")

public class DemoBlockableAreaModelessPopup
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    IPageBean m_currentPopupPageBean;
    ModelessPopup m_currentPopup;
    int m_popupWidth = 800;
    int m_popupHeight = 600;
    int m_popupLeft = -1;
    int m_popupTop = -1;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoBlockableAreaModelessPopup(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnShownInContentArea()
            {
                showCurrentPopup();
            }
            @Override
            public void reactOnHiddenInContentArea()
            {
                hideCurrentPopup();
            }
            
            @Override
            public void closeForced()
            {
                closeCurrentPopup();
                super.closeForced();
            }
        });
    }

    public String getPageName() { return "/workplace/demoblockableareamodelesspopup.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoBlockableAreaModelessPopup}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------


    public boolean getBlocked() 
    { 
        if (m_currentPopupPageBean == null) 
            return false; 
        else
            return true;
    }

    public void onOpenPopupAction(javax.faces.event.ActionEvent event) 
    {
        if (m_currentPopup != null)
            return;
        if (m_currentPopupPageBean == null)
            m_currentPopupPageBean = new DemoHelloWorld(getOwningDispatcher());
        showCurrentPopup();
    }
    
    private void showCurrentPopup()
    {
        if (m_currentPopup == null)
        {
            m_currentPopup = openModelessPopup(m_currentPopupPageBean,"My popup",m_popupWidth,m_popupHeight,new ModelessPopup.IModelessPopupListener()
            {
                public void reactOnPopupClosedByUser()
                {
                    closeCurrentPopup();
                }
            });
            if (m_popupLeft >= 0) m_currentPopup.setLeft(m_popupLeft);
            if (m_popupTop >= 0) m_currentPopup.setTop(m_popupTop);
        }
    }
    
    private void closeCurrentPopup()
    {
        if (m_currentPopupPageBean != null)
        {
            closePopup(m_currentPopupPageBean);
            m_currentPopupPageBean = null;
            m_currentPopup = null;
        }
    }
    
    private void hideCurrentPopup()
    {
        if (m_currentPopupPageBean != null)
        {
            closePopup(m_currentPopupPageBean);
            m_popupWidth = m_currentPopup.getWidth();
            m_popupHeight = m_currentPopup.getHeight();
            m_popupLeft = m_currentPopup.getLeft();
            m_popupTop = m_currentPopup.getTop();
            m_currentPopup = null;
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
