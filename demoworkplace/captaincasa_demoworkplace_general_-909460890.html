The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • Titlebar with extensions (PBC)

The page bean component "CCWorkpageTitle" builds a title bar on top of your workpage that you can extend by own components.

You need to start the corresponding workpage in undecorated mode (WorkpageStartInfo.decorated to be defined as "false").

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.ccaddons.pbc.CCWorkpageTitle;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPBCCCWorkpageTitle}")

public class DemoPBCCCWorkpageTitle
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCWorkpageTitle m_titleUI;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCWorkpageTitle(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_titleUI = new CCWorkpageTitle(getOwningDispatcher());
    }

    public String getPageName() { return "/workplace/demopbcccworkpagetitle.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCWorkpageTitle}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCWorkpageTitle getTitleUI() { return m_titleUI; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
