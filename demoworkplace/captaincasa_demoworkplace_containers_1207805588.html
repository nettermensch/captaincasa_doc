The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Layering and absolute postioning
  • Layered Pane with PAINTAREA

This demo shows a typical use case for the LAYEREDPANE component:

In the background layer some image is positioned.

In the foreground layer some buttons are arrangd using the PAINTAREA component, that allows to absolutely define the position of a component.

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.OKPopup;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoLayeredPanePaintArea}")

public class DemoLayeredPanePaintArea
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoLayeredPanePaintArea(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demolayeredpanepaintarea.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoLayeredPanePaintArea}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onButtonAction(ActionEvent event) 
    {
        BaseActionEvent e = (BaseActionEvent)event;
        String ref = e.getSourceReference();
        OKPopup.createInstance("Button pressed","The reference behind the button is: " + ref);
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
