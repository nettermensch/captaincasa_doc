The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Open combo dialog on focus

All components that open some detail dialog provide the attribute OPENONFOCUS.

If set to true then the detail dialog is directly opened when moving the focus into the component.

Moving the focus can be done by...

(.) clicking the component
(.) tab-ing into the component by keyboard
(.) programmatically requesting the focus

This is the Java code:

package workplace;

import java.io.Serializable;
import java.time.LocalDate;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoOpenOnFocus}")

public class DemoOpenOnFocus
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ValidValuesBinding m_departmentVVS = new ValidValuesBinding();
    String m_department;
    String m_personId;
    String m_someText = "Lorem ipsum dolor sit amet, consetetur sadipscing elitr, sed diam nonumy eirmod tempor invidunt.";
    LocalDate m_someDate = LocalDate.now();
    String m_someColor = "#FF0000";
 
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoOpenOnFocus(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_departmentVVS.addValidValue("SALES","Sales");
        m_departmentVVS.addValidValue("PROD","Production");
    }

    public String getPageName() { return "/workplace/demoopenonfocus.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoOpenOnFocus}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getDepartmentVVS() { return m_departmentVVS; }

    public String getDepartment() { return m_department; }
    public void setDepartment(String value) { this.m_department = value; }

    public String getPersonId() { return m_personId; }
    public void setPersonId(String value) { this.m_personId = value; }

    public String getSomeColor() { return m_someColor; }
    public void setSomeColor(String value) { this.m_someColor = value; }

    public void onPersonAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventValueHelp)
        {
            IdTextSelection idts = IdTextSelection.createInstance();
            idts.addLine("4711","Hirsch, Harry");
            idts.addLine("4712","Smith, Jonathan");
            idts.setCallBack(new ISetId()
            {
                @Override
                public void setId(String id)
                {
                    m_personId = id;
                }
            });
        }
    }

    public String getSomeText() { return m_someText; }
    public void setSomeText(String value) { this.m_someText = value; }

    public LocalDate getSomeDate() { return m_someDate; }
    public void setSomeDate(LocalDate value) { this.m_someDate = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
