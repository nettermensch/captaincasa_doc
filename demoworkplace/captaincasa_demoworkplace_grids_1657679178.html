The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Large Output Lists
  • Multi label - performance

This demo shows the difference in performance between individually rendered grid cell components - and MULTILABEL-based rendering of cells.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.Map;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWDISTANCENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWLINENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.MULTILABELBinding;
import org.eclnt.jsfserver.elements.impl.MULTILABELBinding.TextInfo;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.util.map.LazyLoadingMap;
import org.eclnt.util.map.LazyLoadingMap.ILazyLoader;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoMultilabelGridPerformance}")

public class DemoMultilabelGridPerformance
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Grid1Item extends FIXGRIDItem implements java.io.Serializable
    {
        Map<String,BigDecimal> i_data = new LazyLoadingMap<BigDecimal>();
        public Grid1Item()
        {
            ((LazyLoadingMap<BigDecimal>)i_data).setLoader(new ILazyLoader<BigDecimal>()
            {
                @Override
                public BigDecimal lazyLoad(String key)
                {
                    int v = s_random.nextInt(1000000);
                    return new BigDecimal(v);
                }
            });
        }
        public void onRowSelect() { }
        public void onRowExecute() { }
        public Map<String, BigDecimal> getData() { return i_data; }
    }

    public class Grid2Item extends FIXGRIDItem implements java.io.Serializable
    {
        MULTILABELBinding i_ml1 = new MULTILABELBinding();
        MULTILABELBinding i_ml2 = new MULTILABELBinding();
        public MULTILABELBinding getMl1() { return i_ml1; }
        public MULTILABELBinding getMl2() { return i_ml2; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        private void renderMultiLabels()
        {
            i_ml1.clear();
            i_ml2.clear();
            for (int i=0; i<m_nofColumns; i++)
            {
                {
                    TextInfo ti = new TextInfo();
                    BigDecimal v = new BigDecimal(s_random.nextInt(1000000));
                    String vt = ValueManager.convertObject2ClientDisplayString(v,"bigdecimal","dec2","CET",false,false);
                    ti.setText(vt);
                    ti.setWidth(100);
                    ti.setAlign("right");
                    i_ml1.addTextInfo(ti);
                }
                {
                    TextInfo ti = new TextInfo();
                    BigDecimal v = new BigDecimal(s_random.nextInt(1000000));
                    String vt = ValueManager.convertObject2ClientDisplayString(v,"bigdecimal","dec2","CET",false,false);
                    ti.setText(vt);
                    ti.setWidth(100);
                    ti.setAlign("right");
                    i_ml2.addTextInfo(ti);
                }
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ROWDYNAMICCONTENTBinding m_dynGridContent = new ROWDYNAMICCONTENTBinding();
    FIXGRIDListBinding<Grid2Item> m_grid2 = new FIXGRIDListBinding<Grid2Item>();
    FIXGRIDListBinding<Grid1Item> m_grid1 = new FIXGRIDListBinding<Grid1Item>();
    int m_nofColumns = 25;
    Random s_random = new Random();
    MULTILABELBinding m_mlTitle = new MULTILABELBinding();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoMultilabelGridPerformance(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);  
        fillGrids();
        renderGrid1();
        renderGrid2();
    }

    public String getPageName() { return "/workplace/demomultilabelgridperformance.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoMultilabelGridPerformance}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getDynGridContent() { return m_dynGridContent; }
    public FIXGRIDListBinding<Grid1Item> getGrid1() { return m_grid1; }

    public FIXGRIDListBinding<Grid2Item> getGrid2() { return m_grid2; }
    public MULTILABELBinding getMlTitle() { return m_mlTitle; }


    public int getNofColumns() { return m_nofColumns; }
    public void setNofColumns(int value) { this.m_nofColumns = value; }

    public void onRenderAction(javax.faces.event.ActionEvent event) 
    {
        renderGrid1();
        renderGrid2();
    }

    public int getGrid2Width() 
    { 
        return m_nofColumns * 100; 
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void renderGrid1()
    {
        FIXGRIDNode f = new FIXGRIDNode()
                .setWidth("100%")
                .setHeight("100%")
                .setObjectbinding(pbx("grid1"))
                .setRowheight(45);
        for (int i=0; i<m_nofColumns; i++)
        {
            GRIDCOLNode c = new GRIDCOLNode()
                    .setWidth("100")
                    .setText("Col " + i)
                    .setAlign("right");
            f.addSubNode(c);
            PANENode p = new PANENode();
            c.addSubNode(p);
            p.addSubNode(new ROWDISTANCENode().setHeight("100%"));
            {
                ROWNode r = new ROWNode();
                p.addSubNode(r);
                LABELNode l = new LABELNode()
                        .setText(".{data['value+"+(i*2)+"']}")
                        .setFormat("bigdecimal")
                        .setFormatmask("dec2")
                        .setAlign("right")
                        .setWidth("100%");
                r.addSubNode(l);
            }
            p.addSubNode(new ROWDISTANCENode().setHeight("100%"));
            p.addSubNode(new ROWLINENode().setHeight(1).setBackground("#00000008"));
            p.addSubNode(new ROWDISTANCENode().setHeight("100%"));
            {
                ROWNode r = new ROWNode();
                p.addSubNode(r);
                LABELNode l = new LABELNode()
                        .setText(".{data['value+"+(i*2+1)+"']}")
                        .setFormat("bigdecimal")
                        .setFormatmask("dec2")
                        .setAlign("right")
                        .setWidth("100%");
                r.addSubNode(l);
            }
            p.addSubNode(new ROWDISTANCENode().setHeight("100%"));
        }
        m_dynGridContent.setContentNode(f);
    }

    private void renderGrid2()
    {
        for (int i=0; i<m_nofColumns; i++)
        {
            TextInfo ti = new TextInfo();
            ti.setText("Col " + i);
            ti.setAlign("right");
            ti.setWidth(100);
            ti.setBackground("#00000000");
            ti.setFont("weight:bold");
            m_mlTitle.addTextInfo(ti);
        }
        for (Grid2Item gi: m_grid2.getItems())
        {
            gi.renderMultiLabels();
        }
    }

    private void fillGrids()
    {
        for (int i=0; i<100; i++)
        {
            m_grid1.getItems().add(new Grid1Item());
            m_grid2.getItems().add(new Grid2Item());
        }
    }

}
