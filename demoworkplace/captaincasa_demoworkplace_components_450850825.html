The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Reporting, PDF, SVG
  • Radial

The component RADIAL renders a percentage value as circle (or ellipse).

You can define the line color and line size - and many other parameters...

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoRadial}")

public class DemoRadial
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_value = 50;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRadial(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        initRefComponent("radial"); 
    }

    public String getPageName() { return "/workplace/demoradial.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoRadial}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public int getValue() { return m_value; }
    public void setValue(int value) { this.m_value = value; }

    public String getLineColor() 
    { 
        if (m_value <= 25) return "#800000";
        if (m_value <= 60) return "#FFFF40";
        return "#008000";
    }

}
