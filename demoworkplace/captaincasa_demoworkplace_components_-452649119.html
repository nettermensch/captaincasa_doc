The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • 3D Animations
  • Cube rotation - Turning pages

This demo shows how to use the CUBEROTATOR component for turning pages.

Use the "Next" and "Previous" button to navigate through the pages.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.pagebean.IPageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoCubeRotatorPageTurning}")

public class DemoCubeRotatorPageTurning
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<IPageBean> m_pageBeans = new ArrayList<IPageBean>();
    int m_index = 0;
    String m_direction;
    Trigger m_trigger = new Trigger();
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCubeRotatorPageTurning(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        m_pageBeans.add(new DemoHelloWorld(workpageDispatcher));
        m_pageBeans.add(new DemoMinispread(workpageDispatcher));
        m_pageBeans.add(new DemoBudget(workpageDispatcher));
        m_pageBeans.add(new DemoGrid(workpageDispatcher));
    }

    public String getPageName() { return "/workplace/democuberotatorpageturning.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCubeRotatorPageTurning}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public IPageBean getContentBean() 
    { 
        return m_pageBeans.get(m_index % m_pageBeans.size()); 
    }

    public void onPrevAction(javax.faces.event.ActionEvent event) 
    {
        m_direction = "right";
        m_trigger.trigger();
        m_index--;
        if (m_index < 0)
            m_index = m_pageBeans.size()-1;
    }

    public void onNextAction(javax.faces.event.ActionEvent event) 
    {
        m_direction = "left";
        m_trigger.trigger();
        m_index++;
        if (m_index >= m_pageBeans.size())
            m_index = 0;
    }

    public String getDirection() { return m_direction; }
    public void setDirection(String value) { this.m_direction = value; }

    public Trigger getTrigger() { return m_trigger; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
