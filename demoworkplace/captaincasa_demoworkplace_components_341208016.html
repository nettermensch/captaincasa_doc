The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Reporting, PDF, SVG
  • Chart.js - Dynamic creation

This demo shows how the JSON configuration for the chart is created dynamically.

The code takes use of Google's GSON library for creating JSON out of objects.

Please check the documentation at http://www.chartjs.org/ in order to find details about how to construct the different objects in order to create diverse types of charts.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;

@CCGenClass (expressionBase="#{d.DemoChartJSDynamic}")

/*
 * Details on setting up JSON objects: http://www.chartjs.org/.
 * This program takes use of Google'S GSON-library for creating JSON Strings. 
 */
public class DemoChartJSDynamic
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_region;
        public String getRegion() { return i_region; }
        public void setRegion(String value) { this.i_region = value; }

        BigDecimal[] i_sales = new BigDecimal[4];
        public BigDecimal[] getSales() { return i_sales; }
    }

    // JSON Objects - just for writing shorter code...
    static class JO extends HashMap<String,Object> {}
    static class JOs extends ArrayList<JO> {}
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static final String[] REGIONS = new String[] {"North","West","South","East"}; 
    static final String[] COLORS = new String[] {"#FFC0C0","#C0FFC0","#C0C0FF","#C0C0C0"};
    
    static final String[] CATEGORIES = new String[] {"Hardware","Software","Other"}; 
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    String m_json;

    String m_chartType = "bar";
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoChartJSDynamic(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        int counter = 1;
        for (String region: REGIONS)
        {
            GridItem gi = new GridItem();
            gi.setRegion(region);
            gi.getSales()[0] = new BigDecimal(1*counter);
            gi.getSales()[1] = new BigDecimal(2*counter);
            gi.getSales()[2] = new BigDecimal(3*counter);
            gi.getSales()[3] = new BigDecimal(4*counter);
            m_grid.getItems().add(gi);
            counter++;
        }
        createChart(m_chartType);
    }

    public String getPageName() { return "/workplace/demochartjsdynamic.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoChartJSDynamic}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getJson() { return m_json; }

    public void onUpdateChartAction(ActionEvent event) { createChart(m_chartType); }
    
    public String getChartType() { return m_chartType; }
    public void setChartType(String value) { this.m_chartType = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void createChart(String type)
    {
        JO chart = new JO();
        chart.put("type",type);
        JO data = new JO();
        chart.put("data",data);
        {
            List<String> labels = new ArrayList<String>();
            for (String category: CATEGORIES) labels.add(category);
            data.put("labels",labels);
            JOs datasets = new JOs();
            data.put("datasets",datasets);
            for (int i=0; i<REGIONS.length; i++)
            {
                JO dataset = new JO();
                datasets.add(dataset);
                List<BigDecimal> datasetData = new ArrayList<BigDecimal>();
                for (int j=0; j<CATEGORIES.length; j++)
                    datasetData.add(m_grid.getItems().get(i).getSales()[j]);
                dataset.put("label",REGIONS[i]);
                dataset.put("data",datasetData);
                dataset.put("fill",true);
                dataset.put("backgroundColor",COLORS[i]);
            }
        }
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        m_json = gson.toJson(chart);
    }
    
}
