The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Field Control
  • Date Input
  • Date/Time Touch Input (PBC)

This demo shows the usage of page beans components CCTouchDate and CCTouchTime.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoDateTimeSelection}" rowdistance="#{ccstylevalue['@defaultrowdistance@']}" >
<t:row id="g_2" >

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;

import org.eclnt.ccaddons.pbc.CCTouchDate;
import org.eclnt.ccaddons.pbc.CCTouchTime;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDateTimeSelection}")

public class DemoDateTimeSelection
    extends PageBean
    implements Serializable
{
    CCTouchTime m_timeSelection = new CCTouchTime();
    public CCTouchTime getTimeSelection() { return m_timeSelection; }

    CCTouchDate m_dateSelection = new CCTouchDate();
    public CCTouchDate getDateSelection() { return m_dateSelection; }

    Date m_time = new Date();
    public Date getTime() { return m_time; }
    public void setTime(Date value) { this.m_time = value; }

    Date m_date = new Date();
    public Date getDate() { return m_date; }
    public void setDate(Date value) { this.m_date = value; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    private IListener m_listener;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDateTimeSelection(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_timeSelection.prepare(new CCTouchTime.IListener()
        {
            public void reactOnUpdate()
            {
                m_time = m_timeSelection.updateDate(m_time,null);
            }
        },true);
        m_dateSelection.prepare(new CCTouchDate.IListener()
        {
            public void reactOnUpdate()
            {
                m_date = m_dateSelection.updateDate(m_date,null);
            }
        });
        m_timeSelection.setTime(m_time,null);
        m_dateSelection.setDate(m_date,null);
    }

    public String getPageName() { return "/testdatetimeselection.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.TestDateTimeSelection}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
