The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Workplace Management
  • API to access sizing

This demo shows how to access the tile arrangement of the current workplace.

The API allows e.g. to read (and to set) the divider location of the workplace splitpanes.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkplaceTileManager.WorkplaceTile;
import org.eclnt.workplace.WorkplaceTileManager.WorkplaceTileSplit;
import org.eclnt.workplace.WorkplaceTileManager.WorkplaceTileWorkpageContainer;

@CCGenClass (expressionBase="#{d.DemoWorkplaceAPISizing}")

public class DemoWorkplaceAPISizing
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_protocol;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoWorkplaceAPISizing(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        WorkplaceTileSplit top = (WorkplaceTileSplit)getWorkpageContainer().getTileManager().getTopTile();
    }

    public String getPageName() { return "/workplace/demoworkplaceapisizing.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoWorkplaceAPISizing}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    public void onAnalyzeAction(javax.faces.event.ActionEvent event) 
    {
        StringBuffer sb = new StringBuffer();
        analyze(sb);
        m_protocol = sb.toString();
    }

    public void onSetTop300Action(javax.faces.event.ActionEvent event) 
    {
        try
        {
            // assume top tile is split pane
            // this is dependent on the workplace configuration and must not be true! A workplace can
            // also consist out of a workpage container only.
            WorkplaceTileSplit top = (WorkplaceTileSplit)getWorkpageContainer().getTileManager().getTopTile();
            top.setDividerLocation("300");
        }
        catch (Throwable t)
        {
            Statusbar.outputError("Problem occurred: " + t.toString(),ValueManager.getStackstraceString(t));
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void analyze(StringBuffer sb)
    {
        WorkplaceTile top = getWorkpageContainer().getTileManager().getTopTile();
        if (top == null)
        {
            sb.append("Not in a workplace environment.");
        }
        else
        {
            analyzeTile(0,sb,top);
        }
    }

    private void analyzeTile(int level, StringBuffer sb, WorkplaceTile tile)
    {
        String prefix = "\n" + ValueManager.createEmptyString(level*4);
        sb.append(prefix);
        if (tile == null)
        {
            sb.append("null");
        }
        if (tile instanceof WorkplaceTileSplit)
        {
            WorkplaceTileSplit split = (WorkplaceTileSplit)tile;
            sb.append("SPLIT: Horizontal:" + split.isHorizontal() + ", DividerLocation:" + split.getDividerLocation());
            analyzeTile(level+1,sb,split.getSplit1());
            analyzeTile(level+1,sb,split.getSplit2());
        }
        else if (tile instanceof WorkplaceTileWorkpageContainer)
        {
            WorkplaceTileWorkpageContainer container = (WorkplaceTileWorkpageContainer)tile;
            sb.append("CONTAINER: Id:" + container.getContainerId());
        }
    }

}
