The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • "Shake" animation, popup animation

By using animation type "shake" within the component CAPTUREANIMATUR the content of the corresponding PANE can be animated by using some vibration effect.

You may either directly use "shake" - or you may add two configuration paramters:

1. Iterations of vibrations
2. pixel delta of vibration

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.defaultscreens.OKPopup;
import org.eclnt.jsfserver.elements.util.DefaultModalPopupListener;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.pagebean.IPageBean;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPopupAnimation}")

public class DemoPopupAnimation
    extends PageBean 
    implements Serializable
{
    Trigger m_animationTrigger = new Trigger();
    
    public DemoPopupAnimation(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demopopupanimation.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupAnimation}"; }

    public Trigger getAnimationTrigger() { return m_animationTrigger; }
    
    public void onOpenPopupAnimated(javax.faces.event.ActionEvent event) 
    {
        OKPopup okp = OKPopup.createInstance("Test dialog","Test dialog.");
        okp.getModalPopup().setAnimationtype("shake");
    }

    public void onOpenPopupAnimatedPopup2(javax.faces.event.ActionEvent event) 
    {
        final IPageBean ui = new DemoHelloWorld(getOwningDispatcher());
        ModalPopup mp = openModalPopup(ui,"Demo Hello World!",750,400,new DefaultModalPopupListener(this,ui));
        mp.setAnimationtype("shrinktocenter");
    }

    public void onShakeAction(javax.faces.event.ActionEvent event) 
    {
        m_animationTrigger.trigger();
    }



}
