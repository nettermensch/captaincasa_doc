The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Sorting
  • Initial sorting

You can use the grid's sorting functions from your server side managed bean, too.

The simplest way is to sort by one column only and to pass the sort information directly using the "sort(...)" method.

You may also pass a sort which considers multiple columns by passing corresponding FIXGRIDSortInfo-instances and by calling the "resort()" method afterwards.

Also at later point of time you may always call the resort() method to re-apply the current sorting of the grid - e.g. after having added new items.

package workplace;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoGridInitialSort extends PageBean
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem
    {
        String i_firstName;
        String i_lastName;
        boolean i_married;
        // --------------------------------------------------------------------
        public void setFirstName(String value) { i_firstName = value; }
        public String getFirstName() { return i_firstName; }
        public void setLastName(String value) { i_lastName = value; }
        public String getLastName() { return i_lastName; }
        public void setMarried(boolean value) { i_married = value; }
        public boolean getMarried() { return i_married; }
        // --------------------------------------------------------------------
        public void onRemove(ActionEvent ae)
        {
            m_grid.getItems().remove(this);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGridInitialSort(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        fillGridWithData();
        sortGridByLastNameFirstName();
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public void onAddRow(ActionEvent ae)    
    {
        GridItem row = new GridItem();
        row.setFirstName("new");
        row.setLastName("new");
        m_grid.getItems().add(row);
    }
    
    public void onExplicitlySort(ActionEvent ae)    
    {
        sortGridByLastNameFirstName();
    }
    
    public void onExplicitlySort2(ActionEvent ae)    
    {
        sortGridByLastNameDescending();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void sortGridByLastNameDescending()
    {
        // simple sort by one column
        m_grid.sort("lastName",false);
    }

        (

        );
    }
