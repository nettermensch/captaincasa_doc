The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Combobox examples

A combo box manages a value that is selectable from a list of values. The value is internally represented as value + text, only the text is the one which is shown to the user - whereas the value is the one which is internally set and get into the corresponding property.

A combo box can be filled in three different ways:

(1) STATIC: You can define the items directly as part of the layout definition.

(2) DYNAMIC: You can define the items at backend side.

(3) DYNAMIC with ON DEMAND LOADING: in this case the complete list of values is only sent to the frontend when the user actually operates on the combo box, e.g. when the user clicks on the box to see the valid values.

Have a look into the coding of the adapter side: the valid values that are returned via getDepartmentVvb() method are null until an event from client side tells that the values are required. You also may observe what happens in the client: before the loading only the key of the value ("PROD") is shown - because no value-text information is sent to the client. After loading the full information is shown. If you want to avoid this effect then you can also send exactly one valid-values-item to the client, the one which represents the current value.

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

//import org.eclnt.jsfserver.elements.events.BaseActionEventLoadOnDemand;

import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.workplace.IWorkpageDispatcher;

/**
 * @author bmu
 *
 */
public class DemoCombo extends PageBean
{
    public class ValueText implements Serializable
    {
        String i_value;
        String i_text;
        public ValueText(String value, String text)
        {
            i_value = value;
            i_text = text;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    ValueText[] DEPARTMENTS = new ValueText[]
                                            {
            new ValueText("SALES","Sales & Marketing"),
            new ValueText("FINANCE","Financials"),
            new ValueText("EXEC","Executive Board"),
            new ValueText("PROD","Production"),
            new ValueText("RND","Reserach & Development"),
            new ValueText("HR","Human Resources"),
                                            };    
    
    ValueText[] TITLES = new ValueText[]
                                            {
            new ValueText("1","Dr."),
            new ValueText("2","Prof."),
            new ValueText("3","Earl"),
            new ValueText("4","Prince of"),
                                            };    
    
    String m_creditcard;
    String m_title;
    String m_department;
    String m_number;
    
    ValidValuesBinding m_titleVvb;
    ValidValuesBinding m_departmentVvb;
    ValidValuesBinding m_numberVvb;
    
    COMBOBOXComponent m_titleCB;
    COMBOBOXComponent m_departmentCB;
    boolean m_loadDepartmentCB = false;
    
    String m_result = "Press OK to view result.";
    int m_idCounter = 0;    
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoCombo(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("combobox"); 
        m_department = DEPARTMENTS[3].i_value;
        m_creditcard = "V";
        // titles
        // m_title = TITLES[0].i_value;
        m_titleVvb = new ValidValuesBinding();
        for (int i=0; i<TITLES.length; i++)
            m_titleVvb.addValidValue(TITLES[i].i_value,TITLES[i].i_text);
        // departments
        m_departmentVvb = new ValidValuesBinding();
        for (int i=0; i<DEPARTMENTS.length; i++)
            m_departmentVvb.addValidValue(DEPARTMENTS[i].i_value,DEPARTMENTS[i].i_text);
        m_titleVvb.sortByName();
        m_departmentVvb.sortByValue();
        // numbers
        m_numberVvb = new ValidValuesBinding();
        for (int i=0; i<100; i++)
            m_numberVvb.addValidValue(""+i,"Number " + i);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void setCreditcard(String value) { m_creditcard = value; }
    public String getCreditcard() { return m_creditcard; }
    
    public void setTitle(String value) { m_title = value; }
    public String getTitle() { return m_title; }
    
    public void setDepartment(String value) { m_department = value; }
    public String getDepartment() { return m_department; }

    public void setNumber(String value) { m_number = value; }
    public String getNumber() { return m_number; }
    
    public ValidValuesBinding getTitleVvb() { return m_titleVvb; }
    
    public void setResult(String value) { m_result = value; }
    public String getResult() { return m_result; }
    
    public ValidValuesBinding getNumberVvb() 
    { 
        return m_numberVvb; 
    }
    
    public ValidValuesBinding getDepartmentVvb() 
    {
        if (m_loadDepartmentCB == true)
            return m_departmentVvb;
        else
            return null;
    }
    
//    public void onDepartmentAction(ActionEvent ae)

    public void onOK(ActionEvent ae)
    {
        m_result = "CreditCard: " + m_creditcard + 
                   "\nTitle: " + m_title +
                   "\nDepartment: " + m_department;
    }
    
    public boolean onCreateClientTestLogMessage()
    {
        ClientTestLog.addClientTestLogMessage(m_creditcard + "//" + m_title + "//" + m_department);
        return false;
    }
