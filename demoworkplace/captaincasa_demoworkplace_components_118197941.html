The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Reporting, PDF, SVG
  • Print PDF

The PDFPRINTER component is an invisible component that allows to send PDF documents to the printer. The PDF is passed as hex-string - in the same way as it is passed with the PDFRENDERER component.

On client side there are only limited possibilities for controlling the printing - because by nature of the browser, automated printing is not suitable...! So printing is always showing up some print dialog, there is no possibility to e.g. pre-select a local printer.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventPdfprinterNotification;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.util.file.FileManager;
import org.eclnt.util.valuemgmt.ValueManager;

@CCGenClass (expressionBase="#{d.DemoPdfPrinter}")

public class DemoPdfPrinter
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected Trigger m_trigger = new Trigger();
    protected String m_pdf;
    double m_paperHeight = 11.69d;
    double m_paperWidth = 8.26;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoPdfPrinter(IDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("pdfprinter");
        // first pdf file that is shown
        byte[] initialPdf = WebResourceReader.readFileIntoByteArray("/pdf/pdftest1.pdf");
        m_pdf = ValueManager.encodeHexString(initialPdf);
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public double getPaperHeight() { return m_paperHeight; }
    public void setPaperHeight(double value) { this.m_paperHeight = value; }

    public double getPaperWidth() { return m_paperWidth; }
    public void setPaperWidth(double value) { this.m_paperWidth = value; }

    public Trigger getTrigger() { return m_trigger; }
    public void setTrigger(Trigger value) { m_trigger = value; }

    public String getPdf() { return m_pdf; }
    public void setPdf(String value) { m_pdf = value; }

    public void onTriggerPrint(ActionEvent event) 
    {
        m_trigger.trigger();
    }

    public void onPrintNotification(ActionEvent event) 
    {
        if (event instanceof BaseActionEventPdfprinterNotification)
        {
            BaseActionEventPdfprinterNotification ne = (BaseActionEventPdfprinterNotification)event;
            Statusbar.outputMessage("Notification of client-printing: " + ne.getPrintResult() + "/" + ne.getDetail());
        }
    }

}
