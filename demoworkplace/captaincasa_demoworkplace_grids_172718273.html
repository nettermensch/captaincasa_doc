The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Column Sequence and Widths
  • Column width info & update

This demo shows how to retrieve and update information about the width of the grid columns at runtime.

A grid starts with being sized corresponding to the definition within the layout (XML definition). But the user may change the width of columns, or may change the sequene of columns.

You get notified by corresponding methods, that you may override within the grid processing. And you may override the layout definition's width definition at runtime by corresponding methods.

Please also check the Java-API documentation for the class FIXGRIDBinding.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.text.SimpleDateFormat;
import java.util.Date;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoGridColumnSizing}")

public class DemoGridColumnSizing
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_fourth;
        public String getFourth() { return m_fourth; }
        public void setFourth(String value) { this.m_fourth = value; }

        String m_third;
        public String getThird() { return m_third; }
        public void setThird(String value) { this.m_third = value; }

        String m_second;
        public String getSecond() { return m_second; }
        public void setSecond(String value) { this.m_second = value; }

        String m_first;
        public String getFirst() { return m_first; }
        public void setFirst(String value) { this.m_first = value; }

    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>()
    {
        @Override
        protected void onColumnSequenceUpdated()
        {
            protocol("Column sequence update: " + getColumnsequence());
        }
        @Override
        protected void onColumnWidthsUpdated()
        {
            protocol("Column sequence update: " + getModcolumnwidths());
        }
    };
    String m_widthForUpdate;
    int m_columnIndexForUpdate;
    String m_protocol = "";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridColumnSizing(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirst("First " + i);
            gi.setSecond("Second " + i);
            gi.setThird("Third " + i);
            gi.setFourth("Fourth " + i);
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demogridcolumnsizing.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridColumnSizing}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public String getWidthForUpdate() { return m_widthForUpdate; }
    public void setWidthForUpdate(String value) { this.m_widthForUpdate = value; }

    public int getColumnIndexForUpdate() { return m_columnIndexForUpdate; }
    public void setColumnIndexForUpdate(int value) { this.m_columnIndexForUpdate = value; }

    public void onUpdateWidthAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            if (m_widthForUpdate == null || m_widthForUpdate.trim().length() == 0)
                throw new Exception("Please define a propert width.");
            m_grid.setModcolumnwidth(m_columnIndexForUpdate,m_widthForUpdate);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }
    
    public String getProtocol() { return m_protocol; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void protocol(String s)
    {
        String t = (SimpleDateFormat.getDateTimeInstance()).format(new Date());
        m_protocol = t + ": " + s + "\n" + m_protocol;
    }
    
}
