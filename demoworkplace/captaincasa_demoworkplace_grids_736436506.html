The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Selection Issues
  • Node selection in tree

Input an index and press the "Show" button - the corresponding tree node will be selected and the whole grid will be scrolled to make sure that it is visible. In case of a tree, all nodes above the selected node are opened to make it visible.

The method to be used on server side is "<grid>.ensureItemToBeDisplayed(...)".

A similar method that you might use is "<grid>.selectAndFocusItem(...)" which in addition moves the focus into the corresponding line.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTreePositioning}")

public class DemoTreePositioning
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        public TreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        public void onRowSelect() { }
        public void onRowExecute() { }
        public void onToggle() { }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_index = 0;
    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();
    List<TreeNode> m_nodesList = new ArrayList<TreeNode>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTreePositioning(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher); 
        // build up tree
        for (int i=0; i<20; i++)
        {
            TreeNode top = new TreeNode(m_tree.getRootNode());
            top.setText("Top" + i + "_" + m_nodesList.size());
            top.setStatus(TreeNode.STATUS_CLOSED);
            m_nodesList.add(top);
            for (int j=0; j<5; j++)
            {
                TreeNode sub = new TreeNode(top);
                sub.setText("Sub" + i + "," + j + "_" + m_nodesList.size());
                sub.setStatus(TreeNode.STATUS_CLOSED);
                m_nodesList.add(sub);
                for (int k=0; k<5; k++)
                {
                    TreeNode subsub = new TreeNode(sub);
                    subsub.setText("Subsub" + i + "," + j+"," + k + "_" + m_nodesList.size());
                    subsub.setStatus(TreeNode.STATUS_ENDNODE);
                    m_nodesList.add(subsub);
                }
            }
        }
        Statusbar.outputSuccess("Number of items: " + m_nodesList.size());
    }

    public String getPageName() { return "/workplace/demotreepositioning.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTreePositioning}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onShowAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            TreeNode node = m_nodesList.get(m_index);
            m_tree.deselectCurrentSelection();
            m_tree.selectItem(node);
            m_tree.ensureItemToBeDisplayed(node);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }

    public void onShowAndFocusAction(javax.faces.event.ActionEvent event) 
    {
        try
        {
            TreeNode node = m_nodesList.get(m_index);
            m_tree.deselectCurrentSelection();
            m_tree.selectAndFocusItem(node);
        }
        catch (Throwable t)
        {
            Statusbar.outputAlert(t.toString());
        }
    }

    public int getIndex() { return m_index; }
    public void setIndex(int value) { this.m_index = value; }
    
    public FIXGRIDTreeBinding<TreeNode> getTree() { return m_tree; }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
