The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Scheduling
  • Schedule Grid

This demo shows how to use the components SCHEDULE and SCHEDULEITEM in order to build complex scheduling scenarios. The story behind is very simple:

Each grid row contains one SCHEDULE component. The SCHEDULEITEMS components within the SCHEDUL are created dynamically - dependent on the scheduling situation.

There is a drag & drop management - and there is the possibility to create new items by right mouse button click. In both cases the mouse position that is transferred as percentage value into the server side processing is used to properly position the SCHEDULE component.

For drawing background information (here: break times ) a DRAWAREA can be nicely used.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Random;
import java.util.Set;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.SCHEDULEITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventScheduleSizeChanged;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.jsfserver.util.valuemgmt.IActionListenerDelegation;
import org.eclnt.jsfserver.util.valuemgmt.impl.LongDelegation;

@CCGenClass (expressionBase="#{d.DemoScheduleGrid}")

public class DemoScheduleGrid
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    public class MyOrder implements Serializable
    {
        String i_key = "id"+(s_orderId++) + "";
        String i_name;
        long i_start; // 0 == 8:00, then in minutes
        long i_duration; // in minutes
        MyRow i_assignedRow;
        public MyOrder(String name, long start, long duration, MyRow assignedRow)
        {
            super();
            i_name = name;
            i_start = start;
            i_duration = duration;
            m_orders.put(i_key,this);
            i_assignedRow = assignedRow;
        }
        public long getStart() { return i_start; }
        public void setStart(long start) { i_start = start; }
        public void setDuration(long value) { i_duration = value; }
        public long getDuration() { return i_duration; }
//        public void onInvoke(ActionEvent ae)
//        {
//            if (ae instanceof BaseActionEventScheduleSizeChanged)
//            {
//                BaseActionEventScheduleSizeChanged sscEvent = (BaseActionEventScheduleSizeChanged)ae;
//                i_start = sscEvent.getScheduleLeft();
//                i_duration = sscEvent.getScheduleWidth();
//            }
//        }
    }
    
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        Set<MyOrder> i_orders = new HashSet<MyOrder>(); 
        String m_machine;
        DYNAMICCONTENTBinding m_content = new DYNAMICCONTENTBinding();
        
        
        public String getMachine() { return m_machine; }
        public void setMachine(String value) { m_machine = value; }
        public DYNAMICCONTENTBinding getContent() { return m_content; }
        
        private void updateSchedule()
        {
            // update of the components within the schedule row
            StringBuffer sb = new StringBuffer();
            List<ComponentNode> nodes = new ArrayList<ComponentNode>();
            for (final MyOrder order: i_orders)
            {
                String expression = "#{d.DemoScheduleGrid.orders."+order.i_key+".onInvoke}";
                String background = null;
                if (order.i_name != null && order.i_name.contains("New"))
                    background = "#8da8d0c0";
                else
                    background = "#a8d08dc0";
                SCHEDULEITEMNode node = new SCHEDULEITEMNode()
                    .setScheduleleft(""+ order.i_start)
                    // .setSchedulewidth("#{d.DemoScheduleGrid.orders."+order.i_key+".duration}")
                    .bindSchedulewidth(new LongDelegation()
                    {
                        @Override
                        public void setValue(Long value) { order.setDuration(value); }
                        @Override
                        public Long getValue() { return order.getDuration(); }
                    })
                    .setStylevariant("ccgantt2")
                    .setText(order.i_name + "/" + order.i_start +"/" + order.i_duration)
                    .setValign("middle")
                    .setTooltip(order.i_name + "/" + order.i_start +"/" + order.i_duration)
                    .setBackground(background)
                    .setDragsend("order:"+order.i_key)
                    .setTextwithlinewrap("false")
                    // .setActionListener("#{d.DemoScheduleGrid.orders."+order.i_key+".onInvoke}")
                    .bindActionListener(new IActionListenerDelegation()
                    {
                        @Override
                        public void onAction(ActionEvent event) 
                        { 
                            if (event instanceof BaseActionEventScheduleSizeChanged)
                            {
                                BaseActionEventScheduleSizeChanged sscEvent = (BaseActionEventScheduleSizeChanged)event;
                                order.setStart(sscEvent.getScheduleLeft());
                                order.setDuration(sscEvent.getScheduleWidth());
                            }
                        }
                    })
                    .setSizeratbothsides("true")
                    .setDragsticktocursor("false")
                    .setFocusable("true");
                nodes.add(node);
            }
            m_content.setContentNodes(nodes);
        }
        
        public void addOrder(MyOrder order)
        {
            i_orders.add(order);
            order.i_assignedRow = this;
            updateSchedule();
        }
        
        public void removeOrder(MyOrder order)
        {
            i_orders.remove(order);
            updateSchedule();
        }
        
        public void onScheduleAction(ActionEvent ae)
        {
            if (ae instanceof BaseActionEventDrop)
            {
                BaseActionEventDrop baed = (BaseActionEventDrop)ae;
                String dropInfo = baed.getDragInfo();
                String orderKey = dropInfo.substring("order:".length());
                MyOrder order = m_orders.get(orderKey);
                // new start
                float percX = baed.getPercentageHorizontaAsFloat() + baed.getPercentageDeltaX();
                order.i_start = Math.round(percX / 100f * 480f);
                order.i_start = ((order.i_start+7) / 15) * 15;
                // reassign to new row
                order.i_assignedRow.removeOrder(order);
                addOrder(order);
            }
            else if (ae instanceof BaseActionEventPopupMenuItem)
            {
                BaseActionEventPopupMenuItem bae = (BaseActionEventPopupMenuItem)ae;
                int start = Math.round((float)bae.getPercentageHorizontal() / 100f * 480f);  
                start = ((start+7) / 15) * 15;
                int duration = s_random.nextInt(100);
                MyOrder o = new MyOrder("New Order",start,duration,this);
                addOrder(o);
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static Random s_random = new Random();
    static int s_orderId = 0;
    
    FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    Map<String,MyOrder> m_orders = new HashMap<String,MyOrder>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoScheduleGrid(IDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("schedule");
        for (int i=0; i<20; i++)
        {
            MyRow r = new MyRow();
            r.setMachine("Machine " +  i);
            m_rows.getItems().add(r);
            r.addOrder(new MyOrder("WO",s_random.nextInt(100),s_random.nextInt(100)+20,r));
            r.addOrder(new MyOrder("WO",250+s_random.nextInt(100),s_random.nextInt(120)+20,r));
        }
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    public Map<String,MyOrder> getOrders() { return m_orders; }

}
