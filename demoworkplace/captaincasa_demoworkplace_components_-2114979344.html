The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Menus
  • Combining popup menus
  • Grid item and grid

This example shows how to combine a popup menu on grid level (id "GRID") with a popup menu on item level (id "GRIDITEM").

Please pay attention: the item has to explicitly delegate the popup menu processing to its grid within you program!

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPopupMenuCombinedForGrid}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoPopupMenuCombinedForGrid}" rowdistance="10" >
<t:row id="g_9" >
<t:textpane id="g_10" text="Press the right mouse button inside the grid." width="100%" />
</t:row>
<t:row id="g_6" >
<t:fixgrid id="g_7" height="100%" objectbinding="#{d.DemoPopupMenuCombinedForGrid.grid}" popupmenu="GRID" rowpopupmenu="GRIDITEM;GRID" width="100%" >
<t:gridcol id="g_8" text="First name" width="100%" >
<t:field id="g_17" text=".{firstName}" />
</t:gridcol>
<t:gridcol id="g_11" text="Last name" width="100%" >
<t:field id="g_12" text=".{lastName}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:popupmenu id="GRIDITEM" >
<t:menuitem id="g_14" command="DELETEITEM" text="Delete item" />
</t:popupmenu>
<t:popupmenu id="GRID" >
<t:menuitem id="g_15" command="CREATEITEM" text="Create item" />
</t:popupmenu>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPopupMenuCombinedForGrid}")

public class DemoPopupMenuCombinedForGrid
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public void onRowSelect() { }
        public void onRowExecute() { }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            if ("DELETEITEM".equals(event.getCommand()))
                m_grid.getItems().remove(this);
            else
                // delegate to grid parent processing
                procesGridMenuItem(event.getCommand());
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>()
    {
        @Override
        protected void onPopupMenuItem(BaseActionEventPopupMenuItem popupMenuItemEvent)
        {
            procesGridMenuItem(popupMenuItemEvent.getCommand());
        }
    };

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPopupMenuCombinedForGrid(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demopopupmenucombinedforgrid.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPopupMenuCombinedForGrid}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void procesGridMenuItem(String command)
    {
        if ("CREATEITEM".equals(command))
        {
            GridItem gi = new GridItem();
            gi.setFirstName("new");
            gi.setLastName("new");
            m_grid.getItems().add(gi);
        }
    }
}
