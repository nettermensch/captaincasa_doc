The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Adapter Binding
  • Example: COMBOFIELD referencing bean-objects

The focus of the demo is the COMBOFIELD processing.

The three controls reference to a bean of type "Order". In the "Order" beand there is a direct instance reference to the bean "Customer".

This demo shows how to use the COMBOFIELD accordingly:

1. you can directly input the id of the customer - and it will be checked and transferred into an object which is set into the order
2. you can select the customer from the COMBOFIELD popup
3. In case your input does not match a customer, then the field shows an error background

The interesting issues with the COBOFIELD processing are:

1. always remember: the COMBOFIELD does not know about bean objects! It knows simple data types only (in the example we use String)
2. that's the reason why in the adapter binding you need to tell that "customer" from UI perspective is a String. The conversion into a bean of type "Customer" is "behind the control".
3. because there might be situations in which the user inputs a customer that is not valid, you need to "park" the interim-value outside of the bean object.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ISetId;
import org.eclnt.jsfserver.defaultscreens.IdTextSelection;
import org.eclnt.jsfserver.elements.adapter.ComponentAdapterByAnnotationForBeanProperty;
import org.eclnt.jsfserver.elements.adapter.ComponentAttribute;
import org.eclnt.jsfserver.elements.events.BaseActionEventValueHelp;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoCombofieldBeanReference}")

public class DemoCombofieldBeanReference
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public static class Order
    {
        String i_orderId;
        Customer i_customer;
        String i_orderComment;
        public String getOrderId() { return i_orderId; }
        public void setOrderId(String orderId) { i_orderId = orderId; }
        public Customer getCustomer() { return i_customer; }
        public void setCustomer(Customer customer) { i_customer = customer; }
        public String getOrderComment() { return i_orderComment; }
        public void setOrderComment(String orderComment) { i_orderComment = orderComment; }
    }
    
    public static class Customer
    {
        String i_customerId;
        String i_customerName;
        public String getCustomerId() { return i_customerId; }
        public void setCustomerId(String customerId) { i_customerId = customerId; }
        public String getCustomerName() { return i_customerName; }
        public void setCustomerName(String customerName) { i_customerName = customerName; }
        public String toString() { return i_customerName; }
    }
    
    public class FieldAdapter extends ComponentAdapterByAnnotationForBeanProperty<Order>
    {
        public FieldAdapter(Order bean,String property)
        { 
            super("text",bean,property);
        }
    }
    
    public class ComboFieldAdapter extends ComponentAdapterByAnnotationForBeanProperty<Order>
    {
        String m_parkedText;
        public ComboFieldAdapter(Order bean,String property)
        { 
            super("text",bean,property);
        }
        @Override
        /* This is important! The Adapter by default uses the property type of the
         * referenced property (here: "customer"). But you override: the control is
         * served with String-instances - not with Customer-instances. */
        public Class getPropertyType() 
        { 
            return String.class; 
        }
        @Override
        /* Convert referecned "Customer" into "String", */
        public Object getPropertyValue()
        {
            Customer c = (Customer)super.getPropertyValue();
            if (c == null)
                return m_parkedText;
            else
                return c.toString();
        }
        @Override
        /* Convert String input into Customer. */
        public void setPropertyValue(Object value)
        {
            m_parkedText = null;
            if (value == null)
            {
                super.setPropertyValue(value);
            }
            else
            {
                String s = value.toString();
                Customer c = lookupCustomer(s);
                super.setPropertyValue(c);
                if (c == null)
                    m_parkedText = s;
            }
        }
        @ComponentAttribute
        public String getBgpaint()
        {
            if (m_parkedText != null)
                return "error()";
            else
                return null;
        }
        @Override
        public void onAction(ActionEvent event)
        { 
            if (event instanceof BaseActionEventValueHelp)
            {
                IdTextSelection idts = IdTextSelection.createInstance();
                List<Customer> cs = findCustomers();
                for (Customer c: cs)
                    idts.addLine(c.getCustomerId(),c.toString());
                idts.setCallBack(new ISetId()
                {
                    @Override
                    public void setId(String id)
                    { 
                        setPropertyValue(id);
                    }
                });
            }
        }
        /* This methods should be implemented in the business logic in real life scenarios! */ 
        private Customer lookupCustomer(String s)
        {
            List<Customer> cs = findCustomers();
            for (Customer c: cs)
            {
                if (c.getCustomerId().equals(s))
                    return c;
            }
            return null;
        }
        /* This methods should be implemented in the business logic in real life scenarios! */ 
        private List<Customer> findCustomers()
        {
            List<Customer> result = new ArrayList<Customer>();
            for (int i=0; i<100; i++)
            {
                Customer c = new Customer();
                c.setCustomerId((4700+i)+"");
                c.setCustomerName("Customer " + c.getCustomerId());
                result.add(c);
            }
            return result;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    Order m_order;
    FieldAdapter m_orderIdAdapter;
    FieldAdapter m_orderCommentAdapter;
    ComboFieldAdapter m_customerAdapter;
    String m_resultOutput;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoCombofieldBeanReference(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_order = new Order();
        m_customerAdapter = new ComboFieldAdapter(m_order,"customer");
        m_orderIdAdapter = new FieldAdapter(m_order,"orderId");
        m_orderCommentAdapter = new FieldAdapter(m_order,"orderComment");
    }

    public String getPageName() { return "/workplace/democombofieldbeanreference.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoCombofieldBeanReference}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public Order getOrder() { return m_order; }
    public ComboFieldAdapter getCustomerAdapter() { return m_customerAdapter; }
    public FieldAdapter getOrderIdAdapter() { return m_orderIdAdapter; } 
    public FieldAdapter getOrderCommentAdapter() { return m_orderCommentAdapter; } 
    
    public String getResultOutput() { return m_resultOutput; }
    public void setResultOutput(String value) { this.m_resultOutput = value; }

    public void onOutputAction(javax.faces.event.ActionEvent event) 
    {
        StringBuffer sb = new StringBuffer();
        sb.append(  "Id        : " + m_order.getOrderId());
        sb.append("\nCustomer  : " + m_order.getCustomer());
        sb.append("\nComment   : " + m_order.getOrderComment());
        m_resultOutput = sb.toString();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
}
