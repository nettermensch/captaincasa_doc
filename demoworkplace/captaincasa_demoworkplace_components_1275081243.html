The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Image Capturer & Animator (2)

The CAPTUREANIMATOR component creates an internal screenshot of the component, in which it is placed. This screenshot then is passed to an animation that e.g. hides the screenshot into a certain direction. All this happens if a trigger from the server side is activated.

In the example there is an optical switch between a grid and a detail view. The CAPTURECOMPONENT is triggered every time the switch is done. Dependent on the switch the direction of hiding the captured image is either "to the left" or "to the right".

You see, the CAPTUREANIMATOR component is a very flexibly usable one: you can use it for "softening" any kind of "hard" changes within your screens - with very little programming effort.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoCaptureAnimator}")

public class DemoCaptureAnimator
    extends PageBean 
    implements Serializable
{
    public class MyRow extends FIXGRIDItem implements java.io.Serializable
    {
        protected String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { m_lastName = value; }

        protected String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { m_firstName = value; }
        
        protected String m_image;
        public String getImage() { return m_image; }
        
        @Override
        public void onRowExecute()
        {
            select(this);
        }

    }

    protected String m_animationType;
    protected FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();
    protected boolean m_gridVisible = true;
    protected Trigger m_trigger = new Trigger();
    MyRow m_selRow = null;
    
    public DemoCaptureAnimator(IWorkpageDispatcher owner)
    {
        super(owner);
        initRefComponent("captureanimator");
        for (int i=0; i<25; i++)
        {
            MyRow mr = new MyRow();
            mr.setFirstName("First " + i);
            mr.setLastName("Last " + i);
            mr.m_image = "/images/pictures/photo"+(i%10)+".gif";
            m_rows.getItems().add(mr);
        }
    }

    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }
    public boolean getGridVisible() { return m_gridVisible; }
    public boolean getDetailVisible() { return !m_gridVisible; }
    public Trigger getTrigger() { return m_trigger; }
    public String getAnimationType() { return m_animationType; }
    public MyRow getSelRow() { return m_selRow; }
    
    public void onBack(ActionEvent event) 
    {
        m_animationType = "hidecontenttoright";
        m_trigger.trigger();
        m_gridVisible = true;
    }
    
    public void onLeft(ActionEvent event) 
    {
        int currentIndex = m_rows.getItems().indexOf(m_selRow);
        if (currentIndex == 0)
            return;
        select(m_rows.getItems().get(currentIndex-1),"hidecontenttoright");
    }
    
    public void onRight(ActionEvent event) 
    {
        int currentIndex = m_rows.getItems().indexOf(m_selRow);
        if (currentIndex == (m_rows.getItems().size()-1))
            return;
        select(m_rows.getItems().get(currentIndex+1),"hidecontenttoleft");
    }
    
    private void select(MyRow r)
    {
        select(r,"hidecontenttoleft");
    }
    
    private void select(MyRow r, String animationType)
    {
        m_trigger.trigger();
        m_gridVisible = false;
        m_animationType = animationType;
        m_selRow = r;
    }

}
