The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Grid with Details

Select one item from the article list, either by double clicking or by navigating with the keyboard and pressing Enter. Result: the details of the item will be shown below the grid.

In the implementation of the example you see:

(.) By overriding the onRowExecute() method you can add server side coding, which is executed when the user double-clicks a row. (There is a corresponding onRowSelect() method which is executed when the user double clicks the grid, as well.)

(.) Inside the grid column fields, and inside the detail fields the business object (Article, Stock) coming from the logic layer is simply referenced. There is no repetition of properties within the grid row objects.

(.) The detail area is only rendered when a selected article is available. This is defined within the layout definition of the corresponding row holding the detail area.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridDetail}")

public class DemoGridDetail
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    // BUSINESS LOGIC 
    // (in inner classes, so that it is kept with the UI class - please
    //  do not follow this example!)

    static Random s_random = new Random();
    
    public static class Stock
    {
        String i_location;
        float i_quantity;
        public void setLocation(String value) { i_location = value; }
        public String getLocation() { return i_location; }
        public void setQuantity(float value) { i_quantity = value; }
        public float getQuantity() { return i_quantity; }
    }
    
    static String[] CURRENCIES = new String[] { "ï¿½","$" };
    static String[] VALUES = new String[] { "47","555", "100", "88" };
    public static class Article
    {
        public static List<Article> readArticles()
        {
            List<Article> result = new ArrayList<Article>();
            for (int i=0; i<1000; i++)
            {
                Article a = new Article();
                a.setId(s_random.nextInt(200000)+"");
                a.setText("Text for article " + a.getId());
                a.setSupplier(s_random.nextInt(200000)+"");
                if (i%2 == 0)
                {
                    a.setUom("pc");
                    a.setStockArea("Front");
                }
                else
                {
                    a.setUom("kg");
                    a.setStockArea("Back");
                }
                a.setValue(new BigDecimal(VALUES[i%VALUES.length]));
                a.setCurrency(CURRENCIES[i%CURRENCIES.length]);
                result.add(a);
            }
            return result;
        }
        String i_id;
        String i_text;
        String i_uom;
        String i_supplier;
        String i_stockArea;
        BigDecimal i_value;
        String i_currency;
        public void setId(String value) { i_id = value; }
        public String getId() { return i_id; }
        public void setText(String value) { i_text = value; }
        public String getText() { return i_text; }
        public void setUom(String value) { i_uom = value; }
        public String getUom() { return i_uom; }
        public void setSupplier(String value) { i_supplier = value; }
        public String getSupplier() { return i_supplier; }
        public void setStockArea(String value) { i_stockArea = value; }
        public String getStockArea() { return i_stockArea; }
        public BigDecimal getValue() { return i_value; }
        public void setValue(BigDecimal value) { i_value = value; }
        public String getCurrency() { return i_currency; }
        public void setCurrency(String currency) { i_currency = currency; }
        public List<Stock> getStock()
        {
            List<Stock> result = new ArrayList<Stock>();
            int maxi = s_random.nextInt(25);
            for (int i=0; i<maxi; i++)
            {
                Stock s = new Stock();
                s.setLocation("Loc-"+s_random.nextInt(500));
                s.setQuantity(s_random.nextInt(1000));
                result.add(s);
            }
            return result;
        }
    }
    
    // INTERACTION LOGIC 
    
    public class ArticleRow extends FIXGRIDItem implements java.io.Serializable
    {
        Article i_article;
        public ArticleRow(Article article)
        {
            i_article = article;
        }
        public Article getArticle()
        {
            return i_article;
        }
        @Override
        public void onRowExecute()
        {
            selectArticleRow(this);
        }
        @Override
        public void onRowSelect()
        {
        }
        public String getBgpaintForRow()
        {
            if (this == m_selArticleRow)
                return "bgbackground(#FFC000C0)";
            else
                return null;
        }
    }

    public class StockRow extends FIXGRIDItem implements java.io.Serializable
    {
        Stock i_stock;
        public StockRow(Stock stock)
        {
            i_stock = stock;
        }
        public Stock getStock()
        {
            return i_stock;
        }
    }
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    protected FIXGRIDListBinding<ArticleRow> m_articleRows = new FIXGRIDListBinding<ArticleRow>(true);
    protected FIXGRIDListBinding<StockRow> m_stockRows = new FIXGRIDListBinding<StockRow>(true);
    protected ArticleRow m_selArticleRow;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoGridDetail(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // fill article grid
        List<Article> as = Article.readArticles();
        for (Article a: as) 
            m_articleRows.getItems().add(new ArticleRow(a));
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public FIXGRIDListBinding<ArticleRow> getArticleRows() { return m_articleRows; }
    public FIXGRIDListBinding<StockRow> getStockRows() { return m_stockRows; }
    public ArticleRow getSelArticleRow() { return m_selArticleRow; }
    
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void selectArticleRow(ArticleRow ar)
    {
        if (m_selArticleRow != null) m_selArticleRow.getChangeIndex().indicateChange();
        m_selArticleRow = ar;
        m_selArticleRow.getChangeIndex().indicateChange();
        // fill stock grid
        m_stockRows.getItems().clear();
        for (Stock s: ar.getArticle().getStock()) 
            m_stockRows.getItems().add(new StockRow(s));
    }
    
    

}
