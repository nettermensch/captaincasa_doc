The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Repeat Processing
  • REPEAT inside Grid

The REPEAT component can be used inside a grid cell as well. In the example the number of addresses is different per item - a corresponding ROW-LABEL combination is created by the REPEAT component.

...of course pay attention: using a REPEAT inside a grid may cause quite a lot of components to be generated and managed internally!

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridWithRepeat}")

public class DemoGridWithRepeat
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class AddressesItem extends Object implements java.io.Serializable
    {
        String m_address;
        public AddressesItem(String address)
        {
            m_address = address;
        }
        public String getAddress() { return m_address; }
    }

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_name;
        List<AddressesItem> m_addresses = new ArrayList<AddressesItem>();
        public String getName() { return m_name; }
        public void setName(String value) { this.m_name = value; }
        public List<AddressesItem> getAddresses() { return m_addresses; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGridWithRepeat(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        Random r = new Random();
        for (int i=0; i<10; i++)
        {
            GridItem gi = new GridItem();
            gi.setName("Name " + i);
            m_grid.getItems().add(gi);
            int jMax = r.nextInt(6);
            for (int j=0; j<jMax; j++)
            {
                AddressesItem ai = new AddressesItem("Address " + i + "/" + j);
                gi.m_addresses.add(ai);
            }
        }
    }

    public String getPageName() { return "/workplace/demogridwithrepeat.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGridWithRepeat}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
