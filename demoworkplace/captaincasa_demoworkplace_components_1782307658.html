The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Usage in Grid

This demo shows the usage of combo boxes in a grid environment.

Well, there is nothing "specific" to be aware about - just normal grid processing.

When selecting the left combo box (department) then the right combo box (employee) will be updated correspondingly. To do so the list of valid values on server side is updated. It's important to set the FLUSH attribute of the department-combo-box to "true" so that a selection is directly transferred to the server side and as result triggers the exchange of the items of the employee-combo-box.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoComboInGrid}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" bordercolor="#{ccstylevalue.ccLightBorderColor}" objectbinding="#{d.DemoComboInGrid.grid}" sbvisibleamount="5" >
<t:gridcol id="g_4" text="Department" width="100" >
<t:combobox id="g_5" actionListener=".{onDeparmentAction}" flush="true" validvaluesbinding=".{departmentVVS}" value=".{department}" withvaluecomment="true" />
</t:gridcol>
<t:gridcol id="g_6" text="Employee" width="100" >
<t:combobox id="g_7" validvaluesbinding=".{employeeVVS}" value=".{employee}" />
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.jsfserver.util.IClientLogAware;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoComboInGrid}")

public class DemoComboInGrid
    extends PageBean
    implements Serializable, IClientLogAware
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        public ValidValuesBinding getEmployeeVVS() 
        { 
            if ("D".equals(m_department))
                return s_devEmployeesVVS;
            else if ("S".equals(m_department))
                return s_salesEmployeesVVS;
            else
                return s_emptyVVS;
        }
        
        protected String m_employee;
        public String getEmployee() { return m_employee; }
        public void setEmployee(String value) { m_employee = value; }
        
        public ValidValuesBinding getDepartmentVVS() 
        { 
            return s_departmentVVS; 
        }
        
        protected String m_department;
        public String getDepartment() { return m_department; }
        public void setDepartment(String value) {  m_department = value; }
        
        public void onDeparmentAction(ActionEvent event) 
        {
            m_employee = null;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    static ValidValuesBinding s_departmentVVS = new ValidValuesBinding();
    static ValidValuesBinding s_salesEmployeesVVS = new ValidValuesBinding();
    static ValidValuesBinding s_devEmployeesVVS = new ValidValuesBinding();
    static ValidValuesBinding s_emptyVVS = new ValidValuesBinding();
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    static
    {
        s_departmentVVS.addValidValue("S","Sales","Our fabulous sales department");
        s_departmentVVS.addValidValue("D","Development","Great developers work here");
        s_salesEmployeesVVS.addValidValue("1","John");
        s_salesEmployeesVVS.addValidValue("2","Peter");
        s_salesEmployeesVVS.addValidValue("3","Mary");
        s_devEmployeesVVS.addValidValue("10","Karl");
        s_devEmployeesVVS.addValidValue("11","Klaus");
        s_devEmployeesVVS.addValidValue("12","Harry");
        s_devEmployeesVVS.addValidValue("13","Stephanie");
    }
    
    public DemoComboInGrid(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        for (int i=0; i<10; i++)
            m_grid.getItems().add(new GridItem());
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public boolean onCreateClientTestLogMessage()
    {
        StringBuffer sb = new StringBuffer();
        for (GridItem gi: m_grid.getItems())
            sb.append("/" + gi.getDepartment() + "/" + gi.getEmployee());
        ClientTestLog.addClientTestLogMessage(sb.toString());
        return false;
    }

    
}
