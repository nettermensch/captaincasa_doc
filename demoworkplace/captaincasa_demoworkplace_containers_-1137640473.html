The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Positioning in free style HTML
  • Controls embedded in HTML text

The component TEXTWITHCONTROLS allows to add any component into a free style HTML text.

In the text you need to reference the component by a placeholder with the format "@@<componentId>@@". The component has to be listed as sub-component accordingly.

Please pay attention: if you explicitly size your components, then do not use percentage width and percentage height definitions. They are not supported in the context of TEXTWITHCONTROLS.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoTextWithControls}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoTextWithControls}" >
<t:row id="g_6" >
<t:textwithcontrols id="g_2" height="100%" popupmenu="TEXT" text="&lt;h1&gt;Hello World!&lt;/h1&gt;&lt;h2&gt;Step 1&lt;/h2&gt;&lt;p&gt;Please first input your &lt;b&gt;first name&lt;/b&gt; @@firstname@@ then input your &lt;b&gt;last name&lt;/b&gt; @@lastname@@.&lt;/p&gt;&lt;h2&gt;Step 2&lt;/h2&gt;&lt;p&gt;Then press this button: @@hello@@.&lt;/p&gt;&lt;h2&gt;Step 3&lt;/h2&gt;&lt;p&gt;The result then is output into this label: @@result@@.&lt;br&gt;&lt;br&gt;If you want to clear the content of the fields: press this button @@clear@@.&lt;/p&gt;" width="100%" >
<t:field id="firstname" labeltext="First name" text="#{d.DemoTextWithControls.firstName}" width="150" />
<t:field id="lastname" labeltext="Last name" text="#{d.DemoTextWithControls.lastName}" width="150" />
<t:button id="hello" actionListener="#{d.DemoTextWithControls.onHelloAction}" image="/images/iconssvg/ztest_16x16.svg" text="Say Hello" />
<t:label id="result" background="#00000030" labeltext="Result" text="#{d.DemoTextWithControls.result}" />
<t:button id="clear" actionListener="#{d.DemoTextWithControls.onClearAction}" text="Clear fields" />
</t:textwithcontrols>
</t:row>
</t:rowdemobodypane>
<t:popupmenu id="TEXT" >
<t:menuitem id="g_11" actionListener="#{d.DemoTextWithControls.onClearAction}" text="Clear fields" />
</t:popupmenu>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoTextWithControls}")

public class DemoTextWithControls
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    String m_lastName;
    String m_firstName;
    String m_result = "not yet defined";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTextWithControls(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demotextwithcontrols.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTextWithControls}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public void onClearAction(javax.faces.event.ActionEvent event) 
    {
        m_firstName = null;
        m_lastName = null;
        m_result = "not yet defined";
    }

    public void onHelloAction(javax.faces.event.ActionEvent event) 
    {
        m_result = "Hello world, " + m_firstName + " " + m_lastName + "!";
    }

    public String getResult() { return m_result; }
    public void setResult(String value) { this.m_result = value; }

    public String getLastName() { return m_lastName; }
    public void setLastName(String value) { this.m_lastName = value; }

    public String getFirstName() { return m_firstName; }
    public void setFirstName(String value) { this.m_firstName = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
