The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Layering and absolute postioning
  • Status message area on top of grid

The PAINTAREA is some generic contain that manages content within PAINTAREAITEM elements.

In the example some message area is drawn on top of the grid.

Each PAINTAREAITEM content is absolutely positioned. Positions are passed via the BOUNDS property either in pixels ("100") or as percentage figure ("30%"). In addition there is a BOUNDSANCHOR attribute that is used to define the point of the item to which the x/y position is applied.

Optional part of the BOUNDS definition is a z-index ("x;y;width;height;z-index"). Components with higher z-index are drawn on top of components with lower z-index.

Please make sure that you set attribute MOVETOFRONTONMOUSEOVER to "false" - at least for the components that you want to always be kept in the background (in the example: the PAINTAREAITEM containing the grid).

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoPaintAreaUsage}")

public class DemoPaintAreaUsage
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String m_lastName;
        public String getLastName() { return m_lastName; }
        public void setLastName(String value) { this.m_lastName = value; }

        String m_firstName;
        public String getFirstName() { return m_firstName; }
        public void setFirstName(String value) { this.m_firstName = value; }
    }


    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private boolean m_messageAreaShown = true;
    String m_messageAreaX = "50%";
    String m_messageAreaY = "100%";
    String m_messageAreaWidth = "400";
    String m_messaageAraeHeight = "100";
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPaintAreaUsage(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setFirstName("First " + (i+1));
            gi.setLastName("Last " + (i+1));
            m_grid.getItems().add(gi);
        }
    }

    public String getPageName() { return "/workplace/demopaintareausage.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPaintAreaUsage}"; }

    public String getMessageAreaX() { return m_messageAreaX; }
    public String getMessageAreaY() { return m_messageAreaY; }
    public String getMessageAreaWidth() { return m_messageAreaWidth; }
    public String getMessaageAraeHeight() { return m_messaageAraeHeight; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public void onHideShowAction(javax.faces.event.ActionEvent event) 
    {
        m_messageAreaShown = !m_messageAreaShown;
        if (m_messageAreaShown == true)
        {
            m_messageAreaWidth = "400";
            m_messaageAraeHeight = "100";
        }
        else
        {
            m_messageAreaWidth = "100";
            m_messaageAraeHeight = "0";
        }
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
