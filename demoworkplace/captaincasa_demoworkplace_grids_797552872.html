The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Special Issues
  • Fix Columns (Array Grid)

...the same as in the previous example, but now an ARRAYGRID on the right side, no FIXGRID.

It is not possible to join two ARRAYGRID components in order to form a "fix-columns-grid", you have to join one FIXGRID ans one ARRAYGRID component as consequence.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Collection;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDItem;
import org.eclnt.jsfserver.elements.impl.ARRAYGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.util.ArrayListWithChangeIndex;
import org.eclnt.jsfserver.util.ClientTestLog;
import org.eclnt.jsfserver.util.IClientLogAware;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridFixColumns}")

public class DemoGridFixColumns2
    extends PageBean
    implements Serializable, IClientLogAware
{
    public class GridItem extends ARRAYGRIDItem implements java.io.Serializable
    {
        String m_id;
        String m_lastName;
        String m_firstName;
        String m_street;
        String m_zipCode;
        String m_town;
        public void setId(String value) { m_id = value; }
        public String getId() { return m_id; }
        public void setLastName(String value) { m_lastName = value; }
        public String getLastName() { return m_lastName; }
        @Override
        public Object[] getValues()
        {
            return new Object[] {m_firstName,m_street,m_zipCode,m_town};
        }
        @Override
        public void onRowPopupMenuItem(BaseActionEventPopupMenuItem event)
        {
            super.onRowPopupMenuItem(event);
            if ("CLEAR".equals(event.getCommand()))
            {
                m_id = "";
                m_firstName = "";
                m_lastName = "";
                m_street = "";
                m_zipCode = "";
                m_town = "";
                // by default the ARRAYGRIDListBinding is created with supporting
                // change index management. Consequence: changes on item level
                // need to be explicitly passed!
                // (This is not required if creating the ARRAYGRIDListBinding instance
                // with the constructor in which you can pass "false" in the parameter
                // for change index support.
                getChangeIndex().indicateChange();
                Statusbar.outputSuccess("Cleared click info: " + m_grid.getColumnIndexOfLastSelection());
            }
        }
    }

    public DemoGridFixColumns2(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // there are two grid instances - both sharing the same list
        // of items
        m_gridleft.setItems(m_items);
        m_grid.setItems(m_items);
        // both grids are connected in means of scrolling and selection
        m_gridleft.connectWithBinding(m_grid);
        for (int i=0; i<100; i++)
        {
            GridItem gi = new GridItem();
            gi.setId(""+i);
            gi.m_firstName = "First " + i;
            gi.m_lastName = "Last " + i;
            gi.m_street = "Street " + i;
            gi.m_zipCode = "Zipcode " + i;
            gi.m_town = "Town " + i;
            m_items.add(gi);
        }
        m_grid.setTitles(new String[] {"First Name","Street","Zip Code","Town"});
        m_grid.setWidths(new String[] {"150","150","100","150"});
    }
 
    ArrayListWithChangeIndex<GridItem> m_items = new ArrayListWithChangeIndex<GridItem>();
    
    protected FIXGRIDListBinding<GridItem> m_gridleft = new FIXGRIDListBinding<GridItem>();
    public FIXGRIDListBinding<GridItem> getGridleft() { return m_gridleft; }
    
    protected ARRAYGRIDListBinding<GridItem> m_grid = new ARRAYGRIDListBinding<GridItem>();
    public ARRAYGRIDListBinding<GridItem> getGrid() { return m_grid; }
    
    public boolean onCreateClientTestLogMessage()
    {
        StringBuffer sb = new StringBuffer();
        sb.append(m_grid.getSbvalue()+"//");
        sb.append(m_gridleft.getSbvalue()+"//");
        Collection<GridItem> gis = m_grid.getSelectedItemsAsSequence();
        for (GridItem gi: gis)
            sb.append(gi.getId()+"/");
        sb.append("////");
        gis = m_gridleft.getSelectedItemsAsSequence();
        for (GridItem gi: gis)
            sb.append(gi.getId()+"/");
        ClientTestLog.addClientTestLogMessage(sb.toString());
        return false;
    }
    
}
