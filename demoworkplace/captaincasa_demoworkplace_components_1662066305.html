The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • 3D Animations
  • Rotating content - List-detail-scenario

This demo shows how to use the ROTATECONTENTAREA for a typical list-detail scenario:

While navigating through the grid the focus of the user is on the left side - so the right side is rotated.

Once the user double clickes an item on the left, the optical focus will be moved to the right side and the left part will be rotated.

Please note: in this demo all parts of the screen are implemented within one layout - you normally would separate the list part and the detail part into own page beans.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.ccaddons.pbc.datagridview2.CCDataGridView2;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.Person;
import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemorotatecontentareaUI}")

public class DemoRotateContentArea
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    CCDataGridView2<Person> m_grid = new CCDataGridView2<Person>(Person.class);
    boolean m_rotatedLeft = false;
    boolean m_rotatedRight = true;

    List<Person> m_persons;
    Person m_selectedPerson;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoRotateContentArea(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        m_persons = createGridContent();
        m_grid.prepare(m_persons,new CCDataGridView2.DefaultListener<Person>()
        {
            @Override
            public void reactOnGridSelect(Person object)
            {
                m_selectedPerson = object;
                m_rotatedLeft = false;
                m_rotatedRight = true;
            }
            @Override
            public void reactOnGridExecute(Person object)
            {
                m_rotatedLeft = true;
                m_rotatedRight = false;
            }
        });
        m_grid.getGrid().selectItem(0);
    }

    public String getPageName() { return "/workplace/demorotatecontentarea.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemorotatecontentareaUI}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public CCDataGridView2 getGrid() { return m_grid; }
    public Person getSelectedPerson() { return m_selectedPerson; } 
    public boolean getRotatedLeft() { return m_rotatedLeft; }
    public void setRotatedLeft(boolean value) { this.m_rotatedLeft = value; }
    public boolean getRotatedRight() { return m_rotatedRight; }
    public void setRotatedRight(boolean value) { this.m_rotatedRight = value; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private List<Person> createGridContent()
    {
        List<Person> ps = new ArrayList<Person>();
        for (int i=0; i<100; i++)
            ps.add(RandomLogic.createPerson());
        return ps;
    }

}
