The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Touch Screen
  • Point of Sales

This example shows some point of sales functions that take use of touchscreen components.

Click on one of the article images on the left. A corresponding item will be generated. You may update the quantity by again clicking onto the article or by using the numeric touch keyboard.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.math.BigDecimal;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPos}")

public class DemoPos
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class Article implements Serializable
    {
        String i_image;
        BigDecimal i_price;
        String i_name;
        String i_uom;
        public Article(String name, BigDecimal price, String uom, String image)
        {
            super();
            i_name = name;
            i_price = price;
            i_uom = uom;
            i_image = image;
        }
        public String getImage() { return "/images/touch/" + i_image; }
        public BigDecimal getPrice() { return i_price; }
        public String getName() { return i_name; }
        public String getUom() { return i_uom; }
        public void onAction(ActionEvent ae)
        {
            createNewOrderItem(this);
        }
    }
    
    public class OrderItem extends FIXGRIDItem
    {
        Article i_article;
        int i_quantity = 1;
        public OrderItem(Article article)
        {
            i_article = article;
        }
        public Article getArticle() { return i_article; }
        public int getQuantity() { return i_quantity; }
        public void setQuantity(int value) { i_quantity = value; }
        public BigDecimal getPrice()
        {
            return i_article.getPrice().multiply(new BigDecimal(""+i_quantity));
        }
        public void onRowSelect()
        {
            m_selRow = this;
        }
        
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    Article[] m_articles = new Article[15];
    FIXGRIDListBinding<OrderItem> m_rows = new FIXGRIDListBinding<OrderItem>();
    OrderItem m_selRow = null;
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoPos(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        m_articles[0] = new Article("Apple",new BigDecimal("0.49"),"Pc","apple.png");
        m_articles[1] = new Article("Orange",new BigDecimal("0.49"),"Pc","orange.png");
        m_articles[2] = new Article("Lemmon",new BigDecimal("0.49"),"Pc","citron.png");
        m_articles[3] = new Article("Kiwi",new BigDecimal("0.49"),"Pc","kiwi.png");
        m_articles[4] = new Article("Strawberry",new BigDecimal("0.49"),"Kg","strawberry.png");
        m_articles[5] = new Article("Raspberry",new BigDecimal("0.49"),"Kg","raspberry.png");
        m_articles[6] = new Article("Melon (water)",new BigDecimal("0.49"),"Pc","watermelon.png");
        m_articles[7] = new Article("Melon (honey)",new BigDecimal("0.49"),"Pc","honeymelon.png");
        m_articles[8] = new Article("Potatoe",new BigDecimal("0.49"),"Pc","potatoe.png");
        m_articles[9] = new Article("Onion",new BigDecimal("0.49"),"Pc","onion.png");
        m_articles[10] = new Article("Salad",new BigDecimal("0.49"),"Pc","salad.png");
        m_articles[11] = new Article("Carrot",new BigDecimal("0.49"),"Pc","carrot.png");
        m_articles[12] = new Article("Tomatoe",new BigDecimal("0.49"),"Pc","tomatoe.png");
        m_articles[13] = new Article("Asparagus (white)",new BigDecimal("0.49"),"Pc","asparaguswhite.png");
        m_articles[14] = new Article("Asparagus (green)",new BigDecimal("0.49"),"Pc","asparagusgreen.png");
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public Article[] getArticles() { return m_articles; }
    public FIXGRIDListBinding<OrderItem> getRows() { return m_rows; }
    public OrderItem getSelRow() { return m_selRow; }
    
    public BigDecimal getTotal()
    {
        BigDecimal result = new BigDecimal("0");
        for (int i=0; i<m_rows.getItems().size(); i++)
        {
            OrderItem oi = m_rows.getItems().get(i);
            result = result.add(oi.getPrice());
        }
        return result;
    }
    
    public void onRemove(ActionEvent ae)
    {
        if (m_selRow != null)
        {
            m_rows.deselectCurrentSelection();
            m_rows.getItems().remove(m_selRow);
            m_selRow = null;
        }
    }
    
    public boolean isItemAvailable()
    {
        if (m_selRow == null)
            return false;
        else
            return true;
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void createNewOrderItem(Article article)
    {
        if (m_selRow != null && m_selRow.getArticle() == article)
        {
            m_selRow.i_quantity++;
        }
        else
        {
            OrderItem oi = new OrderItem(article);
            m_rows.getItems().add(oi);
            m_rows.deselectCurrentSelection();
            m_rows.selectItem(oi);
            m_rows.ensureItemToBeDisplayed(oi);
            m_selRow = oi;
        }
    }
    
    
}
