The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Scheduling
  • Week Schedule

This demo shows the usage of SCHEDULE and SCHEDULEITEM in the scenario of a week planner.

Quite a lot of issues are contained in the example:

1. dynamic generation of content (DNYAMICCONTENT)
2. Drag and Drop
3. Popup Menu Management

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.elements.componentnodes.SCHEDULEITEMNode;
import org.eclnt.jsfserver.elements.events.BaseActionEventDrop;
import org.eclnt.jsfserver.elements.events.BaseActionEventInvoke;
import org.eclnt.jsfserver.elements.events.BaseActionEventPopupMenuItem;
import org.eclnt.jsfserver.elements.events.BaseActionEventScheduleSizeChanged;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoScheduleWeek}")

public class DemoScheduleWeek
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class Day
    {
        int i_index;
        public Day(int index)
        {
            i_index = index;
        }
        Map<Integer,Occupation> i_occupations = new HashMap<Integer,Occupation>();
        public Occupation createOccupation(String text, int from, int duration)
        {
            Occupation occ = new Occupation();
            occ.i_day = this;
            occ.i_text = text;
            occ.i_from = from;
            occ.i_duration = duration;
            i_occupations.put(occ.i_id,occ);
            m_allOccs.put(occ.i_id,occ);
            return occ;
        }
    }
    
    public class Occupation
    {
        int i_id = s_idCounter++;
        Day i_day;
        long i_from; // in minutes of day
        long i_duration; // in minutes
        String i_text;
        public String getText() { return i_text; }
        public void setText(String text) { i_text = text; }
        public long getDuration() { return i_duration; }
        public void setDuration(long duration) { i_duration = duration; }
        public void moveTo(Day day, int from)
        {
            i_day.i_occupations.remove(i_id);
            i_day = day;
            i_day.i_occupations.put(i_id,this);
            i_from = from;
        }
        public void remove()
        {
            i_day.i_occupations.remove(i_id);
            m_allOccs.remove(i_id);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static int s_idCounter = 0;
    
    Map<Integer,Occupation> m_allOccs = new HashMap<Integer,Occupation>();
    Day[] m_days = new Day[] {new Day(0),new Day(1),new Day(2),new Day(3),new Day(4),new Day(5),new Day(6)};
    
    DYNAMICCONTENTBinding[] m_dyns = new DYNAMICCONTENTBinding[] 
        {new DYNAMICCONTENTBinding(),new DYNAMICCONTENTBinding(),new DYNAMICCONTENTBinding(),
         new DYNAMICCONTENTBinding(),new DYNAMICCONTENTBinding(),new DYNAMICCONTENTBinding(),
         new DYNAMICCONTENTBinding()};
    
    Occupation m_selectedOcc = null;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoScheduleWeek(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        // put some occupations
        m_days[0].createOccupation("Weekly Meeting",600,120);
        m_days[2].createOccupation("Review",900,240);
        m_days[4].createOccupation("Clean up",1080,120);
        // render
        render();
    }

    public String getPageName() { return "/workplace/demoscheduleweek.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoScheduleWeek}"; }
    public Map<Integer,Occupation> getAllOccs() { return m_allOccs; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public DYNAMICCONTENTBinding[] getDyns() { return m_dyns; }
    public Occupation getSelectedOcc() { return m_selectedOcc; }
    public boolean isSelectedDocAvailable() { return m_selectedOcc != null; }
    
    public void onScheduleItemAction(ActionEvent event) 
    {
        String reference = ((BaseActionEvent)event).getSourceReference();
        int occId = ValueManager.decodeInt(reference,-1);
        if (occId < 0)
            throw new Error("Should never happen...");
        m_selectedOcc = m_allOccs.get(occId);
        if (event instanceof BaseActionEventInvoke)
        {
            // no special activities here...
        }
        else if (event instanceof BaseActionEventScheduleSizeChanged)
        {
            BaseActionEventScheduleSizeChanged e = (BaseActionEventScheduleSizeChanged)event;
            m_selectedOcc.i_from = e.getScheduleLeft();
            m_selectedOcc.i_duration = e.getScheduleWidth();
        }
        else if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("REMOVE".equals(e.getCommand()))
            {
                int dayIndex = m_selectedOcc.i_day.i_index;
                m_selectedOcc.remove();
                m_selectedOcc = null;
                renderDay(dayIndex);
            }
        }
    }

    public void onSelectedDocTextAction(ActionEvent event) 
    {
    }

    public void onSelectedDocDurationAction(ActionEvent event) 
    {
    }

    public void onScheduleAction(ActionEvent event) 
    {
        String reference = ((BaseActionEvent)event).getSourceReference();
        int dayId = ValueManager.decodeInt(reference,-1);
        if (dayId < 0 || dayId > 6)
            throw new Error("Should never happen...");
        if (event instanceof BaseActionEventDrop)
        {
            BaseActionEventDrop e = (BaseActionEventDrop)event;
            String dragInfo = e.getDragInfo();
            if (dragInfo.startsWith("OCC:"))
            {
                int occId = ValueManager.decodeInt(dragInfo.substring(4),-1);
                Occupation occ = m_allOccs.get(occId);
                int newFrom = Math.round((e.getPercentageVerticalAsFloat() + e.getPercentageDeltaY()) * (float)1440 / (float) 100);
                occ.moveTo(m_days[dayId],newFrom);
                render();
            }
        }
        else if (event instanceof BaseActionEventPopupMenuItem)
        {
            BaseActionEventPopupMenuItem e = (BaseActionEventPopupMenuItem)event;
            if ("CREATENEWOCCUPATION".equals(e.getCommand()));
            {
                int newFrom = Math.round(e.getPercentageVerticalAsFloat() * (float)1440 / (float) 100);
                Occupation occ = m_days[dayId].createOccupation("New",newFrom,120);
                m_selectedOcc = occ;
                renderDay(dayId);
            }
        }
    }

    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void render()
    {
        for (int i=0; i<7; i++)
            renderDay(i);
    }
    
    private void renderDay(int dayIndex)
    {
        Day day = m_days[dayIndex];
        List<ComponentNode> lcns = new ArrayList<ComponentNode>();
        for (Occupation occ: day.i_occupations.values())
        {
            SCHEDULEITEMNode sin = new SCHEDULEITEMNode();
            lcns.add(sin);
            sin.setText(getRootExpressionUsedInPage().replace("}",".allOccs["+occ.i_id+"].text}"));
            sin.setScheduleleft(""+occ.i_from);
            sin.setSchedulewidth(getRootExpressionUsedInPage().replace("}",".allOccs["+occ.i_id+"].duration}"));
            sin.setBackground("#FFFFC0");
            sin.setReference(""+occ.i_id);
            sin.setInvokeevent("click");
            sin.setSizeratbothsides(true);
            sin.setDragsend("OCC:"+occ.i_id);
            sin.setPopupmenu("SCHEDULEITEM");
            sin.setActionListener(getRootExpressionUsedInPage().replace("}",".onScheduleItemAction}"));
        }
        m_dyns[dayIndex].setContentNodes(lcns);
    }
    
}
