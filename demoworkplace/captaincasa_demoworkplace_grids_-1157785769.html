The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Dynamic Columns
  • Dynamic Grid - Grid level

This example shows a grid that is dynamically created at server side.

The easiest way that you can use for defining dymamic areas within your screen is to use the ROWDYNAMICCONTENT component - that allows to define the XML layout definition to be places inside dynamically. In the example the dynamic layout definition contains a corresponding grid definition.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoGridDyn}">
<t:rowdynamiccontent id="g_2" contentbinding="#{d.DemoGridDyn.dynRow}" />
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.componentnodes.FIELDNode;
import org.eclnt.jsfserver.elements.componentnodes.FIXGRIDNode;
import org.eclnt.jsfserver.elements.componentnodes.GRIDCOLNode;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoGridDyn}")

public class DemoGridDyn
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class MyRow extends FIXGRIDItem
    {
        String[] m_colValues;
        public void setColValues(String[] value) { m_colValues = value; }
        public String[] getColValues() { return m_colValues; }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private String[] m_columns = new String[] {"firstName", "lastName", "street"};

    private ROWDYNAMICCONTENTBinding m_dynRow = new ROWDYNAMICCONTENTBinding();
    private FIXGRIDListBinding<MyRow> m_rows = new FIXGRIDListBinding<MyRow>();

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------



    public DemoGridDyn(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        buildGrid();
        buildData();
    }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ROWDYNAMICCONTENTBinding getDynRow() { return m_dynRow; }
    public void setDynRow(ROWDYNAMICCONTENTBinding value) { m_dynRow = value; }

    public FIXGRIDListBinding<MyRow> getRows() { return m_rows; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------

    private void buildData()
    {
        m_rows.getItems().clear();
        for (int i=0; i<100; i++)
        {
            MyRow r = new MyRow();
            String[] values = new String[m_columns.length];
            for (int j=0; j<m_columns.length; j++)
                values[j] = "VALUE " + i + "/" + j;
            r.setColValues(values);
            m_rows.getItems().add(r);
        }
    }

    private void buildGrid()
    {
        FIXGRIDNode f = new FIXGRIDNode()
                .setObjectbinding(pbx("rows"))
                .setWidth("100%")
                .setHeight("100%")
                .setSbvisibleamount(35);
        int colCounter = 0;
        for (String column: m_columns)
        {
            GRIDCOLNode gc = new GRIDCOLNode()
                    .setText(column)
                    .setWidth(100);
            f.addSubNode(gc);
            FIELDNode fi = new FIELDNode()
                    .setText(".{colValues[\""+colCounter+"\"]}");
            colCounter++;
            gc.addSubNode(fi);
        }
        m_dynRow.setContentNode(f);
    }

}
