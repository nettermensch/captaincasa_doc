The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Animated change of size

By setting attribute ANIMATECHANGEOFSIZE you can animate a change of the WIDTH and/or the HEIGHT of the component.

Please note: this attribute is only available within the FX client.

This is the Java code:

package workplace;

import java.io.Serializable;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoAnimateChangeOfSize}")

public class DemoAnimateChangeOfSize
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    public void onTogglePaneWidthPercPixel(ActionEvent event) 
    {
        if ("240".equals(m_paneWidthPercPixel))
            m_paneWidthPercPixel = "100%";
        else
            m_paneWidthPercPixel = "240";
    }

    String m_paneWidthPercPixel = "240";
    public String getPaneWidthPercPixel() { return m_paneWidthPercPixel; }
    public void setPaneWidthPercPixel(String value) { this.m_paneWidthPercPixel = value; }

    public void onTogglePaneWidthPerc(ActionEvent event) 
    {
        if ("50%".equals(m_paneWidthPerc))
            m_paneWidthPerc = "100%";
        else
            m_paneWidthPerc = "50%";
    }

    String m_paneWidthPerc = "50%";
    public String getPaneWidthPerc() { return m_paneWidthPerc; }
    public void setPaneWidthPerc(String value) { this.m_paneWidthPerc = value; }

    public void onTogglePaneWidth(ActionEvent event) 
    {
        if (m_paneWidth == 240)
            m_paneWidth = 390;
        else
            m_paneWidth = 240;
    }

    int m_paneWidth = 240;
    public int getPaneWidth() { return m_paneWidth; }
    public void setPaneWidth(int value) { this.m_paneWidth = value; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoAnimateChangeOfSize(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demoanimatechangeofsize.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoAnimateChangeOfSize}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
