The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Blockable Area
  • Use case: Grid with Detail Area

This example shows a typical usage scenario for the BLOCKABLEAREA: when the user selects an item in the grid then the item is opened for editing. While editing, the area around the grid is blocked so that the user first must finish editing the item.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoBlockableAreaWithGrid2}")

public class DemoBlockableAreaWithGrid
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Article implements Cloneable
    {
        String i_number;
        String i_text;
        String i_baseUnitOfMeasure;
        public String getNumber() { return i_number; }
        public void setNumber(String number) { i_number = number; }
        public String getText() { return i_text; }
        public void setText(String text) { i_text = text; }
        public String getBaseUnitOfMeasure() { return i_baseUnitOfMeasure; }
        public void setBaseUnitOfMeasure(String baseUnitOfMeasure) { i_baseUnitOfMeasure = baseUnitOfMeasure; }
        @Override
        protected Object clone() throws CloneNotSupportedException { return super.clone(); }
    }
    
    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        Article i_article;
        public GridItem(Article article)
        {
            i_article = article;
        }
        public Article getArticle() { return i_article; }
        @Override
        public void onRowExecute()
        {
            try
            {
                m_selectedArticle = (Article)i_article.clone();
                m_selectedArticleIsNew = false;
            }
            catch (Throwable t)
            {
                throw new Error(t);
            }
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    Article m_selectedArticle;
    boolean m_selectedArticleIsNew;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoBlockableAreaWithGrid(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        for (int i=0; i<50; i++)
        {
            Article a = new Article();
            a.setText("Article " + i);
            a.setNumber("53000"+i);
            a.setBaseUnitOfMeasure("kg");
            m_grid.getItems().add(new GridItem(a));
        }
    }

    public String getPageName() { return "/workplace/demoblockableareawithgrid2.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoBlockableAreaWithGrid2}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }
    public Article getSelectedArticle() { return m_selectedArticle; }
    public boolean getBoolDetailAvailable() { return m_selectedArticle != null; }
    public boolean getBoolGridBlocked() { return getBoolDetailAvailable(); }

    public void onDetailCancelAction(javax.faces.event.ActionEvent event) 
    {
        m_selectedArticle = null;
    }

    public void onDetailOKAction(javax.faces.event.ActionEvent event) 
    {
        if (m_selectedArticleIsNew == false)
        {
            m_grid.getSelectedItem().i_article = m_selectedArticle;
        }
        else
        {   
            m_grid.getItems().add(0,new GridItem(m_selectedArticle));
        }
        m_selectedArticle = null;
    }

    public void onCreateNewArticle(ActionEvent event)
    {
        m_selectedArticle = new Article();
        m_selectedArticleIsNew = true;
    }
}
