The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • RISC Client Integration
  • Geo location

The component GEOLOCATION transfers the current GPS position of the device to the server side. Whenever the GPS location changes a corresponding signal is sent to the server side.

With the attribute SCANDELAY you can limit the frequency of sending updated positions: if e.g. the value is set to 10000 (10 seconds) then there will be a maximum frequency of one message every 10 seconds.

The component only sends updates! It does not regularly trigger the server - but only if some data change happened.<

The user needs to allow the access to the location information when using this component, otherwise some error message will be sent to server side.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.Date;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.events.BaseActionEventGeoLocationUpdate;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

import javax.faces.event.ActionEvent;

@CCGenClass (expressionBase="#{d.DemoGeoLocation}")

public class DemoGeoLocation
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    String m_protocol = "Protocol:\n";
    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    public void onGeoLocationUpdate(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventGeoLocationUpdate)
        {
            BaseActionEventGeoLocationUpdate e = (BaseActionEventGeoLocationUpdate)event;
            m_protocol += "Event: " + e.getStatus() + " at " + new Date().toString() + "\n";
            m_accuracy = e.getAccuracy(); 
            m_longitude = e.getLongitude(); 
            m_latitude = e.getLatitude(); 
        }
    }

    double m_accuracy;
    public double getAccuracy() { return m_accuracy; }
    public void setAccuracy(double value) { this.m_accuracy = value; }

    double m_latitude;
    public double getLatitude() { return m_latitude; }
    public void setLatitude(double value) { this.m_latitude = value; }

    double m_longitude;
    public double getLongitude() { return m_longitude; }
    public void setLongitude(double value) { this.m_longitude = value; }

    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /* Listener to the user of the page bean. */
    public interface IListener
    {
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    private IListener m_listener;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoGeoLocation(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demogeolocation.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoGeoLocation}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    /* Initialization of the bean. Add any parameter that is required within your scenario. */
    public void prepare(IListener listener)
    {
        m_listener = listener;
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
