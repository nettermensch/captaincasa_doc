The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Grids & Trees
  • Grids and Trees
  • Tree
  • Tree with editable text

Within the TREENODE component you may place any sub-component. So you can also place a FIELD inside.

By setting attribute ONLYSHOWSUBCOMPONENT to true, the TREENODE component does not render its own text anymore but only renders the sub-component.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.DemoTreeEditableText}" >
<t:row id="g_2" >
<t:fixgrid id="g_3" background="#FFFFFF" drawoddevenrows="false" height="100%" objectbinding="#{d.DemoTreeEditableText.tree}" sbvisibleamount="35" suppressheadline="true" width="100%" >
<t:gridcol id="g_4" text="Column" width="100%" >
<t:treenode id="g_5" onlyshowsubcomponent="true" >
<t:field id="g_6" background="#00000000" border="0" text=".{text}" width="100%" />
</t:treenode>
</t:gridcol>
</t:fixgrid>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeBinding;
import org.eclnt.jsfserver.elements.impl.FIXGRIDTreeItem;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoTreeEditableText}")

public class DemoTreeEditableText
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class TreeNode extends FIXGRIDTreeItem implements java.io.Serializable
    {
        public TreeNode(FIXGRIDTreeItem parentNode)
        {
            super(parentNode);
        }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
        public void onToggle()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDTreeBinding<TreeNode> m_tree = new FIXGRIDTreeBinding<TreeNode>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoTreeEditableText(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);
        for (int i=0; i<10; i++)
        {
            TreeNode top = new TreeNode(m_tree.getRootNode());
            top.setText("Some text: Top " + i);
            top.setStatus(TreeNode.STATUS_CLOSED);
            for (int j=0; j<5; j++)
            {
                TreeNode sub = new TreeNode(top);
                sub.setText("Some text: Sub " + i + "/" + j);
                sub.setStatus(TreeNode.STATUS_ENDNODE);
            }
        }
    }

    public String getPageName() { return "/workplace/demotreeeditabletext.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoTreeEditableText}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDTreeBinding<TreeNode> getTree() { return m_tree; }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
