The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Auto-hiding content

The component SIZEANIMATIONAREA that is shown with an initial size and then is resized to its long term size after some duration.

The initial size is defined by attributes FROMWIDTH and FROMHEIGHT. The long term size is defined by attribute TOWIDTH and TOHEIGHT. - If you do not define a value then the size is calculated automatically by its content. You can define any type of sized (pixel sizes, percentage sizes, ...). By using the attribute DURATION you define the duration until the initial size is replaced by the long term size.

The components provides an attribute TRIGGER: when triggering the component then the size is again switched to the initial size for the defined duration.

A typical use case is shown in the demo: some content is shown which then is completely or in parts optically removed.

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.impl.FIXGRIDItem;
import org.eclnt.jsfserver.elements.impl.FIXGRIDListBinding;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.workplace.IWorkpageDispatcher;

import workplace.logic.RandomLogic;

@CCGenClass (expressionBase="#{d.DemoSizeAnimationArea1}")

public class DemoSizeAnimationArea1
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class GridItem extends FIXGRIDItem implements java.io.Serializable
    {
        String i_lastName;
        String i_firstName;
        public String getLastName() { return i_lastName; }
        public void setLastName(String value) { this.i_lastName = value; }
        public String getFirstName() { return i_firstName; }
        public void setFirstName(String value) { this.i_firstName = value; }
        public void onRowSelect()
        {
        }
        public void onRowExecute()
        {
        }
    }

    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    FIXGRIDListBinding<GridItem> m_grid = new FIXGRIDListBinding<GridItem>();
    
    Trigger m_trigger = new Trigger();
    Trigger m_triggerPostIt = new Trigger();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoSizeAnimationArea1(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        fillGrid();
        m_trigger.trigger();
    }

    public String getPageName() { return "/workplace/demosizeanimationarea1.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSizeAnimationArea1}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public FIXGRIDListBinding<GridItem> getGrid() { return m_grid; }

    public Trigger getTrigger() { return m_trigger; }
    public Trigger getTriggerPostIt() { return m_triggerPostIt; }
    
    public void onShowFromSizeAction(ActionEvent event)
    {
        m_trigger.trigger();
    }

    public void onShowPostItAction(javax.faces.event.ActionEvent event) 
    {
        m_triggerPostIt.trigger();
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void fillGrid()
    {
        for (int i=0; i<50; i++)
        {
            GridItem gi = new GridItem();
            m_grid.getItems().add(gi);
            gi.setFirstName(RandomLogic.getFirstName());
            gi.setLastName(RandomLogic.getLastName());
        }
    }

}
