The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Arranging Content
  • Popup Dialogs
  • OK- / Yes-No popup with HTML text

By default the text that you pass into the OK- and YESNO-Popup is treated as plain text.

Use method setTextContentType(...) to indicate that the text that you pass is some formatted HTML text.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoDefaultPopupsHtml}" />
<t:rowdemobodypane id="g_4" objectbinding="#{d.DemoDefaultPopupsHtml}" rowdistance="#{ccstylevalue[&#39;@defaultrowdistance@&#39;]}" >
<t:row id="g_6" >
<t:button id="g_7" actionListener="#{d.DemoDefaultPopupsHtml.onOkPopupDefaultAction}" text="OK popup - Default" width="200+" />
</t:row>
<t:row id="g_9" >
<t:button id="g_10" actionListener="#{d.DemoDefaultPopupsHtml.onOkPopupHtmlAction}" text="OK popup - HTML" width="200+" />
</t:row>
<t:rowdistance id="g_18" height="50" />
<t:row id="g_12" >
<t:button id="g_13" actionListener="#{d.DemoDefaultPopupsHtml.onYesNoPopupDefaultAction}" text="Yes/No popup - Default" width="200+" />
</t:row>
<t:row id="g_15" >
<t:button id="g_16" actionListener="#{d.DemoDefaultPopupsHtml.onYesNoPopupHtmlAction}" text="Yes/No popup - HTML" width="200+" />
</t:row>
</t:rowdemobodypane>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.OKPopup;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup;
import org.eclnt.jsfserver.defaultscreens.YESNOPopup.IYesNoListener;
import org.eclnt.jsfserver.util.ICCServerConstants;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoDefaultPopupsHtml}")

public class DemoDefaultPopupsHtml
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoDefaultPopupsHtml(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
    }

    public String getPageName() { return "/workplace/demodefaultpopupshtml.xml"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoDefaultPopupsHtml}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public void onOkPopupHtmlAction(javax.faces.event.ActionEvent event) 
    {
        String html = createHtmlText();
        OKPopup okp = OKPopup.createInstance("Some html text",html);
        okp.setTextContentType(ICCServerConstants.TEXTPANE_CONTENTTYPE_HTML);
    }

    public void onOkPopupDefaultAction(javax.faces.event.ActionEvent event) 
    {
        String html = createHtmlText();
        OKPopup.createInstance("Some html text",html);
    }

    public void onYesNoPopupHtmlAction(javax.faces.event.ActionEvent event) 
    {
        String html = createHtmlText();
        YESNOPopup ynp = YESNOPopup.createInstance("Some HTML text",html,new IYesNoListener()
        {
            @Override
            public void reactOnYes()
            {
                Statusbar.outputSuccess("Yes! ...was pressed...");
            }
            @Override
            public void reactOnNo()
            {
                Statusbar.outputSuccess("No! ...was pressed...");
            }
        });
        ynp.setTextContentType(ICCServerConstants.TEXTPANE_CONTENTTYPE_HTML);
    }

    public void onYesNoPopupDefaultAction(javax.faces.event.ActionEvent event) 
    {
        String html = createHtmlText();
        YESNOPopup.createInstance("Some HTML text",html,new IYesNoListener()
        {
            @Override
            public void reactOnYes()
            {
                Statusbar.outputSuccess("Yes! ...was pressed...");
            }
            @Override
            public void reactOnNo()
            {
                Statusbar.outputSuccess("No! ...was pressed...");
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private String createHtmlText()
    {
        String html = "Lorem ipsum dolor <span style='font-size:20px;font-weight:bold'>sit amet</span>, consetetur sadipscing elitr, "
            + "sed diam nonumy eirmod tempor invidunt ut labore et "
            + "dolore magna aliquyam erat, <span style='font-size:20px;font-weight:bold'>sed diam voluptua</span>.";
        return html;
    }
}
