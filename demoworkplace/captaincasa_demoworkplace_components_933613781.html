The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Images
  • Image Anonymizer

The component IMAGEANONYMIZERDIRECT allows the drawing on lines on top of an existing image.

The image is passed as hexadecimal information to the client side using attribute HEXDATA. If the user updates the image then the image is passed back as hexadecimal image into the HEXDATA attribute as well.

The control in general is a "mini image editor" but typically is used in the context of anonymizing images (e.g. to hide faces, number plates, ...).

This is the Java code:

package workplace;

import java.io.Serializable;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.util.WebResourceReader;
import org.eclnt.util.file.FileManager;
import org.eclnt.util.valuemgmt.ValueManager;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedBean;

@CCGenClass (expressionBase="#{d.DemoImageAnonymizerDirect}")

public class DemoImageAnonymizerDirect
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    String m_penColor = "#FF0000";
    int m_pensize = 15;
    String m_imageData;
    
    public DemoImageAnonymizerDirect(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        byte[] bs = (new WebResourceReader()).readFileIntoByteArray("/images/pictures/carpicture.jpg");
        m_imageData = ValueManager.encodeHexString(bs);
    }

    public String getImageData() { return m_imageData; }
    public void setImageData(String value) { this.m_imageData = value; }

    public String getPenColor() { return m_penColor; }
    public void setPenColor(String value) { this.m_penColor = value; }

    public int getPensize() { return m_pensize; }
    public void setPensize(int value) { this.m_pensize = value; }

    public void onUndoChanges(ActionEvent event) 
    {
        byte[] bs = (new WebResourceReader()).readFileIntoByteArray("/images/pictures/carpicture.jpg");
        m_imageData = ValueManager.encodeHexString(bs);
    }
    
    public void onReload(ActionEvent event) 
    {
        byte[] bs = (new WebResourceReader()).readFileIntoByteArray("/images/pictures/carpicture.jpg");
        m_imageData = ValueManager.encodeHexString(bs);
    }
    
    public void onReload2(ActionEvent event) 
    {
        byte[] bs = (new WebResourceReader()).readFileIntoByteArray("/images/pictures/wpBubbles.png");
        m_imageData = ValueManager.encodeHexString(bs);
    }
    
    public void onReload3(ActionEvent event) 
    {
        byte[] bs = (new WebResourceReader()).readFileIntoByteArray("/images/pictures/germany.gif");
        m_imageData = ValueManager.encodeHexString(bs);
    }
    
    public void onAnonymizerAction(ActionEvent event) 
    {
    }
}
