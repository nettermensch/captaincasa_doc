The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • File Up/Download
  • Dynamic Download Content - Buffered Content

The content that is provided on server side is dynamic content.

There is a nice function to access dynamic content: create a class that extends interface IBufferedContent by extending class DefaultBufferedContent. In the class you need to implement the method getContent() and the method getContentType().

The methods are called at this point of time, when the user activates the download. In other words: you do not need to build up the download-content in advance (maybe the user does not want to download at all!).

By definint WITHCALLBACK as "true" the server gets notified when the client started the download of the content.

This is the Java code:

package workplace;

import java.io.OutputStream;
import java.io.Serializable;
import java.util.Date;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.bufferedcontent.BufferedContentMgr;
import org.eclnt.jsfserver.bufferedcontent.DefaultBufferedContent;
import org.eclnt.jsfserver.bufferedcontent.DefaultBufferedStreamContent;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;

@CCGenClass (expressionBase="#{d.DemoFileBufferedContent}")

public class DemoFileBufferedContent
    extends WorkpageDispatchedPageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * The getContent() method of this class will be called when the
     * download servlet is accessed.
     */
    class MyBufferedContent extends DefaultBufferedContent
    {
        public byte[] getContent()
        {
            try
            {
                // return result as byte[]
                StringBuffer sb = new StringBuffer();
                for (int i=0; i<1000000; i++)
                    sb.append("1234567890");
                return (sb.toString().getBytes("UTF-8"));
            }
            catch (Throwable t)
            {
                throw new Error("Should never happen!");
            }
        }
        public String getContentType()
        {
            return "application/octet-stream"; // always download, do not show in browser
        }
    }
    
    /**
     * Same implementation as MyBufferedContent, but not using the stream-based
     * implementation. Advantage: the content is not kept in memory as whole.
     */
    class MyBufferedStreamContent extends DefaultBufferedStreamContent
    {
        @Override
        public void writeStream(OutputStream stream)
        {
            try
            {
                // directly write content to stream
                for (int i=0; i<1000000; i++)
                {
                    byte[] bs = "1234567890".getBytes("UTF-8");
                    stream.write(bs);
                }
            }
            catch (Throwable t)
            {
                throw new Error("Should never happen!");
            }
        }
        @Override
        public String getContentType()
        {
             return "application/octet-stream"; // always download, do not show in browser
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    MyBufferedContent m_bufferedContent;
    MyBufferedStreamContent m_bufferedStreamContent;

    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------

    public DemoFileBufferedContent(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // create buffered content
        m_bufferedContent = new MyBufferedContent();
        m_bufferedContent.setFileName("testBufferedContent.txt"); // optional: explicitly assign download file name
        BufferedContentMgr.add(m_bufferedContent);
        // create buffered stream content
        m_bufferedStreamContent = new MyBufferedStreamContent();
        BufferedContentMgr.add(m_bufferedStreamContent);
        // the buffered content is registered in the dialog session; it is automatically released
        // when the session ends. But you can (and should) release it as early as possible in order 
        // to release the associated memory. The point of time to release is, is up to you: here
        // we used the end ot the life cycle of a workpage within the workplace.
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener() 
        {
            public void reactOnDestroyed()
            {
                super.reactOnDestroyed();
                BufferedContentMgr.remove(m_bufferedContent);
                BufferedContentMgr.remove(m_bufferedStreamContent);
            }
        });
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getDownloadURL() { return m_bufferedContent.getURL(); }
    public String getStreamDownloadURL() { return m_bufferedStreamContent.getURL(); }

    public void onCallServer(ActionEvent event) 
    {
        Statusbar.outputSuccess("Download was triggered, timestamp = " + new Date().getTime());
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    
}
