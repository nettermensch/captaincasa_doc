The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Structure Components
  • Outlook Bar

The Outlook bar consists out of a number of items, each rendered as some kind of button. When pressing the button then the items are arranged around the content, so that the button, that was selected, is directly located above the content area.

The server gets notified by the button click and can now itself arrange the content area in any way. In the example the content area is dynamically built using component binding. It is also possible to define different pages to be shown, using the ROWINCLUDE component.

Please note: the outlook bar does not open up one content pane per item, but it has one content pane, that is used by all items.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:rowdemobodypane id="g_1" objectbinding="#{d.demoOutlookbar}" >
<t:row id="g_2" >
<t:outlookbar id="g_3" height="100%" value="#{d.demoOutlookbar.contentpageIndex}" width="150" >
<t:outlookbaritem id="g_4" actionListener="#{d.demoOutlookbar.onMails}" align="left" image="/images/icons/table.png" text="Mails" />
<t:outlookbaritem id="g_5" actionListener="#{d.demoOutlookbar.onFolders}" align="left" image="/images/icons/style.png" text="Folders" />
<t:outlookbaritem id="g_6" actionListener="#{d.demoOutlookbar.onAddresses}" align="left" image="/images/icons/telephone.png" text="Addresses" />
<t:outlookbaritem id="g_7" actionListener="#{d.demoOutlookbar.onFavourites}" align="left" height="40" image="/images/icons/disk.png" text="Favorites" />
<t:outlookbarcontent id="g_8" background="#FFFFFF40" curtaincolor="#F0F0F0" >
<t:rowdynamiccontent id="g_9" contentbinding="#{d.DemoOutlookbar.dynamicContent}" />
</t:outlookbarcontent>
</t:outlookbar>
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.componentnodes.BUTTONNode;
import org.eclnt.jsfserver.elements.componentnodes.LABELNode;
import org.eclnt.jsfserver.elements.componentnodes.PANENode;
import org.eclnt.jsfserver.elements.componentnodes.ROWNode;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.workplace.IWorkpageDispatcher;

public class DemoOutlookbar extends PageBean
{
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    int m_contentPageIndex = 0;
    int m_renderedContentPageIndex = -1;
    ROWDYNAMICCONTENTBinding m_dynamicContent = new ROWDYNAMICCONTENTBinding();
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public DemoOutlookbar(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        initRefComponent("outlookbar");
        buildContentMails();
    }
    
    public ROWDYNAMICCONTENTBinding getDynamicContent() { return m_dynamicContent; }

    public void setContentpageIndex(int value) { m_contentPageIndex = value; }
    public int getContentpageIndex() { return m_contentPageIndex; }
    
    public void onMails(ActionEvent ae)
    {
        Statusbar.outputMessage("onMails " + m_contentPageIndex);
        if (m_renderedContentPageIndex == 0) return;
        buildContentMails();
    }
    
    public void onFolders(ActionEvent ae)
    {
        Statusbar.outputMessage("onFolders " + m_contentPageIndex);
        if (m_renderedContentPageIndex == 1) return;
        buildContentFolders();
    }
    
    public void onAddresses(ActionEvent ae)
    {
        Statusbar.outputMessage("onAddresses " + m_contentPageIndex);
        if (m_renderedContentPageIndex == 2) return;
        buildContentAddresses();
    }
    
    public void onFavourites(ActionEvent ae)
    {
        Statusbar.outputMessage("onFavourites " + m_contentPageIndex);
        if (m_renderedContentPageIndex == 3) return;
        buildContentFavourites();
    }
    
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void buildContentMails()
    {
        m_renderedContentPageIndex = 0;
        PANENode p = new PANENode();
        p.setHeight("100%");
        p.setWidth("100%");
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            BUTTONNode b = new BUTTONNode();
            r.addSubNode(b);
            b.setText("Mails");
            b.setContentareafilled("false");
            b.setImage("/images/icons/bigmails.png");
            b.setImagepressed("/images/icons/bigmailspressed.png");
            b.setImagerollover("/images/icons/bigmailsrollover.png");
            b.setWidth("100%");
            b.setHorizontaltextposition("center");
            b.setVerticaltextposition("bottom");
        }
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            BUTTONNode b = new BUTTONNode();
            r.addSubNode(b);
            b.setText("Voice Mails");
            b.setContentareafilled("false");
            b.setImage("/images/icons/bigspeaker.png");
            b.setImagepressed("/images/icons/bigspeakerpressed.png");
            b.setImagerollover("/images/icons/bigspeakerrollover.png");
            b.setWidth("100%");
            b.setHorizontaltextposition("center");
            b.setVerticaltextposition("bottom");
        }
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            BUTTONNode b = new BUTTONNode();
            r.addSubNode(b);
            b.setText("Calendar");
            b.setContentareafilled("false");
            b.setImage("/images/icons/bigcalendar.png");
            b.setImagepressed("/images/icons/bigcalendarpressed.png");
            b.setImagerollover("/images/icons/bigcalendarrollover.png");
            b.setWidth("100%");
            b.setHorizontaltextposition("center");
            b.setVerticaltextposition("bottom");
        }
        m_dynamicContent.setContentNode(p);
    }
    
    private void buildContentFolders()
    {
        m_renderedContentPageIndex = 1;
        PANENode p = new PANENode();
        p.setHeight("100%");
        p.setWidth("100%");
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            LABELNode b = new LABELNode();
            r.addSubNode(b);
            b.setText("Folders - any content");
        }
        m_dynamicContent.setContentNode(p);
    }
    
    private void buildContentAddresses()
    {
        m_renderedContentPageIndex = 2;
        PANENode p = new PANENode();
        p.setHeight("100%");
        p.setWidth("100%");
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            LABELNode b = new LABELNode();
            r.addSubNode(b);
            b.setText("Addresses - any content");
        }
        m_dynamicContent.setContentNode(p);
    }
    
    private void buildContentFavourites()
    {
        m_renderedContentPageIndex = 3;
        PANENode p = new PANENode();
        p.setHeight("100%");
        p.setWidth("100%");
        {
            ROWNode r = new ROWNode();
            p.addSubNode(r);
            LABELNode b = new LABELNode();
            r.addSubNode(b);
            b.setText("Favourites - any content");
        }
        m_dynamicContent.setContentNode(p);
    }
}
