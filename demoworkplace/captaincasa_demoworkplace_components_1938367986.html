The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Filtered Pick List (PBC) as popup

This example shows how a COMBOFIELD is combined with a filtered pick list.

This is the XML layout definition:

<?xml version="1.0" encoding="utf-8"?>
<t:layoutdefinition xmlns:t="http://www.CaptainCasa.com/controllibrary/t" xmlns:tpbc="http://www.CaptainCasa.com/controllibrary/tpbc" >
<t:beanprocessing id="g_1" beanbinding="#{d.DemoPBCCCFilteredPicklistCombo}" />
<t:rowdemobodypane id="g_2" objectbinding="#{d.DemoPBCCCFilteredPicklistCombo}" >
<t:row id="g_3" >
<t:label id="g_4" text="Persons" width="100" />
<t:combofield id="g_5" actionListener="#{d.DemoPBCCCFilteredPicklistCombo.onPersonsAction}" editable="false" text="#{d.DemoPBCCCFilteredPicklistCombo.persons}" width="100%;200" />
</t:row>
</t:rowdemobodypane>
<t:pageaddons id="g_pa"/>
</t:layoutdefinition>

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import org.eclnt.ccaddons.pbc.CCFilteredPickList;
import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.ModalPopup;
import org.eclnt.jsfserver.elements.util.DefaultModalPopupListener;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoPBCCCFilteredPicklistComboUI}")

public class DemoPBCCCFilteredPicklistCombo
    extends WorkpageDispatchedPageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public static class MyPerson
    {
        String m_name;
        public MyPerson(String name)
        {
            m_name = name;
        }
        public String getName() { return m_name; } 
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    List<MyPerson> m_allPersons = new ArrayList<DemoPBCCCFilteredPicklistCombo.MyPerson>();
    Set<Object> m_selectedPersons = new HashSet<Object>();

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoPBCCCFilteredPicklistCombo(IWorkpageDispatcher workpageDispatcher)
    {
       super(workpageDispatcher);
       for (int i=0; i<20; i++)
           m_allPersons.add(new MyPerson("Person " + i));
    }

    public String getPageName() { return "/workplace/demopbcccfilteredpicklistcombo.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoPBCCCFilteredPicklistComboUI}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    public String getPersons() 
    { 
        StringBuffer sb = new StringBuffer();
        int counter = 0;
        for (Object p: m_selectedPersons)
        {
            if (counter != 0) sb.append(", ");
            sb.append(((MyPerson)p).getName());
            counter++;
        }
        return sb.toString();
    }
    
    public void onPersonsAction(javax.faces.event.ActionEvent event) 
    {
        final CCFilteredPickList ui = new CCFilteredPickList();
        for (MyPerson p: m_allPersons)
            ui.addItem(p,p.getName(),null);
        ui.prepare(new CCFilteredPickList.IListener()
        {
            @Override
            public void reactOnUpdate()
            {
                m_selectedPersons = ui.getSelection();
            }
        });
        ui.setSelection(m_selectedPersons);
        ui.setPanePadding("10");
        ModalPopup pop = openModalPopup(ui,"Filtered selection",450,0,new DefaultModalPopupListener(this,ui));
        pop.setCloseonclickoutside(true);
        pop.setUndecorated(true);
    }


}
