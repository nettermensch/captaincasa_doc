The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Containers
  • Slide Container
  • Interactive time line

The SLIDECONTAINER sends an event to server side when being moved by the user. The event's data includes:

1. the current pixel position of the content inside the container.
2. the current pixel width of the container
3. the current pixel width of the content

In addition the current pixel-shift-value also is automatically transferred into its assigned property (here: "pixelShift"). This property serves both for passing the current shift from the client to the server and for passing some updated value from the server to the client. In the example the client's value is rounded on server side so that the shift is always starting at some quarter of an hour.

In the example this information is used for calculating the "time position" of the time line, that is rendered as content of the SLIDECONTAINER component.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.defaultscreens.Statusbar;
import org.eclnt.jsfserver.elements.events.BaseActionEventSlideContainer;
import org.eclnt.jsfserver.pagebean.PageBean;
import org.eclnt.workplace.IWorkpageContainer;
import org.eclnt.workplace.IWorkpageDispatcher;

@CCGenClass (expressionBase="#{d.DemoSlideContainer}")

public class DemoSlideContainerTimeline
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    public class ScheduleHoursItem extends Object implements java.io.Serializable
    {
        int i_hour;
        String i_hourString;
        public ScheduleHoursItem(int hour)
        {
            i_hour = hour;
            i_hourString = "0" + i_hour+":00";
            if (i_hourString.length() > 5)
                i_hourString = i_hourString.substring(i_hourString.length()-5);
        }
        public int getHour()
        {
            return i_hour;
        }
        public String getHourString() 
        { 
            return i_hourString; 
        }
    }
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    int m_pixelShift = 0;
    long m_pixelShiftAsTime = 0;
    List<ScheduleHoursItem> m_scheduleHours = new ArrayList<ScheduleHoursItem>();
    int m_totalPixels = 0;
    
    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoSlideContainerTimeline(IWorkpageDispatcher dispatcher)
    {
        super(dispatcher);
        // create schedule
        for (int i=0; i<24; i++)
        {
            ScheduleHoursItem shi = new ScheduleHoursItem(i);
            m_scheduleHours.add(shi);
        }
    }

    public String getPageName() { return "/workplace/demoslidecontainer.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoSlideContainer}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<ScheduleHoursItem> getScheduleHours() { return m_scheduleHours; }
    
    public int getPixelShift() { return m_pixelShift; }
    public void setPixelShift(int value) { this.m_pixelShift = value; }

    public long getPixelShiftAsTime() { return m_pixelShiftAsTime; }
   
    public void onSlideContainerAction(javax.faces.event.ActionEvent event) 
    {
        if (event instanceof BaseActionEventSlideContainer)
        {
            BaseActionEventSlideContainer e = (BaseActionEventSlideContainer)event;
            Statusbar.outputSuccess("Slide was executed: " + e.getPixelShift() + "/" + e.getPixelWidth() + "/" + e.getContentPixelWidth());
            m_totalPixels = e.getContentPixelWidth();
            // round to quarters of an hour
            long pat = Math.round(((double)m_pixelShift / (double)m_totalPixels) * 3600000d * 24d);
            pat = Math.round((double)pat/900000) * 900000;
            m_pixelShiftAsTime = pat;
            m_pixelShift = (int)Math.round( (double)pat/(3600000d*24d) * (double)e.getContentPixelWidth());
        }
    }

    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
}
