The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Straight Controls
  • Combobox and List
  • Selection of objects
  • On your own: using COMBOBOX

The COMBOBOX is managing id-Strings and text-Strings (and maybe some comment and image in addition)

Do not hesitate to override the ValidValuesBinding class to add some more references to objects.

In this example the ValidValuesBinding implementation contains some cross reference to objects, so that within the managed bean the Department-object is managed. Of course you now see that in the example all Department objects are held in memory, which may have impacts for big scenarios.

This is the Java code:

package workplace;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.ValidValuesBinding;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoComboBoxWithObjects}")

public class DemoComboBoxWithObjects
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    public class Department
    {
        String i_id;
        String i_text;
        public String getId() { return i_id; }
        public void setId(String id) { i_id = id; }
        public String getText() { return i_text; }
        public void setText(String text) { i_text = text; }
    }
    
    public class ValidValuesBindingWithObject<OBJCLASS extends Object> extends ValidValuesBinding
    {
        Map<String,OBJCLASS> i_objects = new HashMap<>();
        Map<OBJCLASS,String> i_objectIds = new HashMap<>();
        public void addObject(String id, String text, OBJCLASS object)
        {
            i_objects.put(id,object);
            i_objectIds.put(object,id);
            addValidValue(id,text);
        }
        public OBJCLASS lookupObject(String id)
        {
            if (id == null) return null;
            return i_objects.get(id);
        }
        public String lookupId(OBJCLASS object)
        {
            if (object == null) return null;
            return i_objectIds.get(object);
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------

    ValidValuesBindingWithObject<Department> m_departmentVVS = new ValidValuesBindingWithObject();
    Department m_department = null;

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoComboBoxWithObjects(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        loadDepartments();
    }

    public String getPageName() { return "/workplace/democomboboxwithobjects.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoComboBoxWithObjects}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public ValidValuesBinding getDepartmentVVS() { return m_departmentVVS; }

    public String getDepartmentId() { return m_departmentVVS.lookupId(m_department); }
    public void setDepartmentId(String value) { m_department = m_departmentVVS.lookupObject(value); }
    
    public Department getDepartment() { return m_department; }


    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    private void loadDepartments()
    {
        for (int i=0; i<20; i++)
        {
            Department d = new Department();
            d.setId(""+i);
            d.setText("Department " + i);
            m_departmentVVS.addObject(d.getId(),d.getText(),d);
        }
    }
}
