The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • Components
  • Animation & Graphics & Sound
  • Paint Area
  • Slide Show

This demo shows how to use the PAINTAREA component in order to create an animated change of the position of a contained PAINTAREA item.

If changing the position of a PAINTAREA item, then the change is not directly populated to the component - but waits for an explicit TRIGGER event. After firing the trigger, the component is softly animated from its former position to its next position.

This is the Java code:

package workplace;

import java.awt.Rectangle;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.faces.event.ActionEvent;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.BaseActionEvent;
import org.eclnt.jsfserver.elements.componentnodes.PAINTAREAITEMNode;
import org.eclnt.jsfserver.elements.impl.DYNAMICCONTENTBinding;
import org.eclnt.jsfserver.elements.impl.PAINTAREAComponent;
import org.eclnt.jsfserver.elements.impl.PAINTAREAITEMComponentTag;
import org.eclnt.jsfserver.elements.impl.ROWDYNAMICCONTENTBinding.ComponentNode;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.managedbean.IDispatcher;
import org.eclnt.jsfserver.util.HttpSessionAccess;
import org.eclnt.util.valuemgmt.ValueManager;

@CCGenClass (expressionBase="#{d.DemoIntro}")

public class DemoIntro 
    extends PageBean
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    static String[] DEMOIMAGES = new String[]
    {
        "/images/pictures/demoworkplace/demo1.jpg",
        "/images/pictures/demoworkplace/demo2.jpg",
        "/images/pictures/demoworkplace/demo3.jpg"
    };
    
    protected int m_imageIndex = 0;
    protected List<String> m_bounds = new ArrayList<String>();
    protected PAINTAREAComponent m_paintArea = new PAINTAREAComponent();
    protected String m_bgpaintArea;
    protected Trigger m_trigger = new Trigger();
    
    DYNAMICCONTENTBinding m_content = new DYNAMICCONTENTBinding();
    
    // ------------------------------------------------------------------------
    // constructors
    // ------------------------------------------------------------------------
    
    public DemoIntro(IDispatcher dispatcher)
    {
        super(dispatcher);
        buildBounds();
        m_bgpaintArea = "";
        render();
    }
    
    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------

    public List<String> getBounds() { return m_bounds; }
    public DYNAMICCONTENTBinding getContent() { return m_content; }
    
    public String getBgpaintArea() { return m_bgpaintArea; }
    public void setBgpaintArea(String value) { m_bgpaintArea = value; }

    private void render()
    {
        List<ComponentNode> nodes = new ArrayList<ComponentNode>();
        for (int i=0; i<DEMOIMAGES.length; i++) 
        {
            PAINTAREAITEMNode pai = new PAINTAREAITEMNode();
            nodes.add(pai);
            pai.setBgpaint("image(0,0,"+DEMOIMAGES[i]+",lefttop)");
            pai.setInvokeevent("click");
            pai.setMovetofrontonmouseover(false);
            pai.setBounds(pbx("bounds["+i+"]"));
            pai.setActionListener(pbx("onItemAction"));
            pai.setReference(""+i);
        }
        m_content.setContentNodes(nodes);
    }
    
    public Trigger getTrigger() { return m_trigger; }
    
    public void onItemAction(ActionEvent event)
    {
        String refStr = ((BaseActionEvent)event).getSourceReference();
        int ref = ValueManager.decodeInt(refStr,0);
        m_imageIndex = ref;
        buildBounds();
    }
    
    public void onNext(ActionEvent event)
    {
        m_imageIndex++;
        if (m_imageIndex >= DEMOIMAGES.length)
            m_imageIndex = 0;
        buildBounds();
    }
                                        
    // ------------------------------------------------------------------------
    // private usage
    // ------------------------------------------------------------------------
    
    protected void buildBounds()
    {
        m_bounds.clear();
        String[] bounds = new String[DEMOIMAGES.length];
        for (int ii=0; ii<DEMOIMAGES.length; ii++)
        {
            int si = (ii - this.m_imageIndex + DEMOIMAGES.length) % DEMOIMAGES.length;
            Rectangle r = new Rectangle(150-si*50,99 - si*33,500,375);
            bounds[ii] = (ValueManager.encodeRectangle(r)+";"+(10-si));
        }
        for (String bound: bounds)
            m_bounds.add(bound);
        m_trigger.trigger();
    }
    
}
