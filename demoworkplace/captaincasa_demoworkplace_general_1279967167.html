The context of this demo is the creation of a dialog using the CaptainCasa framework for the following purpose:

  • General
  • Long Operations
  • Separate thread

Using a WEBSOCKETPOLLING component is the easiest and most efficient way to set up some processing, in which events from the server side are published to the client.

Imaging the WEBSOCKETPOLLING component to be an invisible button, which always activates its actionListener once a server side event occurs. The connection between the "button" and the server side event is a WebSocket connection.

This is the XML layout definition:

This is the Java code:

package workplace;

import java.io.Serializable;

import org.eclnt.editor.annotations.CCGenClass;
import org.eclnt.jsfserver.elements.util.Trigger;
import org.eclnt.jsfserver.polling.LongPollingMgr;
import org.eclnt.jsfserver.polling.websocket.DefaultLongPollingWebSocket;
import org.eclnt.workplace.IWorkpageDispatcher;
import org.eclnt.workplace.WorkpageDefaultLifecycleListener;
import org.eclnt.workplace.WorkpageDispatchedPageBean;

@CCGenClass (expressionBase="#{d.DemoWebSocketPolling}")

public class DemoWebSocketPolling
    extends PageBean 
    implements Serializable
{
    // ------------------------------------------------------------------------
    // inner classes
    // ------------------------------------------------------------------------

    /**
     * This is some server side "event producing thread". 
     */
    public class MyThread extends Thread
    {
        boolean i_threadStop = false;
        @Override
        public void run()
        {
            for (int i=0; i<100; i++)
            {
                try { Thread.sleep(500); } catch (Throwable t) {}
                if (i_threadStop == true) break;
                m_counter++;
                // waking up == sending an event to the client side
                m_longPollingWebSocket.wakeup(true);
            }
            if (m_thread == this) m_thread = null;
        }
    }
    
    // ------------------------------------------------------------------------
    // members
    // ------------------------------------------------------------------------
    
    /**
     * Server side part of the web socket management.
     */
    DefaultLongPollingWebSocket m_longPollingWebSocket;
    
    MyThread m_thread;
    int m_counter = 0;
    String m_protocol = "Protocol:\n";

    // ------------------------------------------------------------------------
    // constructors & initialization
    // ------------------------------------------------------------------------

    public DemoWebSocketPolling(IWorkpageDispatcher workpageDispatcher)
    {
        super(workpageDispatcher);        
        // first creat the web socket part - then register!
        m_longPollingWebSocket = new DefaultLongPollingWebSocket();
        LongPollingMgr.add(m_longPollingWebSocket);
        // if registering something you have to de-register. In this example: de-registration
        // is done when closing the workpage in the workplace. BUT: there may be other, better
        // points of de-registration in your scenario.
        getWorkpage().addLifecycleListener(new WorkpageDefaultLifecycleListener()
        {
            @Override
            public void reactOnDestroyed()
            {
                if (m_thread != null)
                    m_thread.i_threadStop = true;
                LongPollingMgr.remove(m_longPollingWebSocket);
            }
        });
        m_thread = new MyThread();
        m_thread.start();
    }

    public String getPageName() { return "/workplace/demowebsocketpolling.jsp"; }
    public String getRootExpressionUsedInPage() { return "#{d.DemoWebSocketPolling}"; }

    // ------------------------------------------------------------------------
    // public usage
    // ------------------------------------------------------------------------
    
    /**
     * The URL for the web socket is passed back by the {@link DefaultLongPollingWebSocket} instance.
     */
    public String getWebSocketUrl() { return m_longPollingWebSocket.getURL(); }
    
    public String getProtocol() { return m_protocol; }
    public void setProtocol(String value) { this.m_protocol = value; }

    /**
     * This is the actionListener of the WEBSOCKETPOLLING component. The function
     * is called every time an event is communicated to the client side via web socket.
     */
    public void onWebSocketAction(javax.faces.event.ActionEvent event) 
    {
        m_protocol += "\nCurrent counter: " + m_counter; 
    }

    public void onStopAction(javax.faces.event.ActionEvent event) 
    {
        if (m_thread != null)
            m_thread.i_threadStop = true;
        m_thread = null;
    }

    public void onRestartAction(javax.faces.event.ActionEvent event) 
    {
        if (m_thread != null)
            return;
        m_thread = new MyThread();
        m_thread.start();
    }
}
